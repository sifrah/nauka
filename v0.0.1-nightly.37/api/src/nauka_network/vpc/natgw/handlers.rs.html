8pub fn resource_def() -> ResourceDef {
9    ResourceDef::build(
10        "nat-gateway",
11        "Manage NAT gateways for outbound internet access",
12    )
13    .plural("nat-gateways")
14    .parent("org", "--org", "Organization")
15    .parent("vpc", "--vpc", "VPC")
16    .create()
17    .op(|op| {
18        op.with_arg(OperationArg::optional(
19            "hypervisor",
20            FieldDef::string(
21                "hypervisor",
22                "Hypervisor to provision on (auto-selected if omitted)",
23            ),
24        ))
25        .with_example("nauka vpc nat-gateway create egress --vpc prod-net --org acme")
26    })
27    .list()
28    .get()
29    .delete()
30    .column("NAME", "name")
31    .column("VPC", "vpc_name")
32    .column("PUBLIC IPv6", "public_ipv6")
33    .column_def(ColumnDef::new("STATE", "state").with_format(DisplayFormat::Status))
34    .column("ID", "id")
35    .column_def(ColumnDef::new("CREATED", "created_at").with_format(DisplayFormat::Timestamp))
36    .empty_message("No NAT gateways found.")
37    .detail_section(
38        None,
39        vec![
40            DetailField::new("Name", "name"),
41            DetailField::new("ID", "id"),
42            DetailField::new("VPC", "vpc_name"),
43            DetailField::new("Public IPv6", "public_ipv6"),
44            DetailField::new("Hypervisor", "hypervisor_id"),
45            DetailField::new("State", "state").with_format(DisplayFormat::Status),
46            DetailField::new("Created", "created_at").with_format(DisplayFormat::Timestamp),
47        ],
48    )
49    .done()
50}

52pub fn handler() -> HandlerFn {
53    Box::new(
54        |req: OperationRequest| -> Pin<
55            Box<dyn Future<Output = anyhow::Result<OperationResponse>> + Send>,
56        > {
57            Box::pin(async move {
58                let store = NatGwStore::new(nauka_hypervisor::controlplane::connect().await?);
59                match req.operation.as_str() {
60                    "create" => {
61                        let name = req.name.ok_or_else(|| anyhow::anyhow!("missing name"))?;
62                        let vpc = req
63                            .scope
64                            .get("vpc")
65                            .ok_or_else(|| anyhow::anyhow!("--vpc is required"))?
66                            .to_string();
67                        let org = req
68                            .scope
69                            .get("org")
70                            .ok_or_else(|| anyhow::anyhow!("--org is required"))?
71                            .to_string();
72
73                        nauka_core::validate::name(&name)?;
74
75                        // Resolve hypervisor with an IPv6 block
76                        let (hv_id, ipv6_block) = resolve_hypervisor(
77                            req.fields.get("hypervisor").map(|s| s.as_str()),
78                        )?;
79
80                        let natgw = store
81                            .create(&name, &vpc, &org, &hv_id, &ipv6_block)
82                            .await?;
83                        Ok(OperationResponse::Resource(natgw.to_api_json()))
84                    }
85                    "list" => {
86                        let vpc = req.scope.get("vpc").map(|s| s.to_string());
87                        let natgws = store.list(vpc.as_deref()).await?;
88                        let items: Vec<serde_json::Value> =
89                            natgws.iter().map(|n| n.to_api_json()).collect();
90                        Ok(OperationResponse::ResourceList(items))
91                    }
92                    "get" => {
93                        let name = req
94                            .name
95                            .ok_or_else(|| anyhow::anyhow!("missing name or ID"))?;
96                        let vpc = req.scope.get("vpc").map(|s| s.to_string());
97                        let org = req.scope.get("org").map(|s| s.to_string());
98                        let natgw = store
99                            .get(&name, vpc.as_deref(), org.as_deref())
100                            .await?
101                            .ok_or_else(|| anyhow::anyhow!("nat-gateway '{name}' not found"))?;
102                        Ok(OperationResponse::Resource(natgw.to_api_json()))
103                    }
104                    "delete" => {
105                        let name = req
106                            .name
107                            .ok_or_else(|| anyhow::anyhow!("missing name or ID"))?;
108                        let vpc = req
109                            .scope
110                            .get("vpc")
111                            .ok_or_else(|| anyhow::anyhow!("--vpc is required"))?
112                            .to_string();
113                        let org = req
114                            .scope
115                            .get("org")
116                            .ok_or_else(|| anyhow::anyhow!("--org is required"))?
117                            .to_string();
118                        store.delete(&name, &vpc, &org).await?;
119                        Ok(OperationResponse::Message(format!(
120                            "nat-gateway '{name}' deleted."
121                        )))
122                    }
123                    other => Ok(OperationResponse::Message(format!("unknown: {other}"))),
124                }
125            })
126        },
127    )
128}

138/// Resolve the hypervisor to use for the NAT gateway.
139///
140/// Reads the local fabric state to find a hypervisor with an ipv6_block.
141/// If `preferred` is provided, validates it matches the local node.
142fn resolve_hypervisor(preferred: Option<&str>) -> anyhow::Result<(String, String)> {
143    let dir = nauka_core::process::nauka_dir();
144    let _ = std::fs::create_dir_all(&dir);
145    let db = nauka_state::LayerDb::open("hypervisor")
146        .map_err(|e| anyhow::anyhow!("cannot open hypervisor state: {e}"))?;
147    let state = nauka_hypervisor::fabric::state::FabricState::load(&db)
148        .map_err(|e| anyhow::anyhow!("{e}"))?
149        .ok_or_else(|| {
150            anyhow::anyhow!("hypervisor not initialized. Run 'nauka hypervisor init' first.")
151        })?;
152
153    let hv = &state.hypervisor;
154
155    if let Some(pref) = preferred {
156        if pref != hv.name && pref != hv.id.as_str() {
157            anyhow::bail!(
158                "hypervisor '{pref}' is not the local node. NAT gateways are provisioned locally."
159            );
160        }
161    }
162
163    let ipv6_block = hv.ipv6_block.as_ref().ok_or_else(|| {
164        anyhow::anyhow!(
165            "hypervisor '{}' has no IPv6 block configured. \
166             Re-initialize with --ipv6-block to enable NAT gateways.",
167            hv.name
168        )
169    })?;
170
171    Ok((hv.id.to_string(), ipv6_block.clone()))
172}
