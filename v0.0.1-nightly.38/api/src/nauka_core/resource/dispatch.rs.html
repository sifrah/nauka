13/// Dispatch a parsed CLI invocation to the appropriate resource handler.
14///
15/// Pipeline:
16/// 1. Extract raw values from clap matches
17/// 2. Validate constraints → produce ValidatedRequest
18/// 3. Handle confirmation prompt
19/// 4. Call handler
20/// 5. Render output
21pub async fn dispatch(
22    reg: &ResourceRegistration,
23    op_name: &str,
24    matches: &ArgMatches,
25) -> anyhow::Result<()> {
26    // Check if op_name is a child resource (e.g., "project" under "org")
27    if let Some(child) = reg
28        .children
29        .iter()
30        .find(|c| c.def.identity.cli_name == op_name || c.def.identity.aliases.contains(&op_name))
31    {
32        let (child_op, child_matches) = matches.subcommand().expect("subcommand enforced by clap");
33        return Box::pin(dispatch(child, child_op, child_matches)).await;
34    }
35
36    let def = &reg.def;
37    let op = def
38        .operations
39        .iter()
40        .find(|o| o.name == op_name)
41        .ok_or_else(|| anyhow::anyhow!("unknown operation: {op_name}"))?;
42
43    // ── 1. Extract raw values ──
44
45    let name = matches
46        .try_get_one::<String>("name")
47        .ok()
48        .flatten()
49        .cloned();
50    let json = matches
51        .try_get_one::<bool>("json")
52        .ok()
53        .flatten()
54        .copied()
55        .unwrap_or(false);
56    let yes = matches
57        .try_get_one::<bool>("yes")
58        .ok()
59        .flatten()
60        .copied()
61        .unwrap_or(false);
62
63    let scope = extract_scope(def, matches);
64    let mut fields = extract_fields(def, op, matches);
65
66    // ── 1b. Interactive prompts for missing required args ──
67    //
68    // If stdin is a TTY and required args are missing, prompt interactively
69    // instead of returning an error. Non-TTY (pipe/script) still fails fast.
70
71    if crate::ui::prompt::is_interactive() && !json {
72        // Check operation-specific args
73        for arg in &op.args {
74            if arg.required && !fields.contains_key(arg.name) {
75                if let super::operation::ArgSource::Custom(ref field_def) = arg.source {
76                    if let Ok(Some(val)) = crate::ui::prompt::prompt_field(
77                        arg.name,
78                        arg.description,
79                        &field_def.field_type,
80                        field_def.default,
81                    ) {
82                        fields.insert(arg.name.to_string(), val);
83                    }
84                }
85            }
86        }
87
88        // Check schema fields that are required for create
89        if matches!(op.semantics, super::operation::OperationSemantics::Create) {
90            for field in &def.schema.fields {
91                if field.mutability != crate::resource::Mutability::ReadOnly
92                    && field.mutability != crate::resource::Mutability::Internal
93                    && field.default.is_none()
94                    && !fields.contains_key(field.name)
95                {
96                    if let Ok(Some(val)) = crate::ui::prompt::prompt_field(
97                        field.name,
98                        field.description,
99                        &field.field_type,
100                        field.default,
101                    ) {
102                        fields.insert(field.name.to_string(), val);
103                    }
104                }
105            }
106        }
107    }
108
109    // ── 2. Validate constraints → ValidatedRequest ──
110
111    for constraint in &op.constraints {
112        constraint
113            .validate(&fields)
114            .map_err(|e| anyhow::anyhow!("{e}"))?;
115    }
116
117    let raw_request = OperationRequest {
118        operation: op_name.to_string(),
119        name,
120        scope,
121        fields,
122    };
123
124    let validated = ValidatedRequest::from_raw(def.identity.kind, raw_request.clone());
125
126    // ── 3. Confirmation prompt ──
127
128    if op.confirmable && !yes {
129        let resource_name = raw_request.name.as_deref().unwrap_or("this resource");
130        let prompt = format!(
131            "{} {} '{}'? This cannot be undone. [y/N] ",
132            capitalize(op.name),
133            def.identity.kind,
134            resource_name
135        );
136        print!("{prompt}");
137        io::stdout().flush()?;
138        let mut input = String::new();
139        io::stdin().read_line(&mut input)?;
140        if !input.trim().eq_ignore_ascii_case("y") {
141            println!("Aborted.");
142            return Ok(());
143        }
144    }
145
146    // ── 4. Call handler with validated request ──
147
148    let _ = &validated; // available for future use
149    let response = (reg.handler)(raw_request).await?;
150
151    // ── 5. Render output ──
152
153    let force_json = json || matches!(op.output.kind, OutputKind::JsonOnly);
154
155    match (&op.output.kind, force_json) {
156        (_, true) => match &response {
157            OperationResponse::Resource(v) => {
158                println!("{}", serde_json::to_string_pretty(v)?);
159            }
160            OperationResponse::ResourceList(items) => {
161                println!("{}", serde_json::to_string_pretty(items)?);
162            }
163            OperationResponse::Message(msg) => {
164                println!("{}", serde_json::json!({"message": msg}));
165            }
166            OperationResponse::None => {}
167        },
168        (OutputKind::Resource, false) => {
169            if let OperationResponse::Resource(v) = &response {
170                render_detail(def, v);
171            }
172            if let Some(tpl) = op.output.success_message {
173                let name_val: String = match &response {
174                    OperationResponse::Resource(v) => extract_name(v).unwrap_or_default(),
175                    _ => String::new(),
176                };
177                let rendered = tpl
178                    .replace("{kind}", def.identity.kind)
179                    .replace("{name}", &name_val);
180                println!("{rendered}");
181            }
182        }
183        (OutputKind::ResourceList, false) => {
184            if let OperationResponse::ResourceList(items) = &response {
185                render_table(def, items);
186            }
187        }
188        (OutputKind::Message, false) => {
189            if let OperationResponse::Message(msg) = &response {
190                println!("{msg}");
191            } else if let Some(tpl) = op.output.success_message {
192                let rendered: String = tpl.replace("{kind}", def.identity.kind);
193                println!("{rendered}");
194            }
195        }
196        (OutputKind::None, false) | (OutputKind::JsonOnly, false) => {}
197    }
198
199    Ok(())
200}

202fn extract_scope(def: &ResourceDef, matches: &ArgMatches) -> ScopeValues {
203    let mut scope = ScopeValues::default();
204    for parent in &def.scope.parents {
205        if let Some(val) = matches.get_one::<String>(parent.kind) {
206            scope.set(parent.kind, val.clone());
207        }
208    }
209    scope
210}

212fn extract_fields(
213    def: &ResourceDef,
214    op: &super::operation::OperationDef,
215    matches: &ArgMatches,
216) -> FieldMap {
217    let mut fields = HashMap::new();
218
219    // Schema fields
220    for field in &def.schema.fields {
221        if let Some(val) = matches.try_get_one::<String>(field.name).ok().flatten() {
222            fields.insert(field.name.to_string(), val.clone());
223        }
224        if matches!(field.field_type, super::schema::FieldType::Flag)
225            && matches
226                .try_get_one::<bool>(field.name)
227                .ok()
228                .flatten()
229                .copied()
230                .unwrap_or(false)
231        {
232            fields.insert(field.name.to_string(), "true".to_string());
233        }
234    }
235
236    // Operation-specific args
237    for arg in &op.args {
238        if let super::operation::ArgSource::Custom(field) = &arg.source {
239            if let Some(val) = matches.try_get_one::<String>(arg.name).ok().flatten() {
240                fields.insert(arg.name.to_string(), val.clone());
241            }
242            if matches!(field.field_type, super::schema::FieldType::Flag)
243                && matches
244                    .try_get_one::<bool>(arg.name)
245                    .ok()
246                    .flatten()
247                    .copied()
248                    .unwrap_or(false)
249            {
250                fields.insert(arg.name.to_string(), "true".to_string());
251            }
252        }
253    }
254
255    fields
256}

258fn extract_name(value: &serde_json::Value) -> Option<String> {
259    value
260        .get("name")
261        .and_then(|v| v.as_str())
262        .map(|s| s.to_string())
263}

265fn capitalize(s: &str) -> String {
266    let mut chars = s.chars();
267    match chars.next() {
268        None => String::new(),
269        Some(c) => c.to_uppercase().to_string() + chars.as_str(),
270    }
