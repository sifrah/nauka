12// 5. Request ID

14
15static REQUEST_COUNTER: AtomicU64 = AtomicU64::new(0);
16
17/// Middleware that injects `X-Request-Id` header into every response.
18pub async fn request_id(req: Request, next: Next) -> Response {
19    let id = REQUEST_COUNTER.fetch_add(1, Ordering::Relaxed);
20    let request_id = format!("req-{id:012x}");
21
22    let mut resp = next.run(req).await;
23    resp.headers_mut().insert(
24        "x-request-id",
25        HeaderValue::from_str(&request_id).unwrap_or_else(|_| HeaderValue::from_static("unknown")),
26    );
27    resp
28}

34/// Middleware that injects `X-Nauka-Version` header.
35pub async fn version_header(req: Request, next: Next) -> Response {
36    let mut resp = next.run(req).await;
37    resp.headers_mut().insert(
38        "x-nauka-version",
39        HeaderValue::from_static(env!("CARGO_PKG_VERSION")),
40    );
41    resp
42}

48/// Simple in-memory rate limiter.
49#[derive(Clone)]
50pub struct RateLimiter {
51    max_requests: u64,
52    window_secs: u64,
53    state: Arc<RateLimitState>,
54}
55
56struct RateLimitState {
57    count: AtomicU64,
58    window_start: std::sync::Mutex<Instant>,
59}
60
61impl RateLimiter {
62    /// Create a new rate limiter.
63    pub fn new(max_requests: u64, window_secs: u64) -> Self {
64        Self {
65            max_requests,
66            window_secs,
67            state: Arc::new(RateLimitState {
68                count: AtomicU64::new(0),
69                window_start: std::sync::Mutex::new(Instant::now()),
70            }),
71        }
72    }
73
74    /// Check if a request is allowed. Returns remaining requests.
75    /// Check if a request is allowed. Returns remaining requests or false if rejected.
76    #[allow(clippy::result_unit_err)]
77    pub fn check(&self) -> Result<u64, ()> {
78        let mut start = self.state.window_start.lock().unwrap();
79        let elapsed = start.elapsed().as_secs();
80
81        // Reset window if expired
82        if elapsed >= self.window_secs {
83            *start = Instant::now();
84            self.state.count.store(1, Ordering::Relaxed);
85            return Ok(self.max_requests - 1);
86        }
87
88        let count = self.state.count.fetch_add(1, Ordering::Relaxed) + 1;
89        if count > self.max_requests {
90            Err(())
91        } else {
92            Ok(self.max_requests - count)
93        }
94    }
95
96    /// Reset the limiter (for testing).
97    pub fn reset(&self) {
98        self.state.count.store(0, Ordering::Relaxed);
99        *self.state.window_start.lock().unwrap() = Instant::now();
100    }
101}

107/// Standard pagination parameters extracted from query string.
108#[derive(Debug, Clone, serde::Deserialize)]
109pub struct Pagination {
110    /// Number of items to return (default 50, max 200).
111    #[serde(default = "default_limit")]
112    pub limit: u64,
113    /// Offset for pagination (default 0).
114    #[serde(default)]
115    pub offset: u64,
116    /// Sort field.
117    #[serde(default)]
118    pub sort: Option<String>,
119    /// Sort order: "asc" or "desc".
120    #[serde(default = "default_order")]
121    pub order: String,
122}
123
124fn default_limit() -> u64 {
125    50
126}
127fn default_order() -> String {
128    "asc".into()
129}
130
131impl Pagination {
132    /// Clamp limit to max 200.
133    pub fn clamped_limit(&self) -> u64 {
134        self.limit.min(200)
135    }
136}
137
138/// Paginated list response.
139#[derive(Debug, serde::Serialize)]
140pub struct PaginatedResponse<T: serde::Serialize> {
141    pub items: Vec<T>,
142    pub count: usize,
143    pub total: u64,
144    pub offset: u64,
145    pub limit: u64,
146    pub has_more: bool,
147}
148
149impl<T: serde::Serialize> PaginatedResponse<T> {
150    pub fn new(items: Vec<T>, total: u64, pagination: &Pagination) -> Self {
151        let count = items.len();
152        let has_more = pagination.offset + (count as u64) < total;
153        Self {
154            items,
155            count,
156            total,
157            offset: pagination.offset,
158            limit: pagination.clamped_limit(),
159            has_more,
160        }
161    }
162}
