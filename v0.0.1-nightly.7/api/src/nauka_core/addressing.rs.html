1//! IPv6 ULA addressing for the mesh.
2//!
3//! Each mesh has a random `/48` ULA prefix: `fdXX:XXXX:XXXX::/48`
4//! Each node gets a `/128` address derived from the prefix + its WireGuard public key.
5//!
6//! ```
7//! use nauka_core::addressing;
8//!
9//! let prefix = addressing::generate_mesh_prefix();
10//! let node_addr = addressing::derive_node_address(&prefix, b"wg-public-key-bytes");
11//! assert!(node_addr.to_string().starts_with("fd"));
12//! ```
13
14use sha2::{Digest, Sha256};
15use std::net::Ipv6Addr;
16
17/// Generate a random ULA /48 mesh prefix: `fdXX:XXXX:XXXX::/48`
18///
19/// Uses OS entropy (OsRng) for cryptographic addressing material.
20pub fn generate_mesh_prefix() -> Ipv6Addr {
21    let mut rng_bytes = [0u8; 5];
22    rand::RngCore::fill_bytes(&mut rand::rngs::OsRng, &mut rng_bytes);
23
24    Ipv6Addr::new(
25        0xfd00 | (rng_bytes[0] as u16),
26        ((rng_bytes[1] as u16) << 8) | (rng_bytes[2] as u16),
27        ((rng_bytes[3] as u16) << 8) | (rng_bytes[4] as u16),
28        0,
29        0,
30        0,
31        0,
32        0,
33    )
34}
35
36/// Derive a node's ULA /128 address from the mesh prefix and WireGuard public key.
37///
38/// Takes SHA256(wg_pubkey) and fills the lower 80 bits (segments 3-7),
39/// keeping the /48 prefix (segments 0-2) intact.
40pub fn derive_node_address(mesh_prefix: &Ipv6Addr, wg_public_key: &[u8]) -> Ipv6Addr {
41    let hash = Sha256::digest(wg_public_key);
42    let prefix = mesh_prefix.segments();
43
44    Ipv6Addr::new(
45        prefix[0],
46        prefix[1],
47        prefix[2],
48        ((hash[0] as u16) << 8) | (hash[1] as u16),
49        ((hash[2] as u16) << 8) | (hash[3] as u16),
50        ((hash[4] as u16) << 8) | (hash[5] as u16),
51        ((hash[6] as u16) << 8) | (hash[7] as u16),
52        ((hash[8] as u16) << 8) | (hash[9] as u16),
53    )
54}
55
56/// Extract the /48 prefix from a full /128 address.
57pub fn extract_prefix(addr: &Ipv6Addr) -> Ipv6Addr {
58    let s = addr.segments();
59    Ipv6Addr::new(s[0], s[1], s[2], 0, 0, 0, 0, 0)
60}
61
62/// Format a /48 prefix for display: `fdXX:XXXX:XXXX::/48`
63pub fn format_prefix(prefix: &Ipv6Addr) -> String {
64    let s = prefix.segments();
65    format!("{:x}:{:x}:{:x}::/48", s[0], s[1], s[2])
66}
67
68/// Check if an address is within a given /48 prefix.
69pub fn is_in_prefix(addr: &Ipv6Addr, prefix: &Ipv6Addr) -> bool {
70    let a = addr.segments();
71    let p = prefix.segments();
72    a[0] == p[0] && a[1] == p[1] && a[2] == p[2]
73}

75#[cfg(test)]
76mod tests {
77    use super::*;
78
79    #[test]
80    fn prefix_is_ula() {
81        let p = generate_mesh_prefix();
82        let first = p.segments()[0];
83        assert!((0xfd00..=0xfdff).contains(&first), "not ULA: {p}");
84    }
85
86    #[test]
87    fn prefix_unique() {
88        let a = generate_mesh_prefix();
89        let b = generate_mesh_prefix();
90        assert_ne!(a, b);
91    }
92
93    #[test]
94    fn prefix_lower_bits_zero() {
95        let p = generate_mesh_prefix();
96        let s = p.segments();
97        assert_eq!(s[3], 0);
98        assert_eq!(s[4], 0);
99        assert_eq!(s[5], 0);
100        assert_eq!(s[6], 0);
101        assert_eq!(s[7], 0);
102    }
103
104    #[test]
105    fn derive_address_keeps_prefix() {
106        let prefix = generate_mesh_prefix();
107        let addr = derive_node_address(&prefix, b"test-key");
108        assert!(is_in_prefix(&addr, &prefix));
109    }
110
111    #[test]
112    fn derive_address_deterministic() {
113        let prefix = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
114        let a = derive_node_address(&prefix, b"key1");
115        let b = derive_node_address(&prefix, b"key1");
116        assert_eq!(a, b);
117    }
118
119    #[test]
120    fn derive_address_different_keys() {
121        let prefix = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
122        let a = derive_node_address(&prefix, b"key1");
123        let b = derive_node_address(&prefix, b"key2");
124        assert_ne!(a, b);
125    }
126
127    #[test]
128    fn derive_has_nonzero_lower_bits() {
129        let prefix = generate_mesh_prefix();
130        let addr = derive_node_address(&prefix, b"some-key");
131        let s = addr.segments();
132        // At least some of the lower segments should be non-zero
133        assert!(s[3] != 0 || s[4] != 0 || s[5] != 0);
134    }
135
136    #[test]
137    fn extract_prefix_works() {
138        let prefix = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
139        let addr = derive_node_address(&prefix, b"key");
140        let extracted = extract_prefix(&addr);
141        assert_eq!(extracted, prefix);
142    }
143
144    #[test]
145    fn format_prefix_display() {
146        let p = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
147        assert_eq!(format_prefix(&p), "fd01:2bf2:852d::/48");
148    }
149
150    #[test]
151    fn is_in_prefix_positive() {
152        let p = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
153        let a = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0x1234, 0x5678, 0, 0, 1);
154        assert!(is_in_prefix(&a, &p));
155    }
156
157    #[test]
158    fn is_in_prefix_negative() {
159        let p = Ipv6Addr::new(0xfd01, 0x2bf2, 0x852d, 0, 0, 0, 0, 0);
160        let a = Ipv6Addr::new(0xfd02, 0x2bf2, 0x852d, 0, 0, 0, 0, 1);
161        assert!(!is_in_prefix(&a, &p));
162    }
