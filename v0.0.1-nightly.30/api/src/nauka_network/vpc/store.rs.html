7const NS_VPC: &str = "vpc";
8const NS_VPC_IDX: &str = "vpc-idx";
9const REG_VPCS: (&str, &str) = ("_reg", "vpc-ids");
10const VNI_COUNTER: (&str, &str) = ("_reg", "vni-counter");
11const VNI_START: u32 = 100;
12
13pub struct VpcStore {
14    db: ClusterDb,
15}

17impl VpcStore {
18    pub fn new(db: ClusterDb) -> Self {
19        Self { db }
20    }

22    async fn next_vni(&self) -> anyhow::Result<u32> {
23        let current: Option<u32> = self.db.get(VNI_COUNTER.0, VNI_COUNTER.1).await?;
24        let vni = current.unwrap_or(VNI_START);
25        self.db
26            .put(VNI_COUNTER.0, VNI_COUNTER.1, &(vni + 1))
27            .await?;
28        Ok(vni)
29    }

31    pub async fn create(
32        &self,
33        name: &str,
34        org_name: &str,
35        cidr: &str,
36        project_id: Option<String>,
37        env_id: Option<String>,
38    ) -> anyhow::Result<Vpc> {
39        // Resolve org
40        let org_store = nauka_org::store::OrgStore::new(self.db.clone());
41        let org = org_store
42            .get(org_name)
43            .await?
44            .ok_or_else(|| anyhow::anyhow!("org '{org_name}' not found"))?;
45
46        // Validate CIDR
47        crate::validate::private_cidr(cidr)?;
48
49        // Check uniqueness within org
50        let idx_key = format!("{}/{}", org.meta.id, name);
51        let existing: Option<String> = self.db.get(NS_VPC_IDX, &idx_key).await?;
52        if existing.is_some() {
53            anyhow::bail!("vpc '{name}' already exists in org '{org_name}'");
54        }
55
56        let vni = self.next_vni().await?;
57
58        let vpc = Vpc {
59            meta: ResourceMeta::new(VpcId::generate().to_string(), name),
60            cidr: cidr.to_string(),
61            org_id: org.meta.id.clone().into(),
62            org_name: org.meta.name.clone(),
63            vni,
64            project_id: project_id.map(|s| s.into()),
65            env_id: env_id.map(|s| s.into()),
66        };
67
68        self.db.put(NS_VPC, &vpc.meta.id, &vpc).await?;
69        self.db.put(NS_VPC_IDX, &idx_key, &vpc.meta.id).await?;
70        add_id(&self.db, &vpc.meta.id).await?;
71
72        Ok(vpc)
73    }

75    pub async fn get(
76        &self,
77        name_or_id: &str,
78        org_name: Option<&str>,
79    ) -> anyhow::Result<Option<Vpc>> {
80        if VpcId::looks_like_id(name_or_id) {
81            return self.db.get(NS_VPC, name_or_id).await.map_err(Into::into);
82        }
83        let org_name =
84            org_name.ok_or_else(|| anyhow::anyhow!("--org required to resolve VPC by name"))?;
85        let org_store = nauka_org::store::OrgStore::new(self.db.clone());
86        let org = org_store
87            .get(org_name)
88            .await?
89            .ok_or_else(|| anyhow::anyhow!("org '{org_name}' not found"))?;
90        let idx_key = format!("{}/{}", org.meta.id, name_or_id);
91        let id: Option<String> = self.db.get(NS_VPC_IDX, &idx_key).await?;
92        match id {
93            Some(id) => self.db.get(NS_VPC, &id).await.map_err(Into::into),
94            None => Ok(None),
95        }
96    }

98    pub async fn list(&self, org_name: Option<&str>) -> anyhow::Result<Vec<Vpc>> {
99        let ids = load_ids(&self.db).await?;
100        let mut vpcs = Vec::new();
101        for id in &ids {
102            if let Some(v) = self.db.get::<Vpc>(NS_VPC, id).await? {
103                vpcs.push(v);
104            }
105        }
106        match org_name {
107            Some(name) => Ok(vpcs
108                .into_iter()
109                .filter(|v| v.org_name == name || v.org_id.as_str() == name)
110                .collect()),
111            None => Ok(vpcs),
112        }
113    }
114
115    pub async fn delete(&self, name_or_id: &str, org_name: &str) -> anyhow::Result<()> {
116        let vpc = self
117            .get(name_or_id, Some(org_name))
118            .await?
119            .ok_or_else(|| anyhow::anyhow!("vpc '{name_or_id}' not found in org '{org_name}'"))?;
120
121        // Check for child subnets
122        let sub_store = super::subnet::store::SubnetStore::new(self.db.clone());
123        let subs = sub_store.list(Some(&vpc.meta.name), Some(org_name)).await?;
124        if !subs.is_empty() {
125            anyhow::bail!(
126                "vpc '{}' has {} subnet(s). Delete them first.",
127                vpc.meta.name,
128                subs.len()
129            );
130        }
131
132        let idx_key = format!("{}/{}", vpc.org_id.as_str(), vpc.meta.name);
133        self.db.delete(NS_VPC, &vpc.meta.id).await?;
134        self.db.delete(NS_VPC_IDX, &idx_key).await?;
135        remove_id(&self.db, &vpc.meta.id).await?;
136        Ok(())
137    }
138}

140async fn load_ids(db: &ClusterDb) -> anyhow::Result<Vec<String>> {
141    let ids: Option<Vec<String>> = db.get(REG_VPCS.0, REG_VPCS.1).await?;
142    Ok(ids.unwrap_or_default())
143}
144
145async fn add_id(db: &ClusterDb, id: &str) -> anyhow::Result<()> {
146    let mut ids = load_ids(db).await?;
147    ids.push(id.to_string());
148    db.put(REG_VPCS.0, REG_VPCS.1, &ids).await?;
149    Ok(())
150}
151
152async fn remove_id(db: &ClusterDb, id: &str) -> anyhow::Result<()> {
153    let mut ids = load_ids(db).await?;
154    ids.retain(|i| i != id);
155    db.put(REG_VPCS.0, REG_VPCS.1, &ids).await?;
156    Ok(())
157}
