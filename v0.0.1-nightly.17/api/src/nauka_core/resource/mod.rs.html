31/// The complete definition of a cloud resource.
32/// This is the single source of truth from which CLI, API, validation,
33/// and documentation are all derived.
34#[derive(Debug, Clone)]
35pub struct ResourceDef {
36    pub identity: ResourceIdentity,
37    pub scope: ScopeDef,
38    pub schema: ResourceSchema,
