1//! VM reconciler — ensures VMs assigned to this node are running.
2
3use nauka_compute::vm::types::VmState;
4
5use crate::types::{ReconcileContext, ReconcileResult};
6
7pub struct VmReconciler;
8
9#[async_trait::async_trait]
10impl super::Reconciler for VmReconciler {
11    fn name(&self) -> &str {
12        "vm"
13    }
14
15    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
16        let mut result = ReconcileResult::new("vm");
17
18        // 1. Get VMs assigned to this node that should be running
19        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
20        let all_vms = vm_store.list(None, None, None).await?;
21        let local_vms: Vec<_> = all_vms
22            .iter()
23            .filter(|vm| vm.hypervisor_id.as_deref() == Some(&ctx.hypervisor_id))
24            .collect();
25
26        let should_run: Vec<_> = local_vms
27            .iter()
28            .filter(|vm| vm.state == VmState::Running)
29            .collect();
30
31        result.desired = should_run.len();
32
33        // 2. Check actual state (stub: no processes running)
34        let actual_processes = crate::observer::process::list_vms();
35        result.actual = actual_processes.len();
36
37        // 3. Diff and apply
38        for vm in &should_run {
39            if !actual_processes.iter().any(|p| p == &vm.meta.id) {
40                tracing::info!(
41                    vm_id = vm.meta.id.as_str(),
42                    vm_name = vm.meta.name.as_str(),
43                    image = vm.image.as_str(),
44                    vcpus = vm.vcpus,
45                    memory_mb = vm.memory_mb,
46                    "would start VM"
47                );
48                result.created += 1;
49            }
50        }
51
52        // 4. Check for orphaned processes (running but not in desired state)
53        for process_id in &actual_processes {
54            if !should_run.iter().any(|vm| vm.meta.id == *process_id) {
55                tracing::info!(vm_id = process_id, "would stop orphaned VM");
56                result.deleted += 1;
57            }
58        }
59
60        Ok(result)
61    }
62}
