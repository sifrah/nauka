1//! gVisor runtime — launches VMs as sandboxed containers (no KVM needed).
2//!
3//! Stub implementation: writes PID files to /run/nauka/vms/{vm_id}/pid.
4//! Future: spawns runsc with OCI bundle + network namespace.
5
6use std::path::PathBuf;
7
8use super::{RunningVm, Runtime, VmRunConfig};
9
10const VM_RUN_DIR: &str = "/run/nauka/vms";
11
12pub struct GVisorRuntime;
13
14impl Runtime for GVisorRuntime {
15    fn start(&self, config: &VmRunConfig) -> anyhow::Result<u32> {
16        let vm_dir = PathBuf::from(VM_RUN_DIR).join(&config.vm_id);
17        std::fs::create_dir_all(&vm_dir)?;
18
19        // TODO: Actually spawn runsc here with OCI bundle
20        tracing::info!(
21            vm_id = config.vm_id.as_str(),
22            vm_name = config.vm_name.as_str(),
23            vcpus = config.vcpus,
24            memory_mb = config.memory_mb,
25            image = config.image.as_str(),
26            tap = config.tap_name.as_str(),
27            ip = config.private_ip.as_str(),
28            "would start gVisor container (stub)"
29        );
30
31        let pid = std::process::id();
32        std::fs::write(vm_dir.join("pid"), pid.to_string())?;
33        std::fs::write(vm_dir.join("runtime"), "container")?;
34
35        Ok(pid)
36    }
37
38    fn stop(&self, vm_id: &str) -> anyhow::Result<()> {
39        let vm_dir = PathBuf::from(VM_RUN_DIR).join(vm_id);
40        if vm_dir.exists() {
41            tracing::info!(vm_id, "stopping container (stub)");
42            let _ = std::fs::remove_dir_all(&vm_dir);
43        }
44        Ok(())
45    }
46
47    fn is_running(&self, vm_id: &str) -> Option<u32> {
48        let pid_path = PathBuf::from(VM_RUN_DIR).join(vm_id).join("pid");
49        std::fs::read_to_string(pid_path)
50            .ok()
51            .and_then(|s| s.trim().parse::<u32>().ok())
52            .filter(|&pid| unsafe { libc::kill(pid as i32, 0) == 0 })
53    }
54
55    fn list_running(&self) -> Vec<RunningVm> {
56        let run_dir = PathBuf::from(VM_RUN_DIR);
57        let entries = match std::fs::read_dir(&run_dir) {
58            Ok(e) => e,
59            Err(_) => return vec![],
60        };
61
62        let mut vms = Vec::new();
63        for entry in entries.flatten() {
64            let path = entry.path();
65            if path.is_dir() {
66                let vm_id = entry.file_name().to_string_lossy().to_string();
67                let pid_path = path.join("pid");
68                if let Ok(pid_str) = std::fs::read_to_string(&pid_path) {
69                    if let Ok(pid) = pid_str.trim().parse::<u32>() {
70                        if unsafe { libc::kill(pid as i32, 0) == 0 } {
71                            vms.push(RunningVm { vm_id, pid });
72                        }
73                    }
74                }
75            }
76        }
77        vms
78    }
79}
