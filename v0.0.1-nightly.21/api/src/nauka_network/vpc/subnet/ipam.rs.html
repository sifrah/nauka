1//! IPAM — IP Address Management for subnets.
2//!
3//! Allocates private IPs from a subnet's CIDR range. Uses a simple
4//! list of allocated IPs stored in TiKV (one key per subnet).
5//!
6//! Reserved addresses:
7//! - .0 = network address
8//! - .1 = gateway (assigned to the bridge)
9//! - .255 = broadcast (for /24)
10
11use nauka_hypervisor::controlplane::ClusterDb;
12
13const NS_IPAM: &str = "ipam";
14
15/// Allocated IPs for a subnet, stored as a list in TiKV.
16#[derive(Debug, Clone, serde::Serialize, serde::Deserialize, Default)]
17struct Allocations {
18    ips: Vec<Allocation>,
19}
20
21#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
22struct Allocation {
23    ip: String,
24    vm_id: String,
25}
26
27/// Allocate the next available IP from a subnet.
28///
29/// Skips network (.0), gateway (.1), and broadcast addresses.
30/// Returns the allocated IP as a string (e.g., "10.0.1.2").
31pub async fn allocate(
32    db: &ClusterDb,
33    subnet_id: &str,
34    subnet_cidr: &str,
35    gateway: &str,
36    vm_id: &str,
37) -> anyhow::Result<String> {
38    let net: ipnet::Ipv4Net = subnet_cidr
39        .parse()
40        .map_err(|e| anyhow::anyhow!("invalid subnet CIDR: {e}"))?;
41
42    // Load current allocations
43    let mut allocs: Allocations = db.get(NS_IPAM, subnet_id).await?.unwrap_or_default();
44
45    // Check if this VM already has an IP
46    if let Some(existing) = allocs.ips.iter().find(|a| a.vm_id == vm_id) {
47        return Ok(existing.ip.clone());
48    }
49
50    let allocated_ips: Vec<&str> = allocs.ips.iter().map(|a| a.ip.as_str()).collect();
51
52    // Find next available host IP
53    for host_ip in net.hosts() {
54        let ip_str = host_ip.to_string();
55
56        // Skip gateway
57        if ip_str == gateway {
58            continue;
59        }
60
61        // Skip already allocated
62        if allocated_ips.contains(&ip_str.as_str()) {
63            continue;
64        }
65
66        // Found one — allocate it
67        allocs.ips.push(Allocation {
68            ip: ip_str.clone(),
69            vm_id: vm_id.to_string(),
70        });
71        db.put(NS_IPAM, subnet_id, &allocs).await?;
72
73        return Ok(ip_str);
74    }
75
76    anyhow::bail!(
77        "no IPs available in subnet {} (all {} addresses allocated)",
78        subnet_cidr,
79        allocs.ips.len()
80    )
81}
82
83/// Release an IP allocation for a VM.
84pub async fn release(db: &ClusterDb, subnet_id: &str, vm_id: &str) -> anyhow::Result<()> {
85    let mut allocs: Allocations = db.get(NS_IPAM, subnet_id).await?.unwrap_or_default();
86
87    allocs.ips.retain(|a| a.vm_id != vm_id);
88    db.put(NS_IPAM, subnet_id, &allocs).await?;
89
90    Ok(())
91}
92
93/// Get the allocated IP for a VM in a subnet.
94pub async fn get_allocation(
95    db: &ClusterDb,
96    subnet_id: &str,
97    vm_id: &str,
98) -> anyhow::Result<Option<String>> {
99    let allocs: Allocations = db.get(NS_IPAM, subnet_id).await?.unwrap_or_default();
100
101    Ok(allocs
102        .ips
103        .iter()
104        .find(|a| a.vm_id == vm_id)
105        .map(|a| a.ip.clone()))
106}

108#[cfg(test)]
109mod tests {
110    use super::*;
111
112    #[test]
113    fn allocations_serde_roundtrip() {
114        let allocs = Allocations {
115            ips: vec![
116                Allocation {
117                    ip: "10.0.1.2".to_string(),
118                    vm_id: "vm-01".to_string(),
119                },
120                Allocation {
121                    ip: "10.0.1.3".to_string(),
122                    vm_id: "vm-02".to_string(),
123                },
124            ],
125        };
126        let json = serde_json::to_string(&allocs).unwrap();
127        let back: Allocations = serde_json::from_str(&json).unwrap();
128        assert_eq!(back.ips.len(), 2);
129        assert_eq!(back.ips[0].ip, "10.0.1.2");
130    }
