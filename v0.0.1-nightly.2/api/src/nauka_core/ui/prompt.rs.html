1//! Interactive prompts for missing CLI arguments.
2
3use dialoguer::{Confirm, Input, Password, Select};
4
5use crate::resource::FieldType;
6
7type PromptResult<T> = Result<T, Box<dyn std::error::Error>>;
8
9/// Prompt for a string value with optional default.
10pub fn prompt_string(label: &str, default: Option<&str>) -> PromptResult<String> {
11    let mut input: Input<String> = Input::new().with_prompt(format!("  {label}"));
12    if let Some(d) = default {
13        input = input.default(d.to_string());
14    }
15    Ok(input.interact_text()?)
16}
17
18/// Prompt for an integer value with optional default.
19pub fn prompt_integer(label: &str, default: Option<&str>) -> PromptResult<String> {
20    let input: Input<String> = Input::new()
21        .with_prompt(format!("  {label}"))
22        .default(default.unwrap_or("0").to_string());
23    Ok(input.interact_text()?)
24}
25
26/// Prompt for a yes/no flag.
27pub fn prompt_flag(label: &str, default: bool) -> PromptResult<bool> {
28    Ok(Confirm::new()
29        .with_prompt(format!("  {label}"))
30        .default(default)
31        .interact()?)
32}
33
34/// Prompt for a secret/password value (masked input).
35pub fn prompt_secret(label: &str) -> PromptResult<String> {
36    Ok(Password::new()
37        .with_prompt(format!("  {label}"))
38        .interact()?)
39}
40
41/// Prompt for a selection from a list of options.
42pub fn prompt_enum(label: &str, options: &[&str], default: Option<usize>) -> PromptResult<String> {
43    let mut select = Select::new().with_prompt(format!("  {label}"));
44    for opt in options {
45        select = select.item(opt);
46    }
47    if let Some(d) = default {
48        select = select.default(d);
49    }
50    let idx = select.interact()?;
51    Ok(options[idx].to_string())
52}

54/// Prompt for a value based on its FieldType.
55pub fn prompt_field(
56    name: &str,
57    _description: &str,
58    field_type: &FieldType,
59    default: Option<&str>,
60) -> Result<Option<String>, Box<dyn std::error::Error>> {
61    match field_type {
62        FieldType::String
63        | FieldType::Cidr
64        | FieldType::IpAddr
65        | FieldType::Path
66        | FieldType::Duration
67        | FieldType::KeyValue
68        | FieldType::ResourceRef(_) => {
69            let val = prompt_string(name, default)?;
70            if val.is_empty() {
71                Ok(None)
72            } else {
73                Ok(Some(val))
74            }
75        }
76        FieldType::Integer | FieldType::Port | FieldType::SizeGb | FieldType::SizeMb => {
77            let val = prompt_integer(name, default)?;
78            Ok(Some(val))
79        }
80        FieldType::Flag => {
81            let val = prompt_flag(name, false)?;
82            if val {
83                Ok(Some("true".to_string()))
84            } else {
85                Ok(None)
86            }
87        }
88        FieldType::Secret => {
89            let val = prompt_secret(name)?;
90            if val.is_empty() {
91                Ok(None)
92            } else {
93                Ok(Some(val))
94            }
95        }
96        FieldType::Enum(e) => {
97            let default_idx = e
98                .default
99                .and_then(|d| e.values.iter().position(|v| *v == d));
100            let val = prompt_enum(name, e.values, default_idx)?;
101            Ok(Some(val))
102        }
103    }
104}

106/// Check if stdin is a TTY (interactive terminal).
107pub fn is_interactive() -> bool {
108    use std::io::IsTerminal;
109    std::io::stdin().is_terminal()
110}
