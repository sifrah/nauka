1//! Dump OpenAPI spec as JSON to stdout.
2//! Used by the docs build to generate the REST API reference.
3//!
4//! Layer crates are linked via `extern crate` to ensure their
5//! `inventory::submit!` registrations are collected. Adding a new
6//! layer to Cargo.toml + an extern crate line here is all it takes.
7
8// Force-link layer crates so inventory collects their registrations.
9extern crate nauka_compute;
10extern crate nauka_forge;
11extern crate nauka_hypervisor;
12extern crate nauka_network;
13extern crate nauka_org;
14
15mod registry;
16
17use nauka_core::api::openapi_spec;
18
19fn main() {
20    let registry = registry::build_registry();
21    let spec = openapi_spec(registry.as_slice(), "/admin/v1");
22    println!("{}", serde_json::to_string_pretty(&spec).unwrap());
23}
