1//! Spike binary for P0.1 (sifrah/nauka#185), repurposed in P1.2 to be the
2//! Hetzner-side smoke test for [`nauka_state::EmbeddedDb`].
3//!
4//! The spike's purpose evolves with each phase that touches the same code:
5//!
6//! - **P0.1**: prove that `surrealdb` 3.0.5 with `kv-surrealkv` (+ `kv-tikv`)
7//!   cross-compiles to `x86_64-unknown-linux-musl` and runs as a
8//!   statically-linked binary on a real Hetzner Ubuntu host. ✅ done.
9//! - **P1.1**: drop `kv-tikv` from the production feature set. ✅ done.
10//! - **P1.2**: exercise the new [`EmbeddedDb`] wrapper end-to-end on
11//!   Hetzner — open at a temp path, CRUD round-trip via `db.client()`,
12//!   shut down cleanly. ✅ done.
13//! - **P1.4** (this version, sifrah/nauka#194): also exercise
14//!   [`EmbeddedDb::open_default`], which uses
15//!   `nauka_core::process::nauka_db_path()` to pick the right path for
16//!   the current run mode (CLI: `~/.nauka/bootstrap.skv`, service: root
17//!   → `/var/lib/nauka/bootstrap.skv`). On a Hetzner node we run as root,
18//!   so this exercises the service-mode branch end-to-end and validates
19//!   that the parent state directory is created with 0o700 perms.
20//!
21//! What this binary does on each run:
22//! 1. Print build / runtime info, plus the detected run mode and the
23//!    default DB path
24//! 2. **Phase A** — `EmbeddedDb::open` at `$TMPDIR/nauka-p0-1-spike.skv`
25//!    (the original P1.2 round-trip, kept as a no-side-effects sanity)
26//! 3. **Phase B** — `EmbeddedDb::open_default()` to exercise the
27//!    run-mode-aware path picker, then a quick `INFO FOR DB`, then
28//!    shutdown + wipe of the default datastore. The parent state dir's
29//!    perms are also printed for visual confirmation of the 0o700 target.
30
31use std::path::PathBuf;
32
33use nauka_core::process::{is_service_mode, nauka_db_path};
34use nauka_state::EmbeddedDb;
35use surrealdb::types::SurrealValue;
36
37#[derive(Debug, SurrealValue)]
38struct SpikeRecord {
39    name: String,
40    answer: i32,
41}
42
43#[tokio::main]
44async fn main() -> Result<(), Box<dyn std::error::Error>> {
45    println!("== nauka p0-1 spike (P1.4 — run-mode-aware paths) ==");
46    println!("target_arch    = {}", std::env::consts::ARCH);
47    println!("target_os      = {}", std::env::consts::OS);
48    println!("target_env     = {}", std::env::consts::FAMILY);
49    println!("surrealdb_dep  = 3.0.5 (kv-surrealkv only)");
50    println!(
51        "run_mode       = {}",
52        if is_service_mode() {
53            "service (root)"
54        } else {
55            "cli (user)"
56        }
57    );
58    println!("default_path   = {}", nauka_db_path().display());
59
60    // ─── Phase A: EmbeddedDb::open at a temp path ──────────────────
61    println!("--- Phase A: EmbeddedDb::open(temp_path) ---");
62    let temp_path: PathBuf = std::env::temp_dir().join("nauka-p0-1-spike.skv");
63    println!("skv_path       = {}", temp_path.display());
64
65    let db_a = EmbeddedDb::open(&temp_path).await?;
66    let client_a = db_a.client();
67
68    let created: Option<SpikeRecord> = client_a
69        .create(("spike_record", "first"))
70        .content(SpikeRecord {
71            name: "p0-1".into(),
72            answer: 42,
73        })
74        .await?;
75    println!("created        = {created:?}");
76
77    let fetched: Option<SpikeRecord> = client_a.select(("spike_record", "first")).await?;
78    println!("fetched        = {fetched:?}");
79
80    let all: Vec<SpikeRecord> = client_a.select("spike_record").await?;
81    println!("all_count      = {}", all.len());
82
83    db_a.shutdown().await?;
84    let _ = std::fs::remove_dir_all(&temp_path);
85    println!("phase_a        = OK");
86
87    // ─── Phase B: EmbeddedDb::open_default ──────────────────────────
88    println!("--- Phase B: EmbeddedDb::open_default() ---");
89    let default_path = nauka_db_path();
90    println!("default_path   = {}", default_path.display());
91
92    let db_b = EmbeddedDb::open_default().await?;
93    println!("opened_path    = {}", db_b.path().display());
94
95    // Live no-side-effects query against the default DB to confirm the
96    // round-trip works against the run-mode-resolved location.
97    let _ = db_b.client().query("INFO FOR DB").await?.check()?;
98    println!("info_for_db    = OK");
99
100    db_b.shutdown().await?;
101
102    // Inspect the parent state dir's perms (Unix only) and wipe the
103    // datastore so re-runs are clean. We don't wipe the state dir
104    // itself in case other Nauka state lives next to it.
105    #[cfg(unix)]
106    {
107        use std::os::unix::fs::PermissionsExt;
108        if let Some(parent) = default_path.parent() {
109            if let Ok(meta) = std::fs::metadata(parent) {
110                let mode = meta.permissions().mode() & 0o777;
111                println!("state_dir_perms= 0o{mode:o}");
112            }
113        }
114    }
115    let _ = std::fs::remove_dir_all(&default_path);
116    println!("phase_b        = OK");
117
118    println!("== p0-1 spike OK ==");
119    Ok(())
120}
