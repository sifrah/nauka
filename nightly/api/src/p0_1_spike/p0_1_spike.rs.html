1//! Spike binary for P0.1 (sifrah/nauka#185), repurposed in P1.2 to be the
2//! Hetzner-side smoke test for [`nauka_state::EmbeddedDb`].
3//!
4//! The spike's purpose evolves with each phase that touches the same code:
5//!
6//! - **P0.1**: prove that `surrealdb` 3.0.5 with `kv-surrealkv` (+ `kv-tikv`)
7//!   cross-compiles to `x86_64-unknown-linux-musl` and runs as a
8//!   statically-linked binary on a real Hetzner Ubuntu host. ✅ done.
9//! - **P1.1**: drop `kv-tikv` from the production feature set. ✅ done.
10//! - **P1.2**: exercise the new [`EmbeddedDb`] wrapper end-to-end on
11//!   Hetzner — open at a temp path, CRUD round-trip via `db.client()`,
12//!   shut down cleanly. ✅ done.
13//! - **P1.4** (this version, sifrah/nauka#194): also exercise
14//!   [`EmbeddedDb::open_default`], which uses
15//!   `nauka_core::process::nauka_db_path()` to pick the right path for
16//!   the current run mode (CLI: `~/.nauka/bootstrap.skv`, service: root
17//!   → `/var/lib/nauka/bootstrap.skv`). On a Hetzner node we run as root,
18//!   so this exercises the service-mode branch end-to-end and validates
19//!   that the parent state directory is created with 0o700 perms.
20//!
21//! What this binary does on each run:
22//! 1. Print build / runtime info, plus the detected run mode and the
23//!    default DB path
24//! 2. **Phase A** — `EmbeddedDb::open` at `$TMPDIR/nauka-p0-1-spike.skv`
25//!    (the original P1.2 round-trip, kept as a no-side-effects sanity)
26//! 3. **Phase B** — `EmbeddedDb::open_default()` to exercise the
27//!    run-mode-aware path picker, then a quick `INFO FOR DB`, then
28//!    shutdown + wipe of the default datastore. The parent state dir's
29//!    perms are also printed for visual confirmation of the 0o700 target.
30
31use std::path::PathBuf;
32
33use nauka_core::process::{is_service_mode, nauka_db_path};
34use nauka_state::EmbeddedDb;
35use surrealdb::types::SurrealValue;
36
37#[derive(Debug, SurrealValue)]
38struct SpikeRecord {
39    name: String,
40    answer: i32,
41}
42
43/// Print a short usage message and exit cleanly. Wired to `--help` / `-h`
44/// so the binary can be used as a smoke test for cross-compile validation
45/// (P1.8, sifrah/nauka#198) without producing side effects on disk.
46fn print_help() {
47    println!("nauka p0-1 spike — EmbeddedDb cross-compile + Hetzner smoke test");
48    println!();
49    println!("Exercises the EmbeddedDb<SurrealKv> wrapper end-to-end. Used by");
50    println!("the per-ticket workflow (compile → Hetzner → CI → merge) to");
51    println!("re-validate the build chain on every Phase 1 PR.");
52    println!();
53    println!("USAGE:");
54    println!("    p0-1-spike [FLAGS]");
55    println!();
56    println!("FLAGS:");
57    println!("    -h, --help    Print this help and exit (no side effects)");
58    println!();
59    println!("With no flags the binary runs in two phases against the actual");
60    println!("filesystem:");
61    println!("    Phase A — EmbeddedDb::open at $TMPDIR/nauka-p0-1-spike.skv");
62    println!("              and a CRUD round-trip via db.client().");
63    println!("    Phase B — EmbeddedDb::open_default() to exercise the");
64    println!("              run-mode-aware path picker (CLI → ~/.nauka,");
65    println!("              service mode → /var/lib/nauka), with INFO FOR DB");
66    println!("              and 0o700 perms verification on the parent.");
67}
68
69#[tokio::main]
70async fn main() -> Result<(), Box<dyn std::error::Error>> {
71    // Manual flag handling: only `--help` / `-h` need to be recognised at
72    // P1.8 time, so a tiny if-let is preferable to pulling clap into a
73    // spike binary.
74    if let Some(arg) = std::env::args().nth(1) {
75        match arg.as_str() {
76            "-h" | "--help" => {
77                print_help();
78                return Ok(());
79            }
80            other => {
81                eprintln!("error: unknown flag: {other}");
82                eprintln!("hint:  run `--help` for usage");
83                std::process::exit(2);
84            }
85        }
86    }
87
88    println!("== nauka p0-1 spike (P1.4 — run-mode-aware paths) ==");
89    println!("target_arch    = {}", std::env::consts::ARCH);
90    println!("target_os      = {}", std::env::consts::OS);
91    println!("target_env     = {}", std::env::consts::FAMILY);
92    println!("surrealdb_dep  = 3.0.5 (kv-surrealkv only)");
93    println!(
94        "run_mode       = {}",
95        if is_service_mode() {
96            "service (root)"
97        } else {
98            "cli (user)"
99        }
100    );
101    println!("default_path   = {}", nauka_db_path().display());
102
103    // ─── Phase A: EmbeddedDb::open at a temp path ──────────────────
104    println!("--- Phase A: EmbeddedDb::open(temp_path) ---");
105    let temp_path: PathBuf = std::env::temp_dir().join("nauka-p0-1-spike.skv");
106    println!("skv_path       = {}", temp_path.display());
107
108    let db_a = EmbeddedDb::open(&temp_path).await?;
109    let client_a = db_a.client();
110
111    let created: Option<SpikeRecord> = client_a
112        .create(("spike_record", "first"))
113        .content(SpikeRecord {
114            name: "p0-1".into(),
115            answer: 42,
116        })
117        .await?;
118    println!("created        = {created:?}");
119
120    let fetched: Option<SpikeRecord> = client_a.select(("spike_record", "first")).await?;
121    println!("fetched        = {fetched:?}");
122
123    let all: Vec<SpikeRecord> = client_a.select("spike_record").await?;
124    println!("all_count      = {}", all.len());
125
126    db_a.shutdown().await?;
127    let _ = std::fs::remove_dir_all(&temp_path);
128    println!("phase_a        = OK");
129
130    // ─── Phase B: EmbeddedDb::open_default ──────────────────────────
131    println!("--- Phase B: EmbeddedDb::open_default() ---");
132    let default_path = nauka_db_path();
133    println!("default_path   = {}", default_path.display());
134
135    let db_b = EmbeddedDb::open_default().await?;
136    println!("opened_path    = {}", db_b.path().display());
137
138    // Live no-side-effects query against the default DB to confirm the
139    // round-trip works against the run-mode-resolved location.
140    let _ = db_b.client().query("INFO FOR DB").await?.check()?;
141    println!("info_for_db    = OK");
142
143    db_b.shutdown().await?;
144
145    // Inspect the parent state dir's perms (Unix only) and wipe the
146    // datastore so re-runs are clean. We don't wipe the state dir
147    // itself in case other Nauka state lives next to it.
148    #[cfg(unix)]
149    {
150        use std::os::unix::fs::PermissionsExt;
151        if let Some(parent) = default_path.parent() {
152            if let Ok(meta) = std::fs::metadata(parent) {
153                let mode = meta.permissions().mode() & 0o777;
154                println!("state_dir_perms= 0o{mode:o}");
155            }
156        }
157    }
158    let _ = std::fs::remove_dir_all(&default_path);
159    println!("phase_b        = OK");
160
161    println!("== p0-1 spike OK ==");
162    Ok(())
163}
