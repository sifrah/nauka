3use axum::http::StatusCode;
4use axum::response::{IntoResponse, Response};
5
6use crate::error::NaukaError;
7
8/// Wrapper to convert NaukaError into an axum response.
9pub struct ApiError(pub NaukaError);
10
11impl IntoResponse for ApiError {
12    fn into_response(self) -> Response {
13        let status =
14            StatusCode::from_u16(self.0.http_status()).unwrap_or(StatusCode::INTERNAL_SERVER_ERROR);
15        let body = axum::Json(serde_json::json!({
16            "error": {
17                "code": self.0.code.as_str(),
18                "message": self.0.message,
19            }
20        }));
21        (status, body).into_response()
22    }
23}
24
25impl From<NaukaError> for ApiError {
26    fn from(e: NaukaError) -> Self {
27        Self(e)
28    }
29}

31/// Standard API success response.
32#[derive(Debug, serde::Serialize)]
33pub struct ApiSuccess<T: serde::Serialize> {
34    pub data: T,
35}
36
37impl<T: serde::Serialize> IntoResponse for ApiSuccess<T> {
38    fn into_response(self) -> Response {
39        (StatusCode::OK, axum::Json(self)).into_response()
40    }
41}
42
43/// API response type — either success or error.
44pub type ApiResult<T> = Result<axum::Json<T>, ApiError>;
45
46/// Standard list response with metadata.
47#[derive(Debug, serde::Serialize)]
48pub struct ListResponse<T: serde::Serialize> {
49    pub items: Vec<T>,
50    pub count: usize,
51}
52
53impl<T: serde::Serialize> ListResponse<T> {
54    pub fn new(items: Vec<T>) -> Self {
55        let count = items.len();
56        Self { items, count }
57    }
58}

60/// Standard create/get response.
61#[derive(Debug, serde::Serialize)]
62pub struct ResourceResponse<T: serde::Serialize> {
63    #[serde(flatten)]
64    pub resource: T,
65}
66
67/// Standard delete response.
68#[derive(Debug, serde::Serialize)]
69pub struct DeleteResponse {
70    pub deleted: bool,
71    pub message: String,
72}
73
74/// Standard action response.
75#[derive(Debug, serde::Serialize)]
76pub struct ActionResponse {
77    pub ok: bool,
78    pub message: String,
79    #[serde(skip_serializing_if = "Option::is_none")]
80    pub data: Option<serde_json::Value>,
81}
82
83impl ActionResponse {
84    pub fn ok(message: impl Into<String>) -> Self {
85        Self {
86            ok: true,
87            message: message.into(),
88            data: None,
89        }
90    }
91
92    pub fn ok_with_data(message: impl Into<String>, data: serde_json::Value) -> Self {
93        Self {
94            ok: true,
95            message: message.into(),
96            data: Some(data),
97        }
98    }
99}
