1//! Process utilities — paths, directories, run-mode detection.
2//!
3//! No daemon, no fork, no PID file. Nauka is a CLI orchestrator,
4//! not a daemon. WireGuard runs in the kernel.
5//!
6//! # Run modes
7//!
8//! Nauka can run in one of two modes:
9//!
10//! - **CLI mode** — invoked by a normal user from a shell. State is
11//!   stored under `$HOME/.nauka/`.
12//! - **Service mode** — invoked as `root` by systemd. State is stored
13//!   under `/var/lib/nauka/`.
14//!
15//! The two modes are detected by [`is_service_mode`] (root effective
16//! UID == service mode). Helpers like [`nauka_db_path`] (added in P1.4,
17//! sifrah/nauka#194) use this detection automatically so the same
18//! `nauka` binary picks the right path no matter how it was launched.
19
20use std::path::{Path, PathBuf};
21
22use crate::error::NaukaError;
23
24/// Default nauka directory **for CLI mode**: `~/.nauka`.
25///
26/// Service-mode callers should use [`nauka_state_dir`] or one of the
27/// path helpers ([`nauka_db_path`], etc.) instead — those are
28/// run-mode-aware.
29pub fn nauka_dir() -> PathBuf {
30    dirs::home_dir()
31        .unwrap_or_else(|| PathBuf::from("."))
32        .join(".nauka")
33}
34
35/// Ensure ~/.nauka exists with 0o700 permissions.
36pub fn ensure_nauka_dir() -> Result<(), NaukaError> {
37    let dir = nauka_dir();
38    std::fs::create_dir_all(&dir).map_err(NaukaError::from)?;
39    set_dir_perms(&dir);
40    Ok(())
41}
42
43/// Default control socket path (for future API server).
44pub fn socket_path() -> PathBuf {
45    nauka_dir().join("control.sock")
46}

48// ─────────────────────────────────────────────────────────────────────
49// Run mode detection (P1.4, sifrah/nauka#194)
50// ─────────────────────────────────────────────────────────────────────
51
52/// `true` when the current process is running as root, which is what
53/// "service mode" means for Nauka in practice (systemd starts everything
54/// as root). `false` for normal CLI invocations from a user shell.
55///
56/// On non-Unix targets this always returns `false` because Nauka's only
57/// supported deployment target for service mode is Linux + systemd.
58pub fn is_service_mode() -> bool {
59    #[cfg(unix)]
60    {
61        // SAFETY: `geteuid` has no preconditions and never fails on
62        // any POSIX system. It just reads a thread-local value.
63        unsafe { libc::geteuid() == 0 }
64    }
65    #[cfg(not(unix))]
66    {
67        false
68    }
69}
70
71/// State root directory for the current run mode.
72///
73/// - CLI mode → `$HOME/.nauka` (or `./.nauka` if `$HOME` is unset)
74/// - Service mode → `/var/lib/nauka`
75pub fn nauka_state_dir() -> PathBuf {
76    if is_service_mode() {
77        PathBuf::from("/var/lib/nauka")
78    } else {
79        nauka_dir()
80    }
81}
82
83/// Path of the on-disk SurrealKV datastore used by `EmbeddedDb`.
84///
85/// - CLI mode → `$HOME/.nauka/bootstrap.skv`
86/// - Service mode → `/var/lib/nauka/bootstrap.skv`
87///
88/// The parent directory is **not** created by this function — call
89/// [`ensure_nauka_state_dir`] first if you need it on disk with the
90/// right permissions, or rely on `EmbeddedDb::open` (P1.2,
91/// sifrah/nauka#192) which creates parents on demand.
92pub fn nauka_db_path() -> PathBuf {
93    nauka_state_dir().join("bootstrap.skv")
94}
95
96/// Ensure the run-mode-appropriate state directory exists with
97/// 0o700 permissions on Unix.
98///
99/// Returns the directory path on success.
100pub fn ensure_nauka_state_dir() -> Result<PathBuf, NaukaError> {
101    let dir = nauka_state_dir();
102    std::fs::create_dir_all(&dir).map_err(NaukaError::from)?;
103    set_dir_perms(&dir);
104    Ok(dir)
105}
106
107/// Apply 0o700 permissions to a directory on Unix. Best-effort: a
108/// permission failure is logged via `tracing::warn!` but does not
109/// surface as an error, because the alternative is to refuse to start
110/// over a chmod failure on a directory we just successfully created.
111fn set_dir_perms(dir: &Path) {
112    #[cfg(unix)]
113    {
114        use std::os::unix::fs::PermissionsExt;
115        if let Err(e) = std::fs::set_permissions(dir, std::fs::Permissions::from_mode(0o700)) {
116            tracing::warn!(
117                target: "nauka_core::process",
118                path = %dir.display(),
119                error = %e,
120                "failed to chmod 0o700"
121            );
122        }
123    }
124    #[cfg(not(unix))]
125    {
126        let _ = dir;
127    }
128}

130#[cfg(test)]
131mod tests {
132    use super::*;
133
134    #[test]
135    fn nauka_dir_path() {
136        let d = nauka_dir();
137        assert!(d.to_str().unwrap().contains(".nauka"));
138    }
139
140    #[test]
141    fn socket_path_in_dir() {
142        let p = socket_path();
143        assert!(p.to_str().unwrap().contains(".nauka/control.sock"));
144    }
145
146    #[test]
147    fn ensure_dir_creates() {
148        let _ = ensure_nauka_dir();
149    }
150
151    #[test]
152    fn db_path_filename_is_bootstrap_skv() {
153        let p = nauka_db_path();
154        assert_eq!(
155            p.file_name().and_then(|s| s.to_str()),
156            Some("bootstrap.skv"),
157            "db path must end with bootstrap.skv, got: {}",
158            p.display()
159        );
160    }
161
162    #[test]
163    fn db_path_lives_in_state_dir() {
164        let p = nauka_db_path();
165        let parent = p.parent().expect("db path must have a parent");
166        assert_eq!(parent, nauka_state_dir());
167    }
168
169    #[test]
170    fn state_dir_matches_run_mode() {
171        // Whichever mode the test process happens to run in, the path
172        // must match the expected branch. Tests typically run as the
173        // current user (not root), so this exercises CLI mode in CI;
174        // a manual `sudo cargo test` exercises the other branch.
175        if is_service_mode() {
176            assert_eq!(nauka_state_dir(), PathBuf::from("/var/lib/nauka"));
177        } else {
178            assert_eq!(nauka_state_dir(), nauka_dir());
179        }
180    }
181
182    #[test]
183    fn db_path_is_under_state_dir_in_both_modes() {
184        // Defensive: just confirm the prefix relationship holds.
185        let db = nauka_db_path();
186        assert!(
187            db.starts_with(nauka_state_dir()),
188            "{} must be a child of {}",
189            db.display(),
190            nauka_state_dir().display()
191        );
192    }
