4//! Handlers can assume all inputs are validated.

13/// Validate the resource name based on operation semantics.
14///
15/// Create, Get, Delete, and Update operations require a name that passes
16/// `crate::validate::name()`. List and Action operations skip name validation
17/// (List never has a name; Action may or may not).
18pub fn validate_name(
19    name: &Option<String>,
20    semantics: &OperationSemantics,
21) -> Result<(), NaukaError> {
22    match semantics {
23        OperationSemantics::Create
24        | OperationSemantics::Get
25        | OperationSemantics::Delete
26        | OperationSemantics::Update { .. } => {
27            let n = name
28                .as_deref()
29                .ok_or_else(|| NaukaError::validation("name is required for this operation"))?;
30            crate::validate::name(n)
31        }
32        OperationSemantics::List | OperationSemantics::Action => Ok(()),
33    }
34}

36/// Validate that required scope parents are present.
37///
38/// For Create and Delete operations, every parent with `required_on_create == true`
39/// must be present in the scope values. Delete requires the same scope as Create
40/// to prevent accidental deletion of the wrong resource.
41pub fn validate_scope(
42    def: &ResourceDef,
43    op: &OperationDef,
44    scope: &ScopeValues,
45) -> Result<(), NaukaError> {
46    for parent in &def.scope.parents {
47        let required = match &op.semantics {
48            OperationSemantics::Create | OperationSemantics::Delete => parent.required_on_create,
49            _ => false,
50        };
51
52        if required && scope.get(parent.kind).is_none() {
53            let flag_name = parent.flag.trim_start_matches('-');
54            return Err(NaukaError::validation(format!(
55                "--{} is required",
56                flag_name,
57            )));
58        }
59    }
60    Ok(())
61}

63/// Validate that all required fields are present.
64///
65/// Checks two categories:
66/// 1. Operation args marked as `required: true` must be in the field map.
67/// 2. For Create operations, schema fields with `CreateOnly` or `Mutable` mutability
68///    and no default value must also be present.
69pub fn validate_required_fields(
70    def: &ResourceDef,
71    op: &OperationDef,
72    fields: &FieldMap,
73) -> Result<(), NaukaError> {
74    // Check required operation args.
75    for arg in &op.args {
76        if arg.required && !fields.contains_key(arg.name) {
77            return Err(NaukaError::validation(format!(
78                "missing required field: {}",
79                arg.name,
80            )));
81        }
82    }
83
84    // For Create, schema fields that are settable and have no default must be present.
85    if op.semantics == OperationSemantics::Create {
86        for field in &def.schema.fields {
87            let settable = matches!(
88                field.mutability,
89                Mutability::CreateOnly | Mutability::Mutable
90            );
91            if settable && field.default.is_none() && !fields.contains_key(field.name) {
92                return Err(NaukaError::validation(format!(
93                    "missing required field: {}",
94                    field.name,
95                )));
96            }
97        }
98    }
99
100    Ok(())
101}

103/// Validate field values match their declared types.
104///
105/// Looks up the `FieldDef` for each field (from schema or operation args) and
106/// validates the value against its `FieldType`. Fields without a known definition
107/// are silently skipped (they may be internal or framework-managed).
108pub fn validate_field_types(
109    def: &ResourceDef,
110    op: &OperationDef,
111    fields: &FieldMap,
112) -> Result<(), NaukaError> {
113    for (name, value) in fields {
114        let field_def = match find_field_def(def, op, name) {
115            Some(fd) => fd,
116            None => continue,
117        };
118        validate_single_field(name, value, &field_def.field_type)?;
119    }
120    Ok(())
121}

123/// Apply default values for fields not already present.
124///
125/// Fills in defaults from two sources:
126/// 1. Schema fields with `default: Some(val)`.
127/// 2. Operation args with `ArgSource::Custom(field_def)` where the field def
128///    has a default.
129pub fn apply_defaults(def: &ResourceDef, op: &OperationDef, fields: &mut FieldMap) {
130    // Schema-level defaults.
131    for field in &def.schema.fields {
132        if let Some(default) = field.default {
133            fields
134                .entry(field.name.to_string())
135                .or_insert_with(|| default.to_string());
136        }
137    }
138
139    // Operation arg defaults.
140    for arg in &op.args {
141        if let ArgSource::Custom(ref field_def) = arg.source {
142            if let Some(default) = field_def.default {
143                fields
144                    .entry(arg.name.to_string())
145                    .or_insert_with(|| default.to_string());
146            }
147        }
148    }
149}

151/// Remove fields that clients should never set.
152///
153/// Strips any field whose `FieldDef` has `Mutability::ReadOnly` or
154/// `Mutability::Internal`. This prevents API clients from injecting system
155/// fields like `id` or `created_at`.
156pub fn filter_readonly_fields(def: &ResourceDef, fields: &mut FieldMap) {
157    fields.retain(|name, _| {
158        let Some(field) = def.schema.fields.iter().find(|f| f.name == name) else {
159            // Not a schema field — keep it (might be an operation-specific arg).
160            return true;
161        };
162        !matches!(
163            field.mutability,
164            Mutability::ReadOnly | Mutability::Internal
165        )
166    });
167}

171/// Find the `FieldDef` for a field by name, searching both the resource schema
172/// and the operation's args.
173fn find_field_def<'a>(
174    def: &'a ResourceDef,
175    op: &'a OperationDef,
176    name: &str,
177) -> Option<&'a FieldDef> {
178    // Check schema fields first.
179    if let Some(field) = def.schema.fields.iter().find(|f| f.name == name) {
180        return Some(field);
181    }
182    // Then check operation args with custom field defs.
183    for arg in &op.args {
184        if arg.name == name {
185            if let ArgSource::Custom(ref field_def) = arg.source {
186                return Some(field_def);
187            }
188            // ArgSource::FromSchema — already checked in schema above.
189        }
190    }
191    None
192}

194/// Validate a single field value against its declared type.
195fn validate_single_field(
196    name: &str,
197    value: &str,
198    field_type: &FieldType,
199) -> Result<(), NaukaError> {
200    match field_type {
201        FieldType::Enum(enum_def) => {
202            if !enum_def.values.contains(&value) {
203                return Err(NaukaError::validation(format!(
204                    "--{}: must be one of [{}], got '{}'",
205                    name,
206                    enum_def.values.join(", "),
207                    value,
208                )));
209            }
210        }
211        FieldType::Integer => {
212            value.parse::<i64>().map_err(|_| {
213                NaukaError::validation(format!("--{}: expected an integer, got '{}'", name, value,))
214            })?;
215        }
216        FieldType::Port => {
217            value.parse::<u16>().map_err(|_| {
218                NaukaError::validation(format!(
219                    "--{}: expected a port number (0-65535), got '{}'",
220                    name, value,
221                ))
222            })?;
223        }
224        FieldType::SizeGb | FieldType::SizeMb => {
225            value.parse::<u64>().map_err(|_| {
226                NaukaError::validation(format!(
227                    "--{}: expected a positive number, got '{}'",
228                    name, value,
229                ))
230            })?;
231        }
232        FieldType::Cidr => {
233            crate::validate::cidr(value).map_err(|_| {
234                NaukaError::validation(format!("--{}: invalid CIDR notation '{}'", name, value,))
235            })?;
236        }
237        FieldType::IpAddr => {
238            value.parse::<std::net::IpAddr>().map_err(|_| {
239                NaukaError::validation(format!("--{}: invalid IP address '{}'", name, value,))
240            })?;
241        }
242        FieldType::Flag => {
243            if value != "true" && value != "false" {
244                return Err(NaukaError::validation(format!(
245                    "--{}: expected 'true' or 'false', got '{}'",
246                    name, value,
247                )));
248            }
249        }
250        // String, Path, Secret, Duration, KeyValue, ResourceRef: accept any string.
251        FieldType::String
252        | FieldType::Path
253        | FieldType::Secret
254        | FieldType::Duration
255        | FieldType::KeyValue
256        | FieldType::ResourceRef(_) => {}
257    }
258    Ok(())
259}
