1//! ResourceDef lint engine.
2//!
3//! Enforces structural, behavioral, and UX consistency across all resource
4//! definitions.  Rules are checked at two points:
5//!
6//! 1. **`lint_def()`** — called from `.done()`, validates a single ResourceDef.
7//!    Errors panic immediately (binary won't start with a broken def).
8//! 2. **`lint_registry()`** — called from tests, validates cross-resource rules.
9//!
10//! Severity levels:
11//! - `Error`   — invariant violation, panics in `.done()`.
12//! - `Warning` — convention violation, fails CI test.
13
14use super::operation::{OperationSemantics, OutputKind, ProgressHint};
15use super::presentation::DisplayFormat;
16use super::ResourceDef;
17
18/// Lint severity.
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum Severity {
21    Error,
22    Warning,
23}
24
25/// A single lint violation.
26#[derive(Debug, Clone)]
27pub struct Violation {
28    pub rule: &'static str,
29    pub resource: &'static str,
30    pub message: String,
31    pub severity: Severity,
32}
33
34impl std::fmt::Display for Violation {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        let level = match self.severity {
37            Severity::Error => "ERROR",
38            Severity::Warning => "WARN",
39        };
40        write!(
41            f,
42            "[{}] {} ({}): {}",
43            self.rule, self.resource, level, self.message
44        )
45    }
46}

50/// Validate a single ResourceDef for internal consistency.
51pub fn lint_def(def: &ResourceDef) -> Vec<Violation> {
52    let mut v = Vec::new();
53    let kind = def.identity.kind;
54
55    structural_rules(def, kind, &mut v);
56    ux_rules(def, kind, &mut v);
57    output_rules(def, kind, &mut v);
58    scope_rules(def, kind, &mut v);
59    description_rules(def, kind, &mut v);
60    presentation_alignment_rules(def, kind, &mut v);
61    state_machine_rules(def, kind, &mut v);
62    progress_rules(def, kind, &mut v);
63    naming_rules(def, kind, &mut v);
64
65    v
66}

68/// Validate a flat list of ResourceDefs for cross-resource consistency.
69pub fn lint_registry(defs: &[&ResourceDef]) -> Vec<Violation> {
70    let mut v = Vec::new();
71    registry_rules(defs, &mut v);
72    v
73}

75/// Format a list of violations for display.
76pub fn format_violations(violations: &[Violation]) -> String {
77    violations
78        .iter()
79        .map(|v| v.to_string())
80        .collect::<Vec<_>>()
81        .join("\n")
82}

86fn has_op(def: &ResourceDef, name: &str) -> bool {
87    def.operations.iter().any(|op| op.name == name)
88}
89
90fn is_crud(def: &ResourceDef) -> bool {
91    has_op(def, "create") && has_op(def, "list") && has_op(def, "get") && has_op(def, "delete")
92}
93
94fn has_column(def: &ResourceDef, field: &str) -> bool {
95    def.presentation
96        .table
97        .as_ref()
98        .map(|t| t.columns.iter().any(|c| c.field == field))
99        .unwrap_or(false)
100}
101
102fn has_column_with_format(def: &ResourceDef, field: &str, fmt: &DisplayFormat) -> bool {
103    def.presentation
104        .table
105        .as_ref()
106        .map(|t| {
107            t.columns.iter().any(|c| {
108                c.field == field && std::mem::discriminant(&c.format) == std::mem::discriminant(fmt)
109            })
110        })
111        .unwrap_or(false)
112}
113
114fn has_detail_field(def: &ResourceDef, field: &str) -> bool {
115    def.presentation
116        .detail
117        .as_ref()
118        .map(|d| {
119            d.sections
120                .iter()
121                .any(|s| s.fields.iter().any(|f| f.field == field))
122        })
123        .unwrap_or(false)
124}
125
126fn has_detail_field_with_format(def: &ResourceDef, field: &str, fmt: &DisplayFormat) -> bool {
127    def.presentation
128        .detail
129        .as_ref()
130        .map(|d| {
131            d.sections.iter().any(|s| {
132                s.fields.iter().any(|f| {
133                    f.field == field
134                        && std::mem::discriminant(&f.format) == std::mem::discriminant(fmt)
135                })
136            })
137        })
138        .unwrap_or(false)
139}

141fn is_snake_case(s: &str) -> bool {
142    !s.is_empty()
143        && s.chars()
144            .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_')
145}
146
147fn is_kebab_lower(s: &str) -> bool {
148    !s.is_empty()
149        && s.chars()
150            .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '-')
151}

153// ── Structural (E001–E004, E030–E031, E090, E100) ────────────
154
155fn structural_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
156    // E001: list requires table
157    if has_op(def, "list") && def.presentation.table.is_none() {
158        v.push(Violation {
159            rule: "E001",
160            resource: kind,
161            message: "has 'list' operation but no table columns defined".into(),
162            severity: Severity::Error,
163        });
164    }
165
166    // E002: get requires detail
167    if has_op(def, "get") && def.presentation.detail.is_none() {
168        v.push(Violation {
169            rule: "E002",
170            resource: kind,
171            message: "has 'get' operation but no detail section defined".into(),
172            severity: Severity::Error,
173        });
174    }
175
176    // E003: no duplicate operation names
177    let mut seen_ops = std::collections::HashSet::new();
178    for op in &def.operations {
179        if !seen_ops.insert(op.name) {
180            v.push(Violation {
181                rule: "E003",
182                resource: kind,
183                message: format!("duplicate operation name '{}'", op.name),
184                severity: Severity::Error,
185            });
186        }
187    }
188
189    // E004: no duplicate column headers
190    if let Some(table) = &def.presentation.table {
191        let mut seen_cols = std::collections::HashSet::new();
192        for col in &table.columns {
193            if !seen_cols.insert(col.header) {
194                v.push(Violation {
195                    rule: "E004",
196                    resource: kind,
197                    message: format!("duplicate column header '{}'", col.header),
198                    severity: Severity::Error,
199                });
200            }
201        }
202    }
203
204    // E030: kind must be lowercase kebab
205    if !is_kebab_lower(kind) {
206        v.push(Violation {
207            rule: "E030",
208            resource: kind,
209            message: format!("kind '{}' must be lowercase [a-z0-9-]+", kind),
210            severity: Severity::Error,
211        });
212    }
213
214    // E031: plural must differ from kind
215    if def.identity.plural == kind {
216        v.push(Violation {
217            rule: "E031",
218            resource: kind,
219            message: format!(
220                "plural '{}' must differ from kind '{}'",
221                def.identity.plural, kind
222            ),
223            severity: Severity::Error,
224        });
225    }
226
227    // E090: column and detail fields must be snake_case
228    if let Some(table) = &def.presentation.table {
229        for col in &table.columns {
230            if !is_snake_case(col.field) {
231                v.push(Violation {
232                    rule: "E090",
233                    resource: kind,
234                    message: format!("column field '{}' is not snake_case", col.field),
235                    severity: Severity::Error,
236                });
237            }
238        }
239    }
240    if let Some(detail) = &def.presentation.detail {
241        for section in &detail.sections {
242            for field in &section.fields {
243                if !is_snake_case(field.field) {
244                    v.push(Violation {
245                        rule: "E090",
246                        resource: kind,
247                        message: format!("detail field '{}' is not snake_case", field.field),
248                        severity: Severity::Error,
249                    });
250                }
251            }
252        }
253    }
254
255    // E100: delete must resolve unambiguously
256    if has_op(def, "delete") && !def.scope.parents.is_empty() {
257        let create_required: std::collections::HashSet<&str> = def
258            .scope
259            .parents
260            .iter()
261            .filter(|p| p.required_on_create)
262            .map(|p| p.kind)
263            .collect();
264        let delete_resolvable: std::collections::HashSet<&str> = def
265            .scope
266            .parents
267            .iter()
268            .filter(|p| p.required_on_resolve)
269            .map(|p| p.kind)
270            .collect();
271
272        // If uniqueness is within parent but delete doesn't require the scope parent,
273        // and resolve is not required, it's ambiguous
274        if let super::scope::UniquenessScope::WithinParent(parent) = &def.scope.uniqueness {
275            if create_required.contains(parent) && !delete_resolvable.contains(parent) {
276                // This is OK — name can be resolved by scanning.
277                // Only flag if there's no way to filter.
278                // We allow this pattern (optional scope on resolve) — it's the common case.
279            }
280        }
281    }
282
283    // E040: FromSchema references must resolve to existing schema fields
284    for op in &def.operations {
285        for arg in &op.args {
286            if let super::operation::ArgSource::FromSchema(field_name) = &arg.source {
287                let exists = def.schema.fields.iter().any(|f| f.name == *field_name);
288                if !exists {
289                    v.push(Violation {
290                        rule: "E040",
291                        resource: kind,
292                        message: format!(
293                            "operation '{}' arg '{}' references schema field '{}' which does not exist",
294                            op.name, arg.name, field_name
295                        ),
296                        severity: Severity::Error,
297                    });
298                }
299            }
300        }
301    }
302}

304// ── UX (W001–W007) ───────────────────────────────────────────
305
306fn ux_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
307    // W001: list requires empty_message
308    if has_op(def, "list") {
309        if let Some(table) = &def.presentation.table {
310            if table.empty_message.is_none() {
311                v.push(Violation {
312                    rule: "W001",
313                    resource: kind,
314                    message: "list operation has no empty_message".into(),
315                    severity: Severity::Warning,
316                });
317            }
318        }
319    }
320
321    // W002: CRUD resource should have created_at column with Timestamp
322    if is_crud(def) && !has_column_with_format(def, "created_at", &DisplayFormat::Timestamp) {
323        v.push(Violation {
324            rule: "W002",
325            resource: kind,
326            message: "CRUD resource missing 'created_at' column with Timestamp format".into(),
327            severity: Severity::Warning,
328        });
329    }
330
331    // W003: CRUD resource should have id column
332    if is_crud(def) && !has_column(def, "id") {
333        v.push(Violation {
334            rule: "W003",
335            resource: kind,
336            message: "CRUD resource missing 'id' column".into(),
337            severity: Severity::Warning,
338        });
339    }
340
341    // W004: detail section should include name and id
342    if def.presentation.detail.is_some() {
343        if !has_detail_field(def, "name") {
344            v.push(Violation {
345                rule: "W004",
346                resource: kind,
347                message: "detail section missing 'name' field".into(),
348                severity: Severity::Warning,
349            });
350        }
351        if !has_detail_field(def, "id") {
352            v.push(Violation {
353                rule: "W004",
354                resource: kind,
355                message: "detail section missing 'id' field".into(),
356                severity: Severity::Warning,
357            });
358        }
359    }
360
361    // W005: detail should have created_at with Timestamp
362    if def.presentation.detail.is_some()
363        && !has_detail_field_with_format(def, "created_at", &DisplayFormat::Timestamp)
364    {
365        v.push(Violation {
366            rule: "W005",
367            resource: kind,
368            message: "detail section missing 'created_at' field with Timestamp format".into(),
369            severity: Severity::Warning,
370        });
371    }
372
373    // W006: parent should appear in columns or detail
374    for parent in &def.scope.parents {
375        let parent_field = format!("{}_name", parent.kind);
376        let has_in_table = has_column(def, &parent_field) || has_column(def, parent.kind);
377        let has_in_detail =
378            has_detail_field(def, &parent_field) || has_detail_field(def, parent.kind);
379        if !has_in_table && !has_in_detail {
380            v.push(Violation {
381                rule: "W006",
382                resource: kind,
383                message: format!(
384                    "parent '{}' not visible in table or detail (expected field '{}')",
385                    parent.kind, parent_field
386                ),
387                severity: Severity::Warning,
388            });
389        }
390    }
391
392    // W007: action with required arg "name" — should be positional
393    for op in &def.operations {
394        if matches!(op.semantics, OperationSemantics::Action) {
395            for arg in &op.args {
396                if arg.name == "name" && arg.required {
397                    v.push(Violation {
398                        rule: "W007",
399                        resource: kind,
400                        message: format!(
401                            "action '{}' has required arg 'name' — should use positional semantics (Get/Delete) instead of --name flag",
402                            op.name
403                        ),
404                        severity: Severity::Warning,
405                    });
406                }
407            }
408        }
409    }
410}

412// ── Output (W020–W026) ───────────────────────────────────────
413
414fn output_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
415    for op in &def.operations {
416        // W020: create should have examples
417        if op.semantics == OperationSemantics::Create && op.examples.is_empty() {
418            v.push(Violation {
419                rule: "W020",
420                resource: kind,
421                message: "create operation has no examples".into(),
422                severity: Severity::Warning,
423            });
424        }
425
426        // W021: delete should be confirmable
427        if op.semantics == OperationSemantics::Delete && !op.confirmable {
428            v.push(Violation {
429                rule: "W021",
430                resource: kind,
431                message: "delete operation is not confirmable (missing --yes)".into(),
432                severity: Severity::Warning,
433            });
434        }
435
436        // W022: create output should be Resource
437        if op.semantics == OperationSemantics::Create && op.output.kind != OutputKind::Resource {
438            v.push(Violation {
439                rule: "W022",
440                resource: kind,
441                message: format!(
442                    "create operation output is {:?}, expected Resource",
443                    op.output.kind
444                ),
445                severity: Severity::Warning,
446            });
447        }
448
449        // W023: delete output should be Message
450        if op.semantics == OperationSemantics::Delete && op.output.kind != OutputKind::Message {
451            v.push(Violation {
452                rule: "W023",
453                resource: kind,
454                message: format!(
455                    "delete operation output is {:?}, expected Message",
456                    op.output.kind
457                ),
458                severity: Severity::Warning,
459            });
460        }
461    }
462
463    // W024: NAME should be first column
464    if let Some(table) = &def.presentation.table {
465        if let Some(first) = table.columns.first() {
466            if first.header != "NAME" {
467                v.push(Violation {
468                    rule: "W024",
469                    resource: kind,
470                    message: format!("first column is '{}', expected 'NAME'", first.header),
471                    severity: Severity::Warning,
472                });
473            }
474        }
475    }
476
477    // W025: state/status column should use DisplayFormat::Status
478    if let Some(table) = &def.presentation.table {
479        for col in &table.columns {
480            if (col.field == "state" || col.field == "status")
481                && !matches!(col.format, DisplayFormat::Status)
482            {
483                v.push(Violation {
484                    rule: "W025",
485                    resource: kind,
486                    message: format!("column '{}' should use DisplayFormat::Status", col.header),
487                    severity: Severity::Warning,
488                });
489            }
490        }
491    }
492
493    // W026: list should have default_sort
494    if has_op(def, "list") {
495        if let Some(table) = &def.presentation.table {
496            if table.default_sort.is_none() {
497                v.push(Violation {
498                    rule: "W026",
499                    resource: kind,
500                    message: "list table has no default_sort defined".into(),
501                    severity: Severity::Warning,
502                });
503            }
504        }
505    }
506}

508// ── Scope (E020, W030, W031) ─────────────────────────────────
509
510fn scope_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
511    // E020: create with parent → parent must be required_on_create
512    if has_op(def, "create") {
513        for parent in &def.scope.parents {
514            if !parent.required_on_create {
515                v.push(Violation {
516                    rule: "E020",
517                    resource: kind,
518                    message: format!(
519                        "create operation exists but parent '{}' is not required_on_create",
520                        parent.kind
521                    ),
522                    severity: Severity::Error,
523                });
524            }
525        }
526    }
527
528    // W030: delete should require same scope as create
529    // Check if delete exists and any parent required_on_create is not usable for delete
530    // (This is a heuristic — we check if the handler could resolve via the scope)
531    // Skip if no parents
532    if has_op(def, "delete") && has_op(def, "create") && !def.scope.parents.is_empty() {
533        // We only warn if there are multiple parents (ambiguity risk)
534        let create_parents: Vec<&str> = def
535            .scope
536            .parents
537            .iter()
538            .filter(|p| p.required_on_create)
539            .map(|p| p.kind)
540            .collect();
541        if create_parents.len() > 1 {
542            // Multiple parents on create — delete should also accept them
543            // (We can't enforce this perfectly at the def level since handlers
544            // can do their own resolution, but it's a good heuristic)
545        }
546    }
547
548    // W031: if list accepts optional parent filters, get/delete should too
549    // This checks operation args — if list has optional parent-like args that
550    // get/delete don't have. Hard to check generically without knowing which
551    // args are parent filters vs data filters. Skip for now — covered by
552    // handler review.
553
554    // W080: delete with multiple parents should clarify scope
555    if has_op(def, "delete") && def.scope.parents.len() > 1 {
556        let all_resolve = def.scope.parents.iter().all(|p| p.required_on_resolve);
557        if !all_resolve {
558            v.push(Violation {
559                rule: "W080",
560                resource: kind,
561                message: "delete operation with multiple parents: consider setting required_on_resolve=true for unambiguous deletion".into(),
562                severity: Severity::Warning,
563            });
564        }
565    }
566}

568// ── Description (W040, W041) ─────────────────────────────────
569
570fn description_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
571    let desc = def.identity.description;
572
573    // W040: description should start with uppercase
574    if let Some(first) = desc.chars().next() {
575        if !first.is_uppercase() {
576            v.push(Violation {
577                rule: "W040",
578                resource: kind,
579                message: format!("description '{}' should start with uppercase", desc),
580                severity: Severity::Warning,
581            });
582        }
583    }
584
585    // W041: description should not end with period
586    if desc.ends_with('.') {
587        v.push(Violation {
588            rule: "W041",
589            resource: kind,
590            message: format!("description '{}' should not end with period", desc),
591            severity: Severity::Warning,
592        });
593    }
594}

596// ── Presentation alignment (W050, W051) ──────────────────────
597
598fn presentation_alignment_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
599    // W050: every table column should have a corresponding detail field
600    if let (Some(table), Some(_detail)) = (&def.presentation.table, &def.presentation.detail) {
601        for col in &table.columns {
602            if !has_detail_field(def, col.field) {
603                v.push(Violation {
604                    rule: "W050",
605                    resource: kind,
606                    message: format!(
607                        "table column '{}' (field '{}') has no corresponding detail field",
608                        col.header, col.field
609                    ),
610                    severity: Severity::Warning,
611                });
612            }
613        }
614    }
615
616    // W051: timestamp fields in detail must use DisplayFormat::Timestamp
617    if let Some(detail) = &def.presentation.detail {
618        for section in &detail.sections {
619            for field in &section.fields {
620                if field.field.ends_with("_at") && !matches!(field.format, DisplayFormat::Timestamp)
621                {
622                    v.push(Violation {
623                        rule: "W051",
624                        resource: kind,
625                        message: format!(
626                            "detail field '{}' ends with '_at' but does not use Timestamp format",
627                            field.field
628                        ),
629                        severity: Severity::Warning,
630                    });
631                }
632            }
633        }
634    }
635}

637// ── State machine (W060, W061) ───────────────────────────────
638
639fn state_machine_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
640    let has_start = has_op(def, "start");
641    let has_stop = has_op(def, "stop");
642
643    // W060: resource with start/stop should have STATE column
644    if (has_start || has_stop) && !has_column(def, "state") && !has_column(def, "status") {
645        v.push(Violation {
646            rule: "W060",
647            resource: kind,
648            message: "has start/stop actions but no 'state' or 'status' column".into(),
649            severity: Severity::Warning,
650        });
651    }
652
653    // W061: update action should output Resource
654    for op in &def.operations {
655        if matches!(op.name, "update") && op.output.kind != OutputKind::Resource {
656            v.push(Violation {
657                rule: "W061",
658                resource: kind,
659                message: format!(
660                    "action '{}' mutates state but output is {:?}, expected Resource",
661                    op.name, op.output.kind
662                ),
663                severity: Severity::Warning,
664            });
665        }
666    }
667}

669// ── Progress (W070–W073, E070) ───────────────────────────────
670
671fn progress_rules(def: &ResourceDef, kind: &'static str, v: &mut Vec<Violation>) {
672    for op in &def.operations {
673        match op.semantics {
674            // W070: create should have progress
675            OperationSemantics::Create => {
676                if matches!(op.progress, ProgressHint::None) {
677                    v.push(Violation {
678                        rule: "W070",
679                        resource: kind,
680                        message: "create operation has no ProgressHint".into(),
681                        severity: Severity::Warning,
682                    });
683                }
684            }
685            // W071: delete should have progress
686            OperationSemantics::Delete => {
687                if matches!(op.progress, ProgressHint::None) {
688                    v.push(Violation {
689                        rule: "W071",
690                        resource: kind,
691                        message: "delete operation has no ProgressHint".into(),
692                        severity: Severity::Warning,
693                    });
694                }
695            }
696            _ => {}
697        }
698
699        // W072: state-mutating actions should have progress
700        if matches!(op.semantics, OperationSemantics::Action)
701            && matches!(
702                op.name,
703                "start" | "stop" | "update" | "init" | "join" | "leave"
704            )
705            && matches!(op.progress, ProgressHint::None)
706        {
707            v.push(Violation {
708                rule: "W072",
709                resource: kind,
710                message: format!("action '{}' mutates state but has no ProgressHint", op.name),
711                severity: Severity::Warning,
712            });
713        }
714
715        // W073: list/get should not have progress
716        if matches!(
717            op.semantics,
718            OperationSemantics::List | OperationSemantics::Get
719        ) && !matches!(op.progress, ProgressHint::None)
720        {
721            v.push(Violation {
722                rule: "W073",
723                resource: kind,
724                message: format!("operation '{}' is a read but has ProgressHint set", op.name),
725                severity: Severity::Warning,
726            });
727        }
728
729        // E070: Spinner message must end with "..."
730        if let ProgressHint::Spinner(msg) = op.progress {
731            if !msg.ends_with("...") {
732                v.push(Violation {
733                    rule: "E070",
734                    resource: kind,
735                    message: format!(
736                        "ProgressHint::Spinner message '{}' must end with '...'",
737                        msg
738                    ),
739                    severity: Severity::Error,
740                });
741            }
742        }
743    }
744}

753// ── Registry (E010, E011, W010) ──────────────────────────────
754
755fn registry_rules(defs: &[&ResourceDef], v: &mut Vec<Violation>) {
756    // E010: no duplicate kinds
757    let mut seen = std::collections::HashSet::new();
758    for def in defs {
759        if !seen.insert(def.identity.kind) {
760            v.push(Violation {
761                rule: "E010",
762                resource: def.identity.kind,
763                message: format!("duplicate resource kind '{}'", def.identity.kind),
764                severity: Severity::Error,
765            });
766        }
767    }
768
769    // E011: every parent kind must exist in registry
770    let all_kinds: std::collections::HashSet<&str> = defs.iter().map(|d| d.identity.kind).collect();
771    for def in defs {
772        for parent in &def.scope.parents {
773            if !all_kinds.contains(parent.kind) {
774                v.push(Violation {
775                    rule: "E011",
776                    resource: def.identity.kind,
777                    message: format!("parent kind '{}' not found in registry", parent.kind),
778                    severity: Severity::Error,
779                });
780            }
781        }
782    }
783
784    // W010: child resource should have at least list + one other op
785    for def in defs {
786        if !def.scope.parents.is_empty() && def.operations.len() < 2 {
787            v.push(Violation {
788                rule: "W010",
789                resource: def.identity.kind,
790                message: "child resource should have at least 2 operations".into(),
791                severity: Severity::Warning,
792            });
793        }
794    }
795}
