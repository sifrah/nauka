1//! Org persistence on the SurrealDB-backed cluster store.
2//!
3//! P2.9 (sifrah/nauka#213) migrated this module from the legacy raw-KV
4//! cluster-DB surface (`db.put` / `db.get` / `db.scan_keys`) to the
5//! native SurrealDB SDK on top of [`EmbeddedDb`]. Every method now
6//! reaches the SDK via `self.db.client()` and writes/reads against the
7//! SCHEMAFULL `org` table defined in
8//! `layers/state/schemas/cluster/org.surql` (the canonical schema
9//! shipped by P2.5 in `layers/org/schemas/org.surql`, mirrored into the
10//! state crate by P2.7 so the bootstrap node can `apply_cluster_schemas`
11//! it without an upward layer dependency).
12//!
13//! The store holds an [`EmbeddedDb`] directly rather than the legacy
14//! cluster wrapper type — the previous version stored a
15//! cluster-wrapper value and reached the SDK via a
16//! `.embedded().client()` chain, but that kept the legacy type's name
17//! in the file even though the methods had all migrated. P2.9 drops
18//! the indirection entirely: call sites now pass
19//! `cluster_db.embedded().clone()` at construction time, which gives
20//! the store a cheap `Arc`-shared handle onto the same underlying
21//! `Surreal<Db>` the rest of the cluster path sees.
22//!
23//! The legacy `_reg_v2` registry / `org-idx` sidecar keys that the
24//! pre-P2.9 version of this file maintained are gone: the schema's
25//! unique index on `org.name` is now the source of truth for "have we
26//! seen this name before?", and `SELECT * FROM org` walks every row
27//! directly without a separate key list.
28//!
29//! The Org → SurrealDB JSON bridge is still hand-written rather than
30//! `SurrealValue`-derived because `Org` embeds [`ResourceMeta`] which
31//! in turn carries an `id: String` field that collides with SurrealDB's
32//! reserved record-id slot. We bind each column explicitly in the
33//! `CREATE … SET …` form so the offending `id` field is dropped on the
34//! way in, and we cast the `created_at` / `updated_at` Unix-epoch
35//! values through `<datetime>$iso8601_string` so they match the
36//! schema's `TYPE datetime` constraint without requiring a
37//! `surrealdb::types::SurrealValue` derive cascade. The same
38//! JSON-bridge pattern is documented at length on
39//! `nauka_hypervisor::fabric::state::FabricState::save`.
40

50/// SurrealDB table backing this store. Defined in

53const ORG_TABLE: &str = "org";

63    /// `cluster_db.embedded().clone()` as the argument. The
64    /// [`EmbeddedDb`] is cheap to clone (`Arc`-shared internally), so
65    /// constructing per-request `OrgStore` values off the same cluster
66    /// handle is free.

71    /// Create a new org with the given human-readable name.
72    ///
73    /// Builds an [`Org`] with a freshly-generated [`OrgId`] and writes
74    /// it to the `org` table via
75    /// `CREATE type::record($tbl, $id) SET …`. The record id is the
76    /// ULID-prefixed `OrgId` (e.g. `org-01J…`); the schema's unique
77    /// index on `name` rejects duplicate human names at the database
78    /// level, so the previous "check then insert" race is closed.
79    ///
80    /// Returns the unique-name conflict as a flat `anyhow::Error` with
81    /// a human-friendly message so the CLI keeps producing the same
82    /// "org '<name>' already exists" wording it did before P2.9.
83    pub async fn create(&self, name: &str) -> anyhow::Result<Org> {
84        let org = Org {
85            meta: ResourceMeta::new(OrgId::generate().to_string(), name),
86        };
87
88        let created_at_iso = epoch_to_iso8601(org.meta.created_at);
89        let updated_at_iso = epoch_to_iso8601(org.meta.updated_at);
90
91        // Build the row via explicit SET fields so the SurrealDB
92        // SCHEMAFULL constraints are honoured: `id` is dropped (it
93        // lives in the record id), `created_at`/`updated_at` are cast
94        // through `<datetime>` so the ISO 8601 strings parse into the
95        // schema's native `datetime` type, `labels` is bound as a JSON
96        // object, and `name`/`status` are bound as plain strings.
97        let labels_json = serde_json::to_value(&org.meta.labels)
98            .map_err(|e| anyhow::anyhow!("serialise labels: {e}"))?;
99
100        let query_result = self
101            .db
102            .client()
103            .query(
104                "CREATE type::record($tbl, $id) SET \
105                 name = $name, \
106                 status = $status, \
107                 labels = $labels, \
108                 created_at = <datetime>$created_at, \
109                 updated_at = <datetime>$updated_at",
110            )
111            .bind(("tbl", ORG_TABLE))
112            .bind(("id", org.meta.id.clone()))
113            .bind(("name", org.meta.name.clone()))
114            .bind(("status", org.meta.status.clone()))
115            .bind(("labels", labels_json))
116            .bind(("created_at", created_at_iso))
117            .bind(("updated_at", updated_at_iso))
118            .await;
119        let response = match query_result {
120            Ok(r) => r,
121            Err(e) => {
122                return Err(anyhow::anyhow!(classify_create_error(
123                    "org",
124                    name,
125                    &e.to_string()
126                )))
127            }
128        };
129        if let Err(e) = response.check() {
130            return Err(anyhow::anyhow!(classify_create_error(
131                "org",
132                name,
133                &e.to_string()
134            )));
135        }
136
137        Ok(org)
138    }

140    /// Look up an org by id (when the input looks like an `OrgId`)
141    /// or by human name (otherwise).
142    ///
143    /// The id path is a direct record-id `SELECT`; the name path uses
144    /// the schema's unique `org_name` index for an O(1) lookup. Both
145    /// paths return `Ok(None)` when no row matches — neither one is
146    /// an error.
147    pub async fn get(&self, name_or_id: &str) -> anyhow::Result<Option<Org>> {
148        if OrgId::looks_like_id(name_or_id) {
149            self.get_by_id(name_or_id).await
150        } else {
151            self.get_by_name(name_or_id).await
152        }
153    }
154
155    async fn get_by_id(&self, id: &str) -> anyhow::Result<Option<Org>> {
156        let mut response = self
157            .db
158            .client()
159            .query("SELECT * FROM type::record($tbl, $id)")
160            .bind(("tbl", ORG_TABLE))
161            .bind(("id", id.to_string()))
162            .await
163            .map_err(|e| anyhow::anyhow!("{e}"))?;
164        // Round-trip through `serde_json::Value` so callers don't have
165        // to derive `SurrealValue` on `OrgRow`. See the rationale on
166        // [`OrgRow`] and the `FabricState::load` JSON-bridge pattern.
167        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
168        match raw.into_iter().next() {
169            None => Ok(None),
170            Some(v) => {
171                let row: OrgRow = serde_json::from_value(v)
172                    .map_err(|e| anyhow::anyhow!("deserialise org row: {e}"))?;
173                Ok(Some(row.into_org()))
174            }
175        }
176    }
177
178    async fn get_by_name(&self, name: &str) -> anyhow::Result<Option<Org>> {
179        let mut response = self
180            .db
181            .client()
182            .query("SELECT * FROM org WHERE name = $name LIMIT 1")
183            .bind(("name", name.to_string()))
184            .await
185            .map_err(|e| anyhow::anyhow!("{e}"))?;
186        // Same JSON-bridge as `get_by_id` — see the comment there and
187        // the rationale block on [`OrgRow`].
188        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
189        match raw.into_iter().next() {
190            None => Ok(None),
191            Some(v) => {
192                let row: OrgRow = serde_json::from_value(v)
193                    .map_err(|e| anyhow::anyhow!("deserialise org row: {e}"))?;
194                Ok(Some(row.into_org()))
195            }
196        }
197    }

199    /// List every org in the cluster, in unspecified order.
200    ///
201    /// The SCHEMAFULL `org` table is the source of truth — the legacy
202    /// `_reg_v2` sidecar registry the previous version of this file
203    /// maintained is gone with P2.9.
204    pub async fn list(&self) -> anyhow::Result<Vec<Org>> {
205        let mut response = self
206            .db
207            .client()
208            .query("SELECT * FROM org")
209            .await
210            .map_err(|e| anyhow::anyhow!("{e}"))?;
211        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
212        let orgs: Result<Vec<Org>, _> = raw
213            .into_iter()
214            .map(|v| {
215                serde_json::from_value::<OrgRow>(v)
216                    .map(OrgRow::into_org)
217                    .map_err(|e| anyhow::anyhow!("deserialise org row: {e}"))
218            })
219            .collect();
220        orgs
221    }

223    /// Delete an org by name or id. Refuses to delete an org that
224    /// still owns any project — the caller must remove the children
225    /// first.
226    pub async fn delete(&self, name_or_id: &str) -> anyhow::Result<()> {
227        let org = self
228            .get(name_or_id)
229            .await?
230            .ok_or_else(|| anyhow::anyhow!("org '{name_or_id}' not found"))?;
231
232        // Refuse to delete an org that still has projects. P2.10
233        // (sifrah/nauka#214) migrated the project store to the native
234        // SurrealDB SDK on top of the SCHEMAFULL `project` table, so
235        // we can count the surviving children with a direct query on
236        // the owning-org id. This store still doesn't import
237        // `ProjectStore` because that would force a circular module
238        // dependency (`ProjectStore::delete` reaches back into
239        // `OrgStore::get` to resolve the parent org) — the inline
240        // query is cheaper.
241        let remaining_projects = self.count_projects_in_org(&org.meta.id).await?;
242        if remaining_projects > 0 {
243            anyhow::bail!(
244                "org '{}' has {} project(s). Delete them first.",
245                org.meta.name,
246                remaining_projects
247            );
248        }
249
250        let result = self
251            .db
252            .client()
253            .query("DELETE type::record($tbl, $id)")
254            .bind(("tbl", ORG_TABLE))
255            .bind(("id", org.meta.id.clone()))
256            .await
257            .map_err(|e| anyhow::anyhow!("{e}"))?;
258        result.check().map_err(|e| anyhow::anyhow!("{e}"))?;
259        Ok(())
260    }

262    /// Count the projects currently owned by the org whose record-id
263    /// is `org_id`.
264    ///
265    /// Uses the SCHEMAFULL `project` table that P2.10 migrated to, so
266    /// the query is a single `SELECT name FROM project WHERE org =
267    /// $org`. The `project` table is defined by
268    /// `nauka_state::apply_cluster_schemas` at bootstrap, so a fresh
269    /// database already has the table — no `DEFINE TABLE IF NOT
270    /// EXISTS` dance is needed. If the schema hasn't been applied at
271    /// all (e.g. a pre-bootstrap sanity test), the SurrealDB error
272    /// bubbles up as an `anyhow::Error` and the caller surfaces it.
273    async fn count_projects_in_org(&self, org_id: &str) -> anyhow::Result<usize> {
274        let mut response = self
275            .db
276            .client()
277            .query("SELECT name FROM project WHERE org = $org")
278            .bind(("org", org_id.to_string()))
279            .await
280            .map_err(|e| anyhow::anyhow!("{e}"))?;
281        let rows: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
282        Ok(rows.len())
283    }
284}

286/// Row shape returned by `SELECT * FROM org`. Mirrors the SCHEMAFULL
287/// `org` table's column set with one ergonomic conversion: SurrealDB
288/// hands the record id back as a typed `Thing` (table + id), which
289/// serde_json renders as `{"tb": "org", "id": {"String": "..."}}`. We
290/// bridge through `serde_json::Value` and pull the inner string out in
291/// [`OrgRow::into_org`] so callers see the same flat `org-<ulid>` id
292/// string that they always have.
293///
294/// `created_at` / `updated_at` arrive as RFC 3339 / ISO 8601 strings
295/// from SurrealDB; we parse them back into Unix-epoch seconds with
296/// [`iso8601_to_epoch`] so the resulting [`Org`] is bit-identical to
297/// what `OrgStore::create` originally produced.
298#[derive(Debug, Deserialize)]
299struct OrgRow {
300    id: serde_json::Value,
301    name: String,
302    #[serde(default)]
303    status: Option<String>,
304    #[serde(default)]
305    labels: Option<serde_json::Value>,
306    created_at: String,
307    updated_at: String,
308}
309
310impl OrgRow {
311    fn into_org(self) -> Org {
312        let id = thing_to_id_string("org:", &self.id);
313        let labels = self
314            .labels
315            .and_then(|v| serde_json::from_value(v).ok())
316            .unwrap_or_default();
317        Org {
318            meta: ResourceMeta {
319                id,
320                name: self.name,
321                status: self.status.unwrap_or_else(|| "active".to_string()),
322                labels,
323                created_at: iso8601_to_epoch(&self.created_at),
324                updated_at: iso8601_to_epoch(&self.updated_at),
325            },
326        }
327    }
328}

334    /// Build an in-process `OrgStore` backed by a fresh SurrealKV
335    /// datastore at a temporary path, with the `org` schema applied.
336    ///
337    /// The tempdir handle has to live as long as the store, so we
338    /// return it together. SurrealDB's SCHEMAFULL semantics are
339    /// identical across SurrealKv and TiKv (the Phase-2 contract is
340    /// that one `EmbeddedDb` wrapper fronts both, with the same SDK
341    /// surface), so this is a faithful rehearsal of production cluster
342    /// state without needing a live TiKV cluster on the test host.
343    async fn temp_store() -> (tempfile::TempDir, OrgStore) {
344        let dir = tempfile::tempdir().expect("tempdir");
345        let db = EmbeddedDb::open(&dir.path().join("orgs.skv"))
346            .await
347            .expect("open EmbeddedDb at temp path");
348        // Apply the cluster schema bundle so the SCHEMAFULL `org` table
349        // (and every parent/child it depends on) is live before the
350        // store touches it.
351        nauka_state::apply_cluster_schemas(&db)
352            .await
353            .expect("apply cluster schemas");
354        (dir, OrgStore::new(db))
355    }

453    /// Delete refuses to drop an org that still has a project pointing
454    /// at it.
455    ///
456    /// We seed the SCHEMAFULL `project` table directly (via a `CREATE`
457    /// bypassing `ProjectStore::create`) so this test stays
458    /// self-contained — it doesn't pull in `ProjectStore`, which
459    /// reaches back into `OrgStore::get` to resolve the parent org
460    /// and would force a circular import dance into this file.
461    #[tokio::test]
462    async fn delete_refuses_when_project_remains() {
463        let (_d, store) = temp_store().await;
464        let org = store.create("acme").await.expect("create org");
465
466        // Seed a `project` row directly. The SCHEMAFULL `project`
467        // table is created by `apply_cluster_schemas` in `temp_store`,
468        // so every column has to be populated to satisfy the ASSERT
469        // constraints.
470        store
471            .db
472            .client()
473            .query(
474                "CREATE type::record('project', 'project-fake') SET \
475                 name = 'fakeproj', \
476                 status = 'active', \
477                 labels = {}, \
478                 org = $org, \
479                 org_name = 'acme', \
480                 created_at = time::now(), \
481                 updated_at = time::now()",
482            )
483            .bind(("org", org.meta.id.clone()))
484            .await
485            .expect("seed project row")
486            .check()
487            .expect("seed project row check");
488
489        // Now `delete` should refuse the org — there's a project still
490        // pointing at it via `org = <org.meta.id>`.
491        let err = store
492            .delete("acme")
493            .await
494            .expect_err("delete should refuse while a project still exists");
495        let msg = err.to_string();
496        assert!(
497            msg.contains("project") && msg.contains("Delete them first"),
498            "expected 'has N project(s). Delete them first' message, got: {msg}",
499        );
500
501        // The org is still there — the failed delete must not have
502        // removed it as a side-effect.
503        assert!(store.get("acme").await.unwrap().is_some());
504    }
