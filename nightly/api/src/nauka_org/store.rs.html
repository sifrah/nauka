1//! Org persistence in ClusterDb (TiKV).
2
3use nauka_core::id::OrgId;
4use nauka_core::resource::ResourceMeta;
5use nauka_hypervisor::controlplane::ClusterDb;
6
7use crate::types::Org;
8
9const NS_ORG: &str = "org";
10const NS_ORG_IDX: &str = "org-idx";
11const REG_ORGS: (&str, &str) = ("_reg", "org-ids");
12
13pub struct OrgStore {
14    db: ClusterDb,
15}
16
17impl OrgStore {
18    pub fn new(db: ClusterDb) -> Self {
19        Self { db }
20    }
21
22    pub async fn create(&self, name: &str) -> anyhow::Result<Org> {
23        if self.get_by_name(name).await?.is_some() {
24            anyhow::bail!("org '{name}' already exists");
25        }
26
27        let org = Org {
28            meta: ResourceMeta::new(OrgId::generate().to_string(), name),
29        };
30
31        self.db.put(NS_ORG, &org.meta.id, &org).await?;
32        self.db
33            .put(NS_ORG_IDX, &org.meta.name, &org.meta.id)
34            .await?;
35        add_id(&self.db, &org.meta.id).await?;
36
37        Ok(org)
38    }
39
40    pub async fn get(&self, name_or_id: &str) -> anyhow::Result<Option<Org>> {
41        if OrgId::looks_like_id(name_or_id) {
42            self.db.get(NS_ORG, name_or_id).await.map_err(Into::into)
43        } else {
44            self.get_by_name(name_or_id).await
45        }
46    }
47
48    async fn get_by_name(&self, name: &str) -> anyhow::Result<Option<Org>> {
49        let id: Option<String> = self.db.get(NS_ORG_IDX, name).await?;
50        match id {
51            Some(id) => self.db.get(NS_ORG, &id).await.map_err(Into::into),
52            None => Ok(None),
53        }
54    }
55
56    pub async fn list(&self) -> anyhow::Result<Vec<Org>> {
57        let ids = load_ids(&self.db).await?;
58        let mut orgs = Vec::new();
59        for id in &ids {
60            if let Some(org) = self.db.get::<Org>(NS_ORG, id).await? {
61                orgs.push(org);
62            }
63        }
64        Ok(orgs)
65    }
66
67    pub async fn delete(&self, name_or_id: &str) -> anyhow::Result<()> {
68        let org = self
69            .get(name_or_id)
70            .await?
71            .ok_or_else(|| anyhow::anyhow!("org '{name_or_id}' not found"))?;
72
73        let proj_store = crate::project::store::ProjectStore::new(self.db.clone());
74        let projects = proj_store.list(Some(&org.meta.name)).await?;
75        if !projects.is_empty() {
76            anyhow::bail!(
77                "org '{}' has {} project(s). Delete them first.",
78                org.meta.name,
79                projects.len()
80            );
81        }
82
83        self.db.delete(NS_ORG, &org.meta.id).await?;
84        self.db.delete(NS_ORG_IDX, &org.meta.name).await?;
85        remove_id(&self.db, &org.meta.id).await?;
86        Ok(())
87    }
88}
89
90async fn load_ids(db: &ClusterDb) -> anyhow::Result<Vec<String>> {
91    let ids: Option<Vec<String>> = db.get(REG_ORGS.0, REG_ORGS.1).await?;
92    Ok(ids.unwrap_or_default())
93}
94
95async fn add_id(db: &ClusterDb, id: &str) -> anyhow::Result<()> {
96    let mut ids = load_ids(db).await?;
97    ids.push(id.to_string());
98    db.put(REG_ORGS.0, REG_ORGS.1, &ids).await?;
99    Ok(())
100}
101
102async fn remove_id(db: &ClusterDb, id: &str) -> anyhow::Result<()> {
103    let mut ids = load_ids(db).await?;
104    ids.retain(|i| i != id);
105    db.put(REG_ORGS.0, REG_ORGS.1, &ids).await?;
106    Ok(())
107}
