36/// Check if TiUP is installed.
37fn tiup_available() -> bool {
38    Path::new(&format!("{TIUP_HOME}/bin/tiup")).exists()
39}

41/// Check if PD binary is available.
42fn pd_available() -> bool {
43    // TiUP installs components under TIUP_HOME/components/pd/...
44    Command::new(format!("{TIUP_HOME}/bin/tiup"))
45        .args(["list", "--installed"])
46        .env("TIUP_HOME", TIUP_HOME)
47        .output()
48        .ok()
49        .and_then(|o| String::from_utf8(o.stdout).ok())
50        .map(|s| s.contains("pd"))
51        .unwrap_or(false)
52}

54/// Install TiUP and TiKV components.
55pub fn ensure_installed() -> Result<(), NaukaError> {
56    // Create dirs
57    std::fs::create_dir_all(format!("{TIUP_HOME}/bin")).map_err(NaukaError::from)?;
58    std::fs::create_dir_all("/etc/nauka").map_err(NaukaError::from)?;
59
60    if !tiup_available() {
61        // Map arch: x86_64→amd64, aarch64→arm64
62        let output = Command::new("sh")
63            .args(["-c", &format!(
64                "ARCH=$(uname -m | sed 's/x86_64/amd64/' | sed 's/aarch64/arm64/') && \
65                 curl -fsSL https://tiup-mirrors.pingcap.com/tiup-linux-$ARCH.tar.gz | tar -xz -C {TIUP_HOME}/bin/"
66            )])
67            .output()
68            .map_err(|e| NaukaError::internal(format!("tiup download failed: {e}")))?;
69
70        if !output.status.success() {
71            let stderr = String::from_utf8_lossy(&output.stderr);
72            return Err(NaukaError::internal(format!(
73                "tiup install failed: {stderr}"
74            )));
75        }
76
77        // Initialize TiUP mirror
78        let init = Command::new(format!("{TIUP_HOME}/bin/tiup"))
79            .args(["mirror", "set", "https://tiup-mirrors.pingcap.com"])
80            .env("TIUP_HOME", TIUP_HOME)
81            .output()
82            .map_err(|e| NaukaError::internal(format!("tiup mirror set failed: {e}")))?;
83
84        if !init.status.success() {
85            let stderr = String::from_utf8_lossy(&init.stderr);
86            return Err(NaukaError::internal(format!(
87                "tiup mirror init failed: {stderr}"
88            )));
89        }
90    }
91
92    if !pd_available() {
93        let output = Command::new(format!("{TIUP_HOME}/bin/tiup"))
94            .args(["install", "pd", "tikv"])
95            .env("TIUP_HOME", TIUP_HOME)
96            .output()
97            .map_err(|e| NaukaError::internal(format!("component install failed: {e}")))?;
98
99        if !output.status.success() {
100            let stderr = String::from_utf8_lossy(&output.stderr);
101            return Err(NaukaError::internal(format!(
102                "PD/TiKV install failed: {stderr}"
103            )));
104        }
105    }
106
107    Ok(())
108}

110/// Install specific versions of PD and TiKV via TiUP.
111///
112/// Runs `tiup install pd:VERSION tikv:VERSION` to fetch the requested
113/// component versions. TiUP keeps old versions around, so the upgrade
114/// is non-destructive — rolling back to the previous version only
115/// requires restarting with the old `tiup pd`/`tiup tikv` invocation.
116pub fn install_version(pd_version: &str, tikv_version: &str) -> Result<(), NaukaError> {
117    let output = Command::new(format!("{TIUP_HOME}/bin/tiup"))
118        .args([
119            "install",
120            &format!("pd:{pd_version}"),
121            &format!("tikv:{tikv_version}"),
122        ])
123        .env("TIUP_HOME", TIUP_HOME)
124        .output()
125        .map_err(|e| NaukaError::internal(format!("tiup install failed: {e}")))?;
126
127    if !output.status.success() {
128        let stderr = String::from_utf8_lossy(&output.stderr);
129        return Err(NaukaError::internal(format!(
130            "tiup install pd:{pd_version} tikv:{tikv_version} failed: {stderr}"
131        )));
132    }
133
134    Ok(())
135}

137/// Regenerate and rewrite the systemd unit files so they launch
138/// the currently installed component versions.
139///
140/// TiUP selects the latest installed version by default, so we just
141/// rewrite the units and daemon-reload.
142pub fn regenerate_units(has_pd: bool) -> Result<(), NaukaError> {
143    if has_pd {
144        // Preserve join mode if the unit already exists in join mode
145        let existing = std::fs::read_to_string(PD_UNIT_PATH).unwrap_or_default();
146        let join_url = if existing.contains("--join=") {
147            existing
148                .lines()
149                .find(|l| l.contains("--join="))
150                .and_then(|l| l.split("--join=").nth(1))
151                .map(|s| s.trim().to_string())
152        } else {
153            None
154        };
155        std::fs::write(PD_UNIT_PATH, generate_pd_unit(join_url.as_deref()))
156            .map_err(NaukaError::from)?;
157    }
158
159    std::fs::write(TIKV_UNIT_PATH, generate_tikv_unit()).map_err(NaukaError::from)?;
160    run_systemctl(&["daemon-reload"])?;
161
162    Ok(())
163}

170/// Wait for this node's TiKV store to reach "Up" state in PD.
171/// Polls every 5s, up to `timeout_secs`.
172pub fn wait_store_up(mesh_ipv6: &Ipv6Addr, timeout_secs: u64) -> Result<(), NaukaError> {
173    let our_addr = format!("[{}]:{}", mesh_ipv6, super::TIKV_PORT);
174    let pd_url = format!("http://[{}]:{}", mesh_ipv6, super::PD_CLIENT_PORT);
175    let stores_url = format!("{pd_url}/pd/api/v1/stores");
176    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(timeout_secs);
177
178    while std::time::Instant::now() < deadline {
179        if let Ok(output) = Command::new("curl")
180            .args(["-sf", "--max-time", "5", &stores_url])
181            .output()
182        {
183            if output.status.success() {
184                if let Ok(body) = serde_json::from_slice::<serde_json::Value>(&output.stdout) {
185                    if let Some(stores) = body["stores"].as_array() {
186                        let up = stores.iter().any(|s| {
187                            s["store"]["address"].as_str() == Some(our_addr.as_str())
188                                && s["store"]["state_name"].as_str() == Some("Up")
189                        });
190                        if up {
191                            return Ok(());
192                        }
193                    }
194                }
195            }
196        }
197        std::thread::sleep(std::time::Duration::from_secs(5));
198    }
199
200    Err(NaukaError::timeout("TiKV store Up", timeout_secs))
201}

219/// Generate pd.toml config.
220/// For join mode, initial-cluster is omitted (--join flag handles it).
221pub fn generate_pd_conf(cfg: &PdConfig, is_join: bool) -> String {
222    let cluster_section = if is_join {
223        // join mode: no initial-cluster in config (--join flag on command line)
224        String::new()
225    } else {
226        format!(
227            r#"initial-cluster = "{}"
228initial-cluster-state = "new"
229"#,
230            cfg.initial_cluster
231        )
232    };
233
234    format!(
235        r#"# Nauka PD configuration — auto-generated
236name = "{name}"
237data-dir = "{PD_DATA_DIR}"
238
239client-urls = "http://[{ip}]:{client_port}"
240peer-urls = "http://[{ip}]:{peer_port}"
241advertise-client-urls = "http://[{ip}]:{client_port}"
242advertise-peer-urls = "http://[{ip}]:{peer_port}"
243
244{cluster_section}
245[log]
246level = "warn"
247
248[log.file]
249filename = "/var/log/nauka/pd.log"
250max-size = 50
251"#,
252        name = cfg.name,
253        ip = cfg.mesh_ipv6,
254        client_port = super::PD_CLIENT_PORT,
255        peer_port = super::PD_PEER_PORT,
256    )
257}

267/// Generate tikv.toml config.
268pub fn generate_tikv_conf(cfg: &TikvConfig) -> String {
269    let pd_endpoints: Vec<String> = cfg
270        .pd_endpoints
271        .iter()
272        .map(|e| format!("\"{e}\""))
273        .collect();

278addr = "[{ip}]:{tikv_port}"
279advertise-addr = "[{ip}]:{tikv_port}"
280status-addr = "[{ip}]:{status_port}"
281grpc-keepalive-time = "15s"
282grpc-keepalive-timeout = "30s"
283
284[storage]
285data-dir = "{TIKV_DATA_DIR}"
286
287[pd]
288endpoints = [{pd_list}]
289
290[log]
291level = "warn"
292
293[log.file]
294filename = "/var/log/nauka/tikv.log"
295max-size = 50
296
297[raftstore]
298# Fsync WAL on every write — prevents data loss on crash
299sync-log = true
300# Reduce resource usage for small clusters
301capacity = "0"
302
303[security.encryption]
304data-encryption-method = "aes256-ctr"
305data-key-rotation-period = "168h"
306
307[security.encryption.master-key]
308type = "file"
309path = "{TIKV_MASTER_KEY_PATH}"
310"#,
311        ip = cfg.mesh_ipv6,
312        tikv_port = super::TIKV_PORT,
313        status_port = super::TIKV_STATUS_PORT,
314        pd_list = pd_endpoints.join(", "),
315    )
316}

322fn generate_pd_unit(join_url: Option<&str>) -> String {
323    let exec_start = match join_url {
324        Some(url) => {
325            format!("ExecStart={TIUP_HOME}/bin/tiup pd --config={PD_CONF_PATH} --join={url}")
326        }
327        None => format!("ExecStart={TIUP_HOME}/bin/tiup pd --config={PD_CONF_PATH}"),
328    };
329
330    format!(
331        r#"[Unit]
332Description=Nauka Placement Driver (PD)
333After=network-online.target nauka-wg.service
334Wants=network-online.target
335Requires=nauka-wg.service
336
337[Service]
338Type=simple
339Environment=HOME=/root
340Environment=TIUP_HOME={TIUP_HOME}
341{exec_start}
342Restart=always
343RestartSec=5
344LimitNOFILE=1000000
345OOMScoreAdjust=-999
346
347[Install]
348WantedBy=multi-user.target
349"#
350    )
351}

381/// Ensure the TiKV encryption master key file exists.
382///
383/// Generates a 256-bit random key if the file doesn't already exist.
384/// Sets file permissions to 0600 (owner read/write only).
385pub fn ensure_master_key() -> Result<(), NaukaError> {
386    use std::io::Read;
387    use std::os::unix::fs::PermissionsExt;
388
389    let path = Path::new(TIKV_MASTER_KEY_PATH);
390    if path.exists() {
391        tracing::debug!("TiKV master key already exists at {TIKV_MASTER_KEY_PATH}");
392        return Ok(());
393    }
394
395    std::fs::create_dir_all("/etc/nauka").map_err(NaukaError::from)?;
396
397    // Generate 256-bit (32-byte) random key from /dev/urandom
398    let mut key = [0u8; 32];
399    let mut f = std::fs::File::open("/dev/urandom")
400        .map_err(|e| NaukaError::internal(format!("failed to open /dev/urandom: {e}")))?;
401    f.read_exact(&mut key)
402        .map_err(|e| NaukaError::internal(format!("failed to read random bytes: {e}")))?;
403
404    // Write key as hex string with trailing newline (TiKV expects 65 bytes)
405    let mut hex_key: String = key.iter().map(|b| format!("{b:02x}")).collect();
406    hex_key.push('\n');
407    std::fs::write(path, hex_key.as_bytes()).map_err(NaukaError::from)?;
408
409    // Set permissions to 0600
410    std::fs::set_permissions(path, std::fs::Permissions::from_mode(0o600))
411        .map_err(NaukaError::from)?;
412
413    tracing::info!("Generated TiKV encryption master key at {TIKV_MASTER_KEY_PATH}");
414    Ok(())
415}

417/// Install PD + TiKV configs and systemd units.
418/// `join_url` is Some for joining an existing cluster.
419pub fn install(
420    pd_cfg: &PdConfig,
421    tikv_cfg: &TikvConfig,
422    join_url: Option<&str>,
423) -> Result<(), NaukaError> {
424    // Create directories
425    std::fs::create_dir_all(PD_DATA_DIR).map_err(NaukaError::from)?;
426    std::fs::create_dir_all(TIKV_DATA_DIR).map_err(NaukaError::from)?;
427    std::fs::create_dir_all("/var/log/nauka").map_err(NaukaError::from)?;
428    std::fs::create_dir_all("/etc/nauka").map_err(NaukaError::from)?;
429
430    // Generate encryption master key before TiKV starts
431    ensure_master_key()?;
432
433    // Write configs
434    let is_join = join_url.is_some();
435    std::fs::write(PD_CONF_PATH, generate_pd_conf(pd_cfg, is_join)).map_err(NaukaError::from)?;
436    std::fs::write(TIKV_CONF_PATH, generate_tikv_conf(tikv_cfg)).map_err(NaukaError::from)?;
437
438    // Write systemd units
439    std::fs::write(PD_UNIT_PATH, generate_pd_unit(join_url)).map_err(NaukaError::from)?;
440    std::fs::write(TIKV_UNIT_PATH, generate_tikv_unit()).map_err(NaukaError::from)?;
441
442    // Reload systemd
443    run_systemctl(&["daemon-reload"])?;
444
445    Ok(())
446}

516/// Deregister this TiKV store from PD before leaving.
517/// Reads the local TiKV config to find our address, then deletes the store via PD API.
518pub fn deregister_store(mesh_ipv6: &std::net::Ipv6Addr) -> Result<(), NaukaError> {
519    let our_addr = format!("[{}]:{}", mesh_ipv6, super::TIKV_PORT);
520
521    // Find a reachable PD endpoint from our config
522    let conf = std::fs::read_to_string(TIKV_CONF_PATH).unwrap_or_default();
523    let pd_endpoint = conf
524        .lines()
525        .find(|l| l.contains("http://"))
526        .and_then(|l| {
527            l.trim()
528                .trim_matches('"')
529                .trim_matches(',')
530                .split('"')
531                .find(|s| s.starts_with("http://"))
532        })
533        .map(|s| s.to_string());
534
535    let pd_url = match pd_endpoint {
536        Some(url) => url,
537        None => return Ok(()), // No PD endpoint found, nothing to deregister
538    };
539
540    let client = super::pd_client::PdClient::new(vec![pd_url]);
541    if let Ok(stores) = client.get_stores() {
542        for store in &stores {
543            if store.address == our_addr && store.id > 0 {
544                tracing::info!(store_id = store.id, "deregistering TiKV store");
545                let _ = client.delete_store(store.id);
546            }
547        }
548    }
549
550    Ok(())
551}

553/// Recover from a TiKV data wipe: if TiKV is not running and PD has a stale
554/// store registered at our address, remove it via unsafe recovery API so TiKV
555/// can re-register.
556///
557/// Returns `true` if recovery was performed (TiKV was restarted).
558pub fn recover_stale_store(mesh_ipv6: &std::net::Ipv6Addr) -> bool {
559    // Only act when TiKV is installed but not running
560    if !Path::new(TIKV_UNIT_PATH).exists() || tikv_is_active() {
561        return false;
562    }
563
564    // PD must be reachable
565    if !pd_is_active() {
566        return false;
567    }
568
569    let our_addr = format!("[{}]:{}", mesh_ipv6, super::TIKV_PORT);
570    let client = super::pd_client::PdClient::from_mesh(mesh_ipv6);
571
572    // Find stale stores at our address (any state)
573    let stale_ids: Vec<u64> = client
574        .get_stores_with_states(&[0, 1, 2])
575        .ok()
576        .map(|stores| {
577            stores
578                .iter()
579                .filter(|s| s.address == our_addr)
580                .map(|s| s.id)
581                .collect()
582        })
583        .unwrap_or_default();
584
585    if stale_ids.is_empty() {
586        return false;
587    }
588
589    // Stop TiKV to prevent crash-loop during cleanup
590    let _ = run_systemctl(&["stop", TIKV_SERVICE]);
591
592    tracing::warn!(
593        stores = ?stale_ids,
594        addr = our_addr.as_str(),
595        "removing stale TiKV stores via unsafe recovery"
596    );
597
598    // Step 1: Force-delete each store to move it from Up -> Offline.
599    // PD's unsafe API requires stores to be in a non-Up state.
600    for store_id in &stale_ids {
601        let _ = client.force_delete_store(*store_id);
602    }
603
604    // Brief pause for PD to process the state transitions
605    std::thread::sleep(std::time::Duration::from_secs(2));
606
607    // Step 2: Use PD's unsafe/remove-failed-stores API — this forcefully
608    // evicts dead stores and reassigns their region peers to alive stores.
609    let _ = client.unsafe_remove_failed_stores(&stale_ids);
610
611    // Wait for the unsafe recovery to complete (typically 10-30s)
612    let mut cleared = false;
613    for _ in 0..12 {
614        std::thread::sleep(std::time::Duration::from_secs(5));
615
616        // Check if all stale stores are now Tombstone or gone
617        let still_blocking: Vec<u64> = client
618            .get_stores_with_states(&[0, 1, 2])
619            .ok()
620            .map(|stores| {
621                stores
622                    .iter()
623                    .filter(|s| s.address == our_addr && s.state_name != "Tombstone")
624                    .map(|s| s.id)
625                    .collect()
626            })
627            .unwrap_or_default();
628
629        if still_blocking.is_empty() {
630            // Purge tombstones to fully free the address
631            let _ = client.remove_tombstone();
632            cleared = true;
633            break;
634        }
635    }
636
637    if cleared {
638        tracing::info!("restarting TiKV after stale store removal");
639    } else {
640        tracing::warn!("stale stores not fully cleared, restarting TiKV anyway");
641    }
642    let _ = run_systemctl(&["restart", TIKV_SERVICE]);
643    true
644}

646/// Phase 1 of PD recovery (runs on the SURVIVING node): if our local PD is
647/// healthy but has an unhealthy peer, force the cluster to single-member mode
648/// to restore quorum, then remove the dead member.
649///
650/// Returns `true` if recovery was performed.
651pub fn recover_pd_quorum(mesh_ipv6: &std::net::Ipv6Addr) -> bool {
652    if !pd_is_active() {
653        return false;
654    }
655
656    let client = super::pd_client::PdClient::from_mesh(mesh_ipv6);
657
658    // Check health — if PD unreachable, nothing to do
659    let health = match client.get_health() {
660        Ok(h) => h,
661        Err(_) => return false,
662    };
663
664    // Check if any member is unhealthy
665    let unhealthy: Vec<String> = health
666        .iter()
667        .filter(|m| !m.healthy)
668        .map(|m| m.name.clone())
669        .collect();
670
671    if unhealthy.is_empty() {
672        return false;
673    }
674
675    // Check if the PD members API works (needs quorum)
676    let members_ok = client.get_members().is_ok();
677
678    if members_ok {
679        // Quorum is fine — just remove unhealthy members normally
680        for name in &unhealthy {
681            tracing::warn!(member = name.as_str(), "removing unhealthy PD member");
682            let _ = client.delete_member_by_name(name);
683        }
684        return true;
685    }
686
687    // Quorum is lost — force single-member cluster to restore it.
688    tracing::warn!(
689        unhealthy = ?unhealthy,
690        "PD quorum lost — forcing single-member recovery"
691    );
692
693    let _ = run_systemctl(&["stop", PD_SERVICE]);
694
695    // Run pd-server --force-new-cluster briefly to rewrite etcd state
696    let pd_binary = format!("{TIUP_HOME}/components/pd/{}/pd-server", super::PD_VERSION);
697    let child = Command::new(&pd_binary)
698        .args(["--config", PD_CONF_PATH, "--force-new-cluster"])
699        .stdout(std::process::Stdio::null())
700        .stderr(std::process::Stdio::null())
701        .spawn();
702
703    match child {
704        Ok(mut child) => {
705            // Wait for PD to start and rewrite cluster state
706            std::thread::sleep(std::time::Duration::from_secs(8));
707            let _ = child.kill();
708            let _ = child.wait();
709        }
710        Err(e) => {
711            tracing::error!(error = %e, "failed to run pd-server --force-new-cluster");
712            let _ = run_systemctl(&["start", PD_SERVICE]);
713            return false;
714        }
715    }
716
717    // Restart PD normally (now single-member, quorum restored)
718    tracing::info!("restarting PD after force-new-cluster recovery");
719    let _ = run_systemctl(&["start", PD_SERVICE]);
720
721    true
722}

724/// Phase 2 of PD recovery (runs on the WIPED node): if local PD is not
725/// running, find a healthy remote PD, remove our stale member, and rejoin.
726///
727/// `peer_ipv6s` are mesh IPv6 addresses of other nodes that may run PD.
728/// Returns `true` if recovery was performed (PD was restarted).
729pub fn recover_stale_pd_member(
730    mesh_ipv6: &std::net::Ipv6Addr,
731    node_name: &str,
732    peer_ipv6s: &[std::net::Ipv6Addr],
733) -> bool {
734    // Only act when PD is installed but not running
735    if !Path::new(PD_UNIT_PATH).exists() || pd_is_active() {
736        return false;
737    }
738
739    // Find a reachable remote PD that can serve the members API (has quorum)
740    let remote_client = peer_ipv6s.iter().find_map(|ip| {
741        let client = super::pd_client::PdClient::from_mesh(ip);
742        if client.get_members().is_ok() {
743            Some(client)
744        } else {
745            None
746        }
747    });
748
749    let remote_client = match remote_client {
750        Some(c) => c,
751        None => return false, // No PD with quorum reachable (phase 1 needed first)
752    };
753
754    // Remove our stale member from the cluster
755    let our_ip = mesh_ipv6.to_string();
756    let mut found_stale = false;
757
758    if let Ok(members) = remote_client.get_members() {
759        for member in &members {
760            let peer_urls_joined = member.peer_urls.join(",");
761            if (peer_urls_joined.contains(&our_ip) || member.name == node_name)
762                && member.member_id > 0
763            {
764                tracing::warn!(
765                    member_id = member.member_id,
766                    member_name = member.name.as_str(),
767                    "removing stale PD member for recovery"
768                );
769                let _ = remote_client.delete_member_by_name(&member.name);
770                found_stale = true;
771            }
772        }
773    }
774
775    if !found_stale {
776        // No stale member — we can still try to join
777        tracing::info!("no stale PD member found, attempting join");
778    }
779
780    // Wipe PD data directory (stale etcd state prevents rejoin)
781    let _ = std::fs::remove_dir_all(PD_DATA_DIR);
782    let _ = std::fs::create_dir_all(PD_DATA_DIR);
783
784    // Reconstruct the remote PD URL for the systemd --join flag
785    let remote_pd_url = peer_ipv6s
786        .iter()
787        .map(|ip| format!("http://[{}]:{}", ip, super::PD_CLIENT_PORT))
788        .next()
789        .unwrap_or_default();
790
791    // Rewrite systemd unit in --join mode so PD rejoins the cluster
792    std::fs::write(PD_UNIT_PATH, generate_pd_unit(Some(&remote_pd_url))).ok();
793
794    // Rewrite pd.toml in join mode (no initial-cluster)
795    if let Ok(conf) = std::fs::read_to_string(PD_CONF_PATH) {
796        let new_conf = conf
797            .lines()
798            .filter(|l| {
799                !l.starts_with("initial-cluster") && !l.starts_with("initial-cluster-state")
800            })
801            .collect::<Vec<_>>()
802            .join("\n");
803        let _ = std::fs::write(PD_CONF_PATH, new_conf);
804    }
805
806    let _ = run_systemctl(&["daemon-reload"]);
807
808    std::thread::sleep(std::time::Duration::from_secs(2));
809    tracing::info!("restarting PD in join mode after stale member removal");
810    let _ = run_systemctl(&["restart", PD_SERVICE]);
811
812    true
813}

850/// Wait for all regions to have a leader (post-rebalance).
851/// Polls every 5s, up to timeout_secs.
852pub fn wait_regions_healthy(pd_url: &str, timeout_secs: u64) -> Result<(), NaukaError> {
853    let client = super::pd_client::PdClient::new(vec![pd_url.to_string()]).with_timeout(5);
854    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(timeout_secs);
855
856    while std::time::Instant::now() < deadline {
857        if let Ok(body) = client.get_regions() {
858            if let Some(regions) = body["regions"].as_array() {
859                let total = regions.len();
860                let with_leader = regions
861                    .iter()
862                    .filter(|r| r["leader"]["store_id"].as_u64().unwrap_or(0) > 0)
863                    .count();
864
865                if total > 0 && with_leader == total {
866                    tracing::info!(total, "all regions have leaders");
867                    return Ok(());
868                }
869                tracing::debug!(with_leader, total, "waiting for region leaders");
870            }
871        }
872        std::thread::sleep(std::time::Duration::from_secs(5));
873    }
874
875    // Not fatal — best effort
876    tracing::warn!("timed out waiting for region leaders, proceeding anyway");
877    Ok(())
878}

920/// Wait for all regions to drain off a specific store.
921/// Polls PD every 5s with progress logging. Returns Ok(()) when the store
922/// has 0 regions or is no longer visible (already tombstoned).
923pub fn wait_store_regions_drained(
924    pd_url: &str,
925    store_id: u64,
926    timeout_secs: u64,
927) -> Result<(), NaukaError> {
928    let client = super::pd_client::PdClient::new(vec![pd_url.to_string()]).with_timeout(5);
929    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(timeout_secs);
930    let mut last_count: Option<usize> = None;
931
932    while std::time::Instant::now() < deadline {
933        let region_count = client.count_store_regions(store_id);
934
935        if region_count == 0 {
936            tracing::info!(store_id, "all regions drained from store");
937            return Ok(());
938        }
939
940        // Log progress when count changes
941        if last_count != Some(region_count) {
942            tracing::info!(
943                store_id,
944                region_count,
945                "draining regions: {} remaining...",
946                region_count
947            );
948            last_count = Some(region_count);
949        }
950
951        std::thread::sleep(std::time::Duration::from_secs(5));
952    }
953
954    tracing::warn!(
955        store_id,
956        "timed out waiting for region drain, proceeding anyway"
957    );
958    Ok(())
959}

999        let _ = run_systemctl(&["restart", TIKV_SERVICE]);
1000    }
1001
1002    Ok(())

1069fn run_systemctl(args: &[&str]) -> Result<(), NaukaError> {
1070    let output = Command::new("systemctl")
1071        .args(args)
1072        .output()
1073        .map_err(|e| NaukaError::internal(format!("systemctl failed: {e}")))?;
1074
1075    if !output.status.success() {
1076        let stderr = String::from_utf8_lossy(&output.stderr);
1077        return Err(NaukaError::internal(format!(
1078            "systemctl {} failed: {}",
1079            args.join(" "),
1080            stderr.trim()
1081        )));
1082    }
1083
1084    Ok(())
1085}
