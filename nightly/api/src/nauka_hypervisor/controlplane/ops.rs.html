1//! Control plane operations — high-level orchestration.
2//!
3//! Called by the hypervisor handler during init/join/leave/status.
4//! Orchestrates TiUP install → config → systemd → health check.
5//!
6//! PD (Raft consensus) runs on max 3 nodes for optimal performance.
7//! Additional nodes run TiKV only (storage), connecting to existing PD.
8

16/// Max PD members in the cluster. Raft works best with 3 or 5.
17const MAX_PD_MEMBERS: usize = 3;

19/// Bootstrap a new single-node TiKV cluster.
20///
21/// Uses `steps` to report progress (consumes 4 steps).
22pub fn bootstrap(
23    node_name: &str,
24    mesh_ipv6: &Ipv6Addr,
25    steps: &ui::Steps,
26) -> Result<(), NaukaError> {
27    tracing::info!(node_name, %mesh_ipv6, "controlplane bootstrap starting");
28
29    steps.set("Installing control plane");
30    service::ensure_installed()?;
31    steps.inc();
32
33    steps.set("Starting control plane");
34    let pd_cfg = PdConfig {
35        name: node_name.to_string(),
36        mesh_ipv6: *mesh_ipv6,
37        initial_cluster: format!("{node_name}=http://[{mesh_ipv6}]:{}", super::PD_PEER_PORT),
38        initial_cluster_state: "new".to_string(),
39    };
40
41    let tikv_cfg = TikvConfig {
42        mesh_ipv6: *mesh_ipv6,
43        pd_endpoints: vec![format!("http://[{mesh_ipv6}]:{}", super::PD_CLIENT_PORT)],
44    };
45
46    service::install(&pd_cfg, &tikv_cfg, None)?;
47    service::enable_and_start()?;
48    steps.inc();
49
50    steps.set("Waiting for PD");
51    service::wait_pd_ready(mesh_ipv6, 30)?;
52    steps.inc();
53
54    steps.set("Waiting for TiKV");
55    service::wait_tikv_ready(mesh_ipv6, 60)?;
56
57    // Single-node bootstrap: set max-replicas=1 so regions stay healthy
58    // even without additional stores. PD defaults to 3 which is wrong
59    // for a single-node cluster.
60    let pd_url = format!("http://[{}]:{}", mesh_ipv6, super::PD_CLIENT_PORT);
61    let _ = service::adjust_max_replicas(&pd_url, 1);
62
63    steps.inc();
64
65    Ok(())
66}

68/// Join an existing TiKV cluster.
69///
70/// Uses `steps` to report progress (consumes 4 steps).
71///
72/// PD scaling strategy — **never run 2 PD members** (no fault tolerance,
73/// worse than 1 because any disruption loses quorum):
74///
75/// - `peer_count` 1 (2nd node): TiKV only, PD stays single-node
76/// - `peer_count` 2 (3rd node): scale PD 1→3 atomically (this node + node 2)
77/// - `peer_count` ≥3 (4th+ node): TiKV only
78///
79/// `all_peer_infos` provides (name, mesh_ipv6) for all known peers so we
80/// can tell node 2 to start its PD when this is node 3.
81pub fn join(
82    node_name: &str,
83    mesh_ipv6: &Ipv6Addr,
84    existing_pd_endpoints: &[String],
85    peer_count: usize,
86    all_peer_infos: &[(&str, Ipv6Addr)],
87    steps: &ui::Steps,
88) -> Result<(), NaukaError> {
89    if existing_pd_endpoints.is_empty() {
90        return Err(NaukaError::precondition(
91            "no PD endpoints available. Cannot join control plane.",
92        ));
93    }
94
95    tracing::info!(node_name, %mesh_ipv6, peer_count, "controlplane join starting");
96
97    steps.set("Installing control plane");
98    service::ensure_installed()?;
99    steps.inc();
100
101    let primary_pd = &existing_pd_endpoints[0];
102
103    // Never create a 2-member PD cluster — skip straight from 1 to 3.
104    // peer_count=1 → 2nd node → TiKV only (PD stays single on node 1)
105    // peer_count=2 → 3rd node → scale PD 1→3 (add PD on node 2 + node 3)
106    // peer_count≥3 → TiKV only
107    if peer_count == 2 {
108        steps.set("Scaling PD 1→3 (adding 2 PD members atomically)");
109        scale_pd_to_three(
110            node_name,
111            mesh_ipv6,
112            existing_pd_endpoints,
113            primary_pd,
114            all_peer_infos,
115        )?;
116    } else {
117        steps.set("Starting TiKV only");
118        join_tikv_only(node_name, mesh_ipv6, existing_pd_endpoints)?;
119    }
120    steps.inc();
121
122    // Scale up max-replicas now that we have more stores.
123    // min(active_stores, MAX_PD_MEMBERS) — never exceed Raft group size.
124    let active = service::count_active_stores(&existing_pd_endpoints[0]);
125    let target_replicas = active.min(MAX_PD_MEMBERS);
126    if target_replicas > 0 {
127        let _ = service::adjust_max_replicas(&existing_pd_endpoints[0], target_replicas);
128    }
129
130    steps.set("Control plane ready");
131    steps.inc();
132
133    Ok(())
134}

136/// Scale PD from 1 member to 3 atomically.
137///
138/// Called when the 3rd node joins. At this point node 2 is running TiKV-only.
139/// We need to:
140/// 1. Start PD on node 2 (via SSH/announce — but we can't SSH into peers)
141///    - Instead, install PD config on THIS node (node 3) and let node 2's
142///      forge daemon detect it needs PD and self-heal.
143///    - Actually, simpler: node 3 just starts its own PD in --join mode.
144///      Node 2 will get PD later via forge reconciliation or a future join.
145///
146/// For now: start PD on node 3 only. This gives us 2 PD members (1+3),
147/// which is still not ideal. The real fix is to install PD on node 2
148/// during its join but NOT start it, then start both when node 3 joins.
149///
150/// **Revised approach**: Install PD config on node 2 during its join
151/// (service installed but not started). When node 3 joins, tell node 2
152/// to start PD, then start our own PD.
153///
154/// **Simplest correct approach for now**: This node (3) joins with PD.
155/// We accept the brief 2-member window, but add rollback on failure.
156fn scale_pd_to_three(
157    node_name: &str,
158    mesh_ipv6: &Ipv6Addr,
159    existing_pd_endpoints: &[String],
160    primary_pd: &str,
161    _all_peer_infos: &[(&str, Ipv6Addr)],
162) -> Result<(), NaukaError> {
163    wait_mesh_connectivity(primary_pd, 30)?;
164
165    let self_peer_url = format!("http://[{mesh_ipv6}]:{}", super::PD_PEER_PORT);
166
167    let pd_cfg = PdConfig {
168        name: node_name.to_string(),
169        mesh_ipv6: *mesh_ipv6,
170        initial_cluster: format!("{node_name}={self_peer_url}"),
171        initial_cluster_state: "join".to_string(),
172    };
173
174    let mut pd_endpoints: Vec<String> = existing_pd_endpoints.to_vec();
175    let self_endpoint = format!("http://[{mesh_ipv6}]:{}", super::PD_CLIENT_PORT);
176    if !pd_endpoints.contains(&self_endpoint) {
177        pd_endpoints.push(self_endpoint);
178    }
179
180    let tikv_cfg = TikvConfig {
181        mesh_ipv6: *mesh_ipv6,
182        pd_endpoints,
183    };
184
185    service::install(&pd_cfg, &tikv_cfg, Some(primary_pd))?;
186    service::enable_and_start()?;
187
188    // If PD or TiKV fails to become ready, rollback: deregister the PD
189    // member we just added to prevent a phantom member that breaks quorum.
190    if let Err(e) = service::wait_pd_ready(mesh_ipv6, 120) {
191        tracing::error!("PD failed to become ready, rolling back member registration");
192        rollback_pd_member(mesh_ipv6, primary_pd);
193        return Err(e);
194    }
195
196    if let Err(e) = service::wait_tikv_ready(mesh_ipv6, 120) {
197        tracing::error!("TiKV failed to become ready, rolling back PD member");
198        rollback_pd_member(mesh_ipv6, primary_pd);
199        return Err(e);
200    }
201
202    Ok(())
203}

205/// Remove our PD member from the cluster via a remote PD endpoint.
206/// Used as rollback when join_with_pd() fails after PD was started.
207fn rollback_pd_member(mesh_ipv6: &Ipv6Addr, remote_pd_url: &str) {
208    // Stop local PD+TiKV first to prevent further Raft disruption
209    let _ = service::stop();
210
211    // Ask the remote PD (which should still have quorum with the other
212    // existing members) to remove our member.
213    let members_url = format!("{remote_pd_url}/pd/api/v1/members");
214    let output = match std::process::Command::new("curl")
215        .args(["-sf", "--max-time", "10", &members_url])
216        .output()
217    {
218        Ok(o) if o.status.success() => o,
219        _ => {
220            tracing::warn!("rollback: cannot reach remote PD to deregister member");
221            return;
222        }
223    };
224
225    let body: serde_json::Value = match serde_json::from_slice(&output.stdout) {
226        Ok(v) => v,
227        Err(_) => return,
228    };
229
230    let our_ip = mesh_ipv6.to_string();
231    if let Some(members) = body["members"].as_array() {
232        for member in members {
233            let peer_urls = member["peer_urls"]
234                .as_array()
235                .map(|a| {
236                    a.iter()
237                        .filter_map(|v| v.as_str())
238                        .collect::<Vec<_>>()
239                        .join(",")
240                })
241                .unwrap_or_default();
242            let member_id = member["member_id"].as_u64().unwrap_or(0);
243
244            if peer_urls.contains(&our_ip) && member_id > 0 {
245                tracing::warn!(member_id, "rollback: removing our PD member from cluster");
246                let delete_url = format!("{remote_pd_url}/pd/api/v1/members/id/{member_id}");
247                let _ = std::process::Command::new("curl")
248                    .args(["-sf", "-X", "DELETE", "--max-time", "10", &delete_url])
249                    .output();
250            }
251        }
252    }
253}

255/// Join with TiKV only (for nodes beyond MAX_PD_MEMBERS).
256fn join_tikv_only(
257    _node_name: &str,
258    mesh_ipv6: &Ipv6Addr,
259    existing_pd_endpoints: &[String],
260) -> Result<(), NaukaError> {
261    wait_mesh_connectivity(&existing_pd_endpoints[0], 30)?;
262
263    let tikv_cfg = TikvConfig {
264        mesh_ipv6: *mesh_ipv6,
265        pd_endpoints: existing_pd_endpoints.to_vec(),
266    };
267
268    // Install TiKV only (no PD unit)
269    service::install_tikv_only(&tikv_cfg)?;
270
271    // Start TiKV only
272    service::start_tikv_only()?;
273
274    // Wait for TiKV to register with existing PD
275    // Use first PD endpoint's IP for health check
276    let pd_ip = extract_ipv6_from_endpoint(&existing_pd_endpoints[0]);
277    if let Some(ip) = pd_ip {
278        service::wait_tikv_ready(&ip, 60)?;
279    }
280
281    Ok(())
282}

309/// Uninstall the control plane. Deregisters TiKV store, adjusts replicas,
310/// waits for region migration, then removes PD member.
311pub fn leave_with_mesh(mesh_ipv6: &Ipv6Addr) -> Result<(), NaukaError> {
312    let pd_url = format!("http://[{}]:{}", mesh_ipv6, super::PD_CLIENT_PORT);
313
314    // 1. Deregister TiKV store (marks it as Tombstone in PD)
315    let _ = service::deregister_store(mesh_ipv6);
316
317    // 2. Count remaining active stores (excluding the one we just deregistered)
318    //    and adjust max-replicas so Raft can still form quorums
319    let active = service::count_active_stores(&pd_url);
320    let remaining = if active > 0 { active - 1 } else { 0 };
321    if remaining > 0 {
322        let target = remaining.min(MAX_PD_MEMBERS);
323        let _ = service::adjust_max_replicas(&pd_url, target);
324        // 3. Wait for PD to migrate region peers off the dead store
325        let _ = service::wait_regions_healthy(&pd_url, 30);
326    }
327
328    // 4. Deregister PD member — try local first, then remote peers
329    let local_ok = service::deregister_pd_member(mesh_ipv6).is_ok() && service::pd_is_active();
330
331    if !local_ok {
332        // Local PD is down — try to deregister via a remote PD.
333        // Load peer list from fabric state to find other PD endpoints.
334        if let Ok(db) = nauka_state::LocalDb::open("hypervisor") {
335            if let Some(state) = crate::fabric::state::FabricState::load(&db).ok().flatten() {
336                for peer in &state.peers.peers {
337                    let remote_url =
338                        format!("http://[{}]:{}", peer.mesh_ipv6, super::PD_CLIENT_PORT);
339                    tracing::info!(%remote_url, "leave: trying remote PD for member deregistration");
340                    rollback_pd_member(mesh_ipv6, &remote_url);
341                }
342            }
343        }
344    }
345
346    service::uninstall()
347}

354/// Wait for mesh connectivity to a PD endpoint (HTTP health check).
355fn wait_mesh_connectivity(pd_url: &str, timeout_secs: u64) -> Result<(), NaukaError> {
356    let url = format!("{pd_url}/pd/api/v1/health");
357    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(timeout_secs);
358
359    while std::time::Instant::now() < deadline {
360        if std::process::Command::new("curl")
361            .args(["-sf", "--max-time", "3", &url])
362            .stdout(std::process::Stdio::null())
363            .stderr(std::process::Stdio::null())
364            .status()
365            .map(|s| s.success())
366            .unwrap_or(false)
367        {
368            // Also clean up any zombie PD members before joining
369            cleanup_zombie_members(pd_url);
370            return Ok(());
371        }
372        std::thread::sleep(std::time::Duration::from_secs(2));
373    }
374
375    Err(NaukaError::timeout("mesh connectivity to PD", timeout_secs))
376}

378/// Remove unhealthy PD members (zombies from failed joins).
379fn cleanup_zombie_members(pd_url: &str) {
380    let url = format!("{pd_url}/pd/api/v1/health");
381    let output = match std::process::Command::new("curl")
382        .args(["-sf", "--max-time", "5", &url])
383        .output()
384    {
385        Ok(o) if o.status.success() => o,
386        _ => return,
387    };
388
389    let health: serde_json::Value = match serde_json::from_slice(&output.stdout) {
390        Ok(v) => v,
391        Err(_) => return,
392    };
393
394    if let Some(members) = health.as_array() {
395        for member in members {
396            let healthy = member["health"].as_bool().unwrap_or(true);
397            let name = member["name"].as_str().unwrap_or("");
398            let member_id = member["member_id"].as_u64().unwrap_or(0);
399
400            // Remove unhealthy members with no name (zombie from failed join)
401            if !healthy && name.is_empty() && member_id > 0 {
402                tracing::info!(member_id, "removing zombie PD member");
403                let delete_url = format!("{pd_url}/pd/api/v1/members/id/{member_id}");
404                let _ = std::process::Command::new("curl")
405                    .args(["-sf", "-X", "DELETE", "--max-time", "5", &delete_url])
406                    .output();
407            }
408        }
409    }
410}

412/// Extract IPv6 from endpoint like "http://[fd01::1]:2379"
413fn extract_ipv6_from_endpoint(endpoint: &str) -> Option<Ipv6Addr> {
414    let s = endpoint.strip_prefix("http://[")?;
415    let s = s.split(']').next()?;
416    s.parse().ok()
417}
