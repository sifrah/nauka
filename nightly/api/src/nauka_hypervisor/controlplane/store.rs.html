1//! Distributed state over the cluster-side [`EmbeddedDb`].
2//!
3//! `ClusterDb` is a **thin wrapper** around [`nauka_state::EmbeddedDb`]
4//! configured with the SurrealDB TiKv backend. It preserves the legacy
5//! `put`/`get`/`delete`/`list`/`scan_keys`/`exists`/`batch_put` raw-KV
6//! surface so the ~dozen stores that still live on it
7//! (`layers/org`, `layers/network`, `layers/compute`) keep compiling
8//! unchanged while P2.9–P2.14 (sifrah/nauka#213–#218) migrate them one
9//! at a time to native SurrealDB SDK calls.
10//!
11//! # Why a wrapper, not a direct replacement
12//!
13//! P2.8 (sifrah/nauka#212) is the bridge step: it swaps the underlying
14//! transport from `tikv-client::RawClient` to `EmbeddedDb<TiKv>` without
15//! touching any caller. That keeps the risk surface of this PR small
16//! — all the "migrate one store to SurrealQL" churn lands in its own
17//! dedicated ticket, so each cascade step can be reviewed, tested, and
18//! rolled back independently. P2.16 (sifrah/nauka#220) is the one that
19//! finally deletes this file once the cascade is done.
20//!
21//! New code **must not** use `put`/`get`/`delete`/etc. — prefer the
22//! [`ClusterDb::embedded`] accessor and go through the native SurrealDB
23//! SDK. The legacy methods exist only to keep the in-flight cascade
24//! compiling.
25//!
26//! # Storage shape
27//!
28//! Each `(namespace, key)` pair maps to one SurrealDB record at
29//! `{namespace}:{key}` with the Rust value stored under a single `data`
30//! field. Wrapping the value under `data` keeps the JSON bridge free of
31//! the auto-added SurrealDB `id` field — the caller's struct can have
32//! its own `id` without colliding with the record id.
33//!
34//! The namespace-as-table convention means the catch-all legacy
35//! namespaces (`_reg_v2`, `vm-idx`, `storage/regions`, ...) all live on
36//! their own SCHEMALESS tables, separate from the SCHEMAFULL resource
37//! tables that P2.5 (sifrah/nauka#209) introduced (`org`, `vpc`, `vm`,
38//! ...). SurrealDB is happy to hold SCHEMALESS and SCHEMAFULL tables
39//! side by side in the same database, so the migration tickets can
40//! move each store to its SCHEMAFULL home incrementally without
41//! disturbing the rest.
42//!
43//! # Usage
44//!
45//! ```no_run
46//! use nauka_hypervisor::controlplane::ClusterDb;
47//!
48//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
49//! let db = ClusterDb::connect(&["http://[fd01::1]:2379"]).await?;
50//! db.put("vms", "vm-001", &serde_json::json!({"name": "web-1"})).await?;
51//! let vm: Option<serde_json::Value> = db.get("vms", "vm-001").await?;
52//! # Ok(())
53//! # }
54//! ```

64/// Distributed KV store, backed by an `EmbeddedDb` on the SurrealDB
65/// TiKv engine.
66///
67/// Cheap to clone: [`EmbeddedDb`] is itself `Clone` (the inner
68/// `Surreal<Db>` is reference-counted internally), so every clone
69/// shares the same TiKv connection pool.
70#[derive(Clone)]
71pub struct ClusterDb {
72    db: EmbeddedDb,
73}

75impl ClusterDb {
76    /// Connect to a TiKV cluster via PD endpoints.
77    ///
78    /// `pd_endpoints` are the same `http://[ipv6]:2379` strings that
79    /// the rest of Nauka produces via
80    /// [`nauka_state::pd_endpoints_for`] and
81    /// [`crate::fabric::state::FabricState::pd_endpoints`]. The
82    /// wrapper hands them straight to
83    /// [`EmbeddedDb::open_tikv`](nauka_state::EmbeddedDb::open_tikv),
84    /// which knows how to strip the `http://` prefix and try each
85    /// endpoint in order until one answers.
86    pub async fn connect(pd_endpoints: &[&str]) -> Result<Self, NaukaError> {
87        let db = EmbeddedDb::open_tikv(pd_endpoints)
88            .await
89            .map_err(|e| NaukaError::internal(format!("TiKV connect failed: {e}")))?;
90        Ok(Self { db })
91    }

93    /// Connect to a TiKV cluster from a slice of mesh IPv6 addresses
94    /// plus the PD client port.
95    ///
96    /// Convenience wrapper around [`Self::connect`] that builds the
97    /// `http://[ipv6]:port` list via
98    /// [`nauka_state::pd_endpoints_for`]. `controlplane::connect()`
99    /// uses this to avoid duplicating the endpoint-formatting dance
100    /// at every call site.
101    pub async fn connect_from_addresses(
102        addrs: &[Ipv6Addr],
103        pd_client_port: u16,
104    ) -> Result<Self, NaukaError> {
105        let endpoints = nauka_state::pd_endpoints_for(addrs, pd_client_port);
106        let refs: Vec<&str> = endpoints.iter().map(|s| s.as_str()).collect();
107        Self::connect(&refs).await
108    }

111    /// access.
112    ///
113    /// **Use this for new code.** The legacy
114    /// `put`/`get`/`list`/`delete`/`scan_keys`/`exists`/`batch_put`

137    /// Lazily define the SCHEMALESS catch-all table for a legacy
138    /// namespace.
139    ///
140    /// Idempotent thanks to `IF NOT EXISTS`. The namespace is inlined
141    /// into the SurrealQL string via [`EscapeIdent`] rather than a
142    /// bind parameter because `DEFINE TABLE` resolves its name at
143    /// parse time — we cannot bind it.
144    ///
145    /// Namespaces are always compile-time constants from the caller
146    /// side (`"_reg_v2"`, `"storage/regions"`, ...) so the inlining
147    /// is not a SurrealQL injection risk. `EscapeIdent` wraps any
148    /// non-alphanumeric name in backticks to keep SurrealDB happy
149    /// with the legacy `foo/bar`-shaped namespace literals.
150    async fn ensure_table(&self, namespace: &str) -> Result<(), NaukaError> {
151        let ident = EscapeIdent(namespace);
152        let stmt = format!("DEFINE TABLE IF NOT EXISTS {ident} SCHEMALESS");
153        self.db
154            .client()
155            .query(stmt)
156            .await
157            .map_err(|e| NaukaError::internal(format!("DEFINE TABLE failed: {e}")))?
158            .check()
159            .map_err(|e| NaukaError::internal(format!("DEFINE TABLE check failed: {e}")))?;
160        Ok(())
161    }

163    /// Put a serializable value.
164    ///
165    /// Round-trips the value through `serde_json::Value` then hands
166    /// it to SurrealDB under a `data` wrapper field, mirroring the
167    /// JSON-bridge pattern used by
168    /// [`crate::fabric::state::FabricState::save`].
169    pub async fn put<T: Serialize>(
170        &self,
171        namespace: &str,
172        key: &str,
173        value: &T,
174    ) -> Result<(), NaukaError> {
175        self.ensure_table(namespace).await?;
176
177        let json = serde_json::to_value(value)
178            .map_err(|e| NaukaError::internal(format!("serialize: {e}")))?;
179
180        self.db
181            .client()
182            .query("UPSERT type::record($tbl, $id) CONTENT { data: $data }")
183            .bind(("tbl", namespace.to_string()))
184            .bind(("id", key.to_string()))
185            .bind(("data", json))
186            .await
187            .map_err(|e| NaukaError::internal(format!("UPSERT failed: {e}")))?
188            .check()
189            .map_err(|e| NaukaError::internal(format!("UPSERT check failed: {e}")))?;
190
191        Ok(())
192    }
193
194    /// Get a deserializable value.
195    ///
196    /// Returns `Ok(None)` when the record (or its table) does not
197    /// exist. The "table doesn't exist" path matters for read-mostly
198    /// callers that hit `get` before anything has ever been written
199    /// to the namespace — SurrealDB's SELECT refuses to run against
200    /// an undefined table, so we lazily define it first.
201    pub async fn get<T: DeserializeOwned>(
202        &self,
203        namespace: &str,
204        key: &str,
205    ) -> Result<Option<T>, NaukaError> {
206        self.ensure_table(namespace).await?;
207
208        let mut response = self
209            .db
210            .client()
211            .query("SELECT data FROM type::record($tbl, $id)")
212            .bind(("tbl", namespace.to_string()))
213            .bind(("id", key.to_string()))
214            .await
215            .map_err(|e| NaukaError::internal(format!("SELECT failed: {e}")))?;
216
217        let row: Option<serde_json::Value> = response
218            .take(0)
219            .map_err(|e| NaukaError::internal(format!("take row: {e}")))?;
220
221        let Some(row) = row else {
222            return Ok(None);
223        };
224
225        // The row is `{"data": <value>}`. A missing / null `data`
226        // means the record exists but the wrapper content field is
227        // absent — treat that as "no value" for the legacy callers
228        // that expect `Option<T>`.
229        let Some(data) = row.get("data").cloned() else {
230            return Ok(None);
231        };
232        if data.is_null() {
233            return Ok(None);
234        }
235
236        let value: T = serde_json::from_value(data)
237            .map_err(|e| NaukaError::internal(format!("deserialize: {e}")))?;
238        Ok(Some(value))
239    }

241    /// Delete a key. Idempotent: deleting a missing key is `Ok(())`.
242    pub async fn delete(&self, namespace: &str, key: &str) -> Result<(), NaukaError> {
243        self.ensure_table(namespace).await?;
244
245        self.db
246            .client()
247            .query("DELETE type::record($tbl, $id)")
248            .bind(("tbl", namespace.to_string()))
249            .bind(("id", key.to_string()))
250            .await
251            .map_err(|e| NaukaError::internal(format!("DELETE failed: {e}")))?
252            .check()
253            .map_err(|e| NaukaError::internal(format!("DELETE check failed: {e}")))?;
254
255        Ok(())
256    }

258    /// List all values under `{namespace}/{prefix}…`.
259    ///
260    /// Returns `(key, value)` pairs where `key` is the full record id
261    /// within the namespace (same shape as the legacy TiKV/raw-KV
262    /// `list` that the callers were written against — keys look like
263    /// `"org/abc-123"`, not just `"abc-123"`).
264    ///
265    /// The implementation fetches every record in the table and
266    /// filters client-side on the id prefix. That matches the
267    /// fallback the TiKV `scan` path already used because TiKV's raw
268    /// scan didn't enforce end-key bounds correctly under the Rust
269    /// client's keyspace encoding. Legacy tables are bounded in size
270    /// (a few dozen rows at most), so the cost is negligible.
271    pub async fn list<T: DeserializeOwned>(
272        &self,
273        namespace: &str,
274        prefix: &str,
275    ) -> Result<Vec<(String, T)>, NaukaError> {
276        self.ensure_table(namespace).await?;
277
278        let rows = self.fetch_table_rows(namespace).await?;
279
280        let mut results = Vec::new();
281        for (key, data) in rows {
282            if !key.starts_with(prefix) {
283                continue;
284            }
285            if data.is_null() {
286                continue;
287            }
288            let value: T = serde_json::from_value(data)
289                .map_err(|e| NaukaError::internal(format!("deserialize: {e}")))?;
290            results.push((key, value));
291        }
292        Ok(results)
293    }

295    /// Scan record ids under `{namespace}/{prefix}…`.
296    ///
297    /// Returns the id-only half of [`Self::list`]. Same prefix /
298    /// client-side-filter semantics, same return shape (keys are
299    /// full, un-stripped).
300    pub async fn scan_keys(
301        &self,
302        namespace: &str,
303        prefix: &str,
304    ) -> Result<Vec<String>, NaukaError> {
305        self.ensure_table(namespace).await?;
306
307        let rows = self.fetch_table_rows(namespace).await?;
308        let keys = rows
309            .into_iter()
310            .map(|(k, _)| k)
311            .filter(|k| k.starts_with(prefix))
312            .collect();
313        Ok(keys)
314    }

316    /// Check if a key exists.
317    pub async fn exists(&self, namespace: &str, key: &str) -> Result<bool, NaukaError> {
318        // `get::<serde_json::Value>` already performs the cheapest
319        // existence-check query we have (`SELECT data FROM
320        // type::record(...)`) and will return `Ok(None)` for a
321        // missing row / table. Piggybacking keeps the two code paths
322        // in lockstep — if the `get` query ever changes shape, this
323        // follows along for free.
324        Ok(self
325            .get::<serde_json::Value>(namespace, key)
326            .await?
327            .is_some())
328    }

330    /// Batch put multiple key-value pairs.
331    ///
332    /// The SurrealDB TiKv backend does not expose the tikv-client
333    /// `batch_put` primitive, so this issues one `UPSERT` per entry
334    /// sequentially. The legacy callers (none in-tree as of P2.8)
335    /// only used this for tiny batches, so the throughput loss is
336    /// acceptable for the cascade window.
337    pub async fn batch_put<T: Serialize>(
338        &self,
339        namespace: &str,
340        entries: &[(&str, &T)],
341    ) -> Result<(), NaukaError> {
342        for (key, value) in entries {
343            self.put(namespace, key, *value).await?;
344        }
345        Ok(())
346    }

348    /// Internal: fetch every row in a table as `(id_string, data)`.
349    ///
350    /// Runs `SELECT record::id(id) AS __key, data FROM type::table($tbl)`
351    /// and unpacks the resulting `serde_json::Value` rows into a flat
352    /// `Vec`. The `__key` alias pulls just the id portion via
353    /// `record::id()` — that sidesteps the SurrealQL ident-escaping
354    /// dance (`` `org/abc-123` `` vs `org/abc-123`) that a plain
355    /// `SELECT *` would surface through the JSON bridge.
356    async fn fetch_table_rows(
357        &self,
358        namespace: &str,
359    ) -> Result<Vec<(String, serde_json::Value)>, NaukaError> {
360        let mut response = self
361            .db
362            .client()
363            .query("SELECT record::id(id) AS __key, data FROM type::table($tbl)")
364            .bind(("tbl", namespace.to_string()))
365            .await
366            .map_err(|e| NaukaError::internal(format!("SELECT failed: {e}")))?;
367
368        let rows: Vec<serde_json::Value> = response
369            .take(0)
370            .map_err(|e| NaukaError::internal(format!("take rows: {e}")))?;
371
372        let mut out = Vec::with_capacity(rows.len());
373        for row in rows {
374            let Some(key) = row.get("__key").and_then(|v| v.as_str()) else {
375                continue;
376            };
377            let data = row.get("data").cloned().unwrap_or(serde_json::Value::Null);
378            out.push((key.to_string(), data));
379        }
380        Ok(out)
381    }
382}

384/// SurrealQL-ident escape for catch-all namespace names.
385///
386/// If the name is a plain alphanumeric/underscore identifier it goes
387/// through unquoted; otherwise it is wrapped in backticks (SurrealDB's
388/// standard ident quoting, see
389/// `surrealdb-types::utils::escape::EscapeSqonIdent`). Backticks in the
390/// name itself are backslash-escaped.
391///
392/// This is only used for `DEFINE TABLE IF NOT EXISTS <name>` statements,
393/// where SurrealDB resolves the name at parse time and we cannot bind
394/// it as a parameter. Everywhere else we use `type::record($tbl, $id)`
395/// / `type::table($tbl)` bind-param helpers.
396struct EscapeIdent<'a>(&'a str);
397
398impl std::fmt::Display for EscapeIdent<'_> {
399    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
400        let s = self.0;
401        let needs_quoting = s.is_empty()
402            || s.starts_with(|c: char| c.is_ascii_digit())
403            || s.contains(|c: char| !c.is_ascii_alphanumeric() && c != '_');
404
405        if !needs_quoting {
406            return f.write_str(s);
407        }
408
409        f.write_str("`")?;
410        for c in s.chars() {
411            if c == '`' || c == '\\' {
412                f.write_str("\\")?;
413            }
414            std::fmt::Write::write_char(f, c)?;
415        }
416        f.write_str("`")
417    }
418}

437    /// Open a temp SurrealKV-backed `EmbeddedDb`, wrap it in a
438    /// `ClusterDb`, and return the lot.
439    ///
440    /// Returns `(TempDir, EmbeddedDb, ClusterDb)`. The raw
441    /// `EmbeddedDb` is kept around so we can call `shutdown()` on it
442    /// at end of test — `EmbeddedDb::shutdown(self)` releases the
443    /// SurrealKV OS-level flock on the backing LOCK file, which is a
444    /// hard requirement for the harness to not leak temp dirs across
445    /// tests. Because `EmbeddedDb` is Arc-backed, cloning it into
446    /// `ClusterDb` just bumps the refcount — to actually run the
447    /// shutdown chain (drop the inner `Surreal<Db>` router task) we
448    /// need the last Arc reference, so callers must `drop(db)` the
449    /// `ClusterDb` before awaiting `embedded.shutdown()`.
450    ///
451    /// [`shutdown_cluster_db`] encapsulates that contract so the
452    /// individual test bodies stay readable.
453    async fn temp_cluster_db() -> (tempfile::TempDir, EmbeddedDb, ClusterDb) {
454        let dir = tempfile::tempdir().expect("tempdir");
455        let embedded = EmbeddedDb::open(&dir.path().join("test.skv"))
456            .await
457            .expect("open EmbeddedDb at temp path");
458        let cluster = ClusterDb::for_tests(embedded.clone());
459        (dir, embedded, cluster)
460    }
461
462    /// Drop a `ClusterDb` and shut down its backing `EmbeddedDb`.
463    ///
464    /// The drop-before-shutdown order is what lets SurrealKV's LOCK
465    /// release: `shutdown(self)` only runs the router's
466    /// `kvs.shutdown()` chain when the last Arc reference to the
467    /// inner `Surreal<Db>` is dropped, so we have to get rid of the
468    /// `ClusterDb` clone first. See the doc on
469    /// [`EmbeddedDb::shutdown`](nauka_state::EmbeddedDb::shutdown)
470    /// for the full rationale.
471    async fn shutdown_cluster_db(db: ClusterDb, embedded: EmbeddedDb) {
472        drop(db);
473        embedded.shutdown().await.expect("EmbeddedDb shutdown");
474    }

784    /// P2.8 — `ClusterDb::embedded()` returns a usable `EmbeddedDb`
785    /// handle that new callers (P2.9+) can use to run native
786    /// SurrealDB SDK queries without going through the legacy
787    /// `put/get/list/delete` wrapper.
788    #[tokio::test]
789    async fn embedded_accessor_returns_usable_handle() {
790        let (_d, embedded, db) = temp_cluster_db().await;
791
792        // Go through the wrapper first to define the table and
793        // write a row.
794        db.put(
795            "probe",
796            "row-1",
797            &Item {
798                id: "r".into(),
799                name: "row".into(),
800                count: 3,
801            },
802        )
803        .await
804        .unwrap();
805
806        // Now drop down to the native SurrealDB SDK via the
807        // `embedded()` accessor and read the same row back via a
808        // raw query, proving the handle is live.
809        let mut response = db
810            .embedded()
811            .client()
812            .query("SELECT data FROM type::record($tbl, $id)")
813            .bind(("tbl", "probe".to_string()))
814            .bind(("id", "row-1".to_string()))
815            .await
816            .unwrap();
817
818        let row: Option<serde_json::Value> = response.take(0).unwrap();
819        let row = row.expect("row must exist");
820        assert_eq!(
821            row.get("data")
822                .and_then(|d| d.get("name"))
823                .and_then(|n| n.as_str()),
824            Some("row")
825        );
826
827        shutdown_cluster_db(db, embedded).await;
828    }
