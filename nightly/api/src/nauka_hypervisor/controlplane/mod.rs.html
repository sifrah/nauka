1//! Control plane — TiKV distributed KV store.
2//!
3//! Manages PD + TiKV as systemd services and exposes `ClusterDb`
4//! for distributed state (VMs, VPCs, users, etc.).
5//! All traffic flows over the encrypted WireGuard mesh.
6//!
7//! # Sub-modules
8//!
9//! - `service`: Low-level systemd management (install, start, stop, reload, status)
10//! - `ops`: High-level orchestration (bootstrap, join, leave)
11//! - `store`: `ClusterDb` — async TiKV client
12
13pub mod backup;
14pub mod ops;
15pub mod service;
16pub mod store;
17
18/// Default ports (on mesh IPv6).
19pub const PD_CLIENT_PORT: u16 = 2379;
20pub const PD_PEER_PORT: u16 = 2380;
21pub const TIKV_PORT: u16 = 20160;
22pub const TIKV_STATUS_PORT: u16 = 20180;
23
24pub use store::ClusterDb;

26/// Connect to the TiKV cluster using PD endpoints from local fabric state.
27///
28/// This is the standard way for any layer to get a ClusterDb connection.
29/// Reads the local hypervisor state to discover PD endpoints on the mesh.
30pub async fn connect() -> anyhow::Result<ClusterDb> {
31    let dir = nauka_core::process::nauka_dir();
32    let _ = std::fs::create_dir_all(&dir);
33    let db = nauka_state::LocalDb::open("hypervisor")?;
34
35    let state = crate::fabric::state::FabricState::load(&db)
36        .map_err(|e| anyhow::anyhow!("{e}"))?
37        .ok_or_else(|| {
38            anyhow::anyhow!(
39                "cluster not initialized.\n\n\
40                 Initialize a cluster first with:\n\
41                 \x20 nauka hypervisor init"
42            )
43        })?;
44
45    let self_endpoint = format!("http://[{}]:{}", state.hypervisor.mesh_ipv6, PD_CLIENT_PORT,);
46    let mut endpoints = vec![self_endpoint];
47    for peer in &state.peers.peers {
48        endpoints.push(format!("http://[{}]:{}", peer.mesh_ipv6, PD_CLIENT_PORT,));
49    }
50    let refs: Vec<&str> = endpoints.iter().map(|s| s.as_str()).collect();
51
52    ClusterDb::connect(&refs).await.map_err(Into::into)
53}
