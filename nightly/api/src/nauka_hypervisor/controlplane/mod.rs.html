1//! Control plane — TiKV distributed KV store.
2//!
3//! Manages PD + TiKV as systemd services and exposes `ClusterDb`
4//! for distributed state (VMs, VPCs, users, etc.).
5//! All traffic flows over the encrypted WireGuard mesh.
6//!
7//! # Sub-modules
8//!
9//! - `service`: Low-level systemd management (install, start, stop, reload, status)
10//! - `ops`: High-level orchestration (bootstrap, join, leave)
11//! - `store`: `ClusterDb` — async TiKV client
12

37/// Connect to the TiKV cluster using PD endpoints from local fabric state.
38///
39/// This is the standard way for any layer to get a ClusterDb connection.
40/// Reads the local hypervisor state to discover PD endpoints on the mesh.
41///
42/// P2.3 (sifrah/nauka#207) replaced the per-call `format!` ladder with
43/// the shared [`crate::fabric::state::FabricState::pd_endpoints`] +
44/// [`nauka_state::pd_endpoints_for`] helpers so every Phase-2 call site
45/// lands on the same PD list shape — and so `EmbeddedDb::open_tikv`
46/// (P2.2) and `ClusterDb::connect` both see the exact same endpoints.
47pub async fn connect() -> anyhow::Result<ClusterDb> {
48    let db = nauka_state::EmbeddedDb::open_default().await?;
49
50    let state = crate::fabric::state::FabricState::load(&db)
51        .await
52        .map_err(|e| anyhow::anyhow!("{e}"))?
53        .ok_or_else(|| {
54            anyhow::anyhow!(
55                "cluster not initialized.\n\n\
56                 Initialize a cluster first with:\n\
57                 \x20 nauka hypervisor init"
58            )
59        })?;
60    // Explicit shutdown so the SurrealKV flock is released before the caller
61    // issues its next `connect()` or local-state read.
62    db.shutdown().await?;
63
64    // Self first, peers after. The self-first contract matters for
65    // single-node clusters (nothing in peers) and for always routing
66    // through the cheapest hop when multiple PDs are live.
67    let mesh_addrs = state.pd_endpoints();
68    let endpoints = nauka_state::pd_endpoints_for(&mesh_addrs, PD_CLIENT_PORT);
69    let refs: Vec<&str> = endpoints.iter().map(|s| s.as_str()).collect();
70
71    ClusterDb::connect(&refs).await.map_err(Into::into)
72}
