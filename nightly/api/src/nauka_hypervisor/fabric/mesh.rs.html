1//! Mesh identity and configuration.
2//!
3//! A mesh is defined by:
4//! - A shared secret (syf_sk_...)
5//! - A /48 ULA IPv6 prefix
6//! - Each node gets a /128 address derived from the prefix + its WG key
7
8use std::net::Ipv6Addr;
9
10use nauka_core::addressing;
11use nauka_core::crypto::{self, MeshSecret};
12use nauka_core::id::MeshId;
13use serde::{Deserialize, Serialize};
14
15/// Complete mesh identity — everything needed to describe a mesh.
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct MeshIdentity {
18    /// Unique mesh ID.
19    pub id: MeshId,
20    /// The /48 ULA prefix for this mesh.
21    pub prefix: Ipv6Addr,
22}
23
24/// A node's identity within the mesh.
25#[derive(Debug, Clone, Serialize, Deserialize)]
26pub struct HypervisorIdentity {
27    /// Unique hypervisor ID.
28    pub id: nauka_core::id::HypervisorId,
29    /// Human-readable node name (usually hostname).
30    pub name: String,
31    /// Region label.
32    pub region: String,
33    /// Zone label.
34    pub zone: String,
35    /// WireGuard private key (base64).
36    /// Persisted locally but never shared with peers.
37    #[serde(default)]
38    pub wg_private_key: String,
39    /// WireGuard public key (base64).
40    pub wg_public_key: String,
41    /// WireGuard listen port.
42    pub wg_port: u16,
43    /// Public endpoint (IP:port) for other nodes to connect via WG.
44    pub endpoint: Option<String>,
45    /// Fabric network interface (e.g., "eth1"). Empty = auto.
46    #[serde(default)]
47    pub fabric_interface: String,
48    /// This node's mesh IPv6 address (/128).
49    /// In WG mode: ULA derived from prefix + pubkey.
50    /// In direct mode: real IP of the fabric interface.
51    pub mesh_ipv6: Ipv6Addr,
52    /// Compute runtime: "kvm" (Cloud Hypervisor) or "container" (gVisor).
53    #[serde(default = "default_runtime")]
54    pub runtime: String,
55    /// Public IPv6 /64 block allocated by the hosting provider (e.g., "2a01:4f8:c012:abcd::/64").
56    #[serde(default, skip_serializing_if = "Option::is_none")]
57    pub ipv6_block: Option<String>,
58    /// Public IPv4 address of this server.
59    #[serde(default, skip_serializing_if = "Option::is_none")]
60    pub ipv4_public: Option<String>,
61}
62
63fn default_runtime() -> String {
64    "kvm".to_string()
65}
66
67/// Create a new mesh (called by `hypervisor init`).
68pub fn create_mesh() -> (MeshIdentity, MeshSecret) {
69    let secret = MeshSecret::generate();
70    let prefix = addressing::generate_mesh_prefix();
71    let id = MeshId::generate();
72
73    let mesh = MeshIdentity { id, prefix };
74
75    (mesh, secret)
76}

78/// Configuration for creating a new hypervisor identity.
79pub struct CreateHypervisorConfig<'a> {
80    pub name: &'a str,
81    pub region: &'a str,
82    pub zone: &'a str,
83    pub port: u16,
84    pub endpoint: Option<String>,
85    pub fabric_interface: &'a str,
86    pub mesh_prefix: &'a Ipv6Addr,
87    pub ipv6_block: Option<String>,
88    pub ipv4_public: Option<String>,
89}
90
91/// Create a new node identity (called by both init and join).
92/// Validates name, region, zone, and port.
93pub fn create_hypervisor(
94    cfg: &CreateHypervisorConfig<'_>,
95) -> Result<HypervisorIdentity, nauka_core::error::NaukaError> {
96    nauka_core::validate::name(cfg.name)?;
97    nauka_core::validate::region(cfg.region)?;
98    nauka_core::validate::zone(cfg.zone)?;
99    nauka_core::validate::port(cfg.port)?;
100
101    let (wg_private, wg_public) = crypto::generate_wg_keypair();
102
103    let pub_bytes = base64::Engine::decode(&base64::engine::general_purpose::STANDARD, &wg_public)
104        .unwrap_or_default();
105
106    let mesh_ipv6 = addressing::derive_node_address(cfg.mesh_prefix, &pub_bytes);
107
108    // Detect compute runtime: KVM if /dev/kvm exists, container (gVisor) otherwise
109    let runtime = if std::path::Path::new("/dev/kvm").exists() {
110        "kvm".to_string()
111    } else {
112        "container".to_string()
113    };
114
115    Ok(HypervisorIdentity {
116        id: nauka_core::id::HypervisorId::generate(),
117        name: cfg.name.to_string(),
118        region: cfg.region.to_string(),
119        zone: cfg.zone.to_string(),
120        wg_private_key: wg_private,
121        wg_public_key: wg_public,
122        wg_port: cfg.port,
123        endpoint: cfg.endpoint.clone(),
124        fabric_interface: cfg.fabric_interface.to_string(),
125        mesh_ipv6,
126        runtime,
127        ipv6_block: cfg.ipv6_block.clone(),
128        ipv4_public: cfg.ipv4_public.clone(),
129    })
130}

136    /// Test helper: create a hypervisor with minimal defaults.
137    fn make_hv(
138        name: &str,
139        region: &str,
140        zone: &str,
141        port: u16,
142        endpoint: Option<String>,
143        prefix: &Ipv6Addr,
144    ) -> Result<HypervisorIdentity, nauka_core::error::NaukaError> {
145        create_hypervisor(&CreateHypervisorConfig {
146            name,
147            region,
148            zone,

158    #[test]
159    fn create_mesh_generates_valid_identity() {
160        let (mesh, secret) = create_mesh();
161        assert!(mesh.id.as_str().starts_with("mesh-"));
162        assert!(secret.to_string().starts_with("syf_sk_"));
163        let first = mesh.prefix.segments()[0];
164        assert!((0xfd00..=0xfdff).contains(&first));
165    }
166
167    #[test]
168    fn create_mesh_unique() {
169        let (a, _) = create_mesh();
170        let (b, _) = create_mesh();
171        assert_ne!(a.id.as_str(), b.id.as_str());
172    }
173
174    #[test]
175    fn create_node_has_valid_identity() {
176        let (mesh, _) = create_mesh();
177        let node = make_hv("node-1", "eu", "fsn1", 51820, None, &mesh.prefix).unwrap();
178
179        assert_eq!(node.name, "node-1");
180        assert_eq!(node.region, "eu");
181        assert_eq!(node.zone, "fsn1");
182        assert_eq!(node.wg_port, 51820);
183        assert!(!node.wg_private_key.is_empty());
184        assert!(!node.wg_public_key.is_empty());
185        assert!(addressing::is_in_prefix(&node.mesh_ipv6, &mesh.prefix));
186    }

212    #[test]
213    fn node_identity_serde_roundtrip() {
214        let (mesh, _) = create_mesh();
215        let node = make_hv("node-1", "eu", "fsn1", 51820, None, &mesh.prefix).unwrap();
