1//! Peer announcement — broadcasts new peers to all existing mesh members.
2//!
3//! When a new node joins, the accepting node sends a PeerAnnounce message
4//! to every existing peer so they can add the new node to their WireGuard config.
5//!
6//! Protocol: TCP connect → send PeerAnnounce → close.
7//! Same length-prefixed JSON as the peering protocol.
8//!
9//! Announcements are best-effort: if a peer is unreachable, we skip it
10//! and rely on the next health check or manual sync.
11

23/// Envelope for announce protocol messages.
24#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
25#[serde(tag = "type")]
26pub enum AnnounceMessage {
27    #[serde(rename = "announce")]
28    Announce(PeerAnnounce),
29    #[serde(rename = "remove")]
30    Remove(PeerRemove),
31    #[serde(rename = "state_change")]
32    StateChange(StateChange),
33}
34
35/// Default announce port offset from WireGuard port.
36pub const ANNOUNCE_PORT_OFFSET: u16 = 2;
37
38/// Default timeout for each announce connection.
39const ANNOUNCE_TIMEOUT_SECS: u64 = 5;
40
41/// Broadcast a new peer to all existing peers (in parallel).
42///
43/// Connects to each peer's announce port (mesh_ipv6:wg_port+2) and sends
44/// a PeerAnnounce message. Best-effort: failures are logged but don't
45/// stop the broadcast. Skips peers marked as unreachable.
46///
47/// Returns (successes, failures).
48pub async fn broadcast_new_peer(
49    new_peer: &PeerInfo,
50    announced_by: &str,
51    existing_peers: &[Peer],
52    _wg_port: u16,
53) -> (usize, usize) {
54    let msg = AnnounceMessage::Announce(PeerAnnounce {
55        peer: new_peer.clone(),
56        announced_by: announced_by.to_string(),
57    });
58
59    // Skip unreachable peers — no point waiting for a timeout
60    let reachable: Vec<_> = existing_peers
61        .iter()
62        .filter(|p| p.status != super::peer::PeerStatus::Unreachable)
63        .collect();
64
65    // Broadcast in parallel — don't let one dead peer block the rest
66    let mut handles = Vec::with_capacity(reachable.len());
67    for peer in &reachable {
68        let addr = format!(
69            "[{}]:{}",
70            peer.mesh_ipv6,
71            peer.wg_port + ANNOUNCE_PORT_OFFSET
72        );
73        let peer_name = peer.name.clone();
74        let new_peer_name = new_peer.name.clone();
75        let msg = msg.clone();
76        handles.push(tokio::spawn(async move {
77            match send_message(&addr, &msg).await {
78                Ok(()) => {
79                    tracing::info!(
80                        peer = %peer_name,
81                        new_peer = %new_peer_name,
82                        "announced new peer"
83                    );
84                    true
85                }
86                Err(e) => {
87                    tracing::warn!(
88                        peer = %peer_name,
89                        error = %e,
90                        "failed to announce new peer"
91                    );
92                    false
93                }
94            }
95        }));
96    }
97
98    let mut successes = 0;
99    let mut failures = 0;
100    for handle in handles {
101        match handle.await {
102            Ok(true) => successes += 1,
103            _ => failures += 1,
104        }
105    }
106
107    (successes, failures)
108}

110/// Send an announce message to one peer (over TLS).
111async fn send_message(addr: &str, msg: &AnnounceMessage) -> Result<(), NaukaError> {
112    let timeout = Duration::from_secs(ANNOUNCE_TIMEOUT_SECS);
113
114    let tcp_stream = tokio::time::timeout(timeout, tokio::net::TcpStream::connect(addr))
115        .await
116        .map_err(|_| NaukaError::timeout("announce connection", ANNOUNCE_TIMEOUT_SECS))?
117        .map_err(|e| NaukaError::network(format!("announce connect failed: {e}")))?;
118
119    // TLS handshake (TOFU — same model as peering)
120    let tls_config = super::tls::client_config();
121    let connector = tokio_rustls::TlsConnector::from(tls_config);
122    let server_name = rustls::pki_types::ServerName::try_from("nauka-peering")
123        .map_err(|e| NaukaError::internal(format!("TLS server name: {e}")))?;
124
125    let mut stream = connector
126        .connect(server_name, tcp_stream)
127        .await
128        .map_err(|e| NaukaError::network(format!("announce TLS handshake failed: {e}")))?;
129
130    write_json(&mut stream, msg).await?;
131
132    Ok(())
133}

135/// Broadcast a peer removal to all existing peers in parallel (best-effort).
136/// Called during `leave` before tearing down the network.
137pub async fn broadcast_peer_remove(
138    name: &str,
139    wg_public_key: &str,
140    existing_peers: &[Peer],
141    _wg_port: u16,
142) -> (usize, usize) {
143    let msg = AnnounceMessage::Remove(PeerRemove {
144        name: name.to_string(),
145        wg_public_key: wg_public_key.to_string(),
146    });
147
148    let mut handles = Vec::with_capacity(existing_peers.len());
149    for peer in existing_peers {
150        let addr = format!(
151            "[{}]:{}",
152            peer.mesh_ipv6,
153            peer.wg_port + ANNOUNCE_PORT_OFFSET
154        );
155        let peer_name = peer.name.clone();
156        let leaving_name = name.to_string();
157        let msg = msg.clone();
158        handles.push(tokio::spawn(async move {
159            match send_message(&addr, &msg).await {
160                Ok(()) => {
161                    tracing::info!(peer = %peer_name, leaving = %leaving_name, "sent peer removal");
162                    true
163                }
164                Err(e) => {
165                    tracing::warn!(peer = %peer_name, error = %e, "failed to send peer removal");
166                    false
167                }
168            }
169        }));
170    }
171
172    let mut successes = 0;
173    let mut failures = 0;
174    for handle in handles {
175        match handle.await {
176            Ok(true) => successes += 1,
177            _ => failures += 1,
178        }
179    }
180
181    (successes, failures)
182}

243/// Listen for peer announcements and apply them (over TLS).
244///
245/// Runs on wg_port + 2. When an announcement arrives:
246/// 1. Add the new peer to our PeerList
247/// 2. Update WireGuard config
248/// 3. Persist state
249///
250/// Opens DB per-request to avoid lock contention.
251pub async fn listen<F, Fut>(db_opener: F, bind_addr: std::net::SocketAddr) -> Result<(), NaukaError>
252where
253    F: Fn() -> Fut,
254    Fut: Future<Output = Result<EmbeddedDb, NaukaError>>,
255{
256    let listener = tokio::net::TcpListener::bind(bind_addr)
257        .await
258        .map_err(|e| NaukaError::internal(format!("failed to bind announce port: {e}")))?;
259
260    let tls_config = super::tls::server_config()?;
261    let tls_acceptor = tokio_rustls::TlsAcceptor::from(tls_config);
262
263    tracing::info!(addr = %bind_addr, "announce listener started (TLS)");
264
265    loop {
266        match listener.accept().await {
267            Ok((tcp_stream, peer_addr)) => {
268                // TLS handshake
269                let mut stream = match tls_acceptor.accept(tcp_stream).await {
270                    Ok(s) => s,
271                    Err(e) => {
272                        tracing::warn!(peer = %peer_addr, error = %e, "announce TLS handshake failed");
273                        continue;
274                    }
275                };
276
277                let db = match db_opener().await {
278                    Ok(db) => db,
279                    Err(e) => {
280                        tracing::warn!(error = %e, "announce: failed to open DB");
281                        continue;
282                    }
283                };
284
285                match handle_message(&mut stream, &db).await {
286                    Ok(msg) => {
287                        tracing::info!(from = %peer_addr, result = %msg, "announce handled");
288                    }
289                    Err(e) => {
290                        tracing::warn!(from = %peer_addr, error = %e, "announce failed");
291                    }
292                }
293
294                // Release the SurrealKV flock before the next accept opens a
295                // new handle, so announce/peering/health runs don't contend.
296                let _ = db.shutdown().await;
297            }
298            Err(e) => {
299                tracing::warn!(error = %e, "announce accept error");
300            }
301        }
302    }
303}

305/// Handle an incoming announce message (add or remove).
306async fn handle_message(
307    stream: &mut (impl tokio::io::AsyncReadExt + tokio::io::AsyncWriteExt + Unpin),
308    db: &EmbeddedDb,
309) -> Result<String, NaukaError> {
310    let msg: AnnounceMessage = read_json(stream).await?;
311
312    match msg {
313        AnnounceMessage::Announce(announce) => handle_peer_announce(announce, db).await,
314        AnnounceMessage::Remove(remove) => handle_peer_remove(remove, db).await,
315        AnnounceMessage::StateChange(change) => handle_state_change(change, db).await,
316    }
317}

319/// Handle a PeerRemove — remove a leaving peer.
320async fn handle_peer_remove(remove: PeerRemove, db: &EmbeddedDb) -> Result<String, NaukaError> {
321    super::peering_server::validate_peer_field(&remove.name, "name")?;
322
323    let mut state = super::state::FabricState::load(db)
324        .await
325        .map_err(|e| NaukaError::internal(e.to_string()))?
326        .ok_or_else(|| NaukaError::precondition("not initialized"))?;
327
328    if let Some(removed) = state.peers.remove(&remove.name) {
329        tracing::info!(peer = %remove.name, "removed leaving peer");
330
331        state
332            .save(db)
333            .await
334            .map_err(|e| NaukaError::internal(e.to_string()))?;
335
336        // Update WireGuard config
337        let peers_for_wg: Vec<_> = state
338            .peers
339            .peers
340            .iter()
341            .map(|p| {
342                (
343                    p.wg_public_key.clone(),
344                    "25".to_string(),
345                    p.mesh_ipv6,
346                    p.endpoint.clone(),
347                )
348            })
349            .collect();
350
351        let _ = service::update_config(
352            &state.hypervisor.wg_private_key,
353            state.hypervisor.wg_port,
354            &state.hypervisor.mesh_ipv6,
355            &peers_for_wg,
356        );
357
358        Ok(format!("removed {}", removed.name))
359    } else {
360        Ok(format!("{} not found, skipped", remove.name))
361    }
362}

364/// Handle a StateChange — update a peer's scheduling state.
365async fn handle_state_change(change: StateChange, db: &EmbeddedDb) -> Result<String, NaukaError> {
366    super::peering_server::validate_peer_field(&change.name, "name")?;
367
368    let mut state = super::state::FabricState::load(db)
369        .await
370        .map_err(|e| NaukaError::internal(e.to_string()))?
371        .ok_or_else(|| NaukaError::precondition("not initialized"))?;
372
373    // Find the peer by name or public key and update its node_state
374    let found = state
375        .peers
376        .peers
377        .iter_mut()
378        .find(|p| p.name == change.name || p.wg_public_key == change.wg_public_key);
379
380    if let Some(peer) = found {
381        peer.node_state = change.node_state;
382        tracing::info!(
383            peer = %change.name,
384            state = %change.node_state,
385            "updated peer scheduling state"
386        );
387
388        state
389            .save(db)
390            .await
391            .map_err(|e| NaukaError::internal(e.to_string()))?;
392
393        Ok(format!("{} → {}", change.name, change.node_state))
394    } else {
395        Ok(format!("{} not found, skipped", change.name))
396    }
397}

399/// Handle a PeerAnnounce — add or replace a peer.
400async fn handle_peer_announce(
401    announce: PeerAnnounce,
402    db: &EmbeddedDb,
403) -> Result<String, NaukaError> {
404    // Validate peer data from untrusted source
405    super::peering_server::validate_peer_field(&announce.peer.name, "name")?;
406    super::peering_server::validate_peer_field(&announce.peer.region, "region")?;
407    super::peering_server::validate_peer_field(&announce.peer.zone, "zone")?;
408    super::peering_server::validate_peer_field(&announce.peer.wg_public_key, "wg_public_key")?;
409
410    let mut state = super::state::FabricState::load(db)
411        .await
412        .map_err(|e| NaukaError::internal(e.to_string()))?
413        .ok_or_else(|| NaukaError::precondition("not initialized"))?;
414
415    let peer_name = announce.peer.name.clone();
416
417    // Skip if exact same key already known
418    if state
419        .peers
420        .find_by_key(&announce.peer.wg_public_key)
421        .is_some()
422    {
423        tracing::debug!(peer = %peer_name, "already known (same key), skipping");
424        return Ok(peer_name);
425    }
426
427    // Remove stale entry if same name but different key (leave/rejoin)
428    if state.peers.find_by_name(&peer_name).is_some() {
429        tracing::info!(peer = %peer_name, "replacing stale peer entry (rejoin announce)");
430        state.peers.remove(&peer_name);
431    }
432
433    // Add the new peer
434    let new_peer = Peer::new(
435        announce.peer.name,
436        announce.peer.region,
437        announce.peer.zone,
438        announce.peer.wg_public_key,
439        announce.peer.wg_port,
440        announce.peer.endpoint,
441        announce.peer.mesh_ipv6,
442    );
443    if let Err(e) = state.peers.add(new_peer) {
444        tracing::debug!(error = %e, "announce: peer add failed");
445        return Ok(peer_name);
446    }
447
448    // Persist immediately
449    state
450        .save(db)
451        .await
452        .map_err(|e| NaukaError::internal(e.to_string()))?;
453
454    // Update WireGuard config
455    let peers_for_wg: Vec<_> = state
456        .peers
457        .peers
458        .iter()
459        .map(|p| {
460            (
461                p.wg_public_key.clone(),
462                "25".to_string(),
463                p.mesh_ipv6,
464                p.endpoint.clone(),
465            )
466        })
467        .collect();
468
469    service::update_config(
470        &state.hypervisor.wg_private_key,
471        state.hypervisor.wg_port,
472        &state.hypervisor.mesh_ipv6,
473        &peers_for_wg,
474    )?;
475
476    Ok(peer_name)
477}

483const ANNOUNCE_SERVICE: &str = "nauka-announce";
484const ANNOUNCE_UNIT_PATH: &str = "/etc/systemd/system/nauka-announce.service";
485
486/// Generate the systemd unit for the persistent announce listener.
487fn generate_announce_unit(port: u16) -> String {
488    let announce_port = port + ANNOUNCE_PORT_OFFSET;
489    format!(
490        r#"[Unit]
491Description=Nauka Peer Announce Listener
492After=network-online.target nauka-wg.service
493Wants=network-online.target
494Requires=nauka-wg.service
495
496[Service]
497Type=simple
498ExecStart=/usr/local/bin/nauka hypervisor announce-listen --port {announce_port}
499Restart=on-failure
500RestartSec=5
501
502[Install]
503WantedBy=multi-user.target
504"#
505    )
506}
507
508/// Install and start the announce listener service.
509pub fn install_service(wg_port: u16) -> Result<(), NaukaError> {
510    std::fs::write(ANNOUNCE_UNIT_PATH, generate_announce_unit(wg_port))
511        .map_err(NaukaError::from)?;
512    run_systemctl(&["daemon-reload"])?;
513    run_systemctl(&["enable", "--now", ANNOUNCE_SERVICE])?;
514    Ok(())
515}
516
517/// Start the announce listener service.
518pub fn start_service() -> Result<(), NaukaError> {
519    if !is_service_installed() {
520        return Ok(());
521    }
522    run_systemctl(&["start", ANNOUNCE_SERVICE])
523}
524
525/// Stop the announce listener service.
526pub fn stop_service() -> Result<(), NaukaError> {
527    if !is_service_installed() {
528        return Ok(());
529    }
530    let _ = run_systemctl(&["stop", ANNOUNCE_SERVICE]);
531    Ok(())
532}
533
534/// Uninstall the announce listener service.
535pub fn uninstall_service() -> Result<(), NaukaError> {
536    let _ = run_systemctl(&["disable", "--now", ANNOUNCE_SERVICE]);
537    let _ = std::fs::remove_file(ANNOUNCE_UNIT_PATH);
538    let _ = run_systemctl(&["daemon-reload"]);
539    Ok(())
540}
541
542/// Check if the service is installed.
543pub fn is_service_installed() -> bool {
544    std::path::Path::new(ANNOUNCE_UNIT_PATH).exists()
545}
546
547/// Check if the service is active.
548pub fn is_service_active() -> bool {
549    std::process::Command::new("systemctl")
550        .args(["is-active", "--quiet", ANNOUNCE_SERVICE])
551        .status()
552        .map(|s| s.success())
553        .unwrap_or(false)
554}
555
556fn run_systemctl(args: &[&str]) -> Result<(), NaukaError> {
557    let output = std::process::Command::new("systemctl")
558        .args(args)
559        .output()
560        .map_err(|e| NaukaError::internal(format!("systemctl failed: {e}")))?;
561    if !output.status.success() {
562        let stderr = String::from_utf8_lossy(&output.stderr);
563        return Err(NaukaError::internal(format!(
564            "systemctl {} failed: {}",
565            args.join(" "),
566            stderr.trim()
567        )));
568    }
569    Ok(())
570}
