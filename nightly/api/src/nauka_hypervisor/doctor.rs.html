1//! Hypervisor doctor — diagnose the health of all sub-systems.
2//!
3//! Runs sequential checks against fabric, controlplane, storage, and system.
4//! Outputs a human-readable report with OK / WARN / FAIL for each check.

14// ═══════════════════════════════════════════════════
15// Check result types
16// ═══════════════════════════════════════════════════
17
18#[derive(Debug, Clone, Copy, PartialEq)]
19pub enum CheckStatus {
20    Ok,
21    Warn,
22    Fail,
23    Skip,
24}
25
26#[derive(Debug, Clone)]
27pub struct Check {
28    pub name: String,
29    pub status: CheckStatus,
30    pub detail: String,
31}
32
33#[derive(Debug, Default)]
34pub struct DoctorReport {
35    pub checks: Vec<(String, Vec<Check>)>, // (section_name, checks)
36}
37
38impl DoctorReport {
39    fn add_section(&mut self, name: &str) -> &mut Vec<Check> {
40        self.checks.push((name.to_string(), Vec::new()));
41        &mut self.checks.last_mut().unwrap().1
42    }
43
44    pub fn ok_count(&self) -> usize {
45        self.checks
46            .iter()
47            .flat_map(|(_, c)| c)
48            .filter(|c| c.status == CheckStatus::Ok)
49            .count()
50    }
51
52    pub fn warn_count(&self) -> usize {
53        self.checks
54            .iter()
55            .flat_map(|(_, c)| c)
56            .filter(|c| c.status == CheckStatus::Warn)
57            .count()
58    }
59
60    pub fn fail_count(&self) -> usize {
61        self.checks
62            .iter()
63            .flat_map(|(_, c)| c)
64            .filter(|c| c.status == CheckStatus::Fail)
65            .count()
66    }
67
68    /// Render the report to stderr.
69    pub fn print(&self) {
70        for (section, checks) in &self.checks {
71            eprintln!();
72            eprintln!("  {section}");
73            for check in checks {
74                let icon = match check.status {
75                    CheckStatus::Ok => "\x1b[32m✓\x1b[0m",
76                    CheckStatus::Warn => "\x1b[33m!\x1b[0m",
77                    CheckStatus::Fail => "\x1b[31m✗\x1b[0m",
78                    CheckStatus::Skip => "\x1b[90m-\x1b[0m",
79                };
80                eprintln!("    {icon} {}: {}", check.name, check.detail);
81            }
82        }
83        eprintln!();
84        eprintln!(
85            "  {} passed, {} warnings, {} errors",
86            self.ok_count(),
87            self.warn_count(),
88            self.fail_count()
89        );
90    }
91}
92
93fn ok(name: &str, detail: &str) -> Check {
94    Check {
95        name: name.into(),
96        status: CheckStatus::Ok,
97        detail: detail.into(),
98    }
99}
100
101fn warn(name: &str, detail: &str) -> Check {
102    Check {
103        name: name.into(),
104        status: CheckStatus::Warn,
105        detail: detail.into(),
106    }
107}
108
109fn fail(name: &str, detail: &str) -> Check {
110    Check {
111        name: name.into(),
112        status: CheckStatus::Fail,
113        detail: detail.into(),
114    }
115}
116
117fn skip(name: &str, detail: &str) -> Check {
118    Check {
119        name: name.into(),
120        status: CheckStatus::Skip,
121        detail: detail.into(),
122    }
123}

125// ═══════════════════════════════════════════════════
126// Run all checks
127// ═══════════════════════════════════════════════════
128
129pub fn run() -> DoctorReport {
130    let mut report = DoctorReport::default();
131
132    // Load state
133    let db = nauka_state::LocalDb::open("hypervisor").ok();
134    let state = db
135        .as_ref()
136        .and_then(|db| fabric::state::FabricState::load(db).ok().flatten());
137
138    let mesh_ipv6 = state.as_ref().map(|s| s.hypervisor.mesh_ipv6);
139
140    check_fabric(&mut report, &state);
141    check_controlplane(&mut report, mesh_ipv6.as_ref());
142    check_storage(&mut report, db.as_ref());
143    check_system(&mut report);
144
145    report
146}

152fn check_fabric(report: &mut DoctorReport, state: &Option<fabric::state::FabricState>) {
153    let checks = report.add_section("Fabric");
154
155    let network_mode = state.as_ref().map(|s| s.network_mode).unwrap_or_default();
156    let backend = fabric::backend::create_backend(network_mode);
157
158    match network_mode {
159        fabric::backend::NetworkMode::WireGuard => {
160            // wireguard-tools installed
161            if cmd_exists("wg") {
162                checks.push(ok("wireguard-tools", "installed"));
163            } else {
164                checks.push(fail("wireguard-tools", "not installed"));
165            }
166
167            // Config file
168            if Path::new("/etc/wireguard/nauka0.conf").exists() {
169                checks.push(ok("wg config", "/etc/wireguard/nauka0.conf exists"));
170            } else {
171                checks.push(fail("wg config", "missing"));
172            }
173        }
174        fabric::backend::NetworkMode::Direct => {
175            checks.push(ok("network mode", "direct (no tunnel)"));
176        }
177        fabric::backend::NetworkMode::Mock => {
178            checks.push(ok("network mode", "mock (testing)"));
179        }
180    }
181
182    // Interface/service up (backend-agnostic)
183    if backend.is_up() {
184        checks.push(ok("fabric network", "up"));
185    } else {
186        checks.push(fail("fabric network", "down"));
187    }
188
189    if backend.is_active() {
190        checks.push(ok("fabric service", "active"));
191    } else {
192        checks.push(warn("fabric service", "stopped"));
193    }
194
195    // Peer health
196    if let Some(state) = state {
197        let total = state.peers.len();
198        if total == 0 {
199            checks.push(ok("peers", "no peers configured"));
200        } else {
201            let active = state.peers.active_count();
202            let unreachable = state.peers.unreachable_count();
203            if unreachable == 0 {
204                checks.push(ok("peers", &format!("{total}/{total} reachable")));
205            } else {
206                checks.push(warn(
207                    "peers",
208                    &format!("{active}/{total} reachable, {unreachable} unreachable"),
209                ));
210            }
211
212            // Ping test (sample first 3 peers)
213            let mut ping_ok = 0;
214            let mut ping_fail = 0;
215            for peer in state.peers.peers.iter().take(3) {
216                if ping6(&peer.mesh_ipv6) {
217                    ping_ok += 1;
218                } else {
219                    ping_fail += 1;
220                }
221            }
222            if ping_fail == 0 {
223                checks.push(ok("mesh ping", &format!("{ping_ok}/{ping_ok} responded")));
224            } else {
225                checks.push(warn(
226                    "mesh ping",
227                    &format!("{ping_ok}/{} responded", ping_ok + ping_fail),
228                ));
229            }
230        }
231    } else {
232        checks.push(skip("peers", "not initialized"));
233    }
234}

240fn check_controlplane(report: &mut DoctorReport, mesh_ipv6: Option<&Ipv6Addr>) {
241    let checks = report.add_section("Controlplane");
242
243    // TiUP installed
244    if Path::new("/opt/nauka/tiup/bin/tiup").exists() {
245        checks.push(ok("tiup", "installed"));
246    } else {
247        checks.push(skip("tiup", "not installed"));
248        return;
249    }
250
251    // PD service
252    if controlplane::service::pd_is_active() {
253        checks.push(ok("pd service", "active"));
254    } else if Path::new("/etc/systemd/system/nauka-pd.service").exists() {
255        checks.push(warn("pd service", "installed but stopped"));
256    } else {
257        checks.push(ok("pd service", "not running (tikv-only node)"));
258    }
259
260    // TiKV service
261    if controlplane::service::tikv_is_active() {
262        checks.push(ok("tikv service", "active"));
263    } else if Path::new("/etc/systemd/system/nauka-tikv.service").exists() {
264        checks.push(fail("tikv service", "installed but stopped"));
265    } else {
266        checks.push(fail("tikv service", "not installed"));
267    }
268
269    // PD health + leader
270    if let Some(ip) = mesh_ipv6 {
271        let pd_url = format!("http://[{}]:{}", ip, controlplane::PD_CLIENT_PORT);
272        let health_url = format!("{pd_url}/pd/api/v1/health");
273
274        if let Some(body) = curl_json(&health_url) {
275            if let Some(members) = body.as_array() {
276                let healthy: Vec<_> = members
277                    .iter()
278                    .filter(|m| m["health"].as_bool().unwrap_or(false))
279                    .collect();
280                let total = members.len();
281                if healthy.len() == total {
282                    checks.push(ok("pd health", &format!("{total}/{total} healthy")));
283                } else {
284                    checks.push(warn(
285                        "pd health",
286                        &format!("{}/{total} healthy", healthy.len()),
287                    ));
288                }
289
290                // Leader
291                let leader = members
292                    .iter()
293                    .find(|m| m["health"].as_bool().unwrap_or(false))
294                    .and_then(|m| m["name"].as_str());
295                if let Some(name) = leader {
296                    checks.push(ok("pd leader", name));
297                } else {
298                    checks.push(fail("pd leader", "no leader elected"));
299                }
300            }
301        } else {
302            checks.push(warn("pd health", "could not reach PD API"));
303        }
304
305        // PD quorum health — are all members healthy?
306        let members_url = format!("{pd_url}/pd/api/v1/members");
307        if let Some(body) = curl_json(&members_url) {
308            if let Some(members) = body["members"].as_array() {
309                let total = members.len();
310                if total < 3 {
311                    checks.push(warn(
312                        "pd quorum",
313                        &format!("only {total} member(s) — need 3 for fault tolerance"),
314                    ));
315                } else {
316                    checks.push(ok("pd quorum", &format!("{total} members — quorum intact")));
317                }
318            }
319        }
320
321        // TiKV store registration + health
322        let stores_url = format!("{pd_url}/pd/api/v1/stores");
323        if let Some(body) = curl_json(&stores_url) {
324            let count = body["count"].as_u64().unwrap_or(0);
325            checks.push(ok("tikv stores", &format!("{count} registered")));
326
327            // Check for tombstoned or offline stores
328            if let Some(stores) = body["stores"].as_array() {
329                let mut tombstoned = 0u64;
330                let mut offline = 0u64;
331                let mut down = 0u64;
332                for store in stores {
333                    if let Some(state) = store["store"]["state_name"].as_str() {
334                        match state {
335                            "Tombstone" => tombstoned += 1,
336                            "Offline" => offline += 1,
337                            "Down" => down += 1,
338                            _ => {}
339                        }
340                    }
341                }
342                if tombstoned > 0 {
343                    checks.push(fail(
344                        "tikv tombstoned",
345                        &format!("{tombstoned} store(s) tombstoned — data may be under-replicated"),
346                    ));
347                }
348                if down > 0 {
349                    checks.push(fail("tikv down", &format!("{down} store(s) down")));
350                }
351                if offline > 0 {
352                    checks.push(warn(
353                        "tikv offline",
354                        &format!("{offline} store(s) offline — regions migrating"),
355                    ));
356                }
357                if tombstoned == 0 && offline == 0 && down == 0 {
358                    checks.push(ok("tikv store health", "all stores Up"));
359                }
360            }
361        }
362    }
363}

369fn check_storage(report: &mut DoctorReport, db: Option<&nauka_state::LocalDb>) {
370    let checks = report.add_section("Storage");
371
372    // ZeroFS installed
373    if storage::service::is_installed() {
374        checks.push(ok("zerofs", "installed"));
375    } else {
376        checks.push(skip("zerofs", "not installed"));
377    }
378
379    // Region configs
380    if let Some(db) = db {
381        let statuses = storage::ops::status(db);
382        if statuses.is_empty() {
383            checks.push(ok("regions", "none configured"));
384        } else {
385            for s in &statuses {
386                if s.active {
387                    checks.push(ok(
388                        &format!("region {}", s.region),
389                        &format!("active ({})", s.s3_bucket),
390                    ));
391                } else {
392                    checks.push(warn(
393                        &format!("region {}", s.region),
394                        &format!("stopped ({})", s.s3_bucket),
395                    ));
396                }
397            }
398        }
399    }
400}

402// ═══════════════════════════════════════════════════
403// System checks
404// ═══════════════════════════════════════════════════
405
406fn check_system(report: &mut DoctorReport) {
407    let checks = report.add_section("System");
408
409    // Disk space
410    if let Some(free_mb) = disk_free_mb("/var/lib/nauka") {
411        if free_mb > 1024 {
412            checks.push(ok("disk space", &format!("{}GB free", free_mb / 1024)));
413        } else if free_mb > 256 {
414            checks.push(warn("disk space", &format!("{free_mb}MB free (low)")));
415        } else {
416            checks.push(fail("disk space", &format!("{free_mb}MB free (critical)")));
417        }
418    }
419
420    // Log directory
421    if Path::new("/var/log/nauka").exists() {
422        let log_file = Path::new("/var/log/nauka/nauka.log");
423        if log_file.exists() || std::fs::File::create("/var/log/nauka/.doctor-test").is_ok() {
424            let _ = std::fs::remove_file("/var/log/nauka/.doctor-test");
425            checks.push(ok("logs", "/var/log/nauka writable"));
426        } else {
427            checks.push(warn("logs", "/var/log/nauka not writable"));
428        }
429    } else {
430        checks.push(warn("logs", "/var/log/nauka missing"));
431    }
432
433    // State file
434    let state_path = dirs::home_dir()
435        .unwrap_or_else(|| std::path::PathBuf::from("/root"))
436        .join(".nauka/hypervisor.json");
437    if state_path.exists() {
438        match std::fs::read_to_string(&state_path) {
439            Ok(content) => {
440                if serde_json::from_str::<serde_json::Value>(&content).is_ok() {
441                    checks.push(ok("state file", "valid JSON"));
442                } else {
443                    checks.push(fail("state file", "corrupt (invalid JSON)"));
444                }
445            }
446            Err(e) => checks.push(fail("state file", &format!("unreadable: {e}"))),
447        }
448    } else {
449        checks.push(skip("state file", "not initialized"));
450    }
451}

457fn cmd_exists(cmd: &str) -> bool {
458    Command::new("which")
459        .arg(cmd)
460        .stdout(std::process::Stdio::null())
461        .stderr(std::process::Stdio::null())
462        .status()
463        .map(|s| s.success())
464        .unwrap_or(false)
465}
466
467fn ping6(addr: &Ipv6Addr) -> bool {
468    Command::new("ping6")
469        .args(["-c", "1", "-W", "2", &addr.to_string()])
470        .stdout(std::process::Stdio::null())
471        .stderr(std::process::Stdio::null())
472        .status()
473        .map(|s| s.success())
474        .unwrap_or(false)
475}
476
477fn curl_json(url: &str) -> Option<serde_json::Value> {
478    let output = Command::new("curl")
479        .args(["-sf", "--max-time", "5", url])
480        .output()
481        .ok()?;
482    if output.status.success() {
483        serde_json::from_slice(&output.stdout).ok()
484    } else {
485        None
486    }
487}
488
489fn disk_free_mb(path: &str) -> Option<u64> {
490    let output = Command::new("df")
491        .args(["-m", "--output=avail", path])
492        .output()
493        .ok()?;
494    if output.status.success() {
495        let stdout = String::from_utf8_lossy(&output.stdout);
496        stdout
497            .lines()
498            .last()
499            .and_then(|l| l.trim().parse::<u64>().ok())
500    } else {
501        None
502    }
503}
