223pub fn handler() -> HandlerFn {
224    Box::new(|req: OperationRequest| -> Pin<Box<dyn Future<Output = anyhow::Result<OperationResponse>> + Send>> {
225        Box::pin(async move {
226            match req.operation.as_str() {
227                "init" => handle_init(req).await,
228                "update" => handle_update(req).await,
229                "status" => handle_status().await,
230                "start" => handle_start().await,
231                "stop" => handle_stop().await,
232                "leave" => handle_leave().await,
233                "list" => handle_list().await,
234                "get" => handle_get(req).await,
235                "join" => handle_join(req).await,
236                "peering" => handle_peering(req).await,
237                "doctor" => handle_doctor().await,
238                "announce-listen" => handle_announce_listen(req).await,
239                "drain" => Ok(OperationResponse::Message("drain: not yet implemented".into())),
240                "enable" => Ok(OperationResponse::Message("enable: not yet implemented".into())),
241                other => Ok(OperationResponse::Message(format!("unknown: {other}"))),
242            }
243        })
244    })
245}

259async fn handle_init(req: OperationRequest) -> anyhow::Result<OperationResponse> {
260    let region = req
261        .fields
262        .get("region")
263        .map(|s| s.as_str())
264        .unwrap_or("default");
265    let zone = req
266        .fields
267        .get("zone")
268        .map(|s| s.as_str())
269        .unwrap_or("default");
270    let port: u16 = req
271        .fields
272        .get("port")
273        .and_then(|s| s.parse().ok())
274        .unwrap_or(51820);
275
276    let network_mode: fabric::NetworkMode = req
277        .fields
278        .get("mode")
279        .map(|s| s.as_str())
280        .unwrap_or("wireguard")
281        .parse()
282        .unwrap_or_default();
283
284    let fabric_interface = req
285        .fields
286        .get("interface")
287        .map(|s| s.as_str())
288        .unwrap_or("");
289
290    let endpoint = req.fields.get("endpoint").cloned();
291
292    let s3_endpoint = req
293        .fields
294        .get("s3-endpoint")
295        .ok_or_else(|| anyhow::anyhow!("missing required field: s3-endpoint"))?
296        .clone();
297    let s3_bucket = req
298        .fields
299        .get("s3-bucket")
300        .ok_or_else(|| anyhow::anyhow!("missing required field: s3-bucket"))?
301        .clone();
302    let s3_access_key = req
303        .fields
304        .get("s3-access-key")
305        .ok_or_else(|| anyhow::anyhow!("missing required field: s3-access-key"))?
306        .clone();
307    let s3_secret_key = req
308        .fields
309        .get("s3-secret-key")
310        .ok_or_else(|| anyhow::anyhow!("missing required field: s3-secret-key"))?
311        .clone();
312    let s3_region = req.fields.get("s3-region").cloned().unwrap_or_default();
313
314    let node_name = req
315        .fields
316        .get("name")
317        .filter(|s| !s.is_empty())
318        .cloned()
319        .unwrap_or_else(|| {
320            hostname::get()
321                .ok()
322                .and_then(|h| h.into_string().ok())
323                .map(|h| h.to_lowercase())
324                .unwrap_or_else(|| "node".to_string())
325        });
326
327    let peering = req
328        .fields
329        .get("peering")
330        .map(|s| s == "true")
331        .unwrap_or(false);
332
333    let ipv6_block = req.fields.get("ipv6-block").cloned();
334    let ipv4_public = req.fields.get("ipv4-public").cloned();
335
336    let db = open_db()?;
337    let init_cfg = fabric::ops::InitConfig {
338        node_name: &node_name,
339        region,
340        zone,
341        port,
342        network_mode,
343        fabric_interface,
344        endpoint,
345        ipv6_block,
346        ipv4_public,
347    };
348
349    // Generate a deterministic encryption password from the S3 secret key + region.
350    // All nodes in the same region will derive the same password.
351    let encryption_password = {
352        use sha2::{Digest, Sha256};
353        let mut hasher = Sha256::new();
354        hasher.update(s3_secret_key.as_bytes());
355        hasher.update(b":zerofs:");
356        hasher.update(region.as_bytes());
357        format!("{:x}", hasher.finalize())
358    };
359
360    let region_storage = storage::region::RegionStorage {
361        region: region.to_string(),
362        s3_endpoint,
363        s3_bucket,
364        s3_access_key,
365        s3_secret_key,
366        s3_region,
367        encryption_password,
368        is_default: true,
369    };
370
371    // Init: 2 (fabric) + 4 (control plane) + 2 (storage) + 3 (compute+forge) = 11
372    let step_count = if network_mode == fabric::NetworkMode::WireGuard {
373        11
374    } else {
375        5 // fabric + compute + forge
376    };
377    let steps = ui::Steps::new(step_count);
378
379    let result = fabric::ops::init(&db, &init_cfg, &steps)?;
380    // Track how far we got for rollback on failure
381    let fabric_initialized = true;
382    let mut controlplane_initialized = false;
383    let mut storage_initialized = false;
384
385    // Rollback closure — undoes completed steps in reverse order
386    let rollback = |fabric: bool, cp: bool, stor: bool| {
387        tracing::warn!("init failed — rolling back");
388        if stor {
389            tracing::info!("rollback: uninstalling storage");
390            let _ = storage::ops::leave();
391        }
392        if cp {
393            tracing::info!("rollback: uninstalling control plane");
394            let _ = controlplane::service::uninstall();
395        }
396        if fabric {
397            tracing::info!("rollback: removing fabric state and interface");
398            let backend = fabric::backend::create_backend(network_mode);
399            let _ = backend.teardown();
400            let _ = fabric::state::FabricState::delete(&db);
401        }
402    };
403
404    // Bootstrap control plane (TiKV) on the mesh — only in WireGuard mode
405    if network_mode == fabric::NetworkMode::WireGuard {
406        if let Err(e) =
407            controlplane::ops::bootstrap(&node_name, &result.hypervisor.mesh_ipv6, &steps)
408        {
409            steps.finish_err(&format!("Control plane failed: {e}"));
410            rollback(fabric_initialized, false, false);
411            return Err(e.into());
412        }
413        controlplane_initialized = true;
414    }
415
416    // Publish region storage config to distributed KV, then setup local storage
417    if network_mode == fabric::NetworkMode::WireGuard {
418        steps.set("Publishing storage config");
419        let pd_endpoint = format!(
420            "http://[{}]:{}",
421            result.hypervisor.mesh_ipv6,
422            controlplane::PD_CLIENT_PORT,
423        );
424        if let Err(e) =
425            storage::ops::publish_region_config(&[pd_endpoint.as_str()], &region_storage).await
426        {
427            steps.finish_err(&format!("Storage config failed: {e}"));
428            rollback(fabric_initialized, controlplane_initialized, false);
429            anyhow::bail!("{e}");
430        }
431        steps.inc();
432    }
433
434    if network_mode == fabric::NetworkMode::WireGuard {
435        steps.set("Setting up storage");
436        if let Err(e) = storage::ops::setup_region(&db, region_storage.clone()) {
437            steps.finish_err(&format!("Storage setup failed: {e}"));
438            rollback(fabric_initialized, controlplane_initialized, false);
439            return Err(e.into());
440        }
441        storage_initialized = true;
442        steps.inc();
443    }
444
445    // Non-critical steps — warn but don't rollback
446    if let Err(e) = crate::compute_setup::install(&steps) {
447        tracing::warn!(error = %e, "compute setup failed (VMs won't work until fixed)");
448    }
449
450    if !peering {
451        if let Err(e) = fabric::announce::install_service(port) {
452            tracing::warn!(error = %e, "announce service install failed");
453        }
454    }
455
456    // Suppress unused variable warnings
457    let _ = (fabric_initialized, storage_initialized);
458
459    steps.finish("Hypervisor initialized");
460    eprintln!();
461    eprintln!("  name     {}", result.hypervisor.name);
462    eprintln!("  id       {}", result.hypervisor.id.as_str());
463    eprintln!("  mesh     {}", result.mesh.id);
464    eprintln!("  region   {region}/{zone}");
465    eprintln!("  address  {}", result.hypervisor.mesh_ipv6);
466    eprintln!("  pin      {}", result.pin);
467    eprintln!();
468
469    if peering {
470        let peering_port = port + 1;
471        eprintln!("  Peering active on port {peering_port}");
472        eprintln!("  Nodes can join with:");
473        eprintln!(
474            "    nauka hypervisor join --target <this-ip>:{peering_port} --pin {}",
475            result.pin
476        );
477        eprintln!();
478        eprintln!("  Waiting for joins... (Ctrl+C to stop)");
479        eprintln!();
480
481        // Block here listening for joins (DB opened per-request, no lock held)
482        let accepted = fabric::ops::listen_for_peers(
483            &result.pin,
484            peering_port,
485            3600, // 1 hour timeout
486        )
487        .await?;
488
489        eprintln!("  {} node(s) joined.", accepted);
490
491        // Peering session ended — install announce service for persistent listening
492        if let Err(e) = fabric::announce::install_service(port) {
493            tracing::warn!(error = %e, "announce service install failed");
494        }
495    } else {
496        eprintln!("  To accept joins on this node:");
497        eprintln!("    nauka hypervisor peering");
498        eprintln!();
499        eprintln!("  Or from another node:");
500        eprintln!(
501            "    nauka hypervisor join --target <this-ip> --pin {}",
502            result.pin
503        );
504    }
505
506    Ok(OperationResponse::Resource(serde_json::json!({
507        "name": result.hypervisor.name,
508        "id": result.hypervisor.id.as_str(),
509        "region": region,
510        "zone": zone,
511        "mesh_ipv6": result.hypervisor.mesh_ipv6.to_string(),
512        "state": "available",
513        "pin": result.pin,
514    })))
515}

517async fn handle_join(req: OperationRequest) -> anyhow::Result<OperationResponse> {
518    let target = req
519        .fields
520        .get("target")
521        .ok_or_else(|| anyhow::anyhow!("missing required field: target"))?
522        .clone();
523    let region = req
524        .fields
525        .get("region")
526        .map(|s| s.as_str())
527        .unwrap_or("default");
528    let zone = req
529        .fields
530        .get("zone")
531        .map(|s| s.as_str())
532        .unwrap_or("default");
533    let port: u16 = req
534        .fields
535        .get("port")
536        .and_then(|s| s.parse().ok())
537        .unwrap_or(51820);
538    let pin = req.fields.get("pin").map(|s| s.as_str());
539
540    let network_mode: fabric::NetworkMode = req
541        .fields
542        .get("mode")
543        .map(|s| s.as_str())
544        .unwrap_or("wireguard")
545        .parse()
546        .unwrap_or_default();
547
548    let node_name = req
549        .fields
550        .get("name")
551        .filter(|s| !s.is_empty())
552        .cloned()
553        .unwrap_or_else(|| {
554            hostname::get()
555                .ok()
556                .and_then(|h| h.into_string().ok())
557                .map(|h| h.to_lowercase())
558                .unwrap_or_else(|| "node".to_string())
559        });
560
561    let ipv6_block = req.fields.get("ipv6-block").cloned();
562    let ipv4_public = req.fields.get("ipv4-public").cloned();
563
564    let db = open_db()?;
565    let join_cfg = fabric::ops::JoinConfig {
566        target: &target,
567        node_name: &node_name,
568        region,
569        zone,
570        port,
571        pin,
572        network_mode,
573        ipv6_block,
574        ipv4_public,
575    };
576
577    // Join: 2 (fabric) + 3 (control plane) + 1 (storage) + 3 (compute+forge) + 1 (announce) = 10
578    let step_count = if network_mode == fabric::NetworkMode::WireGuard {
579        10
580    } else {
581        6
582    };
583    let steps = ui::Steps::new(step_count);
584
585    let result = fabric::ops::join(&db, &join_cfg, &steps).await?;
586
587    // Join control plane — only in WireGuard mode
588    if network_mode == fabric::NetworkMode::WireGuard {
589        let state = fabric::state::FabricState::load(&db)
590            .map_err(|e| anyhow::anyhow!("{e}"))?
591            .ok_or_else(|| anyhow::anyhow!("state missing after join"))?;
592        let pd_endpoints: Vec<String> = state
593            .peers
594            .peers
595            .iter()
596            .map(|p| format!("http://[{}]:{}", p.mesh_ipv6, controlplane::PD_CLIENT_PORT))
597            .collect();
598        let peer_count = state.peers.len();
599        controlplane::ops::join(
600            &node_name,
601            &result.hypervisor.mesh_ipv6,
602            &pd_endpoints,
603            peer_count,
604            &steps,
605        )?;
606    }
607
608    // Fetch region storage config from distributed KV and setup locally
609    if network_mode == fabric::NetworkMode::WireGuard {
610        steps.set("Setting up storage");
611        let state = fabric::state::FabricState::load(&db)
612            .map_err(|e| anyhow::anyhow!("{e}"))?
613            .ok_or_else(|| anyhow::anyhow!("state missing after join"))?;
614        let pd_endpoints: Vec<String> = state
615            .peers
616            .peers
617            .iter()
618            .map(|p| format!("http://[{}]:{}", p.mesh_ipv6, controlplane::PD_CLIENT_PORT))
619            .collect();
620        let pd_refs: Vec<&str> = pd_endpoints.iter().map(|s| s.as_str()).collect();
621
622        let region_config = storage::ops::fetch_region_config(&pd_refs, region)
623            .await?
624            .ok_or_else(|| {
625                anyhow::anyhow!(
626                    "no storage backend configured for region '{region}'.\n\n\
627                     An S3 backend must be registered before nodes can join this region.\n\
628                     On the init node, re-initialize with storage flags:\n\n\
629                     \x20 nauka hypervisor init --region {region} \\\n\
630                     \x20   --s3-endpoint <URL> --s3-bucket <BUCKET> \\\n\
631                     \x20   --s3-access-key <KEY> --s3-secret-key <SECRET>"
632                )
633            })?;
634
635        storage::ops::setup_region(&db, region_config)?;
636        steps.inc();
637    }
638
639    // Install compute runtime + base image
640    if let Err(e) = crate::compute_setup::install(&steps) {
641        tracing::warn!(error = %e, "compute setup failed (VMs won't work until fixed)");
642    }
643
644    // Install persistent announce listener
645    if let Err(e) = fabric::announce::install_service(port) {
646        tracing::warn!(error = %e, "announce service install failed");
647    }
648
649    // Self-announce to all peers (ensures they know about us even if the
650    // peering server's announce arrived before their listener was ready).
651    // Re-read state after a delay to include peers discovered via incoming announces.
652    steps.set("Announcing to peers");
653    {
654        tokio::time::sleep(std::time::Duration::from_secs(3)).await;
655        // Re-open DB to get latest state (may include peers from incoming announces)
656        let db = open_db()?;
657        let state = fabric::state::FabricState::load(&db)
658            .map_err(|e| anyhow::anyhow!("{e}"))?
659            .ok_or_else(|| anyhow::anyhow!("state missing after join"))?;
660        let self_info = fabric::peering::PeerInfo {
661            name: state.hypervisor.name.clone(),
662            region: state.hypervisor.region.clone(),
663            zone: state.hypervisor.zone.clone(),
664            wg_public_key: state.hypervisor.wg_public_key.clone(),
665            wg_port: state.hypervisor.wg_port,
666            endpoint: state.hypervisor.endpoint.clone(),
667            mesh_ipv6: state.hypervisor.mesh_ipv6,
668        };
669        let peers: Vec<_> = state.peers.peers.clone();
670        let (ok, fail) = fabric::announce::broadcast_new_peer(
671            &self_info,
672            &state.hypervisor.name,
673            &peers,
674            state.hypervisor.wg_port,
675        )
676        .await;
677        if ok > 0 || fail > 0 {
678            tracing::info!(successes = ok, failures = fail, "self-announce to peers");
679        }
680    }
681    steps.inc();
682
683    steps.finish("Joined cluster");
684    eprintln!();
685    eprintln!("  name     {}", result.hypervisor.name);
686    eprintln!("  id       {}", result.hypervisor.id.as_str());
687    eprintln!("  region   {region}/{zone}");
688    eprintln!("  address  {}", result.hypervisor.mesh_ipv6);
689    eprintln!("  peers    {}", result.peer_count);
690    eprintln!();
691
692    Ok(OperationResponse::Resource(serde_json::json!({
693        "name": result.hypervisor.name,
694        "id": result.hypervisor.id.as_str(),
695        "region": region,
696        "zone": zone,
697        "mesh_ipv6": result.hypervisor.mesh_ipv6.to_string(),
698        "state": "available",
699        "peers": result.peer_count,
700    })))
701}

703async fn handle_peering(req: OperationRequest) -> anyhow::Result<OperationResponse> {
704    let timeout_secs: u64 = req
705        .fields
706        .get("timeout")
707        .and_then(|s| s.parse().ok())
708        .unwrap_or(3600);
709
710    let db = open_db()?;
711    let state = fabric::state::FabricState::load(&db)
712        .map_err(|e| anyhow::anyhow!("{e}"))?
713        .ok_or_else(|| anyhow::anyhow!("not initialized. Run 'nauka hypervisor init' first."))?;
714
715    // Derive PIN from secret
716    let secret: nauka_core::crypto::MeshSecret = state.secret.parse()?;
717    let pin = secret.derive_pin();
718    let peering_port = state.hypervisor.wg_port + 1;
719
720    // Drop DB before starting listener (release lock!)
721    drop(db);
722
723    eprintln!();
724    eprintln!("  Peering active on port {peering_port}");
725    eprintln!("  PIN: {pin}");
726    eprintln!();
727    eprintln!("  Nodes can join with:");
728    eprintln!("    nauka hypervisor join --target <this-ip>:{peering_port} --pin {pin}");
729    eprintln!();
730    eprintln!("  Waiting for joins... (Ctrl+C to stop)");
731    eprintln!();
732
733    let accepted = fabric::ops::listen_for_peers(&pin, peering_port, timeout_secs).await?;
734
735    Ok(OperationResponse::Message(format!(
736        "{accepted} node(s) joined."
737    )))
738}

767async fn handle_status() -> anyhow::Result<OperationResponse> {
768    let db = open_db()?;
769    let s = fabric::ops::status(&db)?;
770
771    // Control plane status (best-effort — may not be installed yet)
772    let mesh_ip: std::net::Ipv6Addr = s
773        .mesh_ipv6
774        .parse()
775        .map_err(|_| anyhow::anyhow!("corrupt state: invalid mesh_ipv6 '{}'", s.mesh_ipv6))?;
776    let cp = controlplane::ops::status(&mesh_ip);
777    let (pd_active, tikv_active, pd_members, tikv_stores, leader) = match cp {
778        Ok(cs) => (
779            cs.pd_active,
780            cs.tikv_active,
781            cs.pd_members,
782            cs.tikv_stores,
783            cs.leader,
784        ),
785        Err(_) => (false, false, 0, 0, None),
786    };
787
788    Ok(OperationResponse::Resource(serde_json::json!({
789        "name": s.hypervisor_name,
790        "id": s.hypervisor_id,
791        "region": s.region,
792        "zone": s.zone,
793        "mesh_ipv6": s.mesh_ipv6,
794        "state": if s.service_active && tikv_active { &s.state } else if s.service_active { "degraded" } else { "down" },
795        "wg": if s.service_active { "running" } else { "stopped" },
796        "wg_interface": s.wg_interface_up,
797        "peers": s.peer_count,
798        "wg_port": s.wg_port,
799        "rx_bytes": s.rx_bytes,
800        "tx_bytes": s.tx_bytes,
801        "pd": if pd_active { "running" } else { "stopped" },
802        "tikv": if tikv_active { "running" } else { "stopped" },
803        "pd_members": pd_members,
804        "tikv_stores": tikv_stores,
805        "leader": leader,
806    })))
807}

823    let port: u16 = req
824        .fields
825        .get("port")
826        .and_then(|s| s.parse().ok())
827        .unwrap_or(51822);
828
829    let bind_addr: std::net::SocketAddr = format!("[::]:{port}")
830        .parse()
831        .map_err(|_| anyhow::anyhow!("invalid port"))?;
832

841    fabric::announce::listen(db_opener, bind_addr).await?;
842
843    Ok(OperationResponse::None)
844}

861async fn handle_leave() -> anyhow::Result<OperationResponse> {
862    let db = open_db()?;
863    // Get mesh IPv6 for TiKV deregistration before leaving
864    let mesh_ipv6 = fabric::state::FabricState::load(&db)
865        .ok()
866        .flatten()
867        .map(|s| s.hypervisor.mesh_ipv6);
868
869    let steps = ui::Steps::new(4);
870
871    // Storage first (stop ZeroFS instances)
872    steps.set("Stopping storage");
873    let _ = storage::ops::leave();
874    steps.inc();
875
876    // Controlplane (deregister TiKV store, then uninstall)
877    steps.set("Leaving control plane");
878    if let Some(ipv6) = mesh_ipv6 {
879        let _ = controlplane::ops::leave_with_mesh(&ipv6);
880    } else {
881        let _ = controlplane::ops::leave();
882    }
883    steps.inc();
884
885    // Notify peers + tear down mesh
886    steps.set("Notifying peers");
887    let _ = fabric::announce::uninstall_service();
888    fabric::ops::leave(&db).await?;
889    steps.inc();
890
891    // Final cleanup
892    steps.set("Removing services");
893    steps.inc();
894
895    steps.finish("Left the cluster");
896    Ok(OperationResponse::None)
897}

899async fn handle_list() -> anyhow::Result<OperationResponse> {
900    let db = open_db()?;
901    let state = match fabric::state::FabricState::load(&db).map_err(|e| anyhow::anyhow!("{e}"))? {
902        Some(s) => s,
903        None => return Ok(OperationResponse::ResourceList(vec![])),
904    };
905
906    let backend = fabric::backend::create_backend(state.network_mode);
907    let self_state = if backend.is_up() { "available" } else { "down" };
908
909    let mut items = vec![serde_json::json!({
910        "name": state.hypervisor.name,
911        "region": state.hypervisor.region,
912        "zone": state.hypervisor.zone,
913        "state": self_state,
914        "cpu": "0/0",
915        "memory": "0/0",
916        "vms": 0,
917    })];
918
919    for peer in &state.peers.peers {
920        items.push(serde_json::json!({
921            "name": peer.name,
922            "region": peer.region,
923            "zone": peer.zone,
924            "state": peer_state_label(peer),
925            "cpu": "0/0",
926            "memory": "0/0",
927            "vms": 0,
928        }));
929    }
930
931    Ok(OperationResponse::ResourceList(items))
932}

934async fn handle_get(req: OperationRequest) -> anyhow::Result<OperationResponse> {
935    let name = req
936        .name
937        .ok_or_else(|| anyhow::anyhow!("missing hypervisor name"))?;
938    let db = open_db()?;
939    let state = fabric::state::FabricState::load(&db)
940        .map_err(|e| anyhow::anyhow!("{e}"))?
941        .ok_or_else(|| anyhow::anyhow!("not initialized"))?;
942
943    let backend = fabric::backend::create_backend(state.network_mode);
944
945    if state.hypervisor.name == name {
946        return Ok(OperationResponse::Resource(serde_json::json!({
947            "name": state.hypervisor.name,
948            "id": state.hypervisor.id.as_str(),
949            "region": state.hypervisor.region,
950            "zone": state.hypervisor.zone,
951            "mesh_ipv6": state.hypervisor.mesh_ipv6.to_string(),
952            "state": if backend.is_up() { "available" } else { "down" },
953        })));
954    }
955
956    if let Some(peer) = state.peers.find_by_name(&name) {
957        return Ok(OperationResponse::Resource(serde_json::json!({
958            "name": peer.name,
959            "id": peer.id.as_str(),
960            "region": peer.region,
961            "zone": peer.zone,
962            "mesh_ipv6": peer.mesh_ipv6.to_string(),
963            "state": peer_state_label(peer),
964        })));
965    }
966
967    anyhow::bail!("hypervisor '{name}' not found")
968}

970/// Map peer status to user-facing label.
971fn peer_state_label(peer: &fabric::peer::Peer) -> &'static str {
972    match peer.status {
973        fabric::peer::PeerStatus::Active => "available",
974        fabric::peer::PeerStatus::Unreachable => "unreachable",
975        fabric::peer::PeerStatus::Removed => "removed",
976    }
977}
