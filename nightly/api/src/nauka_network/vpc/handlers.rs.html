8pub fn resource_def() -> ResourceDef {
9    ResourceDef::build("vpc", "Manage virtual private clouds")
10        .plural("vpcs")
11        .parent("org", "--org", "Organization")
12        .create()
13        .op(|op| {
14            op.with_arg(OperationArg::required(
15                "cidr",
16                FieldDef::cidr("cidr", "VPC CIDR block (e.g. 10.0.0.0/16)"),
17            ))
18            .with_arg(OperationArg::optional(
19                "project",
20                FieldDef::string("project", "Scope to a project"),
21            ))
22            .with_arg(OperationArg::optional(
23                "env",
24                FieldDef::string("env", "Scope to an environment"),
25            ))
26            .with_progress(ProgressHint::Spinner("Creating VPC..."))
27            .with_example("nauka vpc create prod-net --org acme --cidr 10.0.0.0/16")
28        })
29        .list()
30        .get()
31        .delete()
32        .op(|op| op.with_progress(ProgressHint::Spinner("Deleting VPC...")))
33        .column("NAME", "name")
34        .column("CIDR", "cidr")
35        .column("ORG", "org_name")
36        .column("VNI", "vni")
37        .column("ID", "id")
38        .column_def(ColumnDef::new("CREATED", "created_at").with_format(DisplayFormat::Timestamp))
39        .empty_message(
40            "No VPCs found. Create one with: nauka vpc create <name> --org <org> --cidr <cidr>",
41        )
42        .sort_by("name")
43        .detail_section(
44            None,
45            vec![
46                DetailField::new("Name", "name"),
47                DetailField::new("ID", "id"),
48                DetailField::new("CIDR", "cidr"),
49                DetailField::new("VNI", "vni"),
50                DetailField::new("Organization", "org_name"),
51                DetailField::new("Created", "created_at").with_format(DisplayFormat::Timestamp),
52            ],
53        )
54        .done()
55}

57pub fn handler() -> HandlerFn {
58    Box::new(
59        |req: OperationRequest| -> Pin<
60            Box<dyn Future<Output = anyhow::Result<OperationResponse>> + Send>,
61        > {
62            Box::pin(async move {
63                // P2.12 (sifrah/nauka#216): VpcStore now takes an
64                // EmbeddedDb directly; reach the cluster handle via
65                // the wrapper's `.embedded()` accessor.
66                let cluster_db = nauka_hypervisor::controlplane::connect().await?;
67                let store = VpcStore::new(cluster_db.embedded().clone());
68                match req.operation.as_str() {
69                    "create" => {
70                        let name = req.name.ok_or_else(|| anyhow::anyhow!("missing name"))?;
71                        let org = req
72                            .scope
73                            .get("org")
74                            .ok_or_else(|| anyhow::anyhow!("--org is required"))?
75                            .to_string();
76                        let cidr = req
77                            .fields
78                            .get("cidr")
79                            .ok_or_else(|| anyhow::anyhow!("--cidr is required"))?
80                            .clone();
81                        let project = req.fields.get("project").cloned();
82                        let env = req.fields.get("env").cloned();
83                        nauka_core::validate::name(&name)?;
84                        let vpc = store.create(&name, &org, &cidr, project, env).await?;
85                        Ok(OperationResponse::Resource(vpc.to_api_json()))
86                    }
87                    "list" => {
88                        let org = req.scope.get("org").map(|s| s.to_string());
89                        let vpcs = store.list(org.as_deref()).await?;
90                        let items: Vec<serde_json::Value> =
91                            vpcs.iter().map(|v| v.to_api_json()).collect();
92                        Ok(OperationResponse::ResourceList(items))
93                    }
94                    "get" => {
95                        let name = req
96                            .name
97                            .ok_or_else(|| anyhow::anyhow!("missing name or ID"))?;
98                        let org = req.scope.get("org").map(|s| s.to_string());
99                        let vpc = store
100                            .get(&name, org.as_deref())
101                            .await?
102                            .ok_or_else(|| anyhow::anyhow!("vpc '{name}' not found"))?;
103                        Ok(OperationResponse::Resource(vpc.to_api_json()))
104                    }
105                    "delete" => {
106                        let name = req
107                            .name
108                            .ok_or_else(|| anyhow::anyhow!("missing name or ID"))?;
109                        let org = req
110                            .scope
111                            .get("org")
112                            .ok_or_else(|| anyhow::anyhow!("--org is required"))?
113                            .to_string();
114                        store.delete(&name, &org).await?;
115                        Ok(OperationResponse::Message(format!("vpc '{name}' deleted.")))
116                    }
117                    other => Ok(OperationResponse::Message(format!("unknown: {other}"))),
118                }
119            })
120        },
121    )
122}
