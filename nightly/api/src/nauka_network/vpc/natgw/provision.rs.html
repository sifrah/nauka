1//! NAT Gateway provisioning.
2//!
3//! Sets up outbound internet access for VMs in a VPC via a dedicated public IPv6:
4//!
5//! - DNS64 (unbound) on the bridge gateway synthesizes AAAA records using 64:ff9b::/96
6//! - Jool NAT64 translates IPv6 (64:ff9b::) → IPv4 for IPv4-only destinations
7//! - NAT66 (nftables SNAT) pins the public IPv6 as source for all VM IPv6 traffic
8//! - NAT44 (MASQUERADE) as fallback for direct IPv4 outbound
9//!
10//! Flow: VM resolves via DNS64 → gets 64:ff9b::x.x.x.x → sends IPv6 →
11//!       bridge forwards → Jool translates to IPv4 → exits with NAT GW IPv6 as source
12//!       (for IPv6-native sites, NAT66 SNAT applies directly)
13

19/// Jool instance name for a NAT gateway, derived from VPC ID hash.
20pub fn jool_instance_name(vpc_id: &str) -> String {
21    use sha2::{Digest, Sha256};
22    let hash = Sha256::digest(vpc_id.as_bytes());
23    format!("nk-{}", hex::encode(&hash[..3]))
24}

26/// Check if the Jool kernel module is available.
27pub fn jool_available() -> bool {
28    Command::new("modprobe")
29        .args(["--dry-run", "jool"])
30        .output()
31        .map(|o| o.status.success())
32        .unwrap_or(false)
33}

35/// Derive a ULA IPv6 gateway address for internal use on a VPC bridge.
36pub fn bridge_ipv6_gateway(vpc_cidr: &str) -> Ipv6Addr {
37    use sha2::{Digest, Sha256};
38    let hash = Sha256::digest(vpc_cidr.as_bytes());
39    let seg2 = u16::from_be_bytes([hash[0], hash[1]]);
40    let seg3 = u16::from_be_bytes([hash[2], hash[3]]);
41    Ipv6Addr::new(0xfd00, 0x6e61, seg2, seg3, 0, 0, 0, 1)
42}
43
44/// Derive a ULA IPv6 for a VM inside the VPC bridge.
45pub fn bridge_ipv6_vm(vpc_cidr: &str, vm_index: u16) -> Ipv6Addr {
46    use sha2::{Digest, Sha256};
47    let hash = Sha256::digest(vpc_cidr.as_bytes());
48    let seg2 = u16::from_be_bytes([hash[0], hash[1]]);
49    let seg3 = u16::from_be_bytes([hash[2], hash[3]]);
50    Ipv6Addr::new(0xfd00, 0x6e61, seg2, seg3, 0, 0, 0, 2 + vm_index)
51}

53/// Provision NAT gateway for a VPC.
54pub fn ensure_nat_gateway(
55    vpc_id: &str,
56    vpc_cidr: &str,
57    vni: u32,
58    public_ipv6: &Ipv6Addr,
59    public_interface: &str,
60) -> anyhow::Result<()> {
61    let instance = jool_instance_name(vpc_id);
62    let br = bridge_name(vpc_id);
63    let table = vni.to_string();
64
65    // ── 1. Enable forwarding ──
66    let _ = run_cmd("sysctl", &["-w", "net.ipv4.ip_forward=1"]);
67    let _ = run_cmd("sysctl", &["-w", "net.ipv6.conf.all.forwarding=1"]);
68    let _ = run_cmd("sysctl", &["-w", "net.ipv4.conf.all.rp_filter=0"]);
69
70    // ── 2. Default route in VPC routing table via host gateway ──
71    if let Some(gw) = detect_default_gateway() {
72        let _ = run_cmd(
73            "ip",
74            &[
75                "route",
76                "replace",
77                "default",
78                "via",
79                &gw,
80                "dev",
81                public_interface,
82                "table",
83                &table,
84            ],
85        );
86    }
87
88    // ── 2b. ip rule: return traffic for VPC CIDR uses VPC routing table ──
89    let _ = run_cmd(
90        "ip",
91        &[
92            "rule", "add", "to", vpc_cidr, "lookup", &table, "priority", "32763",
93        ],
94    );
95
96    // ── 3. NAT44 MASQUERADE for IPv4 outbound (per-VPC bridge, not wildcard) ──
97    ensure_nft_nat4(&br, public_interface)?;
98
99    // ── 4. Jool NAT64 ──
100    let _ = run_cmd("modprobe", &["jool"]);
101    let addr_cidr = format!("{}/128", public_ipv6);
102    let _ = run_cmd(
103        "ip",
104        &["-6", "addr", "add", &addr_cidr, "dev", public_interface],
105    );
106    let _ = run_cmd(
107        "jool",
108        &[
109            "instance",
110            "add",
111            &instance,
112            "--netfilter",
113            "--pool6",
114            "64:ff9b::/96",
115        ],
116    );
117
118    // ── 5. NAT66 SNAT: VM IPv6 traffic from this VPC exits with its NAT GW public IPv6 ──
119    ensure_nft_nat6(&br, public_ipv6, public_interface)?;
120
121    // ── 6. DNS64 resolver on the bridge ──
122    ensure_dns64(&br, vpc_cidr)?;
123
124    // ── 7. Fix host isolation: allow NDP + DNS + established replies ──
125    fix_host_isolation()?;
126
127    Ok(())
128}

130/// Remove NAT gateway provisioning for a VPC.
131///
132/// Cleans up: Jool instance, public IPv6 address, and nftables NAT rules.
133pub fn remove_nat_gateway(
134    vpc_id: &str,
135    public_ipv6: &Ipv6Addr,
136    public_interface: &str,
137) -> anyhow::Result<()> {
138    let instance = jool_instance_name(vpc_id);
139    let br = bridge_name(vpc_id);
140
141    // Remove Jool NAT64 instance
142    let _ = run_cmd("jool", &["instance", "remove", &instance]);
143
144    // Remove public IPv6 from interface
145    let addr_cidr = format!("{}/128", public_ipv6);
146    let _ = run_cmd(
147        "ip",
148        &["-6", "addr", "del", &addr_cidr, "dev", public_interface],
149    );
150
151    // Remove NAT44 masquerade rule for this bridge
152    remove_nft_rules_matching("ip", "nauka_nat4", "postrouting", &br);
153
154    // Remove NAT66 SNAT rule for this bridge
155    remove_nft_rules_matching("ip6", "nauka_nat", "postrouting", &br);
156
157    Ok(())
158}

160/// Remove all nftables rules in a chain that match a substring.
161fn remove_nft_rules_matching(family: &str, table: &str, chain: &str, pattern: &str) {
162    // List rules with handles
163    let output = Command::new("nft")
164        .args(["-a", "list", "chain", family, table, chain])
165        .output();
166    let text = match output {
167        Ok(ref o) if o.status.success() => String::from_utf8_lossy(&o.stdout).to_string(),
168        _ => return,
169    };
170    // Find handle numbers for matching rules and delete them
171    for line in text.lines() {
172        if line.contains(pattern) {
173            if let Some(handle) = line.rsplit("# handle ").next() {
174                if let Ok(_h) = handle.trim().parse::<u64>() {
175                    let _ = Command::new("nft")
176                        .args([
177                            "delete",
178                            "rule",
179                            family,
180                            table,
181                            chain,
182                            "handle",
183                            handle.trim(),
184                        ])
185                        .stdout(std::process::Stdio::null())
186                        .stderr(std::process::Stdio::null())
187                        .status();
188                }
189            }
190        }
191    }
192}

194/// Remove NAT rules for bridges that no longer have a NAT gateway.
195///
196/// Compares active NAT gateway bridges against existing nftables rules and
197/// removes any rules for bridges that shouldn't have NAT.
198pub fn flush_stale_nat_rules(active_natgws: &[&super::types::NatGw]) {
199    let active_bridges: Vec<String> = active_natgws
200        .iter()
201        .map(|n| bridge_name(n.vpc_id.as_str()))
202        .collect();
203
204    // Clean stale NAT44 masquerade rules
205    flush_stale_rules_in_chain("ip", "nauka_nat4", "postrouting", &active_bridges);
206
207    // Clean stale NAT66 SNAT rules
208    flush_stale_rules_in_chain("ip6", "nauka_nat", "postrouting", &active_bridges);
209}

211fn flush_stale_rules_in_chain(family: &str, table: &str, chain: &str, active_bridges: &[String]) {
212    let output = Command::new("nft")
213        .args(["-a", "list", "chain", family, table, chain])
214        .output();
215    let text = match output {
216        Ok(ref o) if o.status.success() => String::from_utf8_lossy(&o.stdout).to_string(),
217        _ => return,
218    };
219    for line in text.lines() {
220        let trimmed = line.trim();
221        // Only process actual rules with iifname (skip chain header/type lines)
222        if !trimmed.contains("iifname") {
223            continue;
224        }
225        // Check if this rule references a bridge NOT in the active set
226        let is_active = active_bridges
227            .iter()
228            .any(|br| trimmed.contains(br.as_str()));
229        if !is_active {
230            if let Some(handle) = trimmed.rsplit("# handle ").next() {
231                if let Ok(_h) = handle.trim().parse::<u64>() {
232                    let _ = Command::new("nft")
233                        .args([
234                            "delete",
235                            "rule",
236                            family,
237                            table,
238                            chain,
239                            "handle",
240                            handle.trim(),
241                        ])
242                        .stdout(std::process::Stdio::null())
243                        .stderr(std::process::Stdio::null())
244                        .status();
245                }
246            }
247        }
248    }
249}

251/// Detect the host's default IPv4 gateway.
252fn detect_default_gateway() -> Option<String> {
253    let output = Command::new("ip")
254        .args(["route", "show", "default"])
255        .output()
256        .ok()?;
257    let stdout = String::from_utf8_lossy(&output.stdout);
258    let parts: Vec<&str> = stdout.split_whitespace().collect();
259    for window in parts.windows(2) {
260        if window[0] == "via" {
261            return Some(window[1].to_string());
262        }
263    }
264    None
265}

267/// Detect the IPv4 address assigned to a bridge interface.
268fn detect_bridge_ipv4(bridge: &str) -> Option<String> {
269    let output = Command::new("ip")
270        .args(["-4", "addr", "show", "dev", bridge])
271        .output()
272        .ok()?;
273    let stdout = String::from_utf8_lossy(&output.stdout);
274    for line in stdout.lines() {
275        let trimmed = line.trim();
276        if let Some(rest) = trimmed.strip_prefix("inet ") {
277            if let Some(addr) = rest.split('/').next() {
278                return Some(addr.to_string());
279            }
280        }
281    }
282    None
283}

285/// NAT44: MASQUERADE for IPv4 traffic from a specific VPC bridge.
286fn ensure_nft_nat4(vpc_bridge: &str, out_iface: &str) -> anyhow::Result<()> {
287    let _ = run_nft("add table ip nauka_nat4");
288    let _ =
289        run_nft("add chain ip nauka_nat4 postrouting { type nat hook postrouting priority 100 ; }");
290    let existing = Command::new("nft")
291        .args(["list", "chain", "ip", "nauka_nat4", "postrouting"])
292        .output()
293        .map(|o| String::from_utf8_lossy(&o.stdout).contains(vpc_bridge))
294        .unwrap_or(false);
295    if !existing {
296        run_nft(&format!(
297            "add rule ip nauka_nat4 postrouting iifname \"{}\" oifname \"{}\" masquerade",
298            vpc_bridge, out_iface
299        ))?;
300    }
301    Ok(())
302}

304/// NAT66: SNAT VM IPv6 traffic from a specific VPC bridge to the NAT GW's public IPv6.
305///
306/// Removes any stale SNAT rule for this bridge before adding the current one.
307/// This handles the case where a NAT gateway is deleted and recreated with a new IPv6.
308fn ensure_nft_nat6(
309    vpc_bridge: &str,
310    public_ipv6: &Ipv6Addr,
311    out_iface: &str,
312) -> anyhow::Result<()> {
313    let _ = run_nft("add table ip6 nauka_nat");
314    let _ =
315        run_nft("add chain ip6 nauka_nat postrouting { type nat hook postrouting priority 100 ; }");
316
317    // Check if correct rule already exists
318    let ipv6_str = public_ipv6.to_string();
319    let chain_output = Command::new("nft")
320        .args(["list", "chain", "ip6", "nauka_nat", "postrouting"])
321        .output()
322        .ok();
323    let chain_text = chain_output
324        .as_ref()
325        .map(|o| String::from_utf8_lossy(&o.stdout).to_string())
326        .unwrap_or_default();
327
328    let has_correct_rule = chain_text.contains(vpc_bridge) && chain_text.contains(&ipv6_str);
329    let has_stale_rule = chain_text.contains(vpc_bridge) && !chain_text.contains(&ipv6_str);
330
331    if has_stale_rule {
332        // Flush and recreate all rules — stale SNAT for this bridge exists
333        let _ = run_nft("flush chain ip6 nauka_nat postrouting");
334        // Re-add rules for all bridges that aren't this one
335        for line in chain_text.lines() {
336            let trimmed = line.trim();
337            if trimmed.contains("snat to") && !trimmed.contains(vpc_bridge) {
338                let _ = run_nft(&format!("add rule ip6 nauka_nat postrouting {}", trimmed));
339            }
340        }
341        // Add correct rule for this bridge
342        run_nft(&format!(
343            "add rule ip6 nauka_nat postrouting iifname \"{}\" oifname \"{}\" snat to {}",
344            vpc_bridge, out_iface, public_ipv6
345        ))?;
346    } else if !has_correct_rule {
347        run_nft(&format!(
348            "add rule ip6 nauka_nat postrouting iifname \"{}\" oifname \"{}\" snat to {}",
349            vpc_bridge, out_iface, public_ipv6
350        ))?;
351    }
352    Ok(())
353}

355/// Fix the host isolation rules to allow NAT Gateway traffic through.
356///
357/// The default host isolation drops ALL output to VPC bridges. NAT Gateway
358/// needs: NDP (IPv6 neighbor discovery), DNS replies, and conntrack return traffic.
359fn fix_host_isolation() -> anyhow::Result<()> {
360    // Check if the nauka table has the raw drop rule
361    let output = Command::new("nft")
362        .args(["list", "chain", "inet", "nauka", "output"])
363        .output();
364    let has_raw_drop = output
365        .as_ref()
366        .map(|o| {
367            let s = String::from_utf8_lossy(&o.stdout);
368            s.contains("oifname") && s.contains("drop") && !s.contains("ct state")
369        })
370        .unwrap_or(false);
371
372    if has_raw_drop {
373        // Replace with a smarter rule set that allows NAT GW traffic
374        let _ = run_nft("flush chain inet nauka output");
375        let _ = run_nft(
376            "add rule inet nauka output oifname \"nkb-*\" ct state established,related accept",
377        );
378        let _ = run_nft("add rule inet nauka output oifname \"nkb-*\" udp sport 53 accept");
379        let _ = run_nft("add rule inet nauka output oifname \"nkb-*\" tcp sport 53 accept");
380        let _ = run_nft("add rule inet nauka output oifname \"nkb-*\" icmpv6 type { nd-neighbor-solicit, nd-neighbor-advert, nd-router-advert } accept");
381        let _ = run_nft("add rule inet nauka output oifname \"nkb-*\" drop");
382    }
383
384    Ok(())
385}

387/// Install and configure unbound as a DNS64 resolver on the bridge.
388fn ensure_dns64(bridge: &str, vpc_cidr: &str) -> anyhow::Result<()> {
389    if !Command::new("which")
390        .arg("unbound")
391        .output()
392        .map(|o| o.status.success())
393        .unwrap_or(false)
394    {
395        let _ = run_cmd("apt-get", &["install", "-y", "-qq", "unbound"]);
396    }
397
398    let gw_ipv4 = detect_bridge_ipv4(bridge).unwrap_or_else(|| "10.0.1.1".to_string());
399    let gw_ipv6 = bridge_ipv6_gateway(vpc_cidr);
400
401    // Add ULA IPv6 to bridge
402    let _ = run_cmd(
403        "ip",
404        &[
405            "-6",
406            "addr",
407            "add",
408            &format!("{}/64", gw_ipv6),
409            "dev",
410            bridge,
411        ],
412    );
413
414    // Clean old nauka DNS64 configs to avoid conflicts
415    if let Ok(entries) = std::fs::read_dir("/etc/unbound/unbound.conf.d") {
416        for entry in entries.flatten() {
417            let name = entry.file_name();
418            let name_str = name.to_string_lossy();
419            if name_str.starts_with("nauka-dns64-") && !name_str.contains(bridge) {
420                let _ = std::fs::remove_file(entry.path());
421            }
422        }
423    }
424
425    let conf = format!(
426        r#"server:
427    interface: {gw_ipv4}
428    interface: {gw_ipv6}
429    access-control: 0.0.0.0/0 allow
430    access-control: ::/0 allow
431    do-ip6: yes
432    module-config: "dns64 iterator"
433    dns64-prefix: 64:ff9b::/96
434    verbosity: 0
435
436forward-zone:
437    name: "."
438    forward-addr: 8.8.8.8
439    forward-addr: 1.1.1.1
440"#,
441    );
442
443    let conf_path = format!("/etc/unbound/unbound.conf.d/nauka-dns64-{}.conf", bridge);
444    std::fs::write(&conf_path, conf)?;
445
446    let _ = run_cmd("systemctl", &["restart", "unbound"]);
447
448    Ok(())
449}

451fn run_cmd(cmd: &str, args: &[&str]) -> anyhow::Result<()> {
452    let output = Command::new(cmd).args(args).output()?;
453    if !output.status.success() {
454        let stderr = String::from_utf8_lossy(&output.stderr);
455        anyhow::bail!("{} {} failed: {}", cmd, args.join(" "), stderr.trim());
456    }
457    Ok(())
458}
459
460fn run_nft(rule: &str) -> anyhow::Result<()> {
461    let output = Command::new("nft").arg(rule).output()?;
462    if !output.status.success() {
463        let stderr = String::from_utf8_lossy(&output.stderr);
464        anyhow::bail!("nft failed: {}", stderr.trim());
465    }
466    Ok(())
467}
