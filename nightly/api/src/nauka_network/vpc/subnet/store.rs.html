1//! Subnet persistence on the SurrealDB-backed cluster store.
2//!
3//! P2.12 (sifrah/nauka#216) migrated this module from the legacy
4//! raw-KV cluster path to the native SurrealDB SDK on top of
5//! [`EmbeddedDb`]. Every method reaches the SDK via
6//! `self.db.client()` and writes/reads against the SCHEMAFULL
7//! `subnet` table defined in `layers/network/schemas/subnet.surql`.
8//!
9//! The legacy `sub` / `sub-idx` / `_reg_v2` sidecar namespaces are
10//! gone: the schema's composite unique index on `(vpc, name)` plus
11//! the secondary uniqueness on `(vpc, cidr)` enforce every invariant
12//! the pre-P2.12 store checked manually.
13

39    /// Create a new subnet within a VPC.
40    ///
41    /// Resolves the owning VPC, validates the CIDR is inside the
42    /// VPC's address space and doesn't overlap any sibling subnet,
43    /// then writes the row via `CREATE` against the SCHEMAFULL
44    /// `subnet` table. Duplicate-name + duplicate-CIDR conflicts are
45    /// rejected by the schema's composite unique indexes.
46    pub async fn create(
47        &self,
48        name: &str,
49        vpc_name_or_id: &str,
50        org_name: Option<&str>,
51        cidr: &str,
52    ) -> anyhow::Result<Subnet> {
53        let vpc_store = crate::vpc::store::VpcStore::new(self.db.clone());
54        let vpc = vpc_store
55            .get(vpc_name_or_id, org_name)
56            .await?
57            .ok_or_else(|| anyhow::anyhow!("vpc '{vpc_name_or_id}' not found"))?;
58
59        // Validate subnet is within VPC.
60        let subnet_net = crate::validate::subnet_within_vpc(cidr, &vpc.cidr)?;
61
62        // Check overlap with existing sibling subnets in the VPC.
63        let existing_subs = self.list(Some(&vpc.meta.id), None).await?;
64        let existing_cidrs: Vec<String> = existing_subs.iter().map(|s| s.cidr.clone()).collect();
65        crate::validate::no_overlap(&subnet_net, &existing_cidrs)?;
66
67        let gateway = crate::validate::gateway(&subnet_net);
68
69        let subnet = Subnet {
70            meta: ResourceMeta::new(SubnetId::generate().to_string(), name),
71            vpc_id: vpc.meta.id.clone().into(),
72            vpc_name: vpc.meta.name.clone(),
73            cidr: cidr.to_string(),
74            gateway,
75        };
76
77        let created_at_iso = epoch_to_iso8601(subnet.meta.created_at);
78        let updated_at_iso = epoch_to_iso8601(subnet.meta.updated_at);
79        let labels_json = serde_json::to_value(&subnet.meta.labels)
80            .map_err(|e| anyhow::anyhow!("serialise labels: {e}"))?;
81
82        let query_result = self
83            .db
84            .client()
85            .query(
86                "CREATE type::record($tbl, $id) SET \
87                 name = $name, \
88                 status = $status, \
89                 labels = $labels, \
90                 cidr = $cidr, \
91                 gateway = $gateway, \
92                 vpc = $vpc, \
93                 vpc_name = $vpc_name, \
94                 created_at = <datetime>$created_at, \
95                 updated_at = <datetime>$updated_at",
96            )
97            .bind(("tbl", SUBNET_TABLE))
98            .bind(("id", subnet.meta.id.clone()))
99            .bind(("name", subnet.meta.name.clone()))
100            .bind(("status", subnet.meta.status.clone()))
101            .bind(("labels", labels_json))
102            .bind(("cidr", subnet.cidr.clone()))
103            .bind(("gateway", subnet.gateway.clone()))
104            .bind(("vpc", subnet.vpc_id.as_str().to_string()))
105            .bind(("vpc_name", subnet.vpc_name.clone()))
106            .bind(("created_at", created_at_iso))
107            .bind(("updated_at", updated_at_iso))
108            .await;
109        let response = match query_result {
110            Ok(r) => r,
111            Err(e) => return Err(classify_subnet_error(name, &vpc.meta.name, &e.to_string())),
112        };
113        if let Err(e) = response.check() {
114            return Err(classify_subnet_error(name, &vpc.meta.name, &e.to_string()));
115        }
116
117        Ok(subnet)
118    }

120    /// Look up a subnet by id (when the input looks like a
121    /// [`SubnetId`]) or by name (otherwise).
122    pub async fn get(
123        &self,
124        name_or_id: &str,
125        vpc_name_or_id: Option<&str>,
126        org_name: Option<&str>,
127    ) -> anyhow::Result<Option<Subnet>> {
128        if SubnetId::looks_like_id(name_or_id) {
129            return self.get_by_id(name_or_id).await;
130        }
131        let vpc_ref = vpc_name_or_id
132            .ok_or_else(|| anyhow::anyhow!("--vpc required to resolve subnet by name"))?;
133        let vpc_store = crate::vpc::store::VpcStore::new(self.db.clone());
134        let vpc = vpc_store
135            .get(vpc_ref, org_name)
136            .await?
137            .ok_or_else(|| anyhow::anyhow!("vpc '{vpc_ref}' not found"))?;
138
139        self.get_by_vpc_and_name(&vpc.meta.id, name_or_id).await
140    }
141
142    async fn get_by_id(&self, id: &str) -> anyhow::Result<Option<Subnet>> {
143        let mut response = self
144            .db
145            .client()
146            .query("SELECT * FROM type::record($tbl, $id)")
147            .bind(("tbl", SUBNET_TABLE))
148            .bind(("id", id.to_string()))
149            .await
150            .map_err(|e| anyhow::anyhow!("{e}"))?;
151        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
152        decode_first(raw)
153    }
154
155    async fn get_by_vpc_and_name(
156        &self,
157        vpc_id: &str,
158        name: &str,
159    ) -> anyhow::Result<Option<Subnet>> {
160        let mut response = self
161            .db
162            .client()
163            .query("SELECT * FROM subnet WHERE vpc = $vpc AND name = $name LIMIT 1")
164            .bind(("vpc", vpc_id.to_string()))
165            .bind(("name", name.to_string()))
166            .await
167            .map_err(|e| anyhow::anyhow!("{e}"))?;
168        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
169        decode_first(raw)
170    }

172    /// List every subnet, optionally filtered by the owning VPC (by
173    /// name or record id).
174    pub async fn list(
175        &self,
176        vpc_name: Option<&str>,
177        _org_name: Option<&str>,
178    ) -> anyhow::Result<Vec<Subnet>> {
179        let mut response = self
180            .db
181            .client()
182            .query("SELECT * FROM subnet")
183            .await
184            .map_err(|e| anyhow::anyhow!("{e}"))?;
185        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
186        let subnets: Vec<Subnet> = raw
187            .into_iter()
188            .filter_map(|v| {
189                serde_json::from_value::<SubnetRow>(v)
190                    .ok()
191                    .map(SubnetRow::into_subnet)
192            })
193            .collect();
194        match vpc_name {
195            Some(name) => Ok(subnets
196                .into_iter()
197                .filter(|s| s.vpc_name == name || s.vpc_id.as_str() == name)
198                .collect()),
199            None => Ok(subnets),
200        }
201    }

203    /// Delete a subnet. `subnet` is effectively the bottom of the
204    /// network hierarchy — VMs reference a subnet via IPAM rows, but
205    /// those rows are cleared with the subnet's `ipam` side table (P2.13
206    /// will migrate the VM store to take ownership of its IP release),
207    /// so `delete` is a single DELETE today.
208    pub async fn delete(
209        &self,
210        name_or_id: &str,
211        vpc_name: &str,
212        org_name: Option<&str>,
213    ) -> anyhow::Result<()> {
214        let subnet = self
215            .get(name_or_id, Some(vpc_name), org_name)
216            .await?
217            .ok_or_else(|| {
218                anyhow::anyhow!("subnet '{name_or_id}' not found in vpc '{vpc_name}'")
219            })?;
220
221        let result = self
222            .db
223            .client()
224            .query("DELETE type::record($tbl, $id)")
225            .bind(("tbl", SUBNET_TABLE))
226            .bind(("id", subnet.meta.id.clone()))
227            .await
228            .map_err(|e| anyhow::anyhow!("{e}"))?;
229        result.check().map_err(|e| anyhow::anyhow!("{e}"))?;
230        Ok(())
231    }
232}

234/// Row shape returned by `SELECT * FROM subnet`.
235#[derive(Debug, Deserialize)]
236struct SubnetRow {
237    id: serde_json::Value,
238    name: String,
239    #[serde(default)]
240    status: Option<String>,
241    #[serde(default)]
242    labels: Option<serde_json::Value>,
243    cidr: String,
244    gateway: String,
245    vpc: String,
246    vpc_name: String,
247    created_at: String,
248    updated_at: String,
249}
250
251impl SubnetRow {
252    fn into_subnet(self) -> Subnet {
253        let id = thing_to_id_string("subnet:", &self.id);
254        Subnet {
255            meta: ResourceMeta {
256                id,
257                name: self.name,
258                status: self.status.unwrap_or_else(|| "active".to_string()),
259                labels: self
260                    .labels
261                    .and_then(|v| serde_json::from_value(v).ok())
262                    .unwrap_or_default(),
263                created_at: iso8601_to_epoch(&self.created_at),
264                updated_at: iso8601_to_epoch(&self.updated_at),
265            },
266            vpc_id: self.vpc.into(),
267            vpc_name: self.vpc_name,
268            cidr: self.cidr,
269            gateway: self.gateway,
270        }
271    }
272}
273
274fn decode_first(raw: Vec<serde_json::Value>) -> anyhow::Result<Option<Subnet>> {
275    match raw.into_iter().next() {
276        None => Ok(None),
277        Some(v) => {
278            let row: SubnetRow = serde_json::from_value(v)
279                .map_err(|e| anyhow::anyhow!("deserialise subnet row: {e}"))?;
280            Ok(Some(row.into_subnet()))
281        }
282    }
283}

285/// Map a SurrealDB create error into a friendly message. Duplicate
286/// `(vpc, name)` pairs get the classic "subnet '<name>' already exists
287/// in vpc '<vpc>'" wording; everything else falls through the shared
288/// [`classify_create_error`] helper.
289fn classify_subnet_error(name: &str, vpc_name: &str, err_msg: &str) -> anyhow::Error {
290    let lowered = err_msg.to_lowercase();
291    if lowered.contains("already contains")
292        || lowered.contains("already exists")
293        || lowered.contains("duplicate")
294    {
295        anyhow::anyhow!("subnet '{name}' already exists in vpc '{vpc_name}'")
296    } else {
297        anyhow::anyhow!(classify_create_error("subnet", name, err_msg))
298    }
299}

334    #[tokio::test]
335    async fn create_then_get_by_name() {
336        let (_d, subs, vpcs, orgs) = temp_store().await;
337        seed_vpc(&vpcs, &orgs).await;
338        let sub = subs
339            .create("public", "web", Some("acme"), "10.0.1.0/24")
340            .await
341            .expect("create subnet");
342        assert_eq!(sub.meta.name, "public");
343        // `SubnetId` uses the `sub` prefix (not `subnet`); the
344        // SurrealDB record ends up at `subnet:sub-01J…`.
345        assert!(sub.meta.id.starts_with("sub-"), "got: {}", sub.meta.id);
346        assert_eq!(sub.cidr, "10.0.1.0/24");
347        assert_eq!(sub.gateway, "10.0.1.1");
348        let got = subs
349            .get("public", Some("web"), Some("acme"))
350            .await
351            .expect("get")
352            .expect("missing");
353        assert_eq!(got.meta.id, sub.meta.id);
354    }
