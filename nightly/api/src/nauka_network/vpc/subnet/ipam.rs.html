1//! IPAM — IP Address Management for subnets.
2//!
3//! Allocates private IPv4 addresses from a subnet's CIDR range. Each
4//! subnet gets one row in the SCHEMALESS `subnet_ipam` table holding
5//! the list of `(ip, vm_id)` allocations as a JSON blob; that blob is
6//! swapped on every allocate/release.
7//!
8//! P2.12 (sifrah/nauka#216) migrated this helper from the legacy
9//! raw-KV cluster surface to the native SurrealDB SDK. The
10//! `subnet_ipam` table is deliberately SCHEMALESS (not in the
11//! `apply_cluster_schemas` bundle) because it's a transitional
12//! single-blob-per-subnet shape — Phase 3 codegen or a dedicated
13//! IPAM rewrite may normalise it into a `subnet_ip` table with one
14//! row per allocated address, at which point this helper goes away.
15//!
16//! Reserved addresses (skipped by the allocator):
17//! - `.0`  = network address
18//! - `.1`  = gateway (assigned to the bridge)
19//! - `.N`  = any IP already present in the allocation list
20
21use nauka_state::EmbeddedDb;
22use serde::{Deserialize, Serialize};
23
24/// Transitional SCHEMALESS table that holds the IPAM state. One row
25/// per subnet, keyed by the subnet's record id, with a `data` column
26/// containing the JSON blob of allocations.
27const IPAM_TABLE: &str = "subnet_ipam";
28
29/// Allocated IPs for a subnet, stored as a single JSON blob per row.
30#[derive(Debug, Clone, Serialize, Deserialize, Default)]
31struct Allocations {
32    ips: Vec<Allocation>,
33}
34
35#[derive(Debug, Clone, Serialize, Deserialize)]
36struct Allocation {
37    ip: String,
38    vm_id: String,
39}
40
41/// Ensure the transitional SCHEMALESS `subnet_ipam` table exists.
42/// Idempotent thanks to `IF NOT EXISTS`.
43async fn ensure_table(db: &EmbeddedDb) -> anyhow::Result<()> {
44    db.client()
45        .query("DEFINE TABLE IF NOT EXISTS subnet_ipam SCHEMALESS")
46        .await
47        .map_err(|e| anyhow::anyhow!("define subnet_ipam: {e}"))?
48        .check()
49        .map_err(|e| anyhow::anyhow!("define subnet_ipam check: {e}"))?;
50    Ok(())
51}
52
53/// Load the allocation blob for one subnet. Returns an empty
54/// [`Allocations`] if no row exists yet.
55async fn load(db: &EmbeddedDb, subnet_id: &str) -> anyhow::Result<Allocations> {
56    ensure_table(db).await?;
57    let mut response = db
58        .client()
59        .query("SELECT data FROM type::record($tbl, $id)")
60        .bind(("tbl", IPAM_TABLE))
61        .bind(("id", subnet_id.to_string()))
62        .await
63        .map_err(|e| anyhow::anyhow!("{e}"))?;
64    // SurrealDB returns one row per record; `.take("data")` pulls
65    // the `data` column into a flat `Vec<serde_json::Value>`.
66    let rows: Vec<serde_json::Value> = response.take("data").map_err(|e| anyhow::anyhow!("{e}"))?;
67    let first = rows.into_iter().next();
68    match first {
69        None | Some(serde_json::Value::Null) => Ok(Allocations::default()),
70        Some(v) => serde_json::from_value::<Allocations>(v)
71            .map_err(|e| anyhow::anyhow!("deserialise ipam blob: {e}")),
72    }
73}
74
75/// Persist the allocation blob for one subnet. Creates the row on
76/// first write via `UPSERT`.
77async fn save(db: &EmbeddedDb, subnet_id: &str, allocs: &Allocations) -> anyhow::Result<()> {
78    ensure_table(db).await?;
79    let data =
80        serde_json::to_value(allocs).map_err(|e| anyhow::anyhow!("serialise ipam blob: {e}"))?;
81    db.client()
82        .query("UPSERT type::record($tbl, $id) CONTENT { data: $data }")
83        .bind(("tbl", IPAM_TABLE))
84        .bind(("id", subnet_id.to_string()))
85        .bind(("data", data))
86        .await
87        .map_err(|e| anyhow::anyhow!("{e}"))?
88        .check()
89        .map_err(|e| anyhow::anyhow!("{e}"))?;
90    Ok(())
91}

93/// Allocate the next available IP from a subnet.
94///
95/// Skips the network address (`.0`), the gateway, and any IPs that
96/// are already allocated. If the VM already has an allocation in
97/// this subnet, returns the existing IP unchanged (idempotent).
98pub async fn allocate(
99    db: &EmbeddedDb,
100    subnet_id: &str,
101    subnet_cidr: &str,
102    gateway: &str,
103    vm_id: &str,
104) -> anyhow::Result<String> {
105    let net: ipnet::Ipv4Net = subnet_cidr
106        .parse()
107        .map_err(|e| anyhow::anyhow!("invalid subnet CIDR: {e}"))?;
108
109    let mut allocs = load(db, subnet_id).await?;
110
111    // Check if this VM already has an IP in this subnet.
112    if let Some(existing) = allocs.ips.iter().find(|a| a.vm_id == vm_id) {
113        return Ok(existing.ip.clone());
114    }
115
116    let allocated_ips: Vec<&str> = allocs.ips.iter().map(|a| a.ip.as_str()).collect();
117
118    for host_ip in net.hosts() {
119        let ip_str = host_ip.to_string();
120        if ip_str == gateway {
121            continue;
122        }
123        if allocated_ips.contains(&ip_str.as_str()) {
124            continue;
125        }
126        allocs.ips.push(Allocation {
127            ip: ip_str.clone(),
128            vm_id: vm_id.to_string(),
129        });
130        save(db, subnet_id, &allocs).await?;
131        return Ok(ip_str);
132    }
133
134    anyhow::bail!(
135        "no IPs available in subnet {} (all {} addresses allocated)",
136        subnet_cidr,
137        allocs.ips.len()
138    )
139}
140
141/// Release the IP allocation for a VM.
142pub async fn release(db: &EmbeddedDb, subnet_id: &str, vm_id: &str) -> anyhow::Result<()> {
143    let mut allocs = load(db, subnet_id).await?;
144    allocs.ips.retain(|a| a.vm_id != vm_id);
145    save(db, subnet_id, &allocs).await
146}
147
148/// Get the allocated IP for a VM in a subnet, or `None` if no
149/// allocation exists.
150pub async fn get_allocation(
151    db: &EmbeddedDb,
152    subnet_id: &str,
153    vm_id: &str,
154) -> anyhow::Result<Option<String>> {
155    let allocs = load(db, subnet_id).await?;
156    Ok(allocs
157        .ips
158        .iter()
159        .find(|a| a.vm_id == vm_id)
160        .map(|a| a.ip.clone()))
161}

167    async fn temp_db() -> (tempfile::TempDir, EmbeddedDb) {
168        let dir = tempfile::tempdir().expect("tempdir");
169        let db = EmbeddedDb::open(&dir.path().join("ipam.skv"))
170            .await
171            .expect("open EmbeddedDb");
172        (dir, db)
173    }

175    #[test]
176    fn allocations_serde_roundtrip() {
177        let allocs = Allocations {
178            ips: vec![
179                Allocation {
180                    ip: "10.0.1.2".to_string(),
181                    vm_id: "vm-01".to_string(),
182                },
183                Allocation {
184                    ip: "10.0.1.3".to_string(),
185                    vm_id: "vm-02".to_string(),
186                },
187            ],
188        };
189        let json = serde_json::to_string(&allocs).unwrap();
190        let back: Allocations = serde_json::from_str(&json).unwrap();
191        assert_eq!(back.ips.len(), 2);
192        assert_eq!(back.ips[0].ip, "10.0.1.2");
193    }

195    #[tokio::test]
196    async fn allocate_first_ip_skips_gateway() {
197        let (_d, db) = temp_db().await;
198        let ip = allocate(&db, "subnet-test", "10.0.1.0/24", "10.0.1.1", "vm-01")
199            .await
200            .unwrap();
201        assert_eq!(ip, "10.0.1.2");
202    }

204    #[tokio::test]

228    #[tokio::test]
229    async fn release_frees_ip_for_reuse() {
230        let (_d, db) = temp_db().await;
231        allocate(&db, "subnet-test", "10.0.1.0/24", "10.0.1.1", "vm-01")
232            .await
233            .unwrap();
234        release(&db, "subnet-test", "vm-01").await.unwrap();
235        let again = allocate(&db, "subnet-test", "10.0.1.0/24", "10.0.1.1", "vm-02")
236            .await
237            .unwrap();
238        // vm-02 should be able to grab the first available slot
239        // (10.0.1.2), not a later one, because vm-01's release
240        // cleared the slot.
241        assert_eq!(again, "10.0.1.2");
242    }
