1//! VPC persistence on the SurrealDB-backed cluster store.
2//!
3//! P2.12 (sifrah/nauka#216) migrated this module from the legacy
4//! raw-KV cluster path to the native SurrealDB SDK on top of
5//! [`EmbeddedDb`], following the same pattern P2.9/P2.10/P2.11
6//! established for the org-layer stores. Every method now reaches the
7//! SDK via `self.db.client()` and writes/reads against the SCHEMAFULL
8//! `vpc` table defined in `layers/network/schemas/vpc.surql` (applied
9//! to the cluster by `nauka_state::apply_cluster_schemas` at
10//! bootstrap per ADR 0004).
11//!
12//! The store holds an [`EmbeddedDb`] directly. Call sites pass
13//! `db.clone()` at construction time; the clone is
14//! cheap (`Arc`-shared `Surreal<Db>` router).
15//!
16//! The legacy `vpc` / `vpc-idx` / `_reg_v2` sidecar namespaces are
17//! gone: the schema's composite unique index on `(org, name)` is now
18//! the source of truth for "have we seen this (org, name) pair before?",
19//! and `SELECT * FROM vpc` walks every row directly.
20//!
21//! The VNI counter, which the pre-P2.12 version kept in a raw-KV
22//! `_reg:vni-counter` key, is now computed as `max(vni) + 1` across
23//! the `vpc` table, falling back to [`VNI_START`] (100) on an empty
24//! table. Concurrent creators still race, but the SCHEMAFULL `vpc_vni`
25//! unique index guarantees at most one of the racing `CREATE`s wins
26//! and the other gets a clean "already exists" rollback — the store
27//! retries the allocation in a loop to stay obviously-correct under
28//! concurrency.
29

39/// SurrealDB table backing this store. Defined by
40/// `layers/network/schemas/vpc.surql` as `DEFINE TABLE vpc SCHEMAFULL`
41/// and applied at bootstrap via `nauka_state::apply_cluster_schemas`.
42const VPC_TABLE: &str = "vpc";
43
44/// First VNI value handed out on a fresh cluster. VNIs below 100 are
45/// reserved for future system use.
46const VNI_START: u32 = 100;
47
48/// Maximum number of VNI allocation retries before giving up. A real
49/// failure mode is extreme concurrency on `create`, which is not a
50/// production path today (VPCs are created via operator CLI). 8
51/// retries is plenty of headroom without risking a runaway loop.
52const VNI_RETRY_LIMIT: u32 = 8;
53

67    /// Pick the next VNI to allocate.
68    ///
69    /// Reads `math::max(vni)` from the `vpc` table and returns
70    /// `max + 1`, or [`VNI_START`] on an empty table. The SCHEMAFULL
71    /// `vpc_vni` unique index closes the concurrent-allocation race:
72    /// if two creators see the same `max`, one of their `CREATE`s
73    /// will fail the unique-index check, and the caller retries.
74    async fn next_vni(&self) -> anyhow::Result<u32> {
75        let mut response = self
76            .db
77            .client()
78            .query("SELECT math::max(vni) AS max_vni FROM vpc GROUP ALL")
79            .await
80            .map_err(|e| anyhow::anyhow!("{e}"))?;
81        let rows: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
82        let max_vni = rows
83            .into_iter()
84            .next()
85            .and_then(|row| row.get("max_vni").cloned())
86            .and_then(|v| v.as_u64())
87            .map(|v| v as u32);
88        Ok(max_vni.map(|v| v + 1).unwrap_or(VNI_START))
89    }

91    /// Create a new VPC within an org.
92    ///
93    /// The duplicate-name path is closed by the SCHEMAFULL
94    /// `vpc_org_name` composite unique index. The duplicate-VNI path
95    /// is closed by the `vpc_vni` unique index; on a rare race (two
96    /// creators observing the same `max(vni)`) the loser retries the
97    /// whole allocation. [`VNI_RETRY_LIMIT`] caps the retry count.
98    pub async fn create(
99        &self,
100        name: &str,
101        org_name: &str,
102        cidr: &str,
103        project_id: Option<String>,
104        env_id: Option<String>,
105    ) -> anyhow::Result<Vpc> {
106        // Resolve the owning org via the post-P2.9 OrgStore API.
107        let org_store = nauka_org::store::OrgStore::new(self.db.clone());
108        let org = org_store
109            .get(org_name)
110            .await?
111            .ok_or_else(|| anyhow::anyhow!("org '{org_name}' not found"))?;
112
113        // Validate CIDR is in the private-address space.
114        let new_net = crate::validate::private_cidr(cidr)?;
115
116        // Check CIDR doesn't overlap with existing VPCs in this org.
117        let existing_vpcs = self.list(Some(org_name)).await?;
118        let existing_cidrs: Vec<String> = existing_vpcs.iter().map(|v| v.cidr.clone()).collect();
119        crate::validate::no_overlap(&new_net, &existing_cidrs)?;
120
121        // Retry loop: on a VNI collision, re-read `max(vni)` and try
122        // again. Any other error (including the composite `(org,
123        // name)` duplicate) breaks out immediately.
124        let mut attempt = 0u32;
125        loop {
126            let vni = self.next_vni().await?;
127            let vpc = Vpc {
128                meta: ResourceMeta::new(VpcId::generate().to_string(), name),
129                cidr: cidr.to_string(),
130                org_id: org.meta.id.clone().into(),
131                org_name: org.meta.name.clone(),
132                vni,
133                project_id: project_id.clone().map(|s| s.into()),
134                env_id: env_id.clone().map(|s| s.into()),
135            };
136
137            match self.insert(&vpc).await {
138                Ok(()) => return Ok(vpc),
139                Err(e) => {
140                    let msg = e.to_string().to_lowercase();
141                    let is_vni_conflict = msg.contains("vpc_vni");
142                    let is_org_name_conflict = msg.contains("vpc_org_name");
143                    let is_name_conflict = is_org_name_conflict
144                        || (msg.contains("already exists") && !is_vni_conflict);
145
146                    if is_name_conflict {
147                        return Err(anyhow::anyhow!(
148                            "vpc '{name}' already exists in org '{org_name}'"
149                        ));
150                    }
151
152                    if is_vni_conflict && attempt + 1 < VNI_RETRY_LIMIT {
153                        attempt += 1;
154                        continue;
155                    }
156
157                    return Err(anyhow::anyhow!(classify_create_error("vpc", name, &msg)));
158                }
159            }
160        }
161    }

163    /// Write the row. Returns the raw SurrealDB error on failure so
164    /// the caller can classify it.
165    async fn insert(&self, vpc: &Vpc) -> anyhow::Result<()> {
166        let created_at_iso = epoch_to_iso8601(vpc.meta.created_at);
167        let updated_at_iso = epoch_to_iso8601(vpc.meta.updated_at);
168        let labels_json = serde_json::to_value(&vpc.meta.labels)
169            .map_err(|e| anyhow::anyhow!("serialise labels: {e}"))?;
170
171        let response = self
172            .db
173            .client()
174            .query(
175                "CREATE type::record($tbl, $id) SET \
176                 name = $name, \
177                 status = $status, \
178                 labels = $labels, \
179                 cidr = $cidr, \
180                 vni = $vni, \
181                 org = $org, \
182                 org_name = $org_name, \
183                 project = $project, \
184                 env = $env, \
185                 created_at = <datetime>$created_at, \
186                 updated_at = <datetime>$updated_at",
187            )
188            .bind(("tbl", VPC_TABLE))
189            .bind(("id", vpc.meta.id.clone()))
190            .bind(("name", vpc.meta.name.clone()))
191            .bind(("status", vpc.meta.status.clone()))
192            .bind(("labels", labels_json))
193            .bind(("cidr", vpc.cidr.clone()))
194            .bind(("vni", vpc.vni as i64))
195            .bind(("org", vpc.org_id.as_str().to_string()))
196            .bind(("org_name", vpc.org_name.clone()))
197            .bind((
198                "project",
199                vpc.project_id.as_ref().map(|p| p.as_str().to_string()),
200            ))
201            .bind(("env", vpc.env_id.as_ref().map(|e| e.as_str().to_string())))
202            .bind(("created_at", created_at_iso))
203            .bind(("updated_at", updated_at_iso))
204            .await
205            .map_err(|e| anyhow::anyhow!("{e}"))?;
206        response.check().map_err(|e| anyhow::anyhow!("{e}"))?;
207        Ok(())
208    }

210    /// Look up a VPC by id (when the input looks like a [`VpcId`])
211    /// or by name (otherwise).
212    pub async fn get(
213        &self,
214        name_or_id: &str,
215        org_name: Option<&str>,
216    ) -> anyhow::Result<Option<Vpc>> {
217        if VpcId::looks_like_id(name_or_id) {
218            return self.get_by_id(name_or_id).await;
219        }
220        let org_name =
221            org_name.ok_or_else(|| anyhow::anyhow!("--org required to resolve VPC by name"))?;
222
223        let org_store = nauka_org::store::OrgStore::new(self.db.clone());
224        let org = org_store
225            .get(org_name)
226            .await?
227            .ok_or_else(|| anyhow::anyhow!("org '{org_name}' not found"))?;
228
229        self.get_by_org_and_name(&org.meta.id, name_or_id).await
230    }

232    async fn get_by_id(&self, id: &str) -> anyhow::Result<Option<Vpc>> {
233        let mut response = self
234            .db
235            .client()
236            .query("SELECT * FROM type::record($tbl, $id)")
237            .bind(("tbl", VPC_TABLE))
238            .bind(("id", id.to_string()))
239            .await
240            .map_err(|e| anyhow::anyhow!("{e}"))?;
241        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
242        decode_first(raw)
243    }

258    /// List every VPC, optionally filtered by the owning org.
259    pub async fn list(&self, org_name: Option<&str>) -> anyhow::Result<Vec<Vpc>> {
260        let mut response = self
261            .db
262            .client()
263            .query("SELECT * FROM vpc")
264            .await
265            .map_err(|e| anyhow::anyhow!("{e}"))?;
266        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
267        let vpcs: Vec<Vpc> = raw
268            .into_iter()
269            .filter_map(|v| {
270                serde_json::from_value::<VpcRow>(v)
271                    .ok()
272                    .map(VpcRow::into_vpc)
273            })
274            .collect();
275        match org_name {
276            Some(name) => Ok(vpcs
277                .into_iter()
278                .filter(|v| v.org_name == name || v.org_id.as_str() == name)
279                .collect()),
280            None => Ok(vpcs),
281        }
282    }

284    /// Delete a VPC. Refuses to delete if the VPC still has subnets
285    /// or a NAT gateway — the caller must remove the children first.
286    pub async fn delete(&self, name_or_id: &str, org_name: &str) -> anyhow::Result<()> {
287        let vpc = self
288            .get(name_or_id, Some(org_name))
289            .await?
290            .ok_or_else(|| anyhow::anyhow!("vpc '{name_or_id}' not found in org '{org_name}'"))?;
291
292        // Check for child subnets directly on the SCHEMAFULL `subnet`
293        // table — don't build a SubnetStore here, that would force a
294        // circular import dance when tests link it in. Same pattern
295        // OrgStore::delete uses to count projects.
296        let subnet_count = self.count_subnets_in_vpc(&vpc.meta.id).await?;
297        if subnet_count > 0 {
298            anyhow::bail!(
299                "vpc '{}' has {} subnet(s). Delete them first.",
300                vpc.meta.name,
301                subnet_count
302            );
303        }
304
305        // Check for the NAT gateway (at most one per VPC).
306        let natgw_count = self.count_natgws_in_vpc(&vpc.meta.id).await?;
307        if natgw_count > 0 {
308            anyhow::bail!(
309                "vpc '{}' has a NAT gateway. Delete it first.",
310                vpc.meta.name
311            );
312        }
313
314        // Check for peerings that still reference this VPC (on either
315        // side of the pair — peerings are bidirectional, and a live
316        // peering to a VPC that's about to disappear would leave a
317        // dangling row pointing at a dead record id).
318        let peering_count = self.count_peerings_touching_vpc(&vpc.meta.id).await?;
319        if peering_count > 0 {
320            anyhow::bail!(
321                "vpc '{}' has {} peering(s). Delete them first.",
322                vpc.meta.name,
323                peering_count
324            );
325        }
326
327        let result = self
328            .db
329            .client()
330            .query("DELETE type::record($tbl, $id)")
331            .bind(("tbl", VPC_TABLE))
332            .bind(("id", vpc.meta.id.clone()))
333            .await
334            .map_err(|e| anyhow::anyhow!("{e}"))?;
335        result.check().map_err(|e| anyhow::anyhow!("{e}"))?;
336        Ok(())
337    }

379/// Row shape returned by `SELECT * FROM vpc`.
380#[derive(Debug, Deserialize)]
381struct VpcRow {
382    id: serde_json::Value,
383    name: String,
384    #[serde(default)]
385    status: Option<String>,
386    #[serde(default)]
387    labels: Option<serde_json::Value>,
388    cidr: String,
389    vni: i64,
390    org: String,
391    org_name: String,
392    #[serde(default)]
393    project: Option<String>,
394    #[serde(default)]
395    env: Option<String>,
396    created_at: String,
397    updated_at: String,
398}
399
400impl VpcRow {
401    fn into_vpc(self) -> Vpc {
402        let id = thing_to_id_string("vpc:", &self.id);
403        let labels = self
404            .labels
405            .and_then(|v| serde_json::from_value(v).ok())
406            .unwrap_or_default();
407        Vpc {
408            meta: ResourceMeta {
409                id,
410                name: self.name,
411                status: self.status.unwrap_or_else(|| "active".to_string()),
412                labels,
413                created_at: iso8601_to_epoch(&self.created_at),
414                updated_at: iso8601_to_epoch(&self.updated_at),
415            },
416            cidr: self.cidr,
417            org_id: self.org.into(),
418            org_name: self.org_name,
419            vni: self.vni as u32,
420            project_id: self.project.map(Into::into),
421            env_id: self.env.map(Into::into),
422        }
423    }
424}
425
426fn decode_first(raw: Vec<serde_json::Value>) -> anyhow::Result<Option<Vpc>> {
427    match raw.into_iter().next() {
428        None => Ok(None),
429        Some(v) => {
430            let row: VpcRow = serde_json::from_value(v)
431                .map_err(|e| anyhow::anyhow!("deserialise vpc row: {e}"))?;
432            Ok(Some(row.into_vpc()))
433        }
434    }
435}
