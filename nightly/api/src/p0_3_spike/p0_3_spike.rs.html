1//! Spike binary for P0.3 (#187).
2//!
3//! Goal: prove that the SurrealDB SDK with `kv-tikv` (using
4//! `surrealdb-tikv-client v0.3.0-surreal.4`) can connect to the existing
5//! Nauka PD/TiKV v8.5.5 cluster and successfully run a CRUD round-trip via
6//! SurrealQL.
7//!
8//! Usage:
9//!     p0-3-spike <pd_endpoint>
10//!
11//! Where <pd_endpoint> is `host:port` for one PD member, e.g.
12//! `[fdc5:8ba:9b14:6e94:9cbb:1220:ed:af72]:2379`. The TiKv engine takes a
13//! single PD address and discovers the rest of the cluster on its own.
14//!
15//! What this does:
16//!   1. Open a `Surreal<Db>` client with the `TiKv` engine pointing at the
17//!      given PD endpoint.
18//!   2. `use_ns("p0_3_spike").use_db("test")` so we don't pollute any real
19//!      Nauka namespace.
20//!   3. Run a CRUD round-trip: create → select-by-id → select-all → delete.
21//!   4. Measure and print latencies for each step.
22//!   5. Exit 0 on success, non-zero on failure.
23
24use std::time::Instant;
25
26use surrealdb::engine::local::TiKv;
27use surrealdb::types::SurrealValue;
28use surrealdb::Surreal;
29
30#[derive(Debug, SurrealValue)]
31struct SpikeRecord {
32    name: String,
33    answer: i32,
34}
35
36#[tokio::main]
37async fn main() -> Result<(), Box<dyn std::error::Error>> {
38    let pd = std::env::args()
39        .nth(1)
40        .ok_or_else(|| "usage: p0-3-spike <pd_endpoint>  (e.g. [fdc5:...]:2379)".to_string())?;
41
42    println!("== nauka p0-3 spike (kv-tikv compat check) ==");
43    println!("target_arch    = {}", std::env::consts::ARCH);
44    println!("target_os      = {}", std::env::consts::OS);
45    println!("surrealdb_dep  = 3.0.5 (kv-tikv via surrealdb-tikv-client v0.3.0-surreal.4)");
46    println!("pd_endpoint    = {pd}");
47
48    // Step 1: connect.
49    let t0 = Instant::now();
50    let db = Surreal::new::<TiKv>(pd.as_str()).await?;
51    println!("connect_ms     = {}", t0.elapsed().as_millis());
52
53    let t1 = Instant::now();
54    db.use_ns("p0_3_spike").use_db("test").await?;
55    println!("use_ns_db_ms   = {}", t1.elapsed().as_millis());
56
57    // Step 2: create a record with a known id so we can clean up.
58    let t2 = Instant::now();
59    let created: Option<SpikeRecord> = db
60        .create(("p0_3_record", "first"))
61        .content(SpikeRecord {
62            name: "p0-3".into(),
63            answer: 42,
64        })
65        .await?;
66    println!("create_ms      = {}", t2.elapsed().as_millis());
67    println!("create_value   = {:?}", created);
68
69    // Step 3: read it back by id.
70    let t3 = Instant::now();
71    let fetched: Option<SpikeRecord> = db.select(("p0_3_record", "first")).await?;
72    println!("select_id_ms   = {}", t3.elapsed().as_millis());
73    println!("select_id_val  = {:?}", fetched);
74    if fetched.is_none() {
75        return Err("select-by-id returned None — round-trip failed".into());
76    }
77
78    // Step 4: select all (should contain at least 1).
79    let t4 = Instant::now();
80    let all: Vec<SpikeRecord> = db.select("p0_3_record").await?;
81    println!("select_all_ms  = {}", t4.elapsed().as_millis());
82    println!("select_all_n   = {}", all.len());
83
84    // Step 5: cleanup the test record so re-runs are clean.
85    let t5 = Instant::now();
86    let _deleted: Option<SpikeRecord> = db.delete(("p0_3_record", "first")).await?;
87    println!("delete_ms      = {}", t5.elapsed().as_millis());
88
89    println!("total_ms       = {}", t0.elapsed().as_millis());
90    println!("== p0-3 spike OK ==");
91    Ok(())
92}
