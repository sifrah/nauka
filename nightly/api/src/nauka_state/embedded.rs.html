1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, and
7//! an explicit `shutdown` step.
8//!
9//! This is the production backend for Nauka's bootstrap state. The legacy
10//! JSON-file store shipped alongside it during P1.2–P1.10; P1.11
11//! (sifrah/nauka#201) migrated every caller to `EmbeddedDb`, and P1.12
12//! (sifrah/nauka#202) will delete the legacy code outright.
13//!
14//! P1.2 (sifrah/nauka#192) introduces only the wrapper struct and its
15//! lifecycle. The companion deliverables — error mapping (P1.3,
16//! sifrah/nauka#193), default-path helpers (P1.4, sifrah/nauka#194),
17//! comprehensive CRUD/persistence tests (P1.5, sifrah/nauka#195), the
18//! initial `.surql` schema (P1.6, sifrah/nauka#196), and applying that
19//! schema at open time (P1.7, sifrah/nauka#197) — each ship as their own
20//! issue.
21//!
22//! # Example
23//!
24//! ```no_run
25//! # use nauka_state::EmbeddedDb;
26//! # use std::path::Path;
27//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
28//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
29//!
30//! // Use the underlying SurrealDB SDK directly:
31//! let _: Vec<surrealdb::types::Value> = db
32//!     .client()
33//!     .query("INFO FOR DB")
34//!     .await?
35//!     .take(0)?;
36//!
37//! db.shutdown().await?;
38//! # Ok(()) }
39//! ```

50/// The bootstrap-state SurrealQL schema, embedded at compile time.
51///
52/// Defined in `layers/state/schemas/bootstrap.surql` (P1.6, sifrah/nauka#196).
53/// `EmbeddedDb::open` applies this on every open. The schema is fully
54/// idempotent thanks to `IF NOT EXISTS` on every `DEFINE` statement, so
55/// re-applying against an already-initialised database is a no-op.
56const BOOTSTRAP_SCHEMA: &str = include_str!("../schemas/bootstrap.surql");

58/// Embedded SurrealDB instance, persisted to disk via the SurrealKV backend.
59///
60/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
61/// connection (which itself uses an `Arc` internally).
62#[derive(Clone)]
63pub struct EmbeddedDb {
64    inner: Surreal<Db>,
65    path: PathBuf,
66}

68impl EmbeddedDb {
69    /// Open (or create) the SurrealKV-backed database at the run-mode-aware
70    /// default path provided by [`nauka_core::process::nauka_db_path`]:
71    ///
72    /// - CLI mode: `~/.nauka/bootstrap.skv`
73    /// - Service mode (root): `/var/lib/nauka/bootstrap.skv`
74    ///
75    /// The parent state directory is created via
76    /// [`nauka_core::process::ensure_nauka_state_dir`] (with 0o700 perms),
77    /// then the rest of the open path is the same as [`Self::open`].
78    pub async fn open_default() -> Result<Self> {
79        // Create the state directory with the right perms; the SurrealKV
80        // datastore directory itself is created by `Surreal::new::<SurrealKv>`.
81        let _ = nauka_core::process::ensure_nauka_state_dir()
82            .map_err(|e| StateError::Database(format!("ensure state dir: {e}")))?;
83        let path = nauka_core::process::nauka_db_path();
84        Self::open(&path).await
85    }

87    /// Open (or create) the SurrealKV-backed database at `path`, select the
88    /// `nauka` / `bootstrap` namespace/database, and apply the embedded
89    /// bootstrap schema.
90    ///
91    /// The parent directory is created if it doesn't exist. On Unix the
92    /// parent directory is also chmod-ed to 0o700 (best-effort).
93    ///
94    /// The bootstrap schema (`schemas/bootstrap.surql`, P1.6) is applied on
95    /// every open. It is fully idempotent thanks to `IF NOT EXISTS` on
96    /// every `DEFINE` statement, so reopening an existing database is a
97    /// no-op for already-defined tables, fields, and indexes — and any
98    /// new tables/fields/indexes that have appeared since last open are
99    /// added forward-compatibly. P1.7 (sifrah/nauka#197) is the issue that
100    /// wires this in.
101    ///
102    /// For the run-mode-aware default path, use [`Self::open_default`].
103    ///
104    /// # Errors
105    ///
106    /// Returns [`StateError::Io`] if the parent directory cannot be created,
107    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
108    /// [`StateError::Database`] (depending on the underlying SurrealDB
109    /// failure mode) if SurrealDB cannot open the datastore, switch to the
110    /// configured namespace/database, or apply the bootstrap schema. The
111    /// classification is done by the `From<surrealdb::Error>` impl in
112    /// `lib.rs` (P1.3, sifrah/nauka#193).
113    pub async fn open(path: &Path) -> Result<Self> {
114        if let Some(parent) = path.parent() {
115            if !parent.as_os_str().is_empty() {
116                std::fs::create_dir_all(parent)?;
117                #[cfg(unix)]
118                {
119                    use std::os::unix::fs::PermissionsExt;
120                    // Best-effort 0o700 on the parent (P1.4 perms target).
121                    // We don't surface a chmod failure as a hard error: the
122                    // dir was created successfully, refusing to start over a
123                    // chmod refusal would be obnoxious. Logged via tracing
124                    // by nauka_core::process when the helper is used.
125                    let _ =
126                        std::fs::set_permissions(parent, std::fs::Permissions::from_mode(0o700));
127                }
128            }
129        }
130
131        // Acquire the SurrealKV datastore with retry-on-lock-contention.
132        //
133        // # Why this exists
134        //
135        // SurrealKV takes an OS-level exclusive flock on `<path>/LOCK`
136        // whenever a `Datastore` is open. Only one process can hold the
137        // flock at a time. In Nauka, that matters because the fabric
138        // layer has concurrent consumers of the same `bootstrap.skv`:
139        //
140        // - Ad-hoc CLI invocations (`nauka hypervisor status`, etc.)
141        // - The `nauka-forge` daemon running reconcile cycles
142        // - The `nauka-announce` listener
143        //
144        // Each of them opens the DB for a short span (read state, do
145        // work, `shutdown().await`) but their windows overlap, so an
146        // open that happens to collide with another process's brief
147        // write window would fail with
148        // `Other("Database at ... LOCK is already locked by another process")`.
149        //
150        // Before P1.11 (sifrah/nauka#201), this was masked because every
151        // CLI caller used the legacy JSON-file bootstrap backend, which
152        // did not take an exclusive flock. P1.11 migrated every caller
153        // to `EmbeddedDb`, so the contention became real.
154        //
155        // The fix: retry the open on "already locked" with exponential
156        // backoff up to a 5-second deadline, matching the pattern used
157        // by `shutdown` in `wait_for_surrealkv_lock_release`. 5 s is the
158        // same "fast timeout" budget used for health checks — long
159        // enough to ride out a forge reconcile cycle that briefly holds
160        // the lock, short enough that a stuck holder surfaces as a
161        // clear error instead of hanging forever.
162        let path_str = path.to_string_lossy().into_owned();
163        let inner = open_datastore_with_retry(&path_str).await?;
164
165        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
166
167        // Apply the bootstrap schema after switching to the right ns/db.
168        // DEFINE TABLE / FIELD / INDEX are db-scoped, so use_db must
169        // happen first. The schema's IF NOT EXISTS clauses make this
170        // safe to run on every open.
171        inner.query(BOOTSTRAP_SCHEMA).await?.check()?;
172
173        Ok(Self {
174            inner,
175            path: path.to_path_buf(),
176        })
177    }
178
179    /// Borrow the underlying SurrealDB SDK client.
180    ///
181    /// All SurrealQL queries flow through this. The wrapper does not
182    /// re-export every SDK method — call sites that need the full SDK

184    pub fn client(&self) -> &Surreal<Db> {
185        &self.inner
186    }

189    pub fn path(&self) -> &Path {
190        &self.path
191    }

193    /// Shut down the wrapper, dropping the SDK client and waiting for the
194    /// SurrealDB background router task to finish closing the datastore
195    /// so a subsequent `EmbeddedDb::open` at the same path can succeed
196    /// — and, critically, so any writes committed through this handle
197    /// are durably on disk before we return.
198    ///
199    /// # Why this function exists
200    ///
201    /// The SDK's `Surreal<Db>` router runs on a background tokio task
202    /// that owns an `Arc<Datastore>`. `drop(Surreal<Db>)` is
203    /// fire-and-forget: it closes the route channel, the router task
204    /// sees `route_rx.recv()` return `Err`, breaks out of its loop,
205    /// and **then** runs `Datastore::shutdown().await` which:
206    ///
207    /// 1. Shuts down the commit coordinator and WAL background flusher
208    /// 2. Calls `Tree::close()` which flushes all memtables to SSTables
209    /// 3. Closes the WAL
210    /// 4. Drops the `LockFile`, releasing SurrealKV's OS-level exclusive
211    ///    flock on `<path>/LOCK`
212    ///
213    /// None of this is synchronous with `drop(self.inner)`. A caller that
214    /// does `shutdown().await; open(same_path).await` races the
215    /// background task. Previous versions of this method used
216    /// `tokio::time::sleep(50 ms)` as a heuristic — enough on CI Linux,
217    /// **not** enough on Mac arm64 under contention, where it manifested
218    /// as data loss (`load_async` returning `None` after a round-trip
219    /// reopen, reproducible at 77% with `--test-threads=2`). See
220    /// sifrah/nauka#255 for the investigation.
221    ///
222    /// The deterministic signal that step 4 has happened is the release
223    /// of the OS-level flock on `<path>/LOCK`. We poll for that release
224    /// by trying to acquire the same flock ourselves via
225    /// [`std::fs::File::try_lock`]. When `try_lock` succeeds, the
226    /// previous `Datastore::shutdown()` chain has definitively run to
227    /// completion, which implies every prior commit is durable and the
228    /// next open can proceed without a lock-held error.
229    ///
230    /// # Cost
231    ///
232    /// Fast path: a single `yield_now` + `try_lock` (~sub-ms). Slow path:
233    /// exponential backoff from 1 ms to a 50 ms ceiling, up to a hard
234    /// 5-second deadline after which the call returns
235    /// [`StateError::Database`]. 5 s is generous relative to the
236    /// ~10–100 ms that a real `Datastore::shutdown()` takes even on a
237    /// contended runtime, and still well under the "fast timeout" Nauka
238    /// convention for health checks.
239    pub async fn shutdown(self) -> Result<()> {
240        // Explicit drop of the inner client. The compiler would do this
241        // anyway when `self` goes out of scope, but naming the step
242        // makes the handoff to the router task visible.
243        let path = self.path.clone();
244        drop(self.inner);
245
246        wait_for_surrealkv_lock_release(&path).await
247    }
248}

250/// Open the SurrealKV datastore at `path_str`, retrying on flock
251/// contention until a 5-second deadline elapses.
252///
253/// See the rationale block in [`EmbeddedDb::open`] for the full context.
254/// The short version: SurrealKV holds a process-exclusive flock while a
255/// datastore is open, and Nauka's CLI / forge daemon / announce listener
256/// all touch the same `bootstrap.skv`, so brief overlaps are normal. We
257/// retry the open rather than surface a hard error for every such
258/// overlap.
259async fn open_datastore_with_retry(path_str: &str) -> Result<Surreal<Db>> {
260    /// Upper bound on the total wait. Same "fast timeout" budget as
261    /// `shutdown` (see `wait_for_surrealkv_lock_release`). Generous
262    /// relative to the real fast-path cost of a forge reconcile cycle
263    /// (~10–100 ms of flock-held time) but short enough that a truly
264    /// stuck holder surfaces as a clear error instead of a hang.
265    const MAX_WAIT: Duration = Duration::from_secs(5);
266    /// First retry after an initial miss.
267    const MIN_BACKOFF: Duration = Duration::from_millis(5);
268    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
269    const MAX_BACKOFF: Duration = Duration::from_millis(50);
270
271    let deadline = Instant::now() + MAX_WAIT;
272    let mut backoff = MIN_BACKOFF;
273
274    loop {
275        match Surreal::new::<SurrealKv>(path_str).await {
276            Ok(db) => return Ok(db),
277            Err(err) => {
278                // SurrealKV surfaces flock contention as an "other"
279                // backend error with a message containing the literal
280                // path to `<datastore>/LOCK`. There's no structured
281                // variant we can match on, so fall back to a substring
282                // check. If the message shape changes in a future
283                // surrealkv release the test
284                // `open_retries_on_flock_contention` will flag it.
285                let msg = err.to_string();
286                let is_lock_contention = msg.contains("already locked");
287
288                if !is_lock_contention || Instant::now() >= deadline {
289                    return Err(err.into());
290                }
291
292                tokio::time::sleep(backoff).await;
293                backoff = (backoff * 2).min(MAX_BACKOFF);
294            }
295        }
296    }
297}

299/// Poll until SurrealKV's OS-level exclusive flock on `<path>/LOCK` is
300/// released by the previous `Datastore`, or until the 5-second deadline
301/// elapses.
302///
303/// See [`EmbeddedDb::shutdown`] for the full rationale. The short version:
304/// the SurrealDB SDK's `Surreal<Db>` router runs its `kvs.shutdown()`
305/// chain asynchronously after `drop(Surreal<Db>)`, and the release of
306/// the `LockFile` (surrealkv 0.21 / `src/lockfile.rs`) is the last step
307/// of that chain. Successfully acquiring the same flock from here proves
308/// the chain has run to completion.
309async fn wait_for_surrealkv_lock_release(path: &Path) -> Result<()> {
310    /// Upper bound on the total wait. Generous relative to the actual
311    /// `Datastore::shutdown()` cost (10–100 ms typical, even on contended
312    /// runtimes) but well under the "fast timeout" Nauka convention.
313    const MAX_WAIT: Duration = Duration::from_secs(5);
314    /// First retry after the fast-path `try_lock` miss.
315    const MIN_BACKOFF: Duration = Duration::from_millis(1);
316    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
317    const MAX_BACKOFF: Duration = Duration::from_millis(50);
318
319    let lock_path = path.join("LOCK");
320    let deadline = Instant::now() + MAX_WAIT;
321    let mut backoff = MIN_BACKOFF;
322
323    loop {
324        // Give any already-runnable task (the router exit handler) a
325        // chance to make progress before we poll. Cheap in the common
326        // case where the router task is already parked.
327        tokio::task::yield_now().await;
328
329        // If the LOCK file doesn't exist, the datastore was never
330        // initialised far enough to create it (e.g. `open` failed
331        // upstream of the LSM tree build). There is nothing to wait
332        // for — treat as already released.
333        if !lock_path.exists() {
334            return Ok(());
335        }
336
337        // Try to acquire the OS-level exclusive flock that SurrealKV
338        // was holding on the same file. When `try_lock` returns
339        // `Ok(())`, the previous datastore's `kvs.shutdown()` chain
340        // has definitively run to completion and dropped its
341        // `LockFile`; release immediately so a subsequent open() at
342        // the same path can acquire it.
343        //
344        // ENOENT / other `open` errors between the `exists()` check
345        // above and this call are benign races: if the error persists
346        // the deadline check below will surface it, otherwise the next
347        // iteration picks up the real state.
348        if let Ok(file) = OpenOptions::new().read(true).write(true).open(&lock_path) {
349            match file.try_lock() {
350                Ok(()) => {
351                    // Success: release our probe lock and return. The
352                    // explicit `unlock()` isn't strictly necessary
353                    // (dropping `file` closes the fd and releases the
354                    // flock) but makes the intent readable.
355                    let _ = file.unlock();
356                    return Ok(());
357                }
358                Err(TryLockError::WouldBlock) => {
359                    // Router task hasn't reached `LockFile::drop` yet.
360                    // Fall through to the backoff sleep.
361                }
362                Err(TryLockError::Error(io_err)) => {
363                    // Real I/O error (not lock contention). Surface it
364                    // immediately rather than looping against a broken
365                    // filesystem.
366                    return Err(io_err.into());
367                }
368            }
369        }
370
371        if Instant::now() >= deadline {
372            return Err(StateError::Database(format!(
373                "EmbeddedDb::shutdown timed out after {}s waiting for \
374                 SurrealKV LOCK release at {}",
375                MAX_WAIT.as_secs(),
376                lock_path.display(),
377            )));
378        }
379
380        tokio::time::sleep(backoff).await;
381        backoff = (backoff * 2).min(MAX_BACKOFF);
382    }
383}

589        // with the id we wrote to `table_b` returns None — i.e. ids
590        // don't bleed across tables.
591        let cross: Option<Item> = client
592            .select(("table_a", "2"))
593            .await
594            .expect("cross-table select");
595        assert!(
596            cross.is_none(),
597            "table_a should not see records keyed in table_b",
598        );
599
600        db.shutdown().await.expect("shutdown");
601    }

603    /// P1.5 — Concurrent reads from clones of the same client.
604    ///
605    /// `EmbeddedDb` is `Clone`, and the underlying `Surreal<Db>` shares
606    /// state via Arc internally. Multiple readers from clones should see
607    /// the same value without races or panics. Spawns 10 tasks reading
608    /// the same record concurrently and asserts they all observe the
609    /// expected value.
610    #[tokio::test]
611    async fn concurrent_reads_from_clones() {
612        let dir = tempfile::tempdir().expect("tempdir");
613        let db = EmbeddedDb::open(&dir.path().join("concurrent.skv"))
614            .await
615            .expect("open");
616
617        // Seed a single shared record.
618        let _: Option<Item> = db
619            .client()
620            .create(("items", "shared"))
621            .content(Item {
622                name: "concurrent".into(),
623                count: 42,
624            })
625            .await
626            .expect("seed");
627
628        // Spawn concurrent readers.
629        let mut handles = Vec::with_capacity(10);
630        for i in 0..10u32 {
631            let db_clone = db.clone();
632            handles.push(tokio::spawn(async move {
633                let item: Option<Item> = db_clone
634                    .client()
635                    .select(("items", "shared"))
636                    .await
637                    .expect("concurrent select");
638                (i, item.expect("missing on concurrent reader"))
639            }));
640        }
641
642        for h in handles {
643            let (i, item) = h.await.expect("join");
644            assert_eq!(
645                item.count, 42,
646                "concurrent reader {i} observed wrong value: {item:?}"
647            );
648        }
649
650        db.shutdown().await.expect("shutdown");
651    }

653    /// P1.11 — `EmbeddedDb::open` retries on SurrealKV flock contention.
654    ///
655    /// Opens the same datastore twice in sequence. The first open is
656    /// then shut down (which releases the SurrealKV flock), and we
657    /// kick off a parallel open *before* `shutdown` returns. The
658    /// parallel open must not fail immediately with "already locked":
659    /// the retry loop in `open_datastore_with_retry` should re-try
660    /// until the router task's `kvs.shutdown()` chain has released
661    /// the flock, then succeed.
662    ///
663    /// We deliberately DO NOT use `shutdown` on the second handle after
664    /// `drop(first)` — the shutdown's probe-lock would race the second
665    /// open's flock acquisition and make the test flaky. Only the
666    /// second handle is explicitly shut down at the end.
667    #[tokio::test]
668    async fn open_retries_on_flock_contention() {
669        let dir = tempfile::tempdir().expect("tempdir");
670        let path = dir.path().join("contention.skv");
671
672        // First handle — held briefly, then dropped.
673        let db_a = EmbeddedDb::open(&path).await.expect("open #1");
674
675        // Spawn the second open before db_a is gone; it should spin in
676        // the retry loop until db_a's router releases the flock.
677        let path_clone = path.clone();
678        let joiner = tokio::spawn(async move {
679            EmbeddedDb::open(&path_clone)
680                .await
681                .expect("second open should succeed after retry")
682        });
683
684        // Let the second open spin on the retry loop a bit, then drop
685        // db_a (without calling `shutdown` to avoid racing the second
686        // open on the probe-lock — see the doc comment above).
687        tokio::time::sleep(Duration::from_millis(100)).await;
688        drop(db_a);
689
690        let db_b = joiner.await.expect("join second open");
691        db_b.shutdown().await.expect("shutdown #2");
692    }

722    /// P1.7 — `EmbeddedDb::open` applies the bootstrap schema automatically.
723    ///
724    /// Opens a fresh database (no manual schema application) and asserts
725    /// that the four bootstrap tables (`mesh`, `hypervisor`, `peer`,
726    /// `wg_key`) are reachable. With the schema NOT auto-applied, a
727    /// `select` against a never-touched table returns SurrealDB's
728    /// `NotFound` error (verified by `multi_table_isolation` in P1.5).
729    /// With the schema auto-applied, the table is defined-but-empty and
730    /// `select` returns an empty Vec.
731    #[tokio::test]
732    async fn open_applies_bootstrap_schema_automatically() {
733        let dir = tempfile::tempdir().expect("tempdir");
734        let db = EmbeddedDb::open(&dir.path().join("auto_schema.skv"))
735            .await
736            .expect("open should also apply the schema");
737
738        // Each schema-defined table must be reachable on a fresh DB.
739        // We use surrealdb::types::Value to avoid forcing a SurrealValue
740        // derive on a strongly-typed mirror of every table.
741        for table in &["mesh", "hypervisor", "peer", "wg_key"] {
742            let rows: Vec<surrealdb::types::Value> =
743                db.client().select(*table).await.unwrap_or_else(|e| {
744                    panic!("select on {table} should succeed (schema not applied?): {e}")
745                });
746            assert!(
747                rows.is_empty(),
748                "table {table} should be empty on a fresh DB but contained {} rows",
749                rows.len()
750            );
751        }
752
753        // The schema is also idempotent: a second open at the same path
754        // (after explicit shutdown) must succeed and the tables must
755        // still be defined.
756        let path = db.path().to_path_buf();
757        db.shutdown().await.expect("shutdown");
758
759        let db2 = EmbeddedDb::open(&path)
760            .await
761            .expect("reopening an already-initialised DB should succeed");
762        for table in &["mesh", "hypervisor", "peer", "wg_key"] {
763            let rows: Vec<surrealdb::types::Value> =
764                db2.client().select(*table).await.unwrap_or_else(|e| {
765                    panic!("post-reopen select on {table} should succeed: {e}")
766                });
767            assert!(
768                rows.is_empty(),
769                "table {table} should still be empty after reopen"
770            );
771        }
772        db2.shutdown().await.expect("shutdown #2");
773    }
