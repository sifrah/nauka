1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, and
7//! an explicit `shutdown` step.
8//!
9//! This is the long-term replacement for [`crate::LocalDb`] (the
10//! JSON-file-backed bootstrap store). The migration of every existing
11//! `LocalDb` call site is tracked by P1.10 → P1.12 (sifrah/nauka#200,
12//! sifrah/nauka#201, sifrah/nauka#202). Until those land, both backends
13//! coexist.
14//!
15//! P1.2 (sifrah/nauka#192) introduces only the wrapper struct and its
16//! lifecycle. The companion deliverables — error mapping (P1.3,
17//! sifrah/nauka#193), default-path helpers (P1.4, sifrah/nauka#194),
18//! comprehensive CRUD/persistence tests (P1.5, sifrah/nauka#195), the
19//! initial `.surql` schema (P1.6, sifrah/nauka#196), and applying that
20//! schema at open time (P1.7, sifrah/nauka#197) — each ship as their own
21//! issue.
22//!
23//! # Example
24//!
25//! ```no_run
26//! # use nauka_state::EmbeddedDb;
27//! # use std::path::Path;
28//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
29//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
30//!
31//! // Use the underlying SurrealDB SDK directly:
32//! let _: Vec<surrealdb::types::Value> = db
33//!     .client()
34//!     .query("INFO FOR DB")
35//!     .await?
36//!     .take(0)?;
37//!
38//! db.shutdown().await?;
39//! # Ok(()) }
40//! ```
41
42use std::path::{Path, PathBuf};
43
44use surrealdb::engine::local::{Db, SurrealKv};
45use surrealdb::Surreal;
46
47use crate::{Result, StateError, BOOTSTRAP_DB, NAUKA_NS};
48
49/// Embedded SurrealDB instance, persisted to disk via the SurrealKV backend.
50///
51/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
52/// connection (which itself uses an `Arc` internally).
53#[derive(Clone)]
54pub struct EmbeddedDb {
55    inner: Surreal<Db>,
56    path: PathBuf,
57}
58
59impl EmbeddedDb {
60    /// Open (or create) the SurrealKV-backed database at the run-mode-aware
61    /// default path provided by [`nauka_core::process::nauka_db_path`]:
62    ///
63    /// - CLI mode: `~/.nauka/bootstrap.skv`
64    /// - Service mode (root): `/var/lib/nauka/bootstrap.skv`
65    ///
66    /// The parent state directory is created via
67    /// [`nauka_core::process::ensure_nauka_state_dir`] (with 0o700 perms),
68    /// then the rest of the open path is the same as [`Self::open`].
69    pub async fn open_default() -> Result<Self> {
70        // Create the state directory with the right perms; the SurrealKV
71        // datastore directory itself is created by `Surreal::new::<SurrealKv>`.
72        let _ = nauka_core::process::ensure_nauka_state_dir()
73            .map_err(|e| StateError::Database(format!("ensure state dir: {e}")))?;
74        let path = nauka_core::process::nauka_db_path();
75        Self::open(&path).await
76    }
77
78    /// Open (or create) the SurrealKV-backed database at `path` and select
79    /// the `nauka` / `bootstrap` namespace/database.
80    ///
81    /// The parent directory is created if it doesn't exist. On Unix the
82    /// parent directory is also chmod-ed to 0o700 (best-effort).
83    ///
84    /// For the run-mode-aware default path, use [`Self::open_default`].
85    ///
86    /// # Errors
87    ///
88    /// Returns [`StateError::Io`] if the parent directory cannot be created,
89    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
90    /// [`StateError::Database`] (depending on the underlying SurrealDB
91    /// failure mode) if SurrealDB cannot open the datastore or switch to
92    /// the configured namespace/database. The classification is done by the
93    /// `From<surrealdb::Error>` impl in `lib.rs` (P1.3, sifrah/nauka#193).
94    pub async fn open(path: &Path) -> Result<Self> {
95        if let Some(parent) = path.parent() {
96            if !parent.as_os_str().is_empty() {
97                std::fs::create_dir_all(parent)?;
98                #[cfg(unix)]
99                {
100                    use std::os::unix::fs::PermissionsExt;
101                    // Best-effort 0o700 on the parent (P1.4 perms target).
102                    // We don't surface a chmod failure as a hard error: the
103                    // dir was created successfully, refusing to start over a
104                    // chmod refusal would be obnoxious. Logged via tracing
105                    // by nauka_core::process when the helper is used.
106                    let _ =
107                        std::fs::set_permissions(parent, std::fs::Permissions::from_mode(0o700));
108                }
109            }
110        }
111
112        let path_str = path.to_string_lossy().into_owned();
113        let inner = Surreal::new::<SurrealKv>(path_str.as_str()).await?;
114
115        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
116
117        Ok(Self {
118            inner,
119            path: path.to_path_buf(),
120        })
121    }
122
123    /// Borrow the underlying SurrealDB SDK client.
124    ///
125    /// All SurrealQL queries flow through this. The wrapper does not
126    /// re-export every SDK method — call sites that need the full SDK
127    /// surface go through `db.client().query(...)` etc.
128    pub fn client(&self) -> &Surreal<Db> {
129        &self.inner
130    }
131
132    /// Path of the on-disk SurrealKV datastore.
133    pub fn path(&self) -> &Path {
134        &self.path
135    }
136
137    /// Shut down the wrapper, dropping the SDK client.
138    ///
139    /// Dropping the inner [`Surreal<Db>`] is what actually closes the
140    /// datastore — the SurrealDB SDK's background router task exits
141    /// when its sender side goes away, and the SurrealKV engine flushes
142    /// pending writes on drop. This method exists so call sites have an
143    /// explicit, named way to do that, instead of relying on `Drop` order.
144    ///
145    /// Returns `Ok(())` once the inner client has been dropped. Future
146    /// versions may flush, fsync, or wait for outstanding tasks here, so
147    /// the signature is async + `Result` even though the body is trivial
148    /// today.
149    pub async fn shutdown(self) -> Result<()> {
150        // Explicit drop of the inner client. The compiler would do this
151        // anyway when `self` goes out of scope, but naming the step makes
152        // the intent visible at every call site.
153        drop(self.inner);
154        Ok(())
155    }
156}
157
158impl std::fmt::Debug for EmbeddedDb {
159    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
160        f.debug_struct("EmbeddedDb")
161            .field("path", &self.path)
162            .field("ns", &NAUKA_NS)
163            .field("db", &BOOTSTRAP_DB)
164            .finish()
165    }
166}

168#[cfg(test)]
169mod tests {
170    use super::*;
171
172    /// P1.2 smoke test: open the wrapper at a temp path and shut it down.
173    ///
174    /// Comprehensive CRUD / persistence tests live in P1.5
175    /// (sifrah/nauka#195). This one only covers the lifecycle skeleton
176    /// the issue actually adds.
177    #[tokio::test]
178    async fn open_and_shutdown_smoke() {
179        let dir = tempfile::tempdir().expect("tempdir");
180        let path = dir.path().join("smoke.skv");
181
182        let db = EmbeddedDb::open(&path)
183            .await
184            .expect("open should succeed at a writable temp path");
185
186        assert_eq!(db.path(), path);
187
188        // The on-disk datastore should now exist (SurrealKV creates a
189        // directory at the given path).
190        assert!(path.exists(), "datastore path should exist after open");
191
192        // Confirm the SDK client returned a usable handle by issuing a
193        // no-side-effect query.
194        let _ = db
195            .client()
196            .query("INFO FOR DB")
197            .await
198            .expect("INFO FOR DB should succeed against the freshly-opened bootstrap db");
199
200        db.shutdown()
201            .await
202            .expect("shutdown should always succeed in P1.2");
203    }
