1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, an
7//! explicit `shutdown` step that waits for SurrealKV's background router to
8//! release its OS-level flock, and automatic application of the bootstrap
9//! schema (`schemas/bootstrap.surql`) on every open.
10//!
11//! This is the single source of truth for everything a node needs to read
12//! before the cluster is reachable: mesh identity, hypervisor identity, peer
13//! list, WireGuard keypair, and the storage region registry. See the crate
14//! [`README`](https://github.com/sifrah/nauka/blob/main/layers/state/README.md)
15//! for the full guide and on-disk layout.
16//!
17//! # Example
18//!
19//! ```no_run
20//! use nauka_state::EmbeddedDb;
21//! use std::path::Path;
22//!
23//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
24//! // Open (or create) the SurrealKV datastore. The bootstrap schema
25//! // is applied automatically. Use `EmbeddedDb::open_default()` to
26//! // pick up the run-mode-aware default path instead.
27//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
28//!
29//! // Drop down to the SurrealDB SDK for queries.
30//! let _: Vec<surrealdb::types::Value> = db
31//!     .client()
32//!     .query("SELECT * FROM peer")
33//!     .await?
34//!     .take(0)?;
35//!
36//! // Shut down explicitly so the SurrealKV flock is released and a
37//! // subsequent `EmbeddedDb::open` at the same path can succeed.
38//! db.shutdown().await?;
39//! # Ok(()) }
40//! ```

52/// The bootstrap-state SurrealQL schema, embedded at compile time.
53///
54/// Defined in `layers/state/schemas/bootstrap.surql` (P1.6, sifrah/nauka#196).
55/// `EmbeddedDb::open` applies this on every open. The schema is fully
56/// idempotent thanks to `IF NOT EXISTS` on every `DEFINE` statement, so
57/// re-applying against an already-initialised database is a no-op.
58const BOOTSTRAP_SCHEMA: &str = include_str!("../schemas/bootstrap.surql");
59
60/// Which local SurrealDB backend is live behind this handle.
61///
62/// P2.1 (sifrah/nauka#205) enabled the `kv-tikv` feature on `surrealdb`
63/// permanently; P2.2 (sifrah/nauka#206) added the
64/// [`EmbeddedDb::open_tikv`] constructor. From that point onward one
65/// `EmbeddedDb` value can front either an on-disk SurrealKV datastore or
66/// a distributed TiKv cluster, and a few lifecycle steps (in particular
67/// `shutdown`'s LOCK-file poll) are only meaningful for the SurrealKV
68/// branch. This enum lets `shutdown` and `Debug` pick the right path
69/// without keeping the full backend type in the struct.
70#[derive(Debug, Clone, Copy, PartialEq, Eq)]
71enum EmbeddedBackend {
72    /// On-disk SurrealKV datastore, the single source of truth for the
73    /// node's bootstrap state.
74    SurrealKv,
75    /// Distributed TiKv cluster, backing the cluster-side state. No
76    /// process-local LOCK file to reclaim on shutdown — the
77    /// TransactionClient tears itself down over gRPC.
78    TiKv,
79}

81/// Embedded SurrealDB instance, fronting either a local SurrealKV
82/// datastore or a distributed TiKv cluster.
83///
84/// - `open` / `open_default` → `SurrealKv` (bootstrap state).
85/// - `open_tikv` → `TiKv` (cluster state, P2.2 sifrah/nauka#206).
86///
87/// Both variants share the same wrapper so upstream call sites can treat
88/// them uniformly: one `EmbeddedDb` type, one `client()` accessor, one
89/// `shutdown()` contract.
90///
91/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
92/// connection (which itself uses an `Arc` internally).
93#[derive(Clone)]
94pub struct EmbeddedDb {
95    inner: Surreal<Db>,
96    /// Path of the on-disk SurrealKV datastore. Empty for the `TiKv`
97    /// backend — `path()` returns an empty path in that case, which is a
98    /// signal to callers that "the handle has no local file".
99    path: PathBuf,
100    /// Which backend is live behind `inner`. Set at open time, never
101    /// mutated.
102    backend: EmbeddedBackend,
103}
104
105impl EmbeddedDb {
106    /// Open (or create) the SurrealKV-backed database at the run-mode-aware
107    /// default path provided by [`nauka_core::process::nauka_db_path`]:
108    ///
109    /// - CLI mode: `~/.nauka/bootstrap.skv`
110    /// - Service mode (root): `/var/lib/nauka/bootstrap.skv`
111    ///
112    /// The parent state directory is created via
113    /// [`nauka_core::process::ensure_nauka_state_dir`] (with 0o700 perms),
114    /// then the rest of the open path is the same as [`Self::open`].
115    pub async fn open_default() -> Result<Self> {
116        // Create the state directory with the right perms; the SurrealKV
117        // datastore directory itself is created by `Surreal::new::<SurrealKv>`.
118        let _ = nauka_core::process::ensure_nauka_state_dir()
119            .map_err(|e| StateError::Database(format!("ensure state dir: {e}")))?;
120        let path = nauka_core::process::nauka_db_path();
121        Self::open(&path).await
122    }

124    /// Open (or create) the SurrealKV-backed database at `path`, select the
125    /// `nauka` / `bootstrap` namespace/database, and apply the embedded
126    /// bootstrap schema.
127    ///
128    /// The parent directory is created if it doesn't exist. On Unix the
129    /// parent directory is also chmod-ed to 0o700 (best-effort).
130    ///
131    /// The bootstrap schema (`schemas/bootstrap.surql`, P1.6) is applied on
132    /// every open. It is fully idempotent thanks to `IF NOT EXISTS` on
133    /// every `DEFINE` statement, so reopening an existing database is a
134    /// no-op for already-defined tables, fields, and indexes — and any
135    /// new tables/fields/indexes that have appeared since last open are
136    /// added forward-compatibly. P1.7 (sifrah/nauka#197) is the issue that
137    /// wires this in.
138    ///
139    /// For the run-mode-aware default path, use [`Self::open_default`].
140    ///
141    /// # Errors
142    ///
143    /// Returns [`StateError::Io`] if the parent directory cannot be created,
144    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
145    /// [`StateError::Database`] (depending on the underlying SurrealDB
146    /// failure mode) if SurrealDB cannot open the datastore, switch to the
147    /// configured namespace/database, or apply the bootstrap schema. The
148    /// classification is done by the `From<surrealdb::Error>` impl in
149    /// `lib.rs` (P1.3, sifrah/nauka#193).
150    pub async fn open(path: &Path) -> Result<Self> {
151        if let Some(parent) = path.parent() {
152            if !parent.as_os_str().is_empty() {
153                std::fs::create_dir_all(parent)?;
154                #[cfg(unix)]
155                {
156                    use std::os::unix::fs::PermissionsExt;
157                    // Best-effort 0o700 on the parent (P1.4 perms target).
158                    // We don't surface a chmod failure as a hard error: the
159                    // dir was created successfully, refusing to start over a
160                    // chmod refusal would be obnoxious. Logged via tracing
161                    // by nauka_core::process when the helper is used.
162                    let _ =
163                        std::fs::set_permissions(parent, std::fs::Permissions::from_mode(0o700));
164                }
165            }
166        }
167
168        // Acquire the SurrealKV datastore with retry-on-lock-contention.
169        //
170        // # Why this exists
171        //
172        // SurrealKV takes an OS-level exclusive flock on `<path>/LOCK`
173        // whenever a `Datastore` is open. Only one process can hold the
174        // flock at a time. In Nauka, that matters because the fabric
175        // layer has concurrent consumers of the same `bootstrap.skv`:
176        //
177        // - Ad-hoc CLI invocations (`nauka hypervisor status`, etc.)
178        // - The `nauka-forge` daemon running reconcile cycles
179        // - The `nauka-announce` listener
180        //
181        // Each of them opens the DB for a short span (read state, do
182        // work, `shutdown().await`) but their windows overlap, so an
183        // open that happens to collide with another process's brief
184        // write window would fail with
185        // `Other("Database at ... LOCK is already locked by another process")`.
186        //
187        // The fix: retry the open on "already locked" with exponential
188        // backoff up to a 5-second deadline, matching the pattern used
189        // by `shutdown` in `wait_for_surrealkv_lock_release`. 5 s is the
190        // same "fast timeout" budget used for health checks — long
191        // enough to ride out a forge reconcile cycle that briefly holds
192        // the lock, short enough that a stuck holder surfaces as a
193        // clear error instead of hanging forever.
194        let path_str = path.to_string_lossy().into_owned();
195        let inner = open_datastore_with_retry(&path_str).await?;
196
197        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
198
199        // Apply the bootstrap schema after switching to the right ns/db.
200        // DEFINE TABLE / FIELD / INDEX are db-scoped, so use_db must
201        // happen first. The schema's IF NOT EXISTS clauses make this
202        // safe to run on every open.
203        inner.query(BOOTSTRAP_SCHEMA).await?.check()?;
204
205        Ok(Self {
206            inner,
207            path: path.to_path_buf(),
208            backend: EmbeddedBackend::SurrealKv,
209        })
210    }
211
212    /// Open the cluster-side SurrealDB database backed by an existing
213    /// PD/TiKV cluster.
214    ///
215    /// `pd_endpoints` is an ordered list of PD client endpoints. The
216    /// standard format for a Nauka node is `http://[<mesh_ipv6>]:2379`
217    /// (e.g. `http://[fdc5:8ba:9b14::1]:2379`), which matches the format
218    /// that [`nauka_hypervisor::fabric::FabricState::pd_endpoints`]
219    /// hands out. The `http://` prefix is stripped before the endpoint
220    /// is handed to the SurrealDB TiKv engine — the engine's URL parser
221    /// expects the bare `host:port` form.
222    ///
223    /// Endpoints are tried in order; the first one that returns a live
224    /// `Surreal<Db>` handle wins. This is the same fallback contract as
225    /// `tikv-client`'s own `new_with_config(vec![...])`, except that
226    /// SurrealDB's `Surreal::new::<TiKv>` only accepts a single address
227    /// per call, so the iteration lives in this wrapper. If every
228    /// endpoint fails the *last* error is returned, classified through
229    /// the existing `From<surrealdb::Error>` impl.
230    ///
231    /// After the handle is live the wrapper switches to the
232    /// `nauka` / `cluster` namespace/database (per ADR 0003,
233    /// sifrah/nauka#190). No schema is applied — cluster-side schema
234    /// lands in a later Phase 2 ticket.
235    ///
236    /// # Errors
237    ///
238    /// - [`StateError::Database`] (via `From<surrealdb::Error>`) if
239    ///   every PD endpoint fails to answer, or if `use_ns`/`use_db`
240    ///   fails after a handle was acquired.
241    /// - [`StateError::Database`] with a fixed "no PD endpoints" message
242    ///   if `pd_endpoints` is empty.
243    pub async fn open_tikv(pd_endpoints: &[&str]) -> Result<Self> {
244        if pd_endpoints.is_empty() {
245            return Err(StateError::Database(
246                "open_tikv: no PD endpoints provided".to_string(),
247            ));
248        }
249
250        // Try each endpoint in order. The SurrealDB TiKv engine's
251        // `IntoEndpoint` impl only accepts a single address per
252        // `Surreal::new` call, so the fallback loop lives here — match
253        // the fallback contract of `tikv-client::new_with_config(vec![...])`
254        // without forcing callers to duplicate the logic.
255        let mut last_err: Option<surrealdb::Error> = None;
256        for ep in pd_endpoints {
257            // SurrealDB's TiKv endpoint builder wraps the argument in
258            // `tikv://{self}`, and the downstream URL parser expects a
259            // bare `host:port`. Strip any `http://` / `https://` prefix
260            // we were handed so call sites can use the same
261            // `http://[ipv6]:2379` string everywhere.
262            let bare = strip_http_scheme(ep);
263            match Surreal::new::<TiKv>(bare).await {
264                Ok(db) => {
265                    db.use_ns(NAUKA_NS).use_db(CLUSTER_DB).await?;
266                    return Ok(Self {
267                        inner: db,
268                        // The TiKv backend has no local file. `path()`
269                        // returns this empty PathBuf so callers can tell
270                        // "there is nothing on disk to clean up".
271                        path: PathBuf::new(),
272                        backend: EmbeddedBackend::TiKv,
273                    });
274                }
275                Err(err) => {
276                    last_err = Some(err);
277                }
278            }
279        }
280
281        // Every endpoint failed. Surface the most recent error (it
282        // carries the richest detail — DNS failure, gRPC refused, etc.)
283        // through the standard SurrealDB → StateError mapping.
284        Err(last_err
285            .map(StateError::from)
286            .unwrap_or_else(|| StateError::Database("open_tikv: all endpoints failed".into())))
287    }

295        &self.inner

303    /// Shut down the wrapper, dropping the SDK client and waiting for the
304    /// SurrealDB background router task to finish closing the datastore
305    /// so a subsequent `EmbeddedDb::open` at the same path can succeed
306    /// — and, critically, so any writes committed through this handle
307    /// are durably on disk before we return.
308    ///
309    /// # Why this function exists
310    ///
311    /// The SDK's `Surreal<Db>` router runs on a background tokio task
312    /// that owns an `Arc<Datastore>`. `drop(Surreal<Db>)` is
313    /// fire-and-forget: it closes the route channel, the router task
314    /// sees `route_rx.recv()` return `Err`, breaks out of its loop,
315    /// and **then** runs `Datastore::shutdown().await` which:
316    ///
317    /// 1. Shuts down the commit coordinator and WAL background flusher
318    /// 2. Calls `Tree::close()` which flushes all memtables to SSTables
319    /// 3. Closes the WAL
320    /// 4. Drops the `LockFile`, releasing SurrealKV's OS-level exclusive
321    ///    flock on `<path>/LOCK`
322    ///
323    /// None of this is synchronous with `drop(self.inner)`. A caller that
324    /// does `shutdown().await; open(same_path).await` races the
325    /// background task. Previous versions of this method used
326    /// `tokio::time::sleep(50 ms)` as a heuristic — enough on CI Linux,
327    /// **not** enough on Mac arm64 under contention, where it manifested
328    /// as data loss (`load_async` returning `None` after a round-trip
329    /// reopen, reproducible at 77% with `--test-threads=2`). See
330    /// sifrah/nauka#255 for the investigation.
331    ///
332    /// The deterministic signal that step 4 has happened is the release
333    /// of the OS-level flock on `<path>/LOCK`. We poll for that release
334    /// by trying to acquire the same flock ourselves via
335    /// [`std::fs::File::try_lock`]. When `try_lock` succeeds, the
336    /// previous `Datastore::shutdown()` chain has definitively run to
337    /// completion, which implies every prior commit is durable and the
338    /// next open can proceed without a lock-held error.
339    ///
340    /// # Cost
341    ///
342    /// Fast path: a single `yield_now` + `try_lock` (~sub-ms). Slow path:
343    /// exponential backoff from 1 ms to a 50 ms ceiling, up to a hard
344    /// 5-second deadline after which the call returns
345    /// [`StateError::Database`]. 5 s is generous relative to the
346    /// ~10–100 ms that a real `Datastore::shutdown()` takes even on a
347    /// contended runtime, and still well under the "fast timeout" Nauka
348    /// convention for health checks.
349    pub async fn shutdown(self) -> Result<()> {
350        // Explicit drop of the inner client. The compiler would do this
351        // anyway when `self` goes out of scope, but naming the step
352        // makes the handoff to the router task visible.
353        let path = self.path.clone();
354        let backend = self.backend;
355        drop(self.inner);
356
357        match backend {
358            // SurrealKV holds an OS-level exclusive flock on
359            // `<path>/LOCK` that the router task releases as the last
360            // step of `Datastore::shutdown()`. Poll for that release so
361            // a subsequent `open` at the same path can acquire the flock
362            // — and so any writes committed through this handle are
363            // durable before we return. See the doc comment on
364            // `shutdown` and the rationale on `wait_for_surrealkv_lock_release`.
365            EmbeddedBackend::SurrealKv => wait_for_surrealkv_lock_release(&path).await,
366            // TiKv has no process-local LOCK file to reclaim. The
367            // `Surreal<Db>` router drains the in-flight gRPC calls when
368            // the route channel closes, the `TransactionClient` shuts
369            // its connection pool down from its own `Drop`, and nothing
370            // on the local filesystem needs to settle before the next
371            // `open_tikv` call at the same node can succeed. Just
372            // return.
373            EmbeddedBackend::TiKv => Ok(()),
374        }
375    }

378/// Strip a leading `http://` or `https://` from a PD endpoint, leaving
379/// the bare `host:port` form.
380///
381/// SurrealDB's TiKv engine wraps the argument in `tikv://{self}` before
382/// parsing it as a URL. If the caller already prefixed with `http://`
383/// (the internal convention used by everything downstream of
384/// `FabricState::pd_endpoints`, to match `tikv-client::RawClient::new`)
385/// the resulting `tikv://http://...` string is malformed. Strip the
386/// scheme here so both input styles work.
387fn strip_http_scheme(ep: &str) -> &str {
388    if let Some(rest) = ep.strip_prefix("http://") {
389        rest
390    } else if let Some(rest) = ep.strip_prefix("https://") {
391        rest
392    } else {
393        ep
394    }
395}

397/// Build PD client endpoints for a slice of mesh IPv6 addresses.
398///
399/// Produces one `http://[<ipv6>]:<port>` string per input address, in
400/// the same order. The `http://` prefix matches the format used by
401/// [`nauka_hypervisor::fabric::FabricState::pd_endpoints`] and by
402/// `tikv-client::RawClient::new(&[...])`; the SurrealDB TiKv engine
403/// strips the prefix via [`strip_http_scheme`] at `open_tikv` time, so
404/// the same list works for both sides.
405///
406/// Kept in `nauka-state` so callers don't have to redo the
407/// `format!("http://[{ipv6}]:{port}", ...)` dance at every call site.
408/// Added in P2.3 (sifrah/nauka#207) — the same issue that teaches
409/// FabricState to expose the per-node PD addresses.
410pub fn pd_endpoints_for(mesh_addrs: &[Ipv6Addr], pd_client_port: u16) -> Vec<String> {
411    mesh_addrs
412        .iter()
413        .map(|addr| format!("http://[{addr}]:{pd_client_port}"))
414        .collect()
415}

417/// Open the SurrealKV datastore at `path_str`, retrying on flock
418/// contention until a 5-second deadline elapses.
419///
420/// See the rationale block in [`EmbeddedDb::open`] for the full context.
421/// The short version: SurrealKV holds a process-exclusive flock while a
422/// datastore is open, and Nauka's CLI / forge daemon / announce listener
423/// all touch the same `bootstrap.skv`, so brief overlaps are normal. We
424/// retry the open rather than surface a hard error for every such
425/// overlap.
426async fn open_datastore_with_retry(path_str: &str) -> Result<Surreal<Db>> {
427    /// Upper bound on the total wait. Same "fast timeout" budget as
428    /// `shutdown` (see `wait_for_surrealkv_lock_release`). Generous
429    /// relative to the real fast-path cost of a forge reconcile cycle
430    /// (~10–100 ms of flock-held time) but short enough that a truly
431    /// stuck holder surfaces as a clear error instead of a hang.
432    const MAX_WAIT: Duration = Duration::from_secs(5);
433    /// First retry after an initial miss.
434    const MIN_BACKOFF: Duration = Duration::from_millis(5);
435    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
436    const MAX_BACKOFF: Duration = Duration::from_millis(50);
437
438    let deadline = Instant::now() + MAX_WAIT;
439    let mut backoff = MIN_BACKOFF;
440
441    loop {
442        match Surreal::new::<SurrealKv>(path_str).await {
443            Ok(db) => return Ok(db),
444            Err(err) => {
445                // SurrealKV surfaces flock contention as an "other"
446                // backend error with a message containing the literal
447                // path to `<datastore>/LOCK`. There's no structured
448                // variant we can match on, so fall back to a substring
449                // check. If the message shape changes in a future
450                // surrealkv release the test
451                // `open_retries_on_flock_contention` will flag it.
452                let msg = err.to_string();
453                let is_lock_contention = msg.contains("already locked");
454
455                if !is_lock_contention || Instant::now() >= deadline {
456                    return Err(err.into());
457                }
458
459                tokio::time::sleep(backoff).await;
460                backoff = (backoff * 2).min(MAX_BACKOFF);
461            }
462        }
463    }
464}

466/// Poll until SurrealKV's OS-level exclusive flock on `<path>/LOCK` is
467/// released by the previous `Datastore`, or until the 5-second deadline
468/// elapses.
469///
470/// See [`EmbeddedDb::shutdown`] for the full rationale. The short version:
471/// the SurrealDB SDK's `Surreal<Db>` router runs its `kvs.shutdown()`
472/// chain asynchronously after `drop(Surreal<Db>)`, and the release of
473/// the `LockFile` (surrealkv 0.21 / `src/lockfile.rs`) is the last step
474/// of that chain. Successfully acquiring the same flock from here proves
475/// the chain has run to completion.
476async fn wait_for_surrealkv_lock_release(path: &Path) -> Result<()> {
477    /// Upper bound on the total wait. Generous relative to the actual
478    /// `Datastore::shutdown()` cost (10–100 ms typical, even on contended
479    /// runtimes) but well under the "fast timeout" Nauka convention.
480    const MAX_WAIT: Duration = Duration::from_secs(5);
481    /// First retry after the fast-path `try_lock` miss.
482    const MIN_BACKOFF: Duration = Duration::from_millis(1);
483    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
484    const MAX_BACKOFF: Duration = Duration::from_millis(50);
485
486    let lock_path = path.join("LOCK");
487    let deadline = Instant::now() + MAX_WAIT;
488    let mut backoff = MIN_BACKOFF;
489
490    loop {
491        // Give any already-runnable task (the router exit handler) a
492        // chance to make progress before we poll. Cheap in the common
493        // case where the router task is already parked.
494        tokio::task::yield_now().await;
495
496        // If the LOCK file doesn't exist, the datastore was never
497        // initialised far enough to create it (e.g. `open` failed
498        // upstream of the LSM tree build). There is nothing to wait
499        // for — treat as already released.
500        if !lock_path.exists() {
501            return Ok(());
502        }
503
504        // Try to acquire the OS-level exclusive flock that SurrealKV
505        // was holding on the same file. When `try_lock` returns
506        // `Ok(())`, the previous datastore's `kvs.shutdown()` chain
507        // has definitively run to completion and dropped its
508        // `LockFile`; release immediately so a subsequent open() at
509        // the same path can acquire it.
510        //
511        // ENOENT / other `open` errors between the `exists()` check
512        // above and this call are benign races: if the error persists
513        // the deadline check below will surface it, otherwise the next
514        // iteration picks up the real state.
515        if let Ok(file) = OpenOptions::new().read(true).write(true).open(&lock_path) {
516            match file.try_lock() {
517                Ok(()) => {
518                    // Success: release our probe lock and return. The
519                    // explicit `unlock()` isn't strictly necessary
520                    // (dropping `file` closes the fd and releases the
521                    // flock) but makes the intent readable.
522                    let _ = file.unlock();
523                    return Ok(());
524                }
525                Err(TryLockError::WouldBlock) => {
526                    // Router task hasn't reached `LockFile::drop` yet.
527                    // Fall through to the backoff sleep.
528                }
529                Err(TryLockError::Error(io_err)) => {
530                    // Real I/O error (not lock contention). Surface it
531                    // immediately rather than looping against a broken
532                    // filesystem.
533                    return Err(io_err.into());
534                }
535            }
536        }
537
538        if Instant::now() >= deadline {
539            return Err(StateError::Database(format!(
540                "EmbeddedDb::shutdown timed out after {}s waiting for \
541                 SurrealKV LOCK release at {}",
542                MAX_WAIT.as_secs(),
543                lock_path.display(),
544            )));
545        }
546
547        tokio::time::sleep(backoff).await;
548        backoff = (backoff * 2).min(MAX_BACKOFF);
549    }
550}

1112    /// P2.4 — `open_tikv` against an unreachable endpoint surfaces a
1113    /// `StateError::Database` rather than hanging forever or panicking.
1114    ///
1115    /// We point at `127.0.0.1:1`, which is the reserved "TCP muxer" port
1116    /// on every POSIX system — nothing is listening, so the TCP connect
1117    /// races with ECONNREFUSED / gRPC timeout, and the SurrealDB TiKv
1118    /// engine surfaces the failure as an internal `surrealdb::Error`
1119    /// that our `From<surrealdb::Error>` impl classifies as
1120    /// [`StateError::Database`]. The whole call has to return under a
1121    /// 30 s hard timeout — the "fast timeout" budget Nauka uses for
1122    /// health checks — otherwise the test runner would hang CI on a
1123    /// genuine failure.
1124    #[tokio::test]
1125    async fn open_tikv_unreachable_endpoint_returns_state_error() {
1126        // 30 s is the Nauka convention for a health-check timeout and
1127        // is *much* more generous than the real path: gRPC connect to
1128        // a closed port returns ECONNREFUSED in tens of ms. If the
