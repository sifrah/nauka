1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, and
7//! an explicit `shutdown` step.
8//!
9//! This is the long-term replacement for [`crate::LocalDb`] (the
10//! JSON-file-backed bootstrap store). The migration of every existing
11//! `LocalDb` call site is tracked by P1.10 → P1.12 (sifrah/nauka#200,
12//! sifrah/nauka#201, sifrah/nauka#202). Until those land, both backends
13//! coexist.
14//!
15//! P1.2 (sifrah/nauka#192) introduces only the wrapper struct and its
16//! lifecycle. The companion deliverables — error mapping (P1.3,
17//! sifrah/nauka#193), default-path helpers (P1.4, sifrah/nauka#194),
18//! comprehensive CRUD/persistence tests (P1.5, sifrah/nauka#195), the
19//! initial `.surql` schema (P1.6, sifrah/nauka#196), and applying that
20//! schema at open time (P1.7, sifrah/nauka#197) — each ship as their own
21//! issue.
22//!
23//! # Example
24//!
25//! ```no_run
26//! # use nauka_state::EmbeddedDb;
27//! # use std::path::Path;
28//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
29//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
30//!
31//! // Use the underlying SurrealDB SDK directly:
32//! let _: Vec<surrealdb::types::Value> = db
33//!     .client()
34//!     .query("INFO FOR DB")
35//!     .await?
36//!     .take(0)?;
37//!
38//! db.shutdown().await?;
39//! # Ok(()) }
40//! ```

42use std::path::{Path, PathBuf};
43
44use surrealdb::engine::local::{Db, SurrealKv};
45use surrealdb::Surreal;
46
47use crate::{Result, BOOTSTRAP_DB, NAUKA_NS};
48
49/// Embedded SurrealDB instance, persisted to disk via the SurrealKV backend.
50///
51/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
52/// connection (which itself uses an `Arc` internally).
53#[derive(Clone)]
54pub struct EmbeddedDb {
55    inner: Surreal<Db>,
56    path: PathBuf,
57}
58
59impl EmbeddedDb {
60    /// Open (or create) the SurrealKV-backed database at `path` and select
61    /// the `nauka` / `bootstrap` namespace/database.
62    ///
63    /// The parent directory is created if it doesn't exist.
64    ///
65    /// # Errors
66    ///
67    /// Returns [`StateError::Io`] if the parent directory cannot be created,
68    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
69    /// [`StateError::Database`] (depending on the underlying SurrealDB
70    /// failure mode) if SurrealDB cannot open the datastore or switch to
71    /// the configured namespace/database. The classification is done by the
72    /// `From<surrealdb::Error>` impl in `lib.rs` (P1.3, sifrah/nauka#193).
73    pub async fn open(path: &Path) -> Result<Self> {
74        if let Some(parent) = path.parent() {
75            if !parent.as_os_str().is_empty() {
76                std::fs::create_dir_all(parent)?;
77            }
78        }
79
80        let path_str = path.to_string_lossy().into_owned();
81        let inner = Surreal::new::<SurrealKv>(path_str.as_str()).await?;
82
83        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
84
85        Ok(Self {
86            inner,
87            path: path.to_path_buf(),
88        })
89    }
90
91    /// Borrow the underlying SurrealDB SDK client.
92    ///
93    /// All SurrealQL queries flow through this. The wrapper does not
94    /// re-export every SDK method — call sites that need the full SDK
95    /// surface go through `db.client().query(...)` etc.
96    pub fn client(&self) -> &Surreal<Db> {
97        &self.inner
98    }
99
100    /// Path of the on-disk SurrealKV datastore.
101    pub fn path(&self) -> &Path {
102        &self.path
103    }
104
105    /// Shut down the wrapper, dropping the SDK client.
106    ///
107    /// Dropping the inner [`Surreal<Db>`] is what actually closes the
108    /// datastore — the SurrealDB SDK's background router task exits
109    /// when its sender side goes away, and the SurrealKV engine flushes
110    /// pending writes on drop. This method exists so call sites have an
111    /// explicit, named way to do that, instead of relying on `Drop` order.
112    ///
113    /// Returns `Ok(())` once the inner client has been dropped. Future
114    /// versions may flush, fsync, or wait for outstanding tasks here, so
115    /// the signature is async + `Result` even though the body is trivial
116    /// today.
117    pub async fn shutdown(self) -> Result<()> {
118        // Explicit drop of the inner client. The compiler would do this
119        // anyway when `self` goes out of scope, but naming the step makes
120        // the intent visible at every call site.
121        drop(self.inner);
122        Ok(())
123    }
124}
125
126impl std::fmt::Debug for EmbeddedDb {
127    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
128        f.debug_struct("EmbeddedDb")
129            .field("path", &self.path)
130            .field("ns", &NAUKA_NS)
131            .field("db", &BOOTSTRAP_DB)
132            .finish()
133    }
134}

136#[cfg(test)]
137mod tests {
138    use super::*;
139
140    /// P1.2 smoke test: open the wrapper at a temp path and shut it down.
141    ///
142    /// Comprehensive CRUD / persistence tests live in P1.5
143    /// (sifrah/nauka#195). This one only covers the lifecycle skeleton
144    /// the issue actually adds.
145    #[tokio::test]
146    async fn open_and_shutdown_smoke() {
147        let dir = tempfile::tempdir().expect("tempdir");
148        let path = dir.path().join("smoke.skv");
149
150        let db = EmbeddedDb::open(&path)
151            .await
152            .expect("open should succeed at a writable temp path");
153
154        assert_eq!(db.path(), path);
155
156        // The on-disk datastore should now exist (SurrealKV creates a
157        // directory at the given path).
158        assert!(path.exists(), "datastore path should exist after open");
159
160        // Confirm the SDK client returned a usable handle by issuing a
161        // no-side-effect query.
162        let _ = db
163            .client()
164            .query("INFO FOR DB")
165            .await
166            .expect("INFO FOR DB should succeed against the freshly-opened bootstrap db");
167
168        db.shutdown()
169            .await
170            .expect("shutdown should always succeed in P1.2");
171    }
