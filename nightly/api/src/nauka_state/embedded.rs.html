1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, and
7//! an explicit `shutdown` step.
8//!
9//! This is the long-term replacement for [`crate::LocalDb`] (the
10//! JSON-file-backed bootstrap store). The migration of every existing
11//! `LocalDb` call site is tracked by P1.10 → P1.12 (sifrah/nauka#200,
12//! sifrah/nauka#201, sifrah/nauka#202). Until those land, both backends
13//! coexist.
14//!
15//! P1.2 (sifrah/nauka#192) introduces only the wrapper struct and its
16//! lifecycle. The companion deliverables — error mapping (P1.3,
17//! sifrah/nauka#193), default-path helpers (P1.4, sifrah/nauka#194),
18//! comprehensive CRUD/persistence tests (P1.5, sifrah/nauka#195), the
19//! initial `.surql` schema (P1.6, sifrah/nauka#196), and applying that
20//! schema at open time (P1.7, sifrah/nauka#197) — each ship as their own
21//! issue.
22//!
23//! # Example
24//!
25//! ```no_run
26//! # use nauka_state::EmbeddedDb;
27//! # use std::path::Path;
28//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
29//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
30//!
31//! // Use the underlying SurrealDB SDK directly:
32//! let _: Vec<surrealdb::types::Value> = db
33//!     .client()
34//!     .query("INFO FOR DB")
35//!     .await?
36//!     .take(0)?;
37//!
38//! db.shutdown().await?;
39//! # Ok(()) }
40//! ```

49/// The bootstrap-state SurrealQL schema, embedded at compile time.
50///
51/// Defined in `layers/state/schemas/bootstrap.surql` (P1.6, sifrah/nauka#196).
52/// `EmbeddedDb::open` applies this on every open. The schema is fully
53/// idempotent thanks to `IF NOT EXISTS` on every `DEFINE` statement, so
54/// re-applying against an already-initialised database is a no-op.
55const BOOTSTRAP_SCHEMA: &str = include_str!("../schemas/bootstrap.surql");
56
57/// Embedded SurrealDB instance, persisted to disk via the SurrealKV backend.
58///
59/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
60/// connection (which itself uses an `Arc` internally).
61#[derive(Clone)]
62pub struct EmbeddedDb {
63    inner: Surreal<Db>,
64    path: PathBuf,
65}
66
67impl EmbeddedDb {
68    /// Open (or create) the SurrealKV-backed database at the run-mode-aware
69    /// default path provided by [`nauka_core::process::nauka_db_path`]:
70    ///
71    /// - CLI mode: `~/.nauka/bootstrap.skv`
72    /// - Service mode (root): `/var/lib/nauka/bootstrap.skv`
73    ///
74    /// The parent state directory is created via
75    /// [`nauka_core::process::ensure_nauka_state_dir`] (with 0o700 perms),
76    /// then the rest of the open path is the same as [`Self::open`].
77    pub async fn open_default() -> Result<Self> {
78        // Create the state directory with the right perms; the SurrealKV
79        // datastore directory itself is created by `Surreal::new::<SurrealKv>`.
80        let _ = nauka_core::process::ensure_nauka_state_dir()
81            .map_err(|e| StateError::Database(format!("ensure state dir: {e}")))?;
82        let path = nauka_core::process::nauka_db_path();
83        Self::open(&path).await
84    }
85
86    /// Open (or create) the SurrealKV-backed database at `path`, select the
87    /// `nauka` / `bootstrap` namespace/database, and apply the embedded
88    /// bootstrap schema.
89    ///
90    /// The parent directory is created if it doesn't exist. On Unix the
91    /// parent directory is also chmod-ed to 0o700 (best-effort).
92    ///
93    /// The bootstrap schema (`schemas/bootstrap.surql`, P1.6) is applied on
94    /// every open. It is fully idempotent thanks to `IF NOT EXISTS` on
95    /// every `DEFINE` statement, so reopening an existing database is a
96    /// no-op for already-defined tables, fields, and indexes — and any
97    /// new tables/fields/indexes that have appeared since last open are
98    /// added forward-compatibly. P1.7 (sifrah/nauka#197) is the issue that
99    /// wires this in.
100    ///
101    /// For the run-mode-aware default path, use [`Self::open_default`].
102    ///
103    /// # Errors
104    ///
105    /// Returns [`StateError::Io`] if the parent directory cannot be created,
106    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
107    /// [`StateError::Database`] (depending on the underlying SurrealDB
108    /// failure mode) if SurrealDB cannot open the datastore, switch to the
109    /// configured namespace/database, or apply the bootstrap schema. The
110    /// classification is done by the `From<surrealdb::Error>` impl in
111    /// `lib.rs` (P1.3, sifrah/nauka#193).
112    pub async fn open(path: &Path) -> Result<Self> {
113        if let Some(parent) = path.parent() {
114            if !parent.as_os_str().is_empty() {
115                std::fs::create_dir_all(parent)?;
116                #[cfg(unix)]
117                {
118                    use std::os::unix::fs::PermissionsExt;
119                    // Best-effort 0o700 on the parent (P1.4 perms target).
120                    // We don't surface a chmod failure as a hard error: the
121                    // dir was created successfully, refusing to start over a
122                    // chmod refusal would be obnoxious. Logged via tracing
123                    // by nauka_core::process when the helper is used.
124                    let _ =
125                        std::fs::set_permissions(parent, std::fs::Permissions::from_mode(0o700));
126                }
127            }
128        }
129
130        let path_str = path.to_string_lossy().into_owned();
131        let inner = Surreal::new::<SurrealKv>(path_str.as_str()).await?;
132
133        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
134
135        // Apply the bootstrap schema after switching to the right ns/db.
136        // DEFINE TABLE / FIELD / INDEX are db-scoped, so use_db must
137        // happen first. The schema's IF NOT EXISTS clauses make this
138        // safe to run on every open.
139        inner.query(BOOTSTRAP_SCHEMA).await?.check()?;
140
141        Ok(Self {
142            inner,
143            path: path.to_path_buf(),
144        })
145    }
146
147    /// Borrow the underlying SurrealDB SDK client.
148    ///
149    /// All SurrealQL queries flow through this. The wrapper does not
150    /// re-export every SDK method — call sites that need the full SDK
151    /// surface go through `db.client().query(...)` etc.
152    pub fn client(&self) -> &Surreal<Db> {
153        &self.inner
154    }
155
156    /// Path of the on-disk SurrealKV datastore.
157    pub fn path(&self) -> &Path {
158        &self.path
159    }
160
161    /// Shut down the wrapper, dropping the SDK client and waiting briefly
162    /// for the SurrealDB background router task to release the on-disk
163    /// LOCK so a subsequent `EmbeddedDb::open` at the same path can
164    /// succeed.
165    ///
166    /// Why the wait: the SDK's `Surreal<Db>` router runs on a background
167    /// tokio task that holds an `Arc<Datastore>`. The task only exits
168    /// when its route channel closes — i.e. after the last
169    /// `Surreal<Db>` clone is dropped. After it exits, it calls
170    /// `Datastore::shutdown()`, which is what actually flushes SurrealKV
171    /// and removes the LOCK file. All of that happens *asynchronously*
172    /// on the runtime, so without a small wait here a caller that does
173    /// `shutdown().await; open(same_path).await` races against the
174    /// background task and gets `Database is already locked`.
175    ///
176    /// 50 ms is empirically enough on every machine we've tested, and
177    /// nothing in Nauka's hot path calls `shutdown` often enough to
178    /// notice. If a future SDK version exposes a synchronous shutdown
179    /// that joins the router task, we should use it instead.
180    pub async fn shutdown(self) -> Result<()> {
181        // Explicit drop of the inner client. The compiler would do this
182        // anyway when `self` goes out of scope, but naming the step makes
183        // the intent visible.
184        drop(self.inner);
185
186        // Yield once so any task that's already runnable (the router
187        // exit handler) gets a chance to run before we sleep.
188        tokio::task::yield_now().await;
189
190        // Then a short sleep to cover the kvs.shutdown() flush + LOCK
191        // release latency. See the doc comment above for the rationale.
192        tokio::time::sleep(std::time::Duration::from_millis(50)).await;
193
194        Ok(())
195    }
196}

416    /// P1.5 — Concurrent reads from clones of the same client.
417    ///
418    /// `EmbeddedDb` is `Clone`, and the underlying `Surreal<Db>` shares
419    /// state via Arc internally. Multiple readers from clones should see
420    /// the same value without races or panics. Spawns 10 tasks reading
421    /// the same record concurrently and asserts they all observe the
422    /// expected value.
423    #[tokio::test]
424    async fn concurrent_reads_from_clones() {
425        let dir = tempfile::tempdir().expect("tempdir");
426        let db = EmbeddedDb::open(&dir.path().join("concurrent.skv"))
427            .await
428            .expect("open");
429
430        // Seed a single shared record.
431        let _: Option<Item> = db
432            .client()
433            .create(("items", "shared"))
434            .content(Item {
435                name: "concurrent".into(),
436                count: 42,
437            })
438            .await
439            .expect("seed");
440
441        // Spawn concurrent readers.
442        let mut handles = Vec::with_capacity(10);
443        for i in 0..10u32 {
444            let db_clone = db.clone();
445            handles.push(tokio::spawn(async move {
446                let item: Option<Item> = db_clone
447                    .client()
448                    .select(("items", "shared"))
449                    .await
450                    .expect("concurrent select");
451                (i, item.expect("missing on concurrent reader"))
452            }));
453        }
454
455        for h in handles {
456            let (i, item) = h.await.expect("join");
457            assert_eq!(
458                item.count, 42,
459                "concurrent reader {i} observed wrong value: {item:?}"
460            );
461        }
462
463        db.shutdown().await.expect("shutdown");
464    }

547    /// P1.6 — the bootstrap.surql schema must apply cleanly to a fresh
548    /// database, must be idempotent (re-applying is a no-op), and must
549    /// allow inserting one record into every table it defines.
550    ///
551    /// After P1.7 (sifrah/nauka#197) `EmbeddedDb::open` applies the schema
552    /// automatically, so this test now also covers the
553    /// "open + auto-apply + manual re-apply" idempotency contract.
554    #[tokio::test]
555    async fn bootstrap_schema_applies_cleanly() {
556        const SCHEMA: &str = include_str!("../schemas/bootstrap.surql");
557
558        let dir = tempfile::tempdir().expect("tempdir");
559        let db = EmbeddedDb::open(&dir.path().join("schema_test.skv"))
560            .await
561            .expect("open");
562
563        // First application — must succeed against an empty database.
564        db.client()
565            .query(SCHEMA)
566            .await
567            .expect("schema first apply")
568            .check()
569            .expect("schema first apply check");
570
571        // Second application — must be a no-op thanks to IF NOT EXISTS.
572        db.client()
573            .query(SCHEMA)
574            .await
575            .expect("schema second apply")
576            .check()
577            .expect("schema second apply check");
578
579        // Each table is reachable. Inserting a row exercises the field
580        // ASSERTs and confirms the SCHEMAFULL definitions accept the
581        // documented shapes. We use SurrealQL's `time::now()` directly so
582        // the test doesn't need a chrono dev-dep.
583        db.client()
584            .query(
585                "CREATE mesh:current SET ipv6_ula = $u, secret_hash = $h, \
586                 created_at = time::now()",
587            )
588            .bind(("u", "fdc5:8ba:9b14::/48"))
589            .bind(("h", "0123456789abcdef"))
590            .await
591            .expect("insert mesh")
592            .check()
593            .expect("insert mesh check");
594
595        // The record id `hypervisor:hv-01HXXX` IS the ULID — there's no
596        // separate `id` field in the schema (see schemas/bootstrap.surql
597        // for the rationale). P1.10 will create rows the same way:
598        // `CREATE hypervisor:<ulid> SET name = ..., ...`.
599        db.client()
600            .query(
601                "CREATE hypervisor:`hv-01HXXX` SET name = $n, mesh_ipv6 = $m, \
602                 public_key = $p, role = $r",
603            )
604            .bind(("n", "node-1"))
605            .bind(("m", "fdc5:8ba:9b14::1"))
606            .bind(("p", "BASE64=="))
607            .bind(("r", "leader"))
608            .await
609            .expect("insert hypervisor")
610            .check()
611            .expect("insert hypervisor check");
612
613        db.client()
