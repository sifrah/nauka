1//! Embedded SurrealDB wrapper used for Nauka's local bootstrap state.
2//!
3//! Wraps `surrealdb::Surreal<Db>` with a Nauka-friendly lifecycle: a single
4//! `open` constructor that creates the on-disk SurrealKV datastore at a given
5//! path and selects the `nauka` / `bootstrap` namespace/database (per ADR
6//! 0003, sifrah/nauka#190), an accessor for the underlying SDK client, an
7//! explicit `shutdown` step that waits for SurrealKV's background router to
8//! release its OS-level flock, and automatic application of the bootstrap
9//! schema (`schemas/bootstrap.surql`) on every open.
10//!
11//! This is the single source of truth for everything a node needs to read
12//! before the cluster is reachable: mesh identity, hypervisor identity, peer
13//! list, WireGuard keypair, and the storage region registry. See the crate
14//! [`README`](https://github.com/sifrah/nauka/blob/main/layers/state/README.md)
15//! for the full guide and on-disk layout.
16//!
17//! # Example
18//!
19//! ```no_run
20//! use nauka_state::EmbeddedDb;
21//! use std::path::Path;
22//!
23//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
24//! // Open (or create) the SurrealKV datastore. The bootstrap schema
25//! // is applied automatically. Use `EmbeddedDb::open_default()` to
26//! // pick up the run-mode-aware default path instead.
27//! let db = EmbeddedDb::open(Path::new("/var/lib/nauka/bootstrap.skv")).await?;
28//!
29//! // Drop down to the SurrealDB SDK for queries.
30//! let _: Vec<surrealdb::types::Value> = db
31//!     .client()
32//!     .query("SELECT * FROM peer")
33//!     .await?
34//!     .take(0)?;
35//!
36//! // Shut down explicitly so the SurrealKV flock is released and a
37//! // subsequent `EmbeddedDb::open` at the same path can succeed.
38//! db.shutdown().await?;
39//! # Ok(()) }
40//! ```
41

51/// The bootstrap-state SurrealQL schema, embedded at compile time.
52///
53/// Defined in `layers/state/schemas/bootstrap.surql` (P1.6, sifrah/nauka#196).
54/// `EmbeddedDb::open` applies this on every open. The schema is fully
55/// idempotent thanks to `IF NOT EXISTS` on every `DEFINE` statement, so
56/// re-applying against an already-initialised database is a no-op.
57const BOOTSTRAP_SCHEMA: &str = include_str!("../schemas/bootstrap.surql");

59/// Embedded SurrealDB instance, persisted to disk via the SurrealKV backend.
60///
61/// Cheap to clone: every clone shares the same underlying `Surreal<Db>`
62/// connection (which itself uses an `Arc` internally).
63#[derive(Clone)]
64pub struct EmbeddedDb {
65    inner: Surreal<Db>,
66    path: PathBuf,
67}

69impl EmbeddedDb {
70    /// Open (or create) the SurrealKV-backed database at the run-mode-aware
71    /// default path provided by [`nauka_core::process::nauka_db_path`]:
72    ///
73    /// - CLI mode: `~/.nauka/bootstrap.skv`
74    /// - Service mode (root): `/var/lib/nauka/bootstrap.skv`
75    ///
76    /// The parent state directory is created via
77    /// [`nauka_core::process::ensure_nauka_state_dir`] (with 0o700 perms),
78    /// then the rest of the open path is the same as [`Self::open`].
79    pub async fn open_default() -> Result<Self> {
80        // Create the state directory with the right perms; the SurrealKV
81        // datastore directory itself is created by `Surreal::new::<SurrealKv>`.
82        let _ = nauka_core::process::ensure_nauka_state_dir()
83            .map_err(|e| StateError::Database(format!("ensure state dir: {e}")))?;
84        let path = nauka_core::process::nauka_db_path();
85        Self::open(&path).await
86    }

88    /// Open (or create) the SurrealKV-backed database at `path`, select the
89    /// `nauka` / `bootstrap` namespace/database, and apply the embedded
90    /// bootstrap schema.
91    ///
92    /// The parent directory is created if it doesn't exist. On Unix the
93    /// parent directory is also chmod-ed to 0o700 (best-effort).
94    ///
95    /// The bootstrap schema (`schemas/bootstrap.surql`, P1.6) is applied on
96    /// every open. It is fully idempotent thanks to `IF NOT EXISTS` on
97    /// every `DEFINE` statement, so reopening an existing database is a
98    /// no-op for already-defined tables, fields, and indexes — and any
99    /// new tables/fields/indexes that have appeared since last open are
100    /// added forward-compatibly. P1.7 (sifrah/nauka#197) is the issue that
101    /// wires this in.
102    ///
103    /// For the run-mode-aware default path, use [`Self::open_default`].
104    ///
105    /// # Errors
106    ///
107    /// Returns [`StateError::Io`] if the parent directory cannot be created,
108    /// or one of [`StateError::NotFound`] / [`StateError::Schema`] /
109    /// [`StateError::Database`] (depending on the underlying SurrealDB
110    /// failure mode) if SurrealDB cannot open the datastore, switch to the
111    /// configured namespace/database, or apply the bootstrap schema. The
112    /// classification is done by the `From<surrealdb::Error>` impl in
113    /// `lib.rs` (P1.3, sifrah/nauka#193).
114    pub async fn open(path: &Path) -> Result<Self> {
115        if let Some(parent) = path.parent() {
116            if !parent.as_os_str().is_empty() {
117                std::fs::create_dir_all(parent)?;
118                #[cfg(unix)]
119                {
120                    use std::os::unix::fs::PermissionsExt;
121                    // Best-effort 0o700 on the parent (P1.4 perms target).
122                    // We don't surface a chmod failure as a hard error: the
123                    // dir was created successfully, refusing to start over a
124                    // chmod refusal would be obnoxious. Logged via tracing
125                    // by nauka_core::process when the helper is used.
126                    let _ =
127                        std::fs::set_permissions(parent, std::fs::Permissions::from_mode(0o700));
128                }
129            }
130        }
131
132        // Acquire the SurrealKV datastore with retry-on-lock-contention.
133        //
134        // # Why this exists
135        //
136        // SurrealKV takes an OS-level exclusive flock on `<path>/LOCK`
137        // whenever a `Datastore` is open. Only one process can hold the
138        // flock at a time. In Nauka, that matters because the fabric
139        // layer has concurrent consumers of the same `bootstrap.skv`:
140        //
141        // - Ad-hoc CLI invocations (`nauka hypervisor status`, etc.)
142        // - The `nauka-forge` daemon running reconcile cycles
143        // - The `nauka-announce` listener
144        //
145        // Each of them opens the DB for a short span (read state, do
146        // work, `shutdown().await`) but their windows overlap, so an
147        // open that happens to collide with another process's brief
148        // write window would fail with
149        // `Other("Database at ... LOCK is already locked by another process")`.
150        //
151        // The fix: retry the open on "already locked" with exponential
152        // backoff up to a 5-second deadline, matching the pattern used
153        // by `shutdown` in `wait_for_surrealkv_lock_release`. 5 s is the
154        // same "fast timeout" budget used for health checks — long
155        // enough to ride out a forge reconcile cycle that briefly holds
156        // the lock, short enough that a stuck holder surfaces as a
157        // clear error instead of hanging forever.
158        let path_str = path.to_string_lossy().into_owned();
159        let inner = open_datastore_with_retry(&path_str).await?;
160
161        inner.use_ns(NAUKA_NS).use_db(BOOTSTRAP_DB).await?;
162
163        // Apply the bootstrap schema after switching to the right ns/db.
164        // DEFINE TABLE / FIELD / INDEX are db-scoped, so use_db must
165        // happen first. The schema's IF NOT EXISTS clauses make this
166        // safe to run on every open.
167        inner.query(BOOTSTRAP_SCHEMA).await?.check()?;
168
169        Ok(Self {
170            inner,
171            path: path.to_path_buf(),
172        })
173    }

175    /// Borrow the underlying SurrealDB SDK client.

180    pub fn client(&self) -> &Surreal<Db> {
181        &self.inner
182    }

189    /// Shut down the wrapper, dropping the SDK client and waiting for the
190    /// SurrealDB background router task to finish closing the datastore
191    /// so a subsequent `EmbeddedDb::open` at the same path can succeed
192    /// — and, critically, so any writes committed through this handle
193    /// are durably on disk before we return.
194    ///
195    /// # Why this function exists
196    ///
197    /// The SDK's `Surreal<Db>` router runs on a background tokio task
198    /// that owns an `Arc<Datastore>`. `drop(Surreal<Db>)` is
199    /// fire-and-forget: it closes the route channel, the router task
200    /// sees `route_rx.recv()` return `Err`, breaks out of its loop,
201    /// and **then** runs `Datastore::shutdown().await` which:
202    ///
203    /// 1. Shuts down the commit coordinator and WAL background flusher
204    /// 2. Calls `Tree::close()` which flushes all memtables to SSTables
205    /// 3. Closes the WAL
206    /// 4. Drops the `LockFile`, releasing SurrealKV's OS-level exclusive
207    ///    flock on `<path>/LOCK`
208    ///
209    /// None of this is synchronous with `drop(self.inner)`. A caller that
210    /// does `shutdown().await; open(same_path).await` races the
211    /// background task. Previous versions of this method used
212    /// `tokio::time::sleep(50 ms)` as a heuristic — enough on CI Linux,
213    /// **not** enough on Mac arm64 under contention, where it manifested
214    /// as data loss (`load_async` returning `None` after a round-trip
215    /// reopen, reproducible at 77% with `--test-threads=2`). See
216    /// sifrah/nauka#255 for the investigation.
217    ///
218    /// The deterministic signal that step 4 has happened is the release
219    /// of the OS-level flock on `<path>/LOCK`. We poll for that release
220    /// by trying to acquire the same flock ourselves via
221    /// [`std::fs::File::try_lock`]. When `try_lock` succeeds, the
222    /// previous `Datastore::shutdown()` chain has definitively run to
223    /// completion, which implies every prior commit is durable and the
224    /// next open can proceed without a lock-held error.
225    ///
226    /// # Cost
227    ///
228    /// Fast path: a single `yield_now` + `try_lock` (~sub-ms). Slow path:
229    /// exponential backoff from 1 ms to a 50 ms ceiling, up to a hard
230    /// 5-second deadline after which the call returns
231    /// [`StateError::Database`]. 5 s is generous relative to the
232    /// ~10–100 ms that a real `Datastore::shutdown()` takes even on a
233    /// contended runtime, and still well under the "fast timeout" Nauka
234    /// convention for health checks.
235    pub async fn shutdown(self) -> Result<()> {
236        // Explicit drop of the inner client. The compiler would do this
237        // anyway when `self` goes out of scope, but naming the step
238        // makes the handoff to the router task visible.
239        let path = self.path.clone();
240        drop(self.inner);
241
242        wait_for_surrealkv_lock_release(&path).await
243    }
244}

246/// Open the SurrealKV datastore at `path_str`, retrying on flock
247/// contention until a 5-second deadline elapses.
248///
249/// See the rationale block in [`EmbeddedDb::open`] for the full context.
250/// The short version: SurrealKV holds a process-exclusive flock while a
251/// datastore is open, and Nauka's CLI / forge daemon / announce listener
252/// all touch the same `bootstrap.skv`, so brief overlaps are normal. We
253/// retry the open rather than surface a hard error for every such
254/// overlap.
255async fn open_datastore_with_retry(path_str: &str) -> Result<Surreal<Db>> {
256    /// Upper bound on the total wait. Same "fast timeout" budget as
257    /// `shutdown` (see `wait_for_surrealkv_lock_release`). Generous
258    /// relative to the real fast-path cost of a forge reconcile cycle
259    /// (~10–100 ms of flock-held time) but short enough that a truly
260    /// stuck holder surfaces as a clear error instead of a hang.
261    const MAX_WAIT: Duration = Duration::from_secs(5);
262    /// First retry after an initial miss.
263    const MIN_BACKOFF: Duration = Duration::from_millis(5);
264    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
265    const MAX_BACKOFF: Duration = Duration::from_millis(50);
266
267    let deadline = Instant::now() + MAX_WAIT;
268    let mut backoff = MIN_BACKOFF;
269
270    loop {
271        match Surreal::new::<SurrealKv>(path_str).await {
272            Ok(db) => return Ok(db),
273            Err(err) => {
274                // SurrealKV surfaces flock contention as an "other"
275                // backend error with a message containing the literal
276                // path to `<datastore>/LOCK`. There's no structured
277                // variant we can match on, so fall back to a substring
278                // check. If the message shape changes in a future
279                // surrealkv release the test
280                // `open_retries_on_flock_contention` will flag it.
281                let msg = err.to_string();
282                let is_lock_contention = msg.contains("already locked");
283
284                if !is_lock_contention || Instant::now() >= deadline {
285                    return Err(err.into());
286                }
287
288                tokio::time::sleep(backoff).await;
289                backoff = (backoff * 2).min(MAX_BACKOFF);
290            }
291        }
292    }
293}

295/// Poll until SurrealKV's OS-level exclusive flock on `<path>/LOCK` is
296/// released by the previous `Datastore`, or until the 5-second deadline
297/// elapses.
298///
299/// See [`EmbeddedDb::shutdown`] for the full rationale. The short version:
300/// the SurrealDB SDK's `Surreal<Db>` router runs its `kvs.shutdown()`
301/// chain asynchronously after `drop(Surreal<Db>)`, and the release of
302/// the `LockFile` (surrealkv 0.21 / `src/lockfile.rs`) is the last step
303/// of that chain. Successfully acquiring the same flock from here proves
304/// the chain has run to completion.
305async fn wait_for_surrealkv_lock_release(path: &Path) -> Result<()> {
306    /// Upper bound on the total wait. Generous relative to the actual
307    /// `Datastore::shutdown()` cost (10–100 ms typical, even on contended
308    /// runtimes) but well under the "fast timeout" Nauka convention.
309    const MAX_WAIT: Duration = Duration::from_secs(5);
310    /// First retry after the fast-path `try_lock` miss.
311    const MIN_BACKOFF: Duration = Duration::from_millis(1);
312    /// Backoff ceiling. Caps worst-case polling pressure at ~20 Hz.
313    const MAX_BACKOFF: Duration = Duration::from_millis(50);
314
315    let lock_path = path.join("LOCK");
316    let deadline = Instant::now() + MAX_WAIT;
317    let mut backoff = MIN_BACKOFF;
318
319    loop {
320        // Give any already-runnable task (the router exit handler) a
321        // chance to make progress before we poll. Cheap in the common
322        // case where the router task is already parked.
323        tokio::task::yield_now().await;
324
325        // If the LOCK file doesn't exist, the datastore was never
326        // initialised far enough to create it (e.g. `open` failed
327        // upstream of the LSM tree build). There is nothing to wait
328        // for — treat as already released.
329        if !lock_path.exists() {
330            return Ok(());
331        }
332
333        // Try to acquire the OS-level exclusive flock that SurrealKV
334        // was holding on the same file. When `try_lock` returns
335        // `Ok(())`, the previous datastore's `kvs.shutdown()` chain
336        // has definitively run to completion and dropped its
337        // `LockFile`; release immediately so a subsequent open() at
338        // the same path can acquire it.
339        //
340        // ENOENT / other `open` errors between the `exists()` check
341        // above and this call are benign races: if the error persists
342        // the deadline check below will surface it, otherwise the next
343        // iteration picks up the real state.
344        if let Ok(file) = OpenOptions::new().read(true).write(true).open(&lock_path) {
345            match file.try_lock() {
346                Ok(()) => {
347                    // Success: release our probe lock and return. The
348                    // explicit `unlock()` isn't strictly necessary
349                    // (dropping `file` closes the fd and releases the
350                    // flock) but makes the intent readable.
351                    let _ = file.unlock();
352                    return Ok(());
353                }
354                Err(TryLockError::WouldBlock) => {
355                    // Router task hasn't reached `LockFile::drop` yet.
356                    // Fall through to the backoff sleep.
357                }
358                Err(TryLockError::Error(io_err)) => {
359                    // Real I/O error (not lock contention). Surface it
360                    // immediately rather than looping against a broken
361                    // filesystem.
362                    return Err(io_err.into());
363                }
364            }
365        }
366
367        if Instant::now() >= deadline {
368            return Err(StateError::Database(format!(
369                "EmbeddedDb::shutdown timed out after {}s waiting for \
370                 SurrealKV LOCK release at {}",
371                MAX_WAIT.as_secs(),
372                lock_path.display(),
373            )));
374        }
375
376        tokio::time::sleep(backoff).await;
377        backoff = (backoff * 2).min(MAX_BACKOFF);
378    }
379}

599    /// P1.5 — Concurrent reads from clones of the same client.
600    ///
601    /// `EmbeddedDb` is `Clone`, and the underlying `Surreal<Db>` shares
602    /// state via Arc internally. Multiple readers from clones should see
603    /// the same value without races or panics. Spawns 10 tasks reading
604    /// the same record concurrently and asserts they all observe the
605    /// expected value.
606    #[tokio::test]
607    async fn concurrent_reads_from_clones() {
608        let dir = tempfile::tempdir().expect("tempdir");
609        let db = EmbeddedDb::open(&dir.path().join("concurrent.skv"))
610            .await
611            .expect("open");
612
613        // Seed a single shared record.
614        let _: Option<Item> = db
615            .client()
616            .create(("items", "shared"))
617            .content(Item {
618                name: "concurrent".into(),
619                count: 42,
620            })
621            .await
622            .expect("seed");
623
624        // Spawn concurrent readers.
625        let mut handles = Vec::with_capacity(10);
626        for i in 0..10u32 {
627            let db_clone = db.clone();
628            handles.push(tokio::spawn(async move {
629                let item: Option<Item> = db_clone
630                    .client()
631                    .select(("items", "shared"))
632                    .await
633                    .expect("concurrent select");
634                (i, item.expect("missing on concurrent reader"))
635            }));
636        }
637
638        for h in handles {
639            let (i, item) = h.await.expect("join");
640            assert_eq!(
641                item.count, 42,
642                "concurrent reader {i} observed wrong value: {item:?}"
643            );
644        }
645
646        db.shutdown().await.expect("shutdown");
647    }

649    /// P1.11 — `EmbeddedDb::open` retries on SurrealKV flock contention.
650    ///
651    /// Opens the same datastore twice in sequence. The first open is
652    /// then shut down (which releases the SurrealKV flock), and we
653    /// kick off a parallel open *before* `shutdown` returns. The
654    /// parallel open must not fail immediately with "already locked":
655    /// the retry loop in `open_datastore_with_retry` should re-try
656    /// until the router task's `kvs.shutdown()` chain has released
657    /// the flock, then succeed.
658    ///
659    /// We deliberately DO NOT use `shutdown` on the second handle after
660    /// `drop(first)` — the shutdown's probe-lock would race the second
661    /// open's flock acquisition and make the test flaky. Only the
662    /// second handle is explicitly shut down at the end.
663    #[tokio::test]
664    async fn open_retries_on_flock_contention() {
665        let dir = tempfile::tempdir().expect("tempdir");
666        let path = dir.path().join("contention.skv");
667
668        // First handle — held briefly, then dropped.
669        let db_a = EmbeddedDb::open(&path).await.expect("open #1");
670
671        // Spawn the second open before db_a is gone; it should spin in
672        // the retry loop until db_a's router releases the flock.
673        let path_clone = path.clone();
674        let joiner = tokio::spawn(async move {
675            EmbeddedDb::open(&path_clone)
676                .await
677                .expect("second open should succeed after retry")
678        });
679
680        // Let the second open spin on the retry loop a bit, then drop
681        // db_a (without calling `shutdown` to avoid racing the second
682        // open on the probe-lock — see the doc comment above).
683        tokio::time::sleep(Duration::from_millis(100)).await;
684        drop(db_a);
685
686        let db_b = joiner.await.expect("join second open");
687        db_b.shutdown().await.expect("shutdown #2");
688    }

718    /// P1.7 — `EmbeddedDb::open` applies the bootstrap schema automatically.
719    ///
720    /// Opens a fresh database (no manual schema application) and asserts
721    /// that the four bootstrap tables (`mesh`, `hypervisor`, `peer`,
722    /// `wg_key`) are reachable. With the schema NOT auto-applied, a
723    /// `select` against a never-touched table returns SurrealDB's
724    /// `NotFound` error (verified by `multi_table_isolation` in P1.5).
725    /// With the schema auto-applied, the table is defined-but-empty and
726    /// `select` returns an empty Vec.
727    #[tokio::test]
728    async fn open_applies_bootstrap_schema_automatically() {
729        let dir = tempfile::tempdir().expect("tempdir");
730        let db = EmbeddedDb::open(&dir.path().join("auto_schema.skv"))
731            .await
732            .expect("open should also apply the schema");
733
734        // Each schema-defined table must be reachable on a fresh DB.
735        // We use surrealdb::types::Value to avoid forcing a SurrealValue
736        // derive on a strongly-typed mirror of every table.
737        for table in &["mesh", "hypervisor", "peer", "wg_key"] {
738            let rows: Vec<surrealdb::types::Value> =
739                db.client().select(*table).await.unwrap_or_else(|e| {
740                    panic!("select on {table} should succeed (schema not applied?): {e}")
741                });
742            assert!(
743                rows.is_empty(),
744                "table {table} should be empty on a fresh DB but contained {} rows",
745                rows.len()
746            );
747        }
748
749        // The schema is also idempotent: a second open at the same path
750        // (after explicit shutdown) must succeed and the tables must
751        // still be defined.
752        let path = db.path().to_path_buf();
753        db.shutdown().await.expect("shutdown");
754
755        let db2 = EmbeddedDb::open(&path)
756            .await
757            .expect("reopening an already-initialised DB should succeed");
758        for table in &["mesh", "hypervisor", "peer", "wg_key"] {
759            let rows: Vec<surrealdb::types::Value> =
760                db2.client().select(*table).await.unwrap_or_else(|e| {
761                    panic!("post-reopen select on {table} should succeed: {e}")
762                });
763            assert!(
764                rows.is_empty(),
765                "table {table} should still be empty after reopen"
766            );
767        }
768        db2.shutdown().await.expect("shutdown #2");
769    }
