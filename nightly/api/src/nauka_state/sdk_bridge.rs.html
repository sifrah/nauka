1//! Shared SurrealDB SDK → Rust bridge helpers used by every cluster
2//! store in the workspace.
3//!
4//! The pattern was established by
5//! `nauka_hypervisor::fabric::state::FabricState::save` during P1.10:
6//! write via an explicit `CREATE … SET …` statement, read via
7//! `SELECT * FROM …`, and bounce the row through `serde_json::Value`
8//! so the Rust side doesn't need a `surrealdb::types::SurrealValue`
9//! derive cascade over every resource type in the workspace.
10//!
11//! The helpers in this module are the small amount of glue every
12//! store needs for that pattern:
13//!
14//! - [`thing_to_id_string`] extracts the bare `<table>-<ulid>` id
15//!   from the `Thing` value that SurrealDB hands back on `SELECT`.
16//! - [`iso8601_to_epoch`] parses the RFC 3339 `datetime` strings that
17//!   SurrealDB emits back into the `u64` Unix-epoch seconds that
18//!   [`nauka_core::resource::ResourceMeta`] carries.
19//! - [`classify_create_error`] maps a surrealdb-side "unique index
20//!   already contains …" error into a human-friendly
21//!   `"<kind> '<name>' already exists"` flat `anyhow::Error` so every
22//!   store produces the same CLI-facing message shape.
23//!
24//! History: P2.9 (sifrah/nauka#213) introduced inline copies of these
25//! helpers in the org store. P2.10 (sifrah/nauka#214) extracted them
26//! to a private `sdk_bridge` module inside `nauka-org`. P2.12
27//! (sifrah/nauka#216) moved the module here into `nauka-state` — the
28//! lowest crate that holds the [`crate::EmbeddedDb`] wrapper — so
29//! every downstream layer (`nauka-org`, `nauka-network`,
30//! `nauka-compute`, `nauka-forge`, `nauka-hypervisor`) can reuse the
31//! same helpers without drifting into three different copies.
32
33/// Pull the bare id string out of a SurrealDB `Thing` rendered as
34/// `serde_json::Value`.
35///
36/// SurrealDB JSON-encodes a `Thing` as one of several shapes depending
37/// on the SDK version and the variant of the inner id:
38///
39/// 1. `"org:01J…"` or `` "org:`01J…`" `` — a flat string when the id
40///    round-tripped through a SurrealQL response that was already
41///    strings-only. SurrealDB wraps the id in backticks when the raw
42///    id contains any character outside the unquoted-identifier set
43///    (the `-` in our `org-<ulid>` form is the usual culprit).
44/// 2. `{"tb": "org", "id": "01J…"}` — the structured form for a
45///    `String` id.
46/// 3. `{"tb": "org", "id": {"String": "01J…"}}` — the structured form
47///    when SurrealDB tagged the id variant explicitly.
48///
49/// `table_prefix` is the `"<table>:"` string used by Shape 1. For the
50/// org store pass `"org:"`, for the project store `"project:"`, and so
51/// on.
52///
53/// Always returns the bare `<table>-<ulid>` id (without the `tb:`
54/// prefix or any wrapping backticks) because that's what the rest of
55/// Nauka has been carrying around as `ResourceMeta::id` since day one.
56pub fn thing_to_id_string(table_prefix: &str, value: &serde_json::Value) -> String {
57    let trim_backticks = |s: &str| s.trim_start_matches('`').trim_end_matches('`').to_string();
58
59    if let Some(s) = value.as_str() {
60        // Shape 1: flat "org:01J…" string. Strip the "<table>:" prefix
61        // if present so the result matches the legacy `<kind>-<ulid>`
62        // form, then strip any wrapping backticks SurrealDB added
63        // because the id contains a hyphen.
64        let without_prefix = s.strip_prefix(table_prefix).unwrap_or(s);
65        return trim_backticks(without_prefix);
66    }
67    if let Some(obj) = value.as_object() {
68        // Shapes 2 and 3 carry the id under the `id` key. Shape 3
69        // wraps the inner id under `String`.
70        if let Some(inner) = obj.get("id") {
71            if let Some(s) = inner.as_str() {
72                return trim_backticks(s);
73            }
74            if let Some(inner_obj) = inner.as_object() {
75                if let Some(s) = inner_obj.get("String").and_then(|v| v.as_str()) {
76                    return trim_backticks(s);
77                }
78            }
79        }
80    }
81    // Fallback: dump the JSON form so the caller at least sees what
82    // SurrealDB actually returned. The `{}` Display for serde_json::Value
83    // is the JSON encoding.
84    value.to_string()
85}

87/// Parse an ISO 8601 / RFC 3339 datetime back into Unix-epoch seconds.
88///
89/// SurrealDB renders `datetime` values as strings like
90/// `2024-01-02T03:04:05.123456Z`; we round down to whole seconds
91/// because [`nauka_core::resource::ResourceMeta::created_at`] is a
92/// `u64` with second granularity. Any parse failure (e.g. an
93/// unexpected timezone offset) returns `0` so the caller still gets a
94/// value back — the alternative would be to fail a whole list/get path
95/// on a single malformed row, which is strictly worse for operators
96/// trying to clean up bad data.
97pub fn iso8601_to_epoch(s: &str) -> u64 {
98    // Stripped-down parser: YYYY-MM-DDTHH:MM:SS… — accepts a trailing
99    // `Z`, an optional fractional seconds part, and an optional
100    // timezone offset (which we ignore — the only writers are the
101    // org-layer stores, and they always emit `Z`).
102    let bytes = s.as_bytes();
103    if bytes.len() < 19 {
104        return 0;
105    }
106    let year: i64 = std::str::from_utf8(&bytes[0..4])
107        .ok()
108        .and_then(|s| s.parse().ok())
109        .unwrap_or(0);
110    let month: i64 = std::str::from_utf8(&bytes[5..7])
111        .ok()
112        .and_then(|s| s.parse().ok())
113        .unwrap_or(0);
114    let day: i64 = std::str::from_utf8(&bytes[8..10])
115        .ok()
116        .and_then(|s| s.parse().ok())
117        .unwrap_or(0);
118    let hour: i64 = std::str::from_utf8(&bytes[11..13])
119        .ok()
120        .and_then(|s| s.parse().ok())
121        .unwrap_or(0);
122    let minute: i64 = std::str::from_utf8(&bytes[14..16])
123        .ok()
124        .and_then(|s| s.parse().ok())
125        .unwrap_or(0);
126    let second: i64 = std::str::from_utf8(&bytes[17..19])
127        .ok()
128        .and_then(|s| s.parse().ok())
129        .unwrap_or(0);
130    let days = days_from_civil(year, month, day);
131    let total = days * 86400 + hour * 3600 + minute * 60 + second;
132    if total < 0 {
133        0
134    } else {
135        total as u64
136    }
137}

139/// Howard Hinnant's "days from civil" algorithm — converts a
140/// `(year, month, day)` triple to a count of days since 1970-01-01.
141///
142/// Lifted directly from the public-domain reference implementation
143/// (<https://howardhinnant.github.io/date_algorithms.html#days_from_civil>).
144/// We use it instead of a `chrono` / `time` dependency to keep
145/// `nauka-org` build-time small and to mirror the existing date math
146/// already in `nauka_core::resource::api_response::days_to_date`, which
147/// goes the other direction.
148fn days_from_civil(y: i64, m: i64, d: i64) -> i64 {
149    let y = if m <= 2 { y - 1 } else { y };
150    let era = if y >= 0 { y } else { y - 399 } / 400;
151    let yoe = y - era * 400;
152    let doy = (153 * (if m > 2 { m - 3 } else { m + 9 }) + 2) / 5 + d - 1;
153    let doe = yoe * 365 + yoe / 4 - yoe / 100 + doy;
154    era * 146097 + doe - 719468
155}

157/// Classify a SurrealDB error message from a `CREATE` into a
158/// human-friendly flat `String` message.
159///
160/// Each cluster-state store has a unique index on its natural key
161/// (e.g. `org_name`, `project_org_name`) that surfaces duplicate
162/// inserts as `surrealdb::Error::already_exists`, whose `Display`
163/// rendering includes the phrase "already contains" (the substring
164/// is stable across surrealdb-types releases and is what the
165/// `From<surrealdb::Error>` impl in [`crate::lib`] documents as the
166/// `is_already_exists()` signal). We key on that substring rather
167/// than pulling in a direct `surrealdb` dependency at this layer, so
168/// duplicate-name conflicts still surface as the same
169/// `<kind> '<name>' already exists` wording every store returned
170/// before the SurrealDB migration — CLI error-message tests
171/// downstream keep passing without edits.
172///
173/// `kind` is the resource noun used in the message (`"org"`,
174/// `"project"`, `"environment"`, `"vpc"`, `"subnet"`, …). `name` is
175/// the human-readable name that was rejected. Everything else
176/// collapses to the underlying error text verbatim.
177///
178/// Callers wrap the returned `String` in whatever error type they
179/// use — typically `anyhow::anyhow!(classify_create_error(...))`.
180/// The helper deliberately returns `String` rather than
181/// `anyhow::Error` so that `nauka-state` does not need to pull in
182/// `anyhow` just to format a message.
183pub fn classify_create_error(kind: &str, name: &str, err_msg: &str) -> String {
184    let lowered = err_msg.to_lowercase();
185    if lowered.contains("already contains")
186        || lowered.contains("already exists")
187        || lowered.contains("duplicate")
188    {
189        format!("{kind} '{name}' already exists")
190    } else {
191        err_msg.to_string()
192    }
193}

195#[cfg(test)]
196mod tests {
197    use super::*;
198    use nauka_core::resource::epoch_to_iso8601;

200    #[test]
201    fn thing_to_id_string_flat_string_with_prefix() {
202        let v = serde_json::Value::String("org:org-01J00000000000000000000000".into());
203        assert_eq!(
204            thing_to_id_string("org:", &v),
205            "org-01J00000000000000000000000"
206        );
207    }
208
209    #[test]
210    fn thing_to_id_string_flat_string_with_backticks() {
211        let v = serde_json::Value::String("org:`org-01J00000000000000000000000`".into());
212        assert_eq!(
213            thing_to_id_string("org:", &v),
214            "org-01J00000000000000000000000"
215        );
216    }
217
218    #[test]
219    fn thing_to_id_string_structured_string_variant() {
220        let v = serde_json::json!({ "tb": "project", "id": "project-01J…" });
221        assert_eq!(thing_to_id_string("project:", &v), "project-01J…");
222    }
223
224    #[test]
225    fn thing_to_id_string_structured_tagged_string() {
226        let v = serde_json::json!({
227            "tb": "project",
228            "id": { "String": "project-01J…" }
229        });
230        assert_eq!(thing_to_id_string("project:", &v), "project-01J…");
231    }
232
233    #[test]
234    fn iso8601_round_trip_matches_epoch() {
235        for &epoch in &[0u64, 1, 86_399, 86_400, 1_700_000_000, 1_775_665_838] {
236            let iso = epoch_to_iso8601(epoch);
237            let back = iso8601_to_epoch(&iso);
238            assert_eq!(back, epoch, "round trip failed for {epoch}: iso={iso}");
239        }
240    }
241
242    #[test]
243    fn iso8601_too_short_returns_zero() {
244        assert_eq!(iso8601_to_epoch("2024"), 0);
245        assert_eq!(iso8601_to_epoch(""), 0);
246    }
247
248    #[test]
249    fn classify_create_error_detects_duplicate() {
250        let msg = classify_create_error(
251            "project",
252            "web",
253            "database contains already contains an entry for `web`",
254        );
255        assert_eq!(msg, "project 'web' already exists");
256    }
257
258    #[test]
259    fn classify_create_error_passes_through_unrelated() {
260        let msg = classify_create_error("project", "web", "disk full");
261        assert_eq!(msg, "disk full");
