1//! Cluster-state SurrealQL schema application.
2//!
3//! ADR 0004 (sifrah/nauka#210) chose Option A: the bootstrap node applies
4//! the cluster `.surql` schemas to TiKV exactly once, after PD/TiKV come
5//! up and before the storage region is written. Joining nodes do NOT
6//! call this function — they assume the schemas are already in place.
7//!
8//! P2.5 (sifrah/nauka#209) shipped the schema files; this module
9//! consumes them at compile time via `include_str!` so the binary has
10//! no runtime filesystem dependency on the source tree.
11//!
12//! Idempotency: every `DEFINE` in every schema file uses `IF NOT EXISTS`,
13//! so re-running this function against an already-initialised cluster
14//! is a no-op. The test
15//! [`tests::apply_cluster_schemas_is_idempotent`] pins that contract.
16
17use crate::{EmbeddedDb, Result, StateError};
18
19/// Compile-time-embedded copies of every P2.5 cluster schema, keyed by
20/// short name for diagnostics. The order is deliberate: parent tables
21/// come before children, so even a SCHEMAFULL backend that doesn't
22/// support forward references would still apply cleanly.
23///
24/// `include_str!` resolves relative to *this* source file
25/// (`layers/state/src/schema.rs`), so the paths climb out of
26/// `state/src` and back down into each sibling layer's `schemas/`
27/// directory.
28const CLUSTER_SCHEMAS: &[(&str, &str)] = &[
29    ("user", include_str!("../../org/schemas/user.surql")),
30    ("org", include_str!("../../org/schemas/org.surql")),
31    ("project", include_str!("../../org/schemas/project.surql")),
32    ("env", include_str!("../../org/schemas/env.surql")),
33    ("vpc", include_str!("../../network/schemas/vpc.surql")),
34    ("subnet", include_str!("../../network/schemas/subnet.surql")),
35    ("vm", include_str!("../../compute/schemas/vm.surql")),
36];
37
38/// Apply every cluster schema to the given [`EmbeddedDb`] in order.
39///
40/// Idempotent — every `DEFINE` uses `IF NOT EXISTS`, so callers can
41/// invoke this against either a freshly-bootstrapped TiKV cluster or
42/// one that already has the schemas applied. Returns an error on the
43/// first schema that fails to apply, with a message that names the
44/// offending schema for fast triage.
45///
46/// # Errors
47///
48/// Returns [`StateError::Database`] if any schema fails to apply or
49/// if the SurrealDB response's `.check()` surfaces a per-statement
50/// error (e.g. a parse error, an `ASSERT` failure, a SCHEMAFULL
51/// violation). The error message is prefixed with the name of the
52/// offending schema (`"apply vm schema: ..."` or
53/// `"check vm schema: ..."`) so the operator can immediately tell
54/// which file is at fault.
55pub async fn apply_cluster_schemas(db: &EmbeddedDb) -> Result<()> {
56    for (name, schema) in CLUSTER_SCHEMAS {
57        db.client()
58            .query(*schema)
59            .await
60            .map_err(|e| StateError::Database(format!("apply {name} schema: {e}")))?
61            .check()
62            .map_err(|e| StateError::Database(format!("check {name} schema: {e}")))?;
63    }
64    Ok(())
65}

67#[cfg(test)]
68mod tests {
69    use super::*;
70    use crate::EmbeddedDb;
71
72    /// P2.7 — `apply_cluster_schemas` runs cleanly against a fresh
73    /// `EmbeddedDb` (using the SurrealKV backend for the test, since
74    /// we don't have a TiKV cluster in unit tests). Re-running against
75    /// the same handle must also succeed — that's the idempotency
76    /// contract the acceptance criteria calls out as "re-running on an
77    /// existing cluster does not fail".
78    ///
79    /// The schemas define `cluster`-scoped tables that don't collide
80    /// with the bootstrap tables already applied by `EmbeddedDb::open`,
81    /// so we can safely share the same SurrealDB instance here.
82    #[tokio::test]
83    async fn apply_cluster_schemas_is_idempotent() {
84        let dir = tempfile::tempdir().expect("tempdir");
85        let db = EmbeddedDb::open(&dir.path().join("p2_7.skv"))
86            .await
87            .expect("open");
88
89        // First apply — must succeed against an empty db.
90        apply_cluster_schemas(&db)
91            .await
92            .expect("first apply should succeed");
93
94        // Second apply — must be a no-op thanks to `IF NOT EXISTS` on
95        // every DEFINE. Any regression that drops the guard surfaces
96        // here as an "already exists" error.
97        apply_cluster_schemas(&db)
98            .await
99            .expect("second apply (idempotency) should succeed");
100
101        db.shutdown().await.expect("shutdown");
102    }
103}
