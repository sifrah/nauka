1//! Cluster-state SurrealQL schema application.
2//!
3//! ADR 0004 (sifrah/nauka#210) chose Option A: the bootstrap node applies
4//! the cluster `.surql` schemas to TiKV exactly once, after PD/TiKV come
5//! up and before the storage region is written. Joining nodes do NOT
6//! call this function — they assume the schemas are already in place.
7//!
8//! P2.5 (sifrah/nauka#209) shipped the schema files; this module
9//! consumes them at compile time via `include_str!` so the binary has
10//! no runtime filesystem dependency on the source tree.
11//!
12//! Idempotency: every `DEFINE` in every schema file uses `IF NOT EXISTS`,
13//! so re-running this function against an already-initialised cluster
14//! is a no-op. The test
15//! [`tests::apply_cluster_schemas_is_idempotent`] pins that contract.
16
17use crate::{EmbeddedDb, Result, StateError};
18
19/// Compile-time-embedded copies of every P2.5 cluster schema, keyed by
20/// short name for diagnostics. The order is deliberate: parent tables
21/// come before children, so even a SCHEMAFULL backend that doesn't
22/// support forward references would still apply cleanly.
23///
24/// `include_str!` resolves relative to *this* source file
25/// (`layers/state/src/schema.rs`), so the paths climb out of
26/// `state/src` and back down into each sibling layer's `schemas/`
27/// directory.
28/// Table names defined by the cluster schemas, in application order.
29///
30/// Exposed publicly so callers (`nauka hypervisor doctor`, the migration
31/// runner, etc.) can verify that every expected table is present in a
32/// running cluster without having to re-parse the `.surql` files.
33/// Kept in lockstep with [`CLUSTER_SCHEMAS`] by
34/// [`tests::cluster_table_names_matches_schemas`].
35pub const CLUSTER_TABLE_NAMES: &[&str] = &[
36    "user",
37    "org",
38    "project",
39    "env",
40    "vpc",
41    "subnet",
42    "vpc_peering",
43    "natgw",
44    "vm",
45];
46
47const CLUSTER_SCHEMAS: &[(&str, &str)] = &[
48    ("user", include_str!("../../org/schemas/user.surql")),
49    ("org", include_str!("../../org/schemas/org.surql")),
50    ("project", include_str!("../../org/schemas/project.surql")),
51    ("env", include_str!("../../org/schemas/env.surql")),
52    ("vpc", include_str!("../../network/schemas/vpc.surql")),
53    ("subnet", include_str!("../../network/schemas/subnet.surql")),
54    // P2.12 (sifrah/nauka#216): peering + natgw schemas ship alongside
55    // the network layer's SurrealDB-SDK migration so the new stores
56    // write to SCHEMAFULL tables instead of legacy raw-KV catch-alls.
57    (
58        "vpc_peering",
59        include_str!("../../network/schemas/peering.surql"),
60    ),
61    ("natgw", include_str!("../../network/schemas/natgw.surql")),
62    ("vm", include_str!("../../compute/schemas/vm.surql")),
63];

65/// Apply every cluster schema to the given [`EmbeddedDb`] in order.
66///
67/// Idempotent — every `DEFINE` uses `IF NOT EXISTS`, so callers can
68/// invoke this against either a freshly-bootstrapped TiKV cluster or
69/// one that already has the schemas applied. Returns an error on the
70/// first schema that fails to apply, with a message that names the
71/// offending schema for fast triage.
72///
73/// # Errors
74///
75/// Returns [`StateError::Database`] if any schema fails to apply or
76/// if the SurrealDB response's `.check()` surfaces a per-statement
77/// error (e.g. a parse error, an `ASSERT` failure, a SCHEMAFULL
78/// violation). The error message is prefixed with the name of the
79/// offending schema (`"apply vm schema: ..."` or
80/// `"check vm schema: ..."`) so the operator can immediately tell
81/// which file is at fault.
82pub async fn apply_cluster_schemas(db: &EmbeddedDb) -> Result<()> {
83    for (name, schema) in CLUSTER_SCHEMAS {
84        db.client()
85            .query(*schema)
86            .await
87            .map_err(|e| StateError::Database(format!("apply {name} schema: {e}")))?
88            .check()
89            .map_err(|e| StateError::Database(format!("check {name} schema: {e}")))?;
90    }
91    Ok(())
92}

94#[cfg(test)]
95mod tests {
96    use super::*;
97    use crate::EmbeddedDb;
98
99    /// P2.7 — `apply_cluster_schemas` runs cleanly against a fresh
100    /// `EmbeddedDb` (using the SurrealKV backend for the test, since
101    /// we don't have a TiKV cluster in unit tests). Re-running against
102    /// the same handle must also succeed — that's the idempotency
103    /// contract the acceptance criteria calls out as "re-running on an
104    /// existing cluster does not fail".
105    ///
106    /// The schemas define `cluster`-scoped tables that don't collide
107    /// with the bootstrap tables already applied by `EmbeddedDb::open`,
108    /// so we can safely share the same SurrealDB instance here.
109    #[tokio::test]
110    async fn apply_cluster_schemas_is_idempotent() {
111        let dir = tempfile::tempdir().expect("tempdir");
112        let db = EmbeddedDb::open(&dir.path().join("p2_7.skv"))
113            .await
114            .expect("open");
115
116        // First apply — must succeed against an empty db.
117        apply_cluster_schemas(&db)
118            .await
119            .expect("first apply should succeed");
120
121        // Second apply — must be a no-op thanks to `IF NOT EXISTS` on
122        // every DEFINE. Any regression that drops the guard surfaces
123        // here as an "already exists" error.
124        apply_cluster_schemas(&db)
125            .await
126            .expect("second apply (idempotency) should succeed");
127
128        db.shutdown().await.expect("shutdown");
129    }
130
131    /// P2.17 (sifrah/nauka#221) — the public [`CLUSTER_TABLE_NAMES`]
132    /// list must stay in sync with the schemas actually applied. If
133    /// somebody adds a `.surql` file to [`CLUSTER_SCHEMAS`] without
134    /// updating the names list (or vice versa), the doctor's
135    /// schema-presence check would silently miss the new table.
136    #[test]
137    fn cluster_table_names_matches_schemas() {
138        let schema_names: Vec<&str> = CLUSTER_SCHEMAS.iter().map(|(n, _)| *n).collect();
139        assert_eq!(
140            CLUSTER_TABLE_NAMES,
141            &schema_names[..],
142            "CLUSTER_TABLE_NAMES drifted from CLUSTER_SCHEMAS — \
143             update layers/state/src/schema.rs to keep them in lockstep"
144        );
145    }
146}
