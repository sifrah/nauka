1//! Process observer — scan for running VM processes.
2//!
3//! Checks both PID liveness and container runtime state. A VM is only
4//! considered running if the PID file exists, the process is alive, AND
5//! the container runtime reports it as running. This prevents orphaned
6//! processes (e.g. `sleep infinity` surviving after container stop) from
7//! being mistaken for healthy VMs.
8
9use std::path::PathBuf;
10
11const VM_RUN_DIR: &str = "/run/nauka/vms";
12
13/// List running VM IDs on this node.
14pub fn list_vms() -> Vec<String> {
15    let run_dir = PathBuf::from(VM_RUN_DIR);
16    let entries = match std::fs::read_dir(&run_dir) {
17        Ok(e) => e,
18        Err(_) => return vec![],
19    };
20
21    let mut vms = Vec::new();
22    for entry in entries.flatten() {
23        let path = entry.path();
24        if path.is_dir() {
25            let vm_id = entry.file_name().to_string_lossy().to_string();
26            let pid_path = path.join("pid");
27            if let Ok(pid_str) = std::fs::read_to_string(&pid_path) {
28                if let Ok(pid) = pid_str.trim().parse::<i32>() {
29                    let pid_alive = unsafe { libc::kill(pid, 0) == 0 };
30                    if !pid_alive {
31                        continue;
32                    }
33
34                    // PID is alive — check container runtime state.
35                    // If crun says "stopped" the container is dead but the
36                    // process survived as an orphan. Kill it so the
37                    // reconciler can recreate the container cleanly.
38                    match container_status(&vm_id) {
39                        ContainerState::Running => vms.push(vm_id),
40                        ContainerState::Stopped => {
41                            tracing::warn!(
42                                vm_id,
43                                pid,
44                                "container stopped but PID alive — killing orphan"
45                            );
46                            unsafe { libc::kill(pid, libc::SIGKILL) };
47                        }
48                        ContainerState::Unknown => {
49                            // crun state not available (runtime dir cleaned up).
50                            // Trust PID liveness — container may still be fine.
51                            vms.push(vm_id);
52                        }
53                    }
54                }
55            }
56        }
57    }
58    vms
59}
60
61enum ContainerState {
62    Running,
63    Stopped,
64    Unknown,
65}
66
67fn container_status(vm_id: &str) -> ContainerState {
68    let output = std::process::Command::new("crun")
69        .args(["state", vm_id])
70        .stdout(std::process::Stdio::piped())
71        .stderr(std::process::Stdio::null())
72        .output();
73    match output {
74        Ok(o) if o.status.success() => {
75            let state: serde_json::Value = match serde_json::from_slice(&o.stdout) {
76                Ok(v) => v,
77                Err(_) => return ContainerState::Unknown,
78            };
79            match state["status"].as_str() {
80                Some("running") => ContainerState::Running,
81                Some("stopped") | Some("created") => ContainerState::Stopped,
82                _ => ContainerState::Unknown,
83            }
84        }
85        _ => ContainerState::Unknown,
86    }
87}

89/// Get the PID of a running VM by its ID.
90pub fn get_pid(vm_id: &str) -> Option<u32> {
91    let pid_path = PathBuf::from(VM_RUN_DIR).join(vm_id).join("pid");
92    let pid_str = std::fs::read_to_string(&pid_path).ok()?;
93    let pid: i32 = pid_str.trim().parse().ok()?;
94    if unsafe { libc::kill(pid, 0) == 0 } {
95        Some(pid as u32)
96    } else {
97        None
98    }
