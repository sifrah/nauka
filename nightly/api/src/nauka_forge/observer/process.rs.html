1//! Process observer — scan for running VM processes.
2//!
3//! Scans /run/nauka/vms/ for VM directories with PID files.
4//! Checks process liveness with kill(pid, 0).
5
6use std::path::PathBuf;
7
8const VM_RUN_DIR: &str = "/run/nauka/vms";
9
10/// List running VM IDs on this node.
11pub fn list_vms() -> Vec<String> {
12    let run_dir = PathBuf::from(VM_RUN_DIR);
13    let entries = match std::fs::read_dir(&run_dir) {
14        Ok(e) => e,
15        Err(_) => return vec![],
16    };
17
18    let mut vms = Vec::new();
19    for entry in entries.flatten() {
20        let path = entry.path();
21        if path.is_dir() {
22            let vm_id = entry.file_name().to_string_lossy().to_string();
23            let pid_path = path.join("pid");
24            if let Ok(pid_str) = std::fs::read_to_string(&pid_path) {
25                if let Ok(pid) = pid_str.trim().parse::<i32>() {
26                    // kill(pid, 0) checks if process exists without sending a signal
27                    if unsafe { libc::kill(pid, 0) == 0 } {
28                        vms.push(vm_id);
29                    }
30                }
31            }
32        }
33    }
34    vms
35}
36
37/// Get the PID of a running VM by its ID.
38pub fn get_pid(vm_id: &str) -> Option<u32> {
39    let pid_path = PathBuf::from(VM_RUN_DIR).join(vm_id).join("pid");
40    let pid_str = std::fs::read_to_string(&pid_path).ok()?;
41    let pid: i32 = pid_str.trim().parse().ok()?;
42    if unsafe { libc::kill(pid, 0) == 0 } {
43        Some(pid as u32)
44    } else {
45        None
46    }
47}
