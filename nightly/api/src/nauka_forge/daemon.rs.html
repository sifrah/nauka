1//! Forge daemon — the main reconciliation loop.
2
3use nauka_hypervisor::controlplane;
4use nauka_hypervisor::fabric;
5use nauka_state::EmbeddedDb;
6
7use crate::reconciler;
8use crate::types::ReconcileContext;
9
10/// Default reconciliation interval in seconds.
11const RECONCILE_INTERVAL_SECS: u64 = 30;
12
13/// Run the forge daemon (blocking loop).
14pub async fn run() -> anyhow::Result<()> {
15    eprintln!("  Forge starting...");
16
17    let mut cycle: u64 = 0;
18
19    loop {
20        cycle += 1;
21
22        match run_cycle(cycle).await {
23            Ok(_results) => {}
24            Err(e) => {
25                tracing::error!(cycle, error = %e, "reconcile cycle failed");
26            }
27        }
28
29        // Post-reconcile: health checks every 60s (every other cycle).
30        // Runs outside run_cycle so health is reported even when TiKV is down.
31        run_health_if_due(cycle).await;
32
33        tokio::time::sleep(std::time::Duration::from_secs(RECONCILE_INTERVAL_SECS)).await;
34    }
35}
36
37/// Run a single reconciliation cycle (for `nauka forge reconcile`).
38pub async fn run_once() -> anyhow::Result<String> {
39    let results = run_cycle(1).await?;
40
41    let lines: Vec<String> = results.iter().map(|r| format!("  {r}")).collect();
42    Ok(format!("reconciliation complete\n{}", lines.join("\n")))
43}
44
45/// Run health checks if this cycle is eligible.
46///
47/// Loads mesh_ipv6 from local state so it works even when TiKV is down.
48async fn run_health_if_due(cycle: u64) {
49    let local_db = match EmbeddedDb::open_default().await {
50        Ok(db) => db,
51        Err(_) => return,
52    };
53    let state = match fabric::state::FabricState::load(&local_db).await {
54        Ok(Some(s)) => s,
55        _ => return,
56    };
57    // Release the flock before the health helper does its own on-disk
58    // lookups.
59    let _ = local_db.shutdown().await;
60    reconciler::health::run_if_due(cycle, &state.hypervisor.mesh_ipv6).await;
61}
62
63/// Execute one reconciliation cycle.
64async fn run_cycle(cycle: u64) -> anyhow::Result<Vec<crate::types::ReconcileResult>> {
65    // Pre-flight: recover PD/TiKV if they crashed after a data wipe.
66    // This must run before connect() since PD/TiKV being down would fail the connection.
67    {
68        let local_db = EmbeddedDb::open_default().await?;
69        let pre_state = fabric::state::FabricState::load(&local_db)
70            .await
71            .ok()
72            .flatten();
73        // Drop our handle now so `controlplane::connect()` (which opens its
74        // own EmbeddedDb to read PD endpoints) doesn't contend on the flock.
75        let _ = local_db.shutdown().await;
76
77        if let Some(state) = pre_state {
78            let peer_ipv6s: Vec<std::net::Ipv6Addr> =
79                state.peers.peers.iter().map(|p| p.mesh_ipv6).collect();
80
81            // Phase 1: if local PD is healthy but has dead peers, restore quorum
82            if controlplane::service::recover_pd_quorum(&state.hypervisor.mesh_ipv6) {
83                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
84            }
85
86            // Phase 2: if local PD is down, rejoin cluster via a healthy peer
87            if controlplane::service::recover_stale_pd_member(
88                &state.hypervisor.mesh_ipv6,
89                &state.hypervisor.name,
90                &peer_ipv6s,
91            ) {
92                // Give PD time to rejoin the cluster
93                tokio::time::sleep(std::time::Duration::from_secs(10)).await;
94            }
95
96            if controlplane::service::recover_stale_store(&state.hypervisor.mesh_ipv6) {
97                // Give TiKV time to start and register with PD
98                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
99            }
100
101            // Phase 4: remove zombie PD members (unhealthy >5 min)
102            reconciler::pd::cleanup_zombie_members(&state.hypervisor.mesh_ipv6);
103
104            // Phase 5: sync TiKV PD endpoints with fabric state
105            if reconciler::tikv_endpoints::sync_pd_endpoints(
106                &state.hypervisor.mesh_ipv6,
107                &peer_ipv6s,
108            ) {
109                // Give TiKV time to restart and reconnect to PD
110                tokio::time::sleep(std::time::Duration::from_secs(5)).await;
111            }
112        }
113    }
114
115    // Connect to TiKV
116    let db = controlplane::connect().await?;
117
118    // Load this node's identity
119    let local_db = EmbeddedDb::open_default().await?;
120    let state = fabric::state::FabricState::load(&local_db)
121        .await
122        .map_err(|e| anyhow::anyhow!("{e}"))?
123        .ok_or_else(|| anyhow::anyhow!("not initialized"))?;
124    let _ = local_db.shutdown().await;
125
126    let runtime = if state.hypervisor.runtime == "container" {
127        nauka_compute::runtime::RuntimeMode::Container
128    } else {
129        nauka_compute::runtime::RuntimeMode::Kvm
130    };
131
132    // Collect all IDs this node is known by:
133    // - Its own HypervisorId (hv-...)
134    // - Its name (used by peers to identify us)
135    // - Any NodeId that peers assigned to us (node-...)
136    // This ensures VMs scheduled from any node match correctly.
137    let node_ids = vec![
138        state.hypervisor.id.as_str().to_string(),
139        state.hypervisor.name.clone(),
140    ];
141    // Check what IDs peers use for us by looking at peer lists on remote nodes
142    // We can't do that directly, but peers reference us by NodeId.
143    // The scheduler on remote nodes uses the peer's NodeId.
144    // So we need to know our own NodeId as seen by peers.
145    // This is stored in the peer list of OTHER nodes — we can't access it.
146    // Workaround: also check by mesh_ipv6 match, or just match all VMs
147    // assigned to any ID containing our name.
148
149    let ctx = ReconcileContext {
150        db,
151        hypervisor_id: state.hypervisor.id.as_str().to_string(),
152        node_ids,
153        node_name: state.hypervisor.name.clone(),
154        mesh_ipv6: state.hypervisor.mesh_ipv6,
155        runtime,
156        cycle,
157    };
158
159    // Ensure replication is configured correctly.
160    // On every cycle, check if max-replicas matches the number of active stores (capped at 3).
161    let pd_url = format!(
162        "http://[{}]:{}",
163        state.hypervisor.mesh_ipv6,
164        controlplane::PD_CLIENT_PORT
165    );
166    let active_stores = controlplane::service::count_active_stores(&pd_url);
167    if active_stores >= 3 {
168        let _ = controlplane::service::adjust_max_replicas(&pd_url, 3);
169    }
170
171    tracing::debug!(
172        cycle,
173        node = ctx.node_name.as_str(),
174        "reconcile cycle start"
175    );
176
177    let results = reconciler::run_all(&ctx).await;
178
179    let total_actions: usize = results
180        .iter()
181        .map(|r| r.created + r.deleted + r.updated)
182        .sum();
183    let total_failures: usize = results.iter().map(|r| r.failed).sum();
184
185    if total_actions > 0 || total_failures > 0 {
186        tracing::info!(
187            cycle,
188            actions = total_actions,
189            failures = total_failures,
190            "reconcile cycle done"
191        );
192    } else {
193        tracing::debug!(cycle, "reconcile cycle done — in sync");
194    }
195
196    Ok(results)
197}
