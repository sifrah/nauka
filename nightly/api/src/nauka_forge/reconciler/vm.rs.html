1//! VM reconciler — ensures VMs assigned to this node have TAPs and processes.
2
3use nauka_compute::runtime::{
4    gvisor::GVisorRuntime, kvm::KvmRuntime, Runtime, RuntimeMode, VmRunConfig,
5};
6use nauka_compute::vm::provision;
7use nauka_compute::vm::types::VmState;
8use nauka_network::vpc::provision as vpc_provision;

10use crate::types::{ReconcileContext, ReconcileResult};

12/// Pure logic: returns true if this VM is assigned to one of the local node IDs.
13fn is_local_vm(hypervisor_id: Option<&str>, node_ids: &[String]) -> bool {
14    hypervisor_id
15        .map(|hid| node_ids.iter().any(|nid| nid == hid))
16        .unwrap_or(false)
17}
18
19/// Pure logic: returns true if this VM's state means it should have a running
20/// process on the node (i.e., it is either `Pending` or `Running`).
21fn should_vm_exist(state: &VmState) -> bool {
22    matches!(state, VmState::Pending | VmState::Running)
23}
24
25/// Pure logic: given the current VM state and whether the process is actually
26/// running, returns the action the reconciler should take.
27#[derive(Debug, PartialEq)]
28pub(crate) enum VmAction {
29    /// Process is running but state is `Pending` — correct to `Running`,
30    /// then run health checks.
31    CorrectStateAndHealthCheck,
32    /// Process is running and state is already `Running` — just health check.
33    HealthCheck,
34    /// Process is not running — start it.
35    Start,
36}
37
38/// Pure logic: decide what action to take for a VM that should exist.
39pub(crate) fn decide_vm_action(state: &VmState, process_running: bool) -> VmAction {
40    if process_running {
41        if *state == VmState::Pending {
42            VmAction::CorrectStateAndHealthCheck
43        } else {
44            VmAction::HealthCheck
45        }
46    } else {
47        VmAction::Start
48    }
49}
50
51/// Pure logic: returns the IDs of processes that are orphaned — they are running
52/// but no longer have a corresponding VM that should exist on this node.
53fn find_orphaned_processes(actual_processes: &[String], needed_ids: &[&str]) -> Vec<String> {
54    actual_processes
55        .iter()
56        .filter(|vm_id| !needed_ids.contains(&vm_id.as_str()))
57        .cloned()
58        .collect()
59}
60
61/// Pure logic: returns true if the expected network interface name is present
62/// in the list of actual interfaces.
63fn is_interface_present(expected: &str, actual: &[String]) -> bool {
64    actual.iter().any(|v| v == expected)
65}

67pub struct VmReconciler;
68
69impl VmReconciler {
70    /// Get the right runtime for this node.
71    fn runtime(ctx: &ReconcileContext) -> Box<dyn Runtime> {
72        if ctx.runtime == RuntimeMode::Kvm {
73            Box::new(KvmRuntime)
74        } else {
75            Box::new(GVisorRuntime)
76        }
77    }
78}

80#[async_trait::async_trait]
81impl super::Reconciler for VmReconciler {
82    fn name(&self) -> &str {
83        "vm"
84    }
85
86    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
87        let mut result = ReconcileResult::new("vm");
88        let rt = Self::runtime(ctx);
89
90        // 1. Get VMs assigned to this node
91        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
92        let all_vms = vm_store.list(None, None, None).await?;
93        let local_vms: Vec<_> = all_vms
94            .iter()
95            .filter(|vm| is_local_vm(vm.hypervisor_id.as_deref(), &ctx.node_ids))
96            .collect();
97
98        // VMs that should be running (pending = needs starting, running = should be alive).
99        // Skip orphaned VMs whose org no longer exists (cascading orphan).
100        // P2.9 (sifrah/nauka#213) migrated `OrgStore` to take an
101        // `EmbeddedDb` directly; we hand it the cluster-DB wrapper's
102        // internal SurrealDB handle via `.embedded().clone()`.
103        let org_store = nauka_org::store::OrgStore::new(ctx.db.embedded().clone());
104        let mut should_exist: Vec<_> = Vec::new();
105        for vm in &local_vms {
106            if !should_vm_exist(&vm.state) {
107                continue;
108            }
109            if org_store.get(vm.org_id.as_str()).await?.is_none() {
110                tracing::warn!(
111                    vm_id = vm.meta.id.as_str(),
112                    org_id = vm.org_id.as_str(),
113                    "skipping orphaned VM (org deleted)"
114                );
115                continue;
116            }
117            should_exist.push(vm);
118        }
119
120        result.desired = should_exist.len();
121
122        // 2. Check actual state
123        let actual_processes = crate::observer::process::list_vms();
124        result.actual = actual_processes.len();
125
126        // 3. Create network interface + start processes for missing VMs
127        let use_veth = ctx.runtime == RuntimeMode::Container;
128        for vm in &should_exist {
129            let bridge = vpc_provision::bridge_name(vm.vpc_id.as_str());
130
131            // Ensure network interface exists (veth for containers, TAP for KVM)
132            if use_veth {
133                let expected = provision::veth_host_name(&vm.meta.id);
134                let actual = provision::list_veths();
135                let veth_was_missing = !is_interface_present(&expected, &actual);
136                if veth_was_missing {
137                    if let Err(e) = provision::ensure_veth(&vm.meta.id, &bridge) {
138                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to create veth");
139                        result.failed += 1;
140                        result.errors.push(format!("veth {}: {e}", vm.meta.id));
141                        continue;
142                    }
143
144                    // If container is running, inject the new veth into its netns
145                    if actual_processes.contains(&vm.meta.id) {
146                        if let Some(pid) = crate::observer::process::get_pid(&vm.meta.id) {
147                            let ip = vm.private_ip.as_deref().unwrap_or("0.0.0.0");
148                            let mac =
149                                nauka_network::vpc::provision::mac_from_ip(ip).unwrap_or_default();
150                            let subnet_store =
151                                nauka_network::vpc::subnet::store::SubnetStore::new(ctx.db.clone());
152                            let gateway = subnet_store
153                                .get(vm.subnet_id.as_str(), None, None)
154                                .await?
155                                .map(|s| s.gateway.clone())
156                                .unwrap_or_else(|| "0.0.0.0".to_string());
157                            let vpc_store =
158                                nauka_network::vpc::store::VpcStore::new(ctx.db.clone());
159                            let vpc_cidr = vpc_store
160                                .get(vm.vpc_id.as_str(), None)
161                                .await?
162                                .map(|v| v.cidr);
163                            match provision::setup_container_net(
164                                &vm.meta.id,
165                                pid,
166                                ip,
167                                &gateway,
168                                &mac,
169                                vpc_cidr.as_deref(),
170                            ) {
171                                Ok(()) => {
172                                    tracing::info!(
173                                        vm_id = vm.meta.id.as_str(),
174                                        "veth injected into container netns"
175                                    );
176                                    result.updated += 1;
177                                }
178                                Err(e) => {
179                                    tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to inject veth");
180                                    result.failed += 1;
181                                    result
182                                        .errors
183                                        .push(format!("veth inject {}: {e}", vm.meta.id));
184                                }
185                            }
186                        }
187                    }
188                }
189            } else {
190                let expected = provision::tap_name(&vm.meta.id);
191                let actual = provision::list_taps();
192                if !is_interface_present(&expected, &actual) {
193                    if let Err(e) = provision::ensure_tap(&vm.meta.id, &bridge) {
194                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to create TAP");
195                        result.failed += 1;
196                        result.errors.push(format!("tap {}: {e}", vm.meta.id));
197                        continue;
198                    }
199                }
200            }
201
202            let action = decide_vm_action(&vm.state, actual_processes.contains(&vm.meta.id));
203
204            // If process is already running, check health and correct state
205            if matches!(
206                action,
207                VmAction::CorrectStateAndHealthCheck | VmAction::HealthCheck
208            ) {
209                if action == VmAction::CorrectStateAndHealthCheck {
210                    if let Err(e) = vm_store
211                        .update_state(&vm.meta.id, VmState::Running, None, None, None)
212                        .await
213                    {
214                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to correct VM state");
215                    } else {
216                        tracing::info!(
217                            vm_id = vm.meta.id.as_str(),
218                            "corrected state: pending -> running"
219                        );
220                        result.updated += 1;
221                    }
222                }
223
224                // Validate OCI config.json — regenerate if corrupted
225                if use_veth {
226                    let config_path = format!("/run/nauka/vms/{}/bundle/config.json", vm.meta.id);
227                    let config_valid = std::fs::read_to_string(&config_path)
228                        .ok()
229                        .and_then(|s| serde_json::from_str::<serde_json::Value>(&s).ok())
230                        .and_then(|v| v["process"]["args"].as_array().cloned())
231                        .is_some();
232
233                    if !config_valid {
234                        let rootfs_dir = format!("/run/nauka/vms/{}/bundle/rootfs", vm.meta.id);
235                        let has_tini = std::path::Path::new(&rootfs_dir)
236                            .join("usr/bin/tini")
237                            .exists();
238                        let subnet_store =
239                            nauka_network::vpc::subnet::store::SubnetStore::new(ctx.db.clone());
240                        let subnet = subnet_store.get(vm.subnet_id.as_str(), None, None).await?;
241                        let (gateway, cidr) = match &subnet {
242                            Some(s) => (s.gateway.clone(), s.cidr.clone()),
243                            None => ("0.0.0.0".to_string(), "0.0.0.0/0".to_string()),
244                        };
245                        let vpc_store = nauka_network::vpc::store::VpcStore::new(ctx.db.clone());
246                        let vpc_cidr = vpc_store
247                            .get(vm.vpc_id.as_str(), None)
248                            .await?
249                            .map(|v| v.cidr);
250                        let run_config = VmRunConfig {
251                            vm_id: vm.meta.id.clone(),
252                            vm_name: vm.meta.name.clone(),
253                            vcpus: vm.vcpus,
254                            memory_mb: vm.memory_mb,
255                            disk_gb: vm.disk_gb,
256                            image: vm.image.clone(),
257                            tap_name: provision::veth_guest_name(&vm.meta.id),
258                            private_ip: vm.private_ip.clone().unwrap_or_default(),
259                            gateway,
260                            subnet_cidr: cidr,
261                            vpc_cidr,
262                        };
263                        let oci = nauka_compute::runtime::gvisor::generate_oci_config(
264                            &run_config,
265                            has_tini,
266                        );
267                        if let Err(e) = std::fs::write(&config_path, oci) {
268                            tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to regenerate config.json");
269                        } else {
270                            tracing::info!(
271                                vm_id = vm.meta.id.as_str(),
272                                "regenerated corrupted config.json"
273                            );
274                            result.updated += 1;
275                        }
276                    }
277                }
278
279                // Verify base image exists — re-pull if deleted
280                if !nauka_compute::image::registry::exists(&vm.image) {
281                    tracing::warn!(
282                        vm_id = vm.meta.id.as_str(),
283                        image = vm.image.as_str(),
284                        "base image missing — pulling"
285                    );
286                    match nauka_compute::image::registry::pull(&vm.image).await {
287                        Ok(_) => {
288                            tracing::info!(
289                                vm_id = vm.meta.id.as_str(),
290                                image = vm.image.as_str(),
291                                "base image restored"
292                            );
293                            result.updated += 1;
294                        }
295                        Err(e) => {
296                            tracing::error!(vm_id = vm.meta.id.as_str(), image = vm.image.as_str(), error = %e, "failed to pull base image");
297                            result.failed += 1;
298                            result.errors.push(format!("image {}: {e}", vm.image));
299                        }
300                    }
301                }
302
303                // Health check: ensure sshd is alive inside containers
304                if use_veth && !crate::observer::health::is_sshd_alive(&vm.meta.id) {
305                    match crate::observer::health::restart_sshd(&vm.meta.id) {
306                        Ok(()) => {
307                            tracing::info!(vm_id = vm.meta.id.as_str(), "restarted sshd");
308                            result.updated += 1;
309                        }
310                        Err(e) => {
311                            tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to restart sshd");
312                            result.failed += 1;
313                            result.errors.push(format!("sshd {}: {e}", vm.meta.id));
314                        }
315                    }
316                }
317
318                continue;
319            }
320
321            // Ensure process is running — clean up stale state first
322            if action == VmAction::Start {
323                // Kill orphaned processes and clean stale container state
324                // before recreating. This handles the case where the container
325                // died but `sleep infinity` or tini survived as an orphan.
326                let _ = rt.stop(&vm.meta.id);
327
328                // Remove stale veth pair — the guest side may be stuck in the
329                // dead container's netns and can't be reused.
330                if use_veth {
331                    let _ = provision::remove_veth(&vm.meta.id);
332                    if let Err(e) = provision::ensure_veth(&vm.meta.id, &bridge) {
333                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to recreate veth");
334                        result.failed += 1;
335                        result.errors.push(format!("veth {}: {e}", vm.meta.id));
336                        continue;
337                    }
338                }
339
340                let subnet_store =
341                    nauka_network::vpc::subnet::store::SubnetStore::new(ctx.db.clone());
342                let subnet = subnet_store.get(vm.subnet_id.as_str(), None, None).await?;
343
344                let (gateway, cidr) = match &subnet {
345                    Some(s) => (s.gateway.clone(), s.cidr.clone()),
346                    None => ("0.0.0.0".to_string(), "0.0.0.0/0".to_string()),
347                };
348
349                // Resolve VPC CIDR for DNS64/NAT64 setup
350                let vpc_store = nauka_network::vpc::store::VpcStore::new(ctx.db.clone());
351                let vpc_cidr = vpc_store
352                    .get(vm.vpc_id.as_str(), None)
353                    .await?
354                    .map(|v| v.cidr);
355
356                let config = VmRunConfig {
357                    vm_id: vm.meta.id.clone(),
358                    vm_name: vm.meta.name.clone(),
359                    vcpus: vm.vcpus,
360                    memory_mb: vm.memory_mb,
361                    disk_gb: vm.disk_gb,
362                    image: vm.image.clone(),
363                    tap_name: if use_veth {
364                        provision::veth_guest_name(&vm.meta.id)
365                    } else {
366                        provision::tap_name(&vm.meta.id)
367                    },
368                    private_ip: vm.private_ip.clone().unwrap_or_default(),
369                    gateway,
370                    subnet_cidr: cidr,
371                    vpc_cidr,
372                };
373
374                match rt.start(&config) {
375                    Ok(pid) => {
376                        tracing::info!(
377                            vm_id = vm.meta.id.as_str(),
378                            pid,
379                            runtime = %ctx.runtime,
380                            "VM process started"
381                        );
382                        if vm.state == VmState::Pending {
383                            if let Err(e) = vm_store
384                                .update_state(&vm.meta.id, VmState::Running, None, None, None)
385                                .await
386                            {
387                                tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to update VM state");
388                            }
389                        }
390                        result.created += 1;
391                    }
392                    Err(e) => {
393                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to start VM");
394                        result.failed += 1;
395                        result.errors.push(format!("start {}: {e}", vm.meta.id));
396                    }
397                }
398            }
399        }
400
401        // 4. Remove orphaned processes + network interfaces
402        let needed_ids: Vec<&str> = should_exist.iter().map(|vm| vm.meta.id.as_str()).collect();
403        let orphaned = find_orphaned_processes(&actual_processes, &needed_ids);
404        for vm_id in &orphaned {
405            tracing::info!(vm_id = vm_id.as_str(), "stopping orphaned VM");
406            let _ = rt.stop(vm_id);
407            if use_veth {
408                let _ = provision::remove_veth(vm_id);
409            } else {
410                let _ = provision::remove_tap(vm_id);
411            }
412            result.deleted += 1;
413        }
414
415        Ok(result)
416    }
417}
