1//! VM reconciler — ensures TAP interfaces exist for VMs on this node.
2
3use nauka_compute::vm::provision;
4use nauka_compute::vm::types::VmState;
5use nauka_network::vpc::provision as vpc_provision;
6
7use crate::types::{ReconcileContext, ReconcileResult};
8
9pub struct VmReconciler;
10
11#[async_trait::async_trait]
12impl super::Reconciler for VmReconciler {
13    fn name(&self) -> &str {
14        "vm"
15    }
16
17    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
18        let mut result = ReconcileResult::new("vm");
19
20        // 1. Get VMs assigned to this node
21        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
22        let all_vms = vm_store.list(None, None, None).await?;
23        let local_vms: Vec<_> = all_vms
24            .iter()
25            .filter(|vm| vm.hypervisor_id.as_deref() == Some(&ctx.hypervisor_id))
26            .collect();
27
28        // VMs that should have TAPs (pending or running — anything assigned to this node)
29        let need_tap: Vec<_> = local_vms
30            .iter()
31            .filter(|vm| vm.state == VmState::Pending || vm.state == VmState::Running)
32            .collect();
33
34        result.desired = need_tap.len();
35
36        // 2. Check actual TAPs
37        let actual_taps = provision::list_taps();
38        result.actual = actual_taps.len();
39
40        // 3. Create missing TAPs
41        for vm in &need_tap {
42            let expected_tap = provision::tap_name(&vm.meta.id);
43            if !actual_taps.iter().any(|t| t == &expected_tap) {
44                let bridge = vpc_provision::bridge_name(vm.vpc_id.as_str());
45                match provision::ensure_tap(&vm.meta.id, &bridge) {
46                    Ok(_) => result.created += 1,
47                    Err(e) => {
48                        tracing::error!(
49                            vm_id = vm.meta.id.as_str(),
50                            error = %e,
51                            "failed to create TAP"
52                        );
53                        result.failed += 1;
54                        result.errors.push(format!("vm {}: {e}", vm.meta.id));
55                    }
56                }
57            }
58        }
59
60        // 4. Remove orphaned TAPs
61        let needed_tap_names: Vec<String> = need_tap
62            .iter()
63            .map(|vm| provision::tap_name(&vm.meta.id))
64            .collect();
65        for tap in &actual_taps {
66            if !needed_tap_names.contains(tap) {
67                tracing::info!(tap, "removing orphaned TAP");
68                let _ = std::process::Command::new("ip")
69                    .args(["link", "del", tap])
70                    .status();
71                result.deleted += 1;
72            }
73        }
74
75        Ok(result)
76    }
77}
