1//! VPC reconciler — ensures VXLAN bridges exist for VPCs with local VMs.
2
3use crate::types::{ReconcileContext, ReconcileResult};
4
5pub struct VpcReconciler;
6
7#[async_trait::async_trait]
8impl super::Reconciler for VpcReconciler {
9    fn name(&self) -> &str {
10        "vpc"
11    }
12
13    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
14        let mut result = ReconcileResult::new("vpc");
15
16        // 1. Find which VMs are on this node
17        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
18        let all_vms = vm_store.list(None, None, None).await?;
19        let local_vms: Vec<_> = all_vms
20            .iter()
21            .filter(|vm| vm.hypervisor_id.as_deref() == Some(&ctx.hypervisor_id))
22            .collect();
23
24        // 2. Collect unique VPC IDs needed on this node
25        let mut needed_vpc_ids: Vec<String> = local_vms
26            .iter()
27            .map(|vm| vm.vpc_id.as_str().to_string())
28            .collect();
29        needed_vpc_ids.sort();
30        needed_vpc_ids.dedup();
31
32        result.desired = needed_vpc_ids.len();
33
34        // 3. Check actual state (stub: nothing exists yet)
35        let actual_bridges = crate::observer::network::list_bridges();
36        result.actual = actual_bridges.len();
37
38        // 4. Diff and apply
39        for vpc_id in &needed_vpc_ids {
40            if !actual_bridges.iter().any(|b| b == vpc_id) {
41                tracing::info!(vpc_id, "would create bridge for VPC");
42                result.created += 1;
43            }
44        }
45
46        Ok(result)
47    }
48}
