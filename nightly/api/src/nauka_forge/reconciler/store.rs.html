1//! TiKV store reconciler — offlines stores whose fabric peer is unreachable.
2//!
3//! When a fabric peer has been unreachable for more than 60 seconds, the TiKV
4//! store running on that peer is moved offline via the PD API. This prevents
5//! stuck regions caused by PD waiting for a store that will never return.
6
7use std::process::Command;
8use std::time::{SystemTime, UNIX_EPOCH};
9
10use nauka_hypervisor::controlplane;
11use nauka_hypervisor::fabric;
12use nauka_state::LocalDb;
13
14use crate::types::{ReconcileContext, ReconcileResult};
15
16/// How long a peer must be unreachable before we offline its store (seconds).
17const UNREACHABLE_THRESHOLD_SECS: u64 = 60;
18
19pub struct StoreReconciler;
20
21#[async_trait::async_trait]
22impl super::Reconciler for StoreReconciler {
23    fn name(&self) -> &str {
24        "store"
25    }
26
27    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
28        let mut result = ReconcileResult::new("store");
29
30        // Load fabric state to get peer list
31        let local_db = LocalDb::open("hypervisor")?;
32        let state = match fabric::state::FabricState::load(&local_db)
33            .map_err(|e| anyhow::anyhow!("{e}"))?
34        {
35            Some(s) => s,
36            None => return Ok(result), // not initialized
37        };
38
39        let now = SystemTime::now()
40            .duration_since(UNIX_EPOCH)
41            .unwrap_or_default()
42            .as_secs();
43
44        // Find peers that have been unreachable for >60s
45        let unreachable_peers: Vec<_> = state
46            .peers
47            .peers
48            .iter()
49            .filter(|p| p.status == fabric::peer::PeerStatus::Unreachable)
50            .filter(|p| {
51                // Use last_handshake as the reference point for how long the peer
52                // has been unreachable. If last_handshake is 0, use added_at instead.
53                let ref_time = if p.last_handshake > 0 {
54                    p.last_handshake
55                } else {
56                    p.added_at
57                };
58                now.saturating_sub(ref_time) > UNREACHABLE_THRESHOLD_SECS
59            })
60            .collect();
61
62        if unreachable_peers.is_empty() {
63            return Ok(result);
64        }
65
66        // Build the set of TiKV addresses for unreachable peers
67        let unreachable_addrs: Vec<String> = unreachable_peers
68            .iter()
69            .map(|p| format!("[{}]:{}", p.mesh_ipv6, controlplane::TIKV_PORT))
70            .collect();
71
72        // Get store list from PD
73        let pd_url = format!(
74            "http://[{}]:{}",
75            ctx.mesh_ipv6,
76            controlplane::PD_CLIENT_PORT,
77        );
78        let stores_url = format!("{pd_url}/pd/api/v1/stores");
79        let output = match Command::new("curl")
80            .args(["-sf", "--max-time", "5", &stores_url])
81            .output()
82        {
83            Ok(o) if o.status.success() => o,
84            _ => return Ok(result), // PD not reachable, skip
85        };
86
87        let body: serde_json::Value = match serde_json::from_slice(&output.stdout) {
88            Ok(v) => v,
89            Err(_) => return Ok(result),
90        };
91
92        let stores = match body["stores"].as_array() {
93            Some(s) => s,
94            None => return Ok(result),
95        };
96
97        result.desired = stores
98            .iter()
99            .filter(|s| s["store"]["state_name"].as_str() == Some("Up"))
100            .count();
101
102        // For each "Up" store whose address matches an unreachable peer, offline it
103        for store in stores {
104            let addr = store["store"]["address"].as_str().unwrap_or("");
105            let state_name = store["store"]["state_name"].as_str().unwrap_or("");
106            let store_id = store["store"]["id"].as_u64().unwrap_or(0);
107
108            if state_name != "Up" || store_id == 0 {
109                continue;
110            }
111
112            if !unreachable_addrs.iter().any(|a| a == addr) {
113                continue;
114            }
115
116            // Find the peer name for logging
117            let peer_name = unreachable_peers
118                .iter()
119                .find(|p| format!("[{}]:{}", p.mesh_ipv6, controlplane::TIKV_PORT) == addr)
120                .map(|p| p.name.as_str())
121                .unwrap_or("unknown");
122
123            tracing::warn!(
124                store_id,
125                addr,
126                peer = peer_name,
127                "offlining TiKV store — peer unreachable >60s"
128            );
129
130            let delete_url = format!("{pd_url}/pd/api/v1/store/{store_id}");
131            match Command::new("curl")
132                .args(["-sf", "-X", "DELETE", "--max-time", "5", &delete_url])
133                .output()
134            {
135                Ok(o) if o.status.success() => {
136                    result.deleted += 1;
137                }
138                _ => {
139                    result.failed += 1;
140                    result
141                        .errors
142                        .push(format!("failed to offline store {store_id} ({addr})"));
143                }
144            }
145        }
146
147        Ok(result)
148    }
149}
