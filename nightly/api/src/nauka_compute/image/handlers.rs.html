1//! Image resource definition + handler.
2//!
3//! CLI: nauka vm image pull/list/delete
4
5use std::future::Future;
6use std::pin::Pin;
7
8use nauka_core::resource::*;
9
10use super::registry;

12pub fn resource_def() -> ResourceDef {
13    ResourceDef::build("image", "Manage OS images")
14        .plural("images")
15        .action("pull", "Pull an image from the registry")
16        .op(|op| {
17            op.with_arg(OperationArg::required(
18                "name",
19                FieldDef::string("name", "Image name (e.g., ubuntu-24.04)"),
20            ))
21            .with_output(OutputKind::Resource)
22            .with_example("nauka vm image pull ubuntu-24.04")
23        })
24        .action("list", "List locally available images")
25        .op(|op| op.with_output(OutputKind::ResourceList))
26        .action("catalog", "List images available in the registry")
27        .op(|op| op.with_output(OutputKind::ResourceList))
28        .action("delete", "Delete a local image")
29        .op(|op| {
30            op.with_arg(OperationArg::required(
31                "name",
32                FieldDef::string("name", "Image name to delete"),
33            ))
34        })
35        .column("NAME", "name")
36        .column("TYPE", "type")
37        .column("SIZE", "size")
38        .column("ARCH", "arch")
39        .column("LOCAL", "local")
40        .empty_message("No images found. Pull one with: nauka vm image pull ubuntu-24.04")
41        .detail_section(
42            None,
43            vec![
44                DetailField::new("Name", "name"),
45                DetailField::new("Size", "size"),
46            ],
47        )
48        .done()
49}

51pub fn handler() -> HandlerFn {
52    Box::new(
53        |req: OperationRequest| -> Pin<
54            Box<dyn Future<Output = anyhow::Result<OperationResponse>> + Send>,
55        > {
56            Box::pin(async move {
57                match req.operation.as_str() {
58                    "pull" => {
59                        let name = req
60                            .fields
61                            .get("name")
62                            .ok_or_else(|| anyhow::anyhow!("--name is required"))?
63                            .clone();
64                        let size = registry::pull(&name).await?;
65                        Ok(OperationResponse::Resource(serde_json::json!({
66                            "name": name,
67                            "status": "ready",
68                            "size_bytes": size,
69                            "size": format_size(size),
70                        })))
71                    }
72                    "list" => {
73                        let arch = match std::env::consts::ARCH {
74                            "x86_64" => "amd64",
75                            "aarch64" => "arm64",
76                            other => other,
77                        };
78                        let image_type = if std::path::Path::new("/dev/kvm").exists() {
79                            "vm"
80                        } else {
81                            "container"
82                        };
83                        let images = registry::list();
84                        let items: Vec<serde_json::Value> = images
85                            .iter()
86                            .map(|(name, size)| {
87                                serde_json::json!({
88                                    "name": name,
89                                    "type": image_type,
90                                    "size": format_size(*size),
91                                    "arch": arch,
92                                    "local": "✓",
93                                })
94                            })
95                            .collect();
96                        Ok(OperationResponse::ResourceList(items))
97                    }
98                    "catalog" => {
99                        let entries = registry::catalog().await?;
100                        let items: Vec<serde_json::Value> = entries
101                            .iter()
102                            .map(|e| {
103                                serde_json::json!({
104                                    "name": e.name,
105                                    "type": e.image_type,
106                                    "size": format_size(e.size),
107                                    "arch": e.arch,
108                                    "local": if e.local { "✓" } else { "✗" },
109                                })
110                            })
111                            .collect();
112                        Ok(OperationResponse::ResourceList(items))
113                    }
114                    "delete" => {
115                        let name = req
116                            .fields
117                            .get("name")
118                            .ok_or_else(|| anyhow::anyhow!("--name is required"))?
119                            .clone();
120                        registry::delete(&name)?;
121                        Ok(OperationResponse::Message(format!(
122                            "image '{name}' deleted."
123                        )))
124                    }
125                    other => Ok(OperationResponse::Message(format!("unknown: {other}"))),
126                }
127            })
128        },
129    )
130}
131
132pub fn registration() -> ResourceRegistration {
133    ResourceRegistration {
134        def: resource_def(),
135        handler: handler(),
136        children: vec![],
137    }
138}

140fn format_size(bytes: u64) -> String {
141    if bytes >= 1_073_741_824 {
142        format!("{:.1} GB", bytes as f64 / 1_073_741_824.0)
143    } else if bytes >= 1_048_576 {
144        format!("{:.0} MB", bytes as f64 / 1_048_576.0)
145    } else {
146        format!("{} KB", bytes / 1024)
147    }
