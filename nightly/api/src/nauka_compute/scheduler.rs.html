1//! VM scheduler — assigns VMs to hypervisors.
2//!
3//! Minimal implementation: picks the current node (or filters by zone).
4//! Future: capacity-based scoring, anti-affinity, taints/tolerations.
5
6use nauka_hypervisor::fabric;
7use nauka_state::EmbeddedDb;
8
9/// A candidate hypervisor for VM placement.
10#[derive(Debug)]
11#[allow(dead_code)]
12struct Candidate {
13    id: String,
14    name: String,
15    region: String,
16    zone: String,
17}
18
19/// Schedule a VM to a hypervisor. Returns the hypervisor ID.
20///
21/// Reads the local fabric state to build the list of available nodes.
22/// Filters by region/zone if specified. Picks the first match.
23pub async fn schedule(region: &str, zone: &str) -> anyhow::Result<String> {
24    let db = EmbeddedDb::open_default().await?;
25
26    let state = fabric::state::FabricState::load(&db)
27        .await
28        .map_err(|e| anyhow::anyhow!("{e}"))?
29        .ok_or_else(|| anyhow::anyhow!("cluster not initialized"))?;
30    // Release the flock before the caller does anything else on-disk.
31    db.shutdown().await?;
32
33    // Build candidate list: self + peers
34    // Use node NAME as the canonical identifier — it's the only ID
35    // that all nodes can resolve (peers know each other by name).
36    let mut candidates = vec![Candidate {
37        id: state.hypervisor.name.clone(),
38        name: state.hypervisor.name.clone(),
39        region: state.hypervisor.region.clone(),
40        zone: state.hypervisor.zone.clone(),
41    }];
42
43    for peer in &state.peers.peers {
44        if peer.status == fabric::peer::PeerStatus::Active {
45            // Use peer name as the identifier — each node knows its own name
46            // and the Forge matches VMs by name. Node IDs (node-...) are not
47            // known by the remote node itself, only by this node's peer list.
48            candidates.push(Candidate {
49                id: peer.name.clone(),
50                name: peer.name.clone(),
51                region: peer.region.clone(),
52                zone: peer.zone.clone(),
53            });
54        }
55    }
56
57    // Filter by zone (if not "default")
58    if zone != "default" {
59        let filtered: Vec<&Candidate> = candidates.iter().filter(|c| c.zone == zone).collect();
60        if filtered.is_empty() {
61            anyhow::bail!("no hypervisor available in zone '{zone}'");
62        }
63        // Pick first match (future: least loaded)
64        return Ok(filtered[0].id.clone());
65    }
66
67    // Filter by region (if not "default")
68    if region != "default" {
69        let filtered: Vec<&Candidate> = candidates.iter().filter(|c| c.region == region).collect();
70        if filtered.is_empty() {
71            anyhow::bail!("no hypervisor available in region '{region}'");
72        }
73        return Ok(filtered[0].id.clone());
74    }
75
76    // No filter — pick first (self)
77    Ok(candidates[0].id.clone())
78}
