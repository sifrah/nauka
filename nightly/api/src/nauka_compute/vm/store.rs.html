1//! VM persistence on the SurrealDB-backed cluster store.
2//!
3//! P2.13 (sifrah/nauka#217) migrated this module from the legacy
4//! raw-KV cluster path to the native SurrealDB SDK on top of
5//! [`EmbeddedDb`], completing the compute-layer migration. Every
6//! method reaches the SDK via `self.db.client()` and writes/reads
7//! against the SCHEMAFULL `vm` table defined in
8//! `layers/compute/schemas/vm.surql` (applied by
9//! `nauka_state::apply_cluster_schemas` at bootstrap per ADR 0004).
10//!
11//! The legacy `vm` / `vm-idx` / `_reg_v2` sidecar namespaces are
12//! gone: the schema's composite unique index on `(env, name)` is
13//! now the source of truth for "have we seen this (env, name) pair
14//! before?", and `SELECT * FROM vm` walks every row directly.
15//!
16//! The composite unique index on `(subnet, private_ip)` closes the
17//! allocation-collision race that the pre-P2.13 store enforced only
18//! at the IPAM helper level.
19//!
20//! State transitions continue to use the validated-FSM pattern from
21//! the pre-P2.13 version: `update_state` reads the row, checks the
22//! `(from, to)` pair against the allowed transitions, then writes
23//! back via `UPDATE type::record($tbl, $id) MERGE …`.
24

52    /// Create a new VM in the given environment.
53    ///
54    /// Resolves the full `(org → project → env, vpc → subnet)` path,
55    /// allocates a private IP via the subnet's IPAM helper, then
56    /// writes the row through `CREATE type::record($tbl, $id) SET …`.
57    /// The schema's composite unique index on `(env, name)` rejects
58    /// duplicates.
59    ///
60    /// `hypervisor_id` is passed in by the caller (typically the CLI
61    /// handler) so the store stays decoupled from
62    /// `crate::scheduler::schedule`, which needs live fabric state
63    /// and can't run in unit tests. The CLI path resolves the id via
64    /// the scheduler immediately before calling `create`; tests pass
65    /// a placeholder id directly.
66    #[allow(clippy::too_many_arguments)]
67    pub async fn create(
68        &self,
69        name: &str,
70        org_name: &str,
71        project_name: &str,
72        env_name: &str,
73        vpc_name: &str,
74        subnet_name: &str,
75        vcpus: u32,
76        memory_mb: u32,
77        disk_gb: u32,
78        image: &str,
79        region: &str,
80        zone: &str,
81        hypervisor_id: String,
82    ) -> anyhow::Result<Vm> {
83        // Resolve org → project → env. Every upstream store already
84        // takes an `EmbeddedDb` directly after P2.9-P2.11.
85        let org_store = nauka_org::store::OrgStore::new(self.db.clone());
86        let org = org_store
87            .get(org_name)
88            .await?
89            .ok_or_else(|| anyhow::anyhow!("org '{org_name}' not found"))?;
90
91        let proj_store = nauka_org::project::store::ProjectStore::new(self.db.clone());
92        let project = proj_store
93            .get(project_name, Some(org_name))
94            .await?
95            .ok_or_else(|| {
96                anyhow::anyhow!("project '{project_name}' not found in org '{org_name}'")
97            })?;
98
99        let env_store = nauka_org::project::env::store::EnvStore::new(self.db.clone());
100        let env = env_store
101            .get(env_name, Some(project_name), Some(org_name))
102            .await?
103            .ok_or_else(|| {
104                anyhow::anyhow!("environment '{env_name}' not found in project '{project_name}'")
105            })?;
106
107        // Resolve vpc → subnet. Both stores take an `EmbeddedDb`
108        // directly after P2.12.
109        let vpc_store = nauka_network::vpc::store::VpcStore::new(self.db.clone());
110        let vpc = vpc_store
111            .get(vpc_name, Some(org_name))
112            .await?
113            .ok_or_else(|| anyhow::anyhow!("vpc '{vpc_name}' not found"))?;
114
115        let sub_store = nauka_network::vpc::subnet::store::SubnetStore::new(self.db.clone());
116        let subnet = sub_store
117            .get(subnet_name, Some(vpc_name), Some(org_name))
118            .await?
119            .ok_or_else(|| {
120                anyhow::anyhow!("subnet '{subnet_name}' not found in vpc '{vpc_name}'")
121            })?;
122
123        // Generate VM ID first — needed for IPAM allocation.
124        let vm_id = VmId::generate().to_string();
125
126        // Allocate private IP from subnet IPAM. P2.12 migrated the
127        // helper to take an `EmbeddedDb` directly.
128        let private_ip = nauka_network::vpc::subnet::ipam::allocate(
129            &self.db,
130            &subnet.meta.id,
131            &subnet.cidr,
132            &subnet.gateway,
133            &vm_id,
134        )
135        .await?;
136
137        let vm = Vm {
138            meta: ResourceMeta::new(vm_id.clone(), name),
139            org_id: org.meta.id.clone().into(),
140            org_name: org.meta.name.clone(),
141            project_id: project.meta.id.clone().into(),
142            project_name: project.meta.name.clone(),
143            env_id: env.meta.id.clone().into(),
144            env_name: env.meta.name.clone(),
145            vpc_id: vpc.meta.id.clone().into(),
146            vpc_name: vpc.meta.name.clone(),
147            subnet_id: subnet.meta.id.clone().into(),
148            subnet_name: subnet.meta.name.clone(),
149            vcpus,
150            memory_mb,
151            disk_gb,
152            image: image.to_string(),
153            region: region.to_string(),
154            zone: zone.to_string(),
155            private_ip: Some(private_ip),
156            hypervisor_id: Some(hypervisor_id),
157            state: VmState::Pending,
158        };
159
160        // Write the row through the SDK. If the DB write fails we
161        // release the IPAM allocation so the address slot doesn't
162        // leak — same rollback pattern as `NatGwStore::create`.
163        if let Err(e) = self.insert(&vm).await {
164            let msg = e.to_string().to_lowercase();
165            let _ = nauka_network::vpc::subnet::ipam::release(
166                &self.db,
167                vm.subnet_id.as_str(),
168                &vm.meta.id,
169            )
170            .await;
171
172            if msg.contains("already contains")
173                || msg.contains("already exists")
174                || msg.contains("duplicate")
175            {
176                return Err(anyhow::anyhow!(
177                    "vm '{name}' already exists in environment '{env_name}'"
178                ));
179            }
180            return Err(e);
181        }
182
183        Ok(vm)
184    }

186    /// Write the row via SurrealQL `CREATE`. Extracted from `create`
187    /// so the rollback path there can cleanly handle the error.
188    async fn insert(&self, vm: &Vm) -> anyhow::Result<()> {
189        let created_at_iso = epoch_to_iso8601(vm.meta.created_at);
190        let updated_at_iso = epoch_to_iso8601(vm.meta.updated_at);
191        let labels_json = serde_json::to_value(&vm.meta.labels)
192            .map_err(|e| anyhow::anyhow!("serialise labels: {e}"))?;
193
194        let response = self
195            .db
196            .client()
197            .query(
198                "CREATE type::record($tbl, $id) SET \
199                 name = $name, \
200                 status = $status, \
201                 labels = $labels, \
202                 org = $org, \
203                 org_name = $org_name, \
204                 project = $project, \
205                 project_name = $project_name, \
206                 env = $env, \
207                 env_name = $env_name, \
208                 vpc = $vpc, \
209                 vpc_name = $vpc_name, \
210                 subnet = $subnet, \
211                 subnet_name = $subnet_name, \
212                 vcpus = $vcpus, \
213                 memory_mb = $memory_mb, \
214                 disk_gb = $disk_gb, \
215                 image = $image, \
216                 region = $region, \
217                 zone = $zone, \
218                 private_ip = $private_ip, \
219                 hypervisor = $hypervisor, \
220                 state = $state, \
221                 created_at = <datetime>$created_at, \
222                 updated_at = <datetime>$updated_at",
223            )
224            .bind(("tbl", VM_TABLE))
225            .bind(("id", vm.meta.id.clone()))
226            .bind(("name", vm.meta.name.clone()))
227            .bind(("status", vm.meta.status.clone()))
228            .bind(("labels", labels_json))
229            .bind(("org", vm.org_id.as_str().to_string()))
230            .bind(("org_name", vm.org_name.clone()))
231            .bind(("project", vm.project_id.as_str().to_string()))
232            .bind(("project_name", vm.project_name.clone()))
233            .bind(("env", vm.env_id.as_str().to_string()))
234            .bind(("env_name", vm.env_name.clone()))
235            .bind(("vpc", vm.vpc_id.as_str().to_string()))
236            .bind(("vpc_name", vm.vpc_name.clone()))
237            .bind(("subnet", vm.subnet_id.as_str().to_string()))
238            .bind(("subnet_name", vm.subnet_name.clone()))
239            .bind(("vcpus", vm.vcpus as i64))
240            .bind(("memory_mb", vm.memory_mb as i64))
241            .bind(("disk_gb", vm.disk_gb as i64))
242            .bind(("image", vm.image.clone()))
243            .bind(("region", vm.region.clone()))
244            .bind(("zone", vm.zone.clone()))
245            .bind(("private_ip", vm.private_ip.clone()))
246            .bind(("hypervisor", vm.hypervisor_id.clone()))
247            .bind(("state", vm.state.to_string()))
248            .bind(("created_at", created_at_iso))
249            .bind(("updated_at", updated_at_iso))
250            .await
251            .map_err(|e| anyhow::anyhow!("{e}"))?;
252        response.check().map_err(|e| anyhow::anyhow!("{e}"))?;
253        Ok(())
254    }

256    /// Look up a VM by id (when the input looks like a [`VmId`]) or
257    /// by name (otherwise, requires the full `(org, project, env)`
258    /// scope).
259    pub async fn get(
260        &self,
261        name_or_id: &str,
262        org_name: Option<&str>,
263        project_name: Option<&str>,
264        env_name: Option<&str>,
265    ) -> anyhow::Result<Option<Vm>> {
266        if VmId::looks_like_id(name_or_id) {
267            return self.get_by_id(name_or_id).await;
268        }
269
270        let org_name =
271            org_name.ok_or_else(|| anyhow::anyhow!("--org required to resolve VM by name"))?;
272        let project_name = project_name
273            .ok_or_else(|| anyhow::anyhow!("--project required to resolve VM by name"))?;
274        let env_name =
275            env_name.ok_or_else(|| anyhow::anyhow!("--env required to resolve VM by name"))?;
276
277        // Resolve the owning env so we can query by its record id
278        // via the composite `(env, name)` unique index.
279        let env_store = nauka_org::project::env::store::EnvStore::new(self.db.clone());
280        let env = env_store
281            .get(env_name, Some(project_name), Some(org_name))
282            .await?
283            .ok_or_else(|| anyhow::anyhow!("environment '{env_name}' not found"))?;
284
285        self.get_by_env_and_name(&env.meta.id, name_or_id).await
286    }

288    async fn get_by_id(&self, id: &str) -> anyhow::Result<Option<Vm>> {
289        let mut response = self
290            .db
291            .client()
292            .query("SELECT * FROM type::record($tbl, $id)")
293            .bind(("tbl", VM_TABLE))
294            .bind(("id", id.to_string()))
295            .await
296            .map_err(|e| anyhow::anyhow!("{e}"))?;
297        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
298        decode_first(raw)
299    }

301    async fn get_by_env_and_name(&self, env_id: &str, name: &str) -> anyhow::Result<Option<Vm>> {
302        let mut response = self

305            .query("SELECT * FROM vm WHERE env = $env AND name = $name LIMIT 1")

311        decode_first(raw)

314    /// List every VM, with optional `(org, project, env)` filters
315    /// applied progressively by the application layer after the
316    /// `SELECT *`.
317    pub async fn list(
318        &self,
319        org_name: Option<&str>,
320        project_name: Option<&str>,
321        env_name: Option<&str>,
322    ) -> anyhow::Result<Vec<Vm>> {
323        let mut response = self
324            .db
325            .client()
326            .query("SELECT * FROM vm")
327            .await
328            .map_err(|e| anyhow::anyhow!("{e}"))?;
329        let raw: Vec<serde_json::Value> = response.take(0).map_err(|e| anyhow::anyhow!("{e}"))?;
330        let mut vms: Vec<Vm> = raw
331            .into_iter()
332            .filter_map(|v| serde_json::from_value::<VmRow>(v).ok().map(VmRow::into_vm))
333            .collect();
334
335        if let Some(org) = org_name {
336            vms.retain(|v| v.org_name == org || v.org_id.as_str() == org);
337        }
338        if let Some(proj) = project_name {
339            vms.retain(|v| v.project_name == proj || v.project_id.as_str() == proj);
340        }
341        if let Some(env) = env_name {
342            vms.retain(|v| v.env_name == env || v.env_id.as_str() == env);
343        }
344
345        Ok(vms)
346    }

348    /// Apply a state transition to a VM.
349    ///
350    /// Validates the `(from, to)` pair against the allowed VM FSM
351    /// edges, then rewrites `state` + `updated_at` atomically via
352    /// `UPDATE … MERGE`.
353    pub async fn update_state(
354        &self,
355        name_or_id: &str,
356        new_state: VmState,
357        org_name: Option<&str>,
358        project_name: Option<&str>,
359        env_name: Option<&str>,
360    ) -> anyhow::Result<Vm> {
361        let mut vm = self
362            .get(name_or_id, org_name, project_name, env_name)
363            .await?
364            .ok_or_else(|| anyhow::anyhow!("vm '{name_or_id}' not found"))?;
365
366        match (&vm.state, &new_state) {
367            (VmState::Pending, VmState::Running) => {}
368            (VmState::Stopped, VmState::Running) => {}
369            (VmState::Running, VmState::Stopped) => {}
370            (VmState::Pending, VmState::Deleted) => {}
371            (VmState::Stopped, VmState::Deleted) => {}
372            (from, to) => anyhow::bail!("invalid state transition: {from} -> {to}"),
373        }
374
375        vm.state = new_state;
376        vm.meta.updated_at = std::time::SystemTime::now()
377            .duration_since(std::time::UNIX_EPOCH)
378            .unwrap_or_default()
379            .as_secs();
380
381        let updated_at_iso = epoch_to_iso8601(vm.meta.updated_at);
382        let response = self
383            .db
384            .client()
385            .query(
386                "UPDATE type::record($tbl, $id) MERGE { \
387                 state: $state, \
388                 updated_at: <datetime>$updated_at \
389                 }",
390            )
391            .bind(("tbl", VM_TABLE))
392            .bind(("id", vm.meta.id.clone()))
393            .bind(("state", vm.state.to_string()))
394            .bind(("updated_at", updated_at_iso))
395            .await
396            .map_err(|e| anyhow::anyhow!("{e}"))?;
397        response.check().map_err(|e| anyhow::anyhow!("{e}"))?;
398        Ok(vm)
399    }

401    /// Delete a VM. Refuses to delete unless in `pending` / `stopped`.
402    /// Releases the IPAM allocation on success.
403    pub async fn delete(
404        &self,
405        name_or_id: &str,
406        org_name: Option<&str>,
407        project_name: Option<&str>,
408        env_name: Option<&str>,
409    ) -> anyhow::Result<()> {
410        let vm = self
411            .get(name_or_id, org_name, project_name, env_name)
412            .await?
413            .ok_or_else(|| anyhow::anyhow!("vm '{name_or_id}' not found"))?;
414
415        if vm.state != VmState::Stopped && vm.state != VmState::Pending {
416            anyhow::bail!(
417                "vm must be stopped or pending to delete (current state: {})",
418                vm.state
419            );
420        }
421
422        // Release IPAM allocation. The helper is idempotent — a
423        // retry after a partial failure just removes the entry
424        // from the blob.
425        nauka_network::vpc::subnet::ipam::release(&self.db, vm.subnet_id.as_str(), &vm.meta.id)
426            .await?;
427
428        let result = self
429            .db
430            .client()
431            .query("DELETE type::record($tbl, $id)")
432            .bind(("tbl", VM_TABLE))
433            .bind(("id", vm.meta.id.clone()))
434            .await
435            .map_err(|e| anyhow::anyhow!("{e}"))?;
436        result.check().map_err(|e| anyhow::anyhow!("{e}"))?;
437        Ok(())
438    }

441/// Row shape returned by `SELECT * FROM vm`.
442#[derive(Debug, Deserialize)]
443struct VmRow {
444    id: serde_json::Value,
445    name: String,
446    #[serde(default)]
447    status: Option<String>,
448    #[serde(default)]
449    labels: Option<serde_json::Value>,
450    org: String,
451    org_name: String,
452    project: String,
453    project_name: String,
454    env: String,
455    env_name: String,
456    vpc: String,
457    vpc_name: String,
458    subnet: String,
459    subnet_name: String,
460    vcpus: i64,
461    memory_mb: i64,
462    disk_gb: i64,
463    image: String,
464    region: String,
465    zone: String,
466    #[serde(default)]
467    private_ip: Option<String>,
468    #[serde(default)]
469    hypervisor: Option<String>,
470    state: String,
471    created_at: String,
472    updated_at: String,
473}
474
475impl VmRow {
476    fn into_vm(self) -> Vm {
477        let id = thing_to_id_string("vm:", &self.id);
478        let state = match self.state.as_str() {
479            "pending" => VmState::Pending,
480            "creating" => VmState::Creating,
481            "running" => VmState::Running,
482            "stopped" => VmState::Stopped,
483            "deleting" => VmState::Deleting,
484            "deleted" => VmState::Deleted,
485            // Default to Pending for anything else so a bad
486            // migration doesn't crash the list. The schema's
487            // ASSERT keeps this branch unreachable in practice.
488            _ => VmState::Pending,
489        };
490        Vm {
491            meta: ResourceMeta {
492                id,
493                name: self.name,
494                status: self.status.unwrap_or_else(|| "active".to_string()),
495                labels: self
496                    .labels
497                    .and_then(|v| serde_json::from_value(v).ok())
498                    .unwrap_or_default(),
499                created_at: iso8601_to_epoch(&self.created_at),
500                updated_at: iso8601_to_epoch(&self.updated_at),
501            },
502            org_id: OrgId::from(self.org),
503            org_name: self.org_name,
504            project_id: ProjectId::from(self.project),
505            project_name: self.project_name,
506            env_id: EnvId::from(self.env),
507            env_name: self.env_name,
508            vpc_id: VpcId::from(self.vpc),
509            vpc_name: self.vpc_name,
510            subnet_id: SubnetId::from(self.subnet),
511            subnet_name: self.subnet_name,
512            vcpus: self.vcpus as u32,
513            memory_mb: self.memory_mb as u32,
514            disk_gb: self.disk_gb as u32,
515            image: self.image,
516            region: self.region,
517            zone: self.zone,
518            private_ip: self.private_ip,
519            hypervisor_id: self.hypervisor,
520            state,
521        }
522    }
523}
524
525fn decode_first(raw: Vec<serde_json::Value>) -> anyhow::Result<Option<Vm>> {
526    match raw.into_iter().next() {
527        None => Ok(None),
528        Some(v) => {
529            let row: VmRow = serde_json::from_value(v)
530                .map_err(|e| anyhow::anyhow!("deserialise vm row: {e}"))?;
531            Ok(Some(row.into_vm()))
532        }
533    }
534}
