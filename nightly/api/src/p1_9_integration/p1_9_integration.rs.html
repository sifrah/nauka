1//! Hetzner integration test binary for P1.9 (sifrah/nauka#199).
2//!
3//! Demonstrates that data written by `EmbeddedDb<SurrealKv>` survives
4//! a process restart on a real Hetzner Ubuntu host. The same binary is
5//! invoked twice from the test runner:
6//!
7//! 1. **`p1-9-integration seed`** — open the datastore at
8//!    `/var/lib/nauka/p1-9-integration.skv`, create three records,
9//!    select them back to verify the in-process round-trip, exit.
10//! 2. **`p1-9-integration verify`** — re-open the *same* datastore
11//!    in a *new* process, list every record, assert that all three
12//!    survived the restart with their original values, exit.
13//!
14//! Between the two invocations the binary process exits, the
15//! kernel reaps it, and the test runner re-launches it. That gap is
16//! the "restart" that P1.5's in-process `persistence_across_reopen`
17//! test cannot exercise — only an actual process boundary proves
18//! the data is on disk and not just in some long-lived in-memory
19//! buffer.
20//!
21//! Cleanup is the test runner's job: provision the VM, run the
22//! binary in both phases, capture stdout, then delete the VM.
23
24use std::path::PathBuf;
25
26use nauka_state::EmbeddedDb;
27use surrealdb::types::SurrealValue;
28
29#[derive(Debug, Clone, PartialEq, SurrealValue)]
30struct Record {
31    name: String,
32    count: i64,
33}
34
35/// The fixed test fixture: three records the seed phase writes and the
36/// verify phase asserts on. Defined as a constant so any drift between
37/// the phases would be a compile error rather than a runtime mismatch.
38const RECORDS: &[(&str, &str, i64)] = &[
39    ("alpha", "first", 1),
40    ("beta", "second", 2),
41    ("gamma", "third", 3),
42];
43
44const TEST_PATH: &str = "/var/lib/nauka/p1-9-integration.skv";
45const TABLE: &str = "p1_9_record";
46
47fn print_help() {
48    println!("nauka p1-9 integration — EmbeddedDb<SurrealKv> persist-across-restart");
49    println!();
50    println!("Used by P1.9 (sifrah/nauka#199) to prove that data written via");
51    println!("EmbeddedDb<SurrealKv> survives a process restart on a real");
52    println!("Hetzner host. Run by the test runner in two phases.");
53    println!();
54    println!("USAGE:");
55    println!("    p1-9-integration <PHASE>");
56    println!();
57    println!("PHASES:");
58    println!("    seed     Open the datastore, create 3 records, select them");
59    println!("             back, exit. Datastore is at:");
60    println!("             {TEST_PATH}");
61    println!("    verify   Open the datastore in a fresh process, list all");
62    println!("             records, assert the 3 records from `seed` are");
63    println!("             still there with their original values, exit 0.");
64    println!("    -h       Print this help and exit.");
65    println!("    --help");
66    println!();
67    println!("The test runner is responsible for the surrounding ritual:");
68    println!("provision the VM, scp the binary, run `seed`, run `verify`, then");
69    println!("delete the VM. The binary itself does not touch hcloud.");
70}

72#[tokio::main]
73async fn main() -> Result<(), Box<dyn std::error::Error>> {
74    let phase = match std::env::args().nth(1).as_deref() {
75        Some("-h") | Some("--help") => {
76            print_help();
77            return Ok(());
78        }
79        Some("seed") => "seed",
80        Some("verify") => "verify",
81        Some(other) => {
82            eprintln!("error: unknown phase: {other}");
83            eprintln!("hint:  run `--help` for usage");
84            std::process::exit(2);
85        }
86        None => {
87            eprintln!("error: phase required (seed | verify)");
88            eprintln!("hint:  run `--help` for usage");
89            std::process::exit(2);
90        }
91    };
92
93    let path = PathBuf::from(TEST_PATH);
94    println!("== nauka p1-9 integration ==");
95    println!("phase          = {phase}");
96    println!("path           = {}", path.display());
97    println!("target_arch    = {}", std::env::consts::ARCH);
98    println!("target_os      = {}", std::env::consts::OS);
99    println!("expected_count = {}", RECORDS.len());
100
101    let db = EmbeddedDb::open(&path).await?;
102    println!("opened         = {}", db.path().display());
103
104    match phase {
105        "seed" => seed(&db).await?,
106        "verify" => verify(&db).await?,
107        _ => unreachable!(),
108    }
109
110    db.shutdown().await?;
111    println!("== p1-9 {phase} OK ==");
112    Ok(())
113}

115async fn seed(db: &EmbeddedDb) -> Result<(), Box<dyn std::error::Error>> {
116    // The test runner is responsible for delivering a clean
117    // /var/lib/nauka/p1-9-integration.skv before invoking `seed` —
118    // attempting to pre-clean records here would actually error,
119    // because SurrealDB returns NotFound on `delete` against a
120    // never-touched table. Trust the runner.
121
122    // Create the three fixture records.
123    for (id, name, count) in RECORDS {
124        let created: Option<Record> = db
125            .client()
126            .create((TABLE, *id))
127            .content(Record {
128                name: (*name).into(),
129                count: *count,
130            })
131            .await?;
132        let created = created.expect("create returned None");
133        println!(
134            "created        = id={id} name={} count={}",
135            created.name, created.count
136        );
137    }
138
139    // In-process round-trip: select-by-id for each record and assert
140    // the round-trip works before we even consider the restart case.
141    for (id, name, count) in RECORDS {
142        let fetched: Option<Record> = db.client().select((TABLE, *id)).await?;
143        let fetched =
144            fetched.unwrap_or_else(|| panic!("seed: in-process select for id={id} returned None"));
145        assert_eq!(fetched.name, *name, "seed: name mismatch for id={id}");
146        assert_eq!(fetched.count, *count, "seed: count mismatch for id={id}");
147    }
148
149    let all: Vec<Record> = db.client().select(TABLE).await?;
150    println!("seed_count     = {}", all.len());
151    assert_eq!(
152        all.len(),
153        RECORDS.len(),
154        "seed: expected {} records, got {}",
155        RECORDS.len(),
156        all.len()
157    );
158    Ok(())
159}

161async fn verify(db: &EmbeddedDb) -> Result<(), Box<dyn std::error::Error>> {
162    // Cross-process verification: this binary was started fresh after
163    // the seed phase exited. Reading the same records here proves the
164    // data is on disk and not in some long-lived in-memory state.
165
166    let all: Vec<Record> = db.client().select(TABLE).await?;
167    println!("verify_count   = {}", all.len());
168    assert_eq!(
169        all.len(),
170        RECORDS.len(),
171        "verify: expected {} records after restart, got {}",
172        RECORDS.len(),
173        all.len()
174    );
175
176    for (id, name, count) in RECORDS {
177        let fetched: Option<Record> = db.client().select((TABLE, *id)).await?;
178        let fetched = fetched.unwrap_or_else(|| {
179            panic!("verify: post-restart select for id={id} returned None — data lost?")
180        });
181        assert_eq!(
182            fetched.name, *name,
183            "verify: name mismatch for id={id} (lost write?)"
184        );
185        assert_eq!(
186            fetched.count, *count,
187            "verify: count mismatch for id={id} (corrupted?)"
188        );
189        println!(
190            "verified       = id={id} name={} count={}",
191            fetched.name, fetched.count
192        );
193    }
194
195    println!(
196        "verify_status  = all {} records intact across restart",
197        RECORDS.len()
198    );
199    Ok(())
200}
