1//! Runtime abstraction — Cloud Hypervisor (KVM) or gVisor (container).
2//!
3//! The runtime trait abstracts how a VM is actually executed:
4//! - On bare-metal with /dev/kvm → Cloud Hypervisor (hardware isolation)
5//! - On VPS without KVM → gVisor/runsc (userspace kernel isolation)
6//!
7//! The user never sees the difference — `nauka vm create` works the same.
8
9pub mod detect;
10pub mod gvisor;
11pub mod kvm;
12
13use serde::{Deserialize, Serialize};
14
15/// Which runtime this node uses.
16#[derive(Debug, Clone, Serialize, Deserialize, Default, PartialEq)]
17#[serde(rename_all = "lowercase")]
18pub enum RuntimeMode {
19    /// Cloud Hypervisor — bare-metal with /dev/kvm (hardware isolation)
20    #[default]
21    Kvm,
22    /// gVisor/runsc — VPS without KVM (userspace kernel isolation)
23    Container,
24}
25
26impl std::fmt::Display for RuntimeMode {
27    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
28        match self {
29            RuntimeMode::Kvm => write!(f, "kvm"),
30            RuntimeMode::Container => write!(f, "container"),
31        }
32    }
33}

35/// Info about a running VM process.
36#[derive(Debug)]
37pub struct RunningVm {
38    pub vm_id: String,
39    pub pid: u32,
40}
41
42/// Runtime trait — how to start/stop a VM.
43pub trait Runtime: Send + Sync {
44    /// Start a VM. Returns the PID of the process.
45    fn start(&self, config: &VmRunConfig) -> anyhow::Result<u32>;
46
47    /// Stop a VM by its ID.
48    fn stop(&self, vm_id: &str) -> anyhow::Result<()>;
49
50    /// Check if a VM is running. Returns the PID if so.
51    fn is_running(&self, vm_id: &str) -> Option<u32>;
52
53    /// List all running VMs managed by this runtime.
54    fn list_running(&self) -> Vec<RunningVm>;
55}
56
57/// Configuration passed to the runtime to start a VM.

59pub struct VmRunConfig {
60    pub vm_id: String,

62    pub vcpus: u32,
63    pub memory_mb: u32,
64    pub disk_gb: u32,
65    pub image: String,
66    pub tap_name: String,
67    pub private_ip: String,
68    pub gateway: String,
69    pub subnet_cidr: String,
70}
