1//! Forge daemon — the main reconciliation loop.
2
3use nauka_hypervisor::controlplane;
4use nauka_hypervisor::fabric;
5use nauka_state::LocalDb;
6
7use crate::reconciler;
8use crate::types::ReconcileContext;
9
10/// Default reconciliation interval in seconds.
11const RECONCILE_INTERVAL_SECS: u64 = 30;
12
13/// Run the forge daemon (blocking loop).
14pub async fn run() -> anyhow::Result<()> {
15    eprintln!("  Forge starting...");
16
17    let mut cycle: u64 = 0;
18
19    loop {
20        cycle += 1;
21
22        match run_cycle(cycle).await {
23            Ok(_results) => {}
24            Err(e) => {
25                tracing::error!(cycle, error = %e, "reconcile cycle failed");
26            }
27        }
28
29        tokio::time::sleep(std::time::Duration::from_secs(RECONCILE_INTERVAL_SECS)).await;
30    }
31}
32
33/// Run a single reconciliation cycle (for `nauka forge reconcile`).
34pub async fn run_once() -> anyhow::Result<String> {
35    let results = run_cycle(1).await?;
36
37    let lines: Vec<String> = results.iter().map(|r| format!("  {r}")).collect();
38    Ok(format!("reconciliation complete\n{}", lines.join("\n")))
39}
40
41/// Execute one reconciliation cycle.
42async fn run_cycle(cycle: u64) -> anyhow::Result<Vec<crate::types::ReconcileResult>> {
43    // Connect to TiKV
44    let db = controlplane::connect().await?;
45
46    // Load this node's identity
47    let local_db = LocalDb::open("hypervisor")?;
48    let state = fabric::state::FabricState::load(&local_db)
49        .map_err(|e| anyhow::anyhow!("{e}"))?
50        .ok_or_else(|| anyhow::anyhow!("not initialized"))?;
51
52    let runtime = if state.hypervisor.runtime == "container" {
53        nauka_compute::runtime::RuntimeMode::Container
54    } else {
55        nauka_compute::runtime::RuntimeMode::Kvm
56    };
57
58    // Collect all IDs this node is known by:
59    // - Its own HypervisorId (hv-...)
60    // - Its name (used by peers to identify us)
61    // - Any NodeId that peers assigned to us (node-...)
62    // This ensures VMs scheduled from any node match correctly.
63    let node_ids = vec![
64        state.hypervisor.id.as_str().to_string(),
65        state.hypervisor.name.clone(),
66    ];
67    // Check what IDs peers use for us by looking at peer lists on remote nodes
68    // We can't do that directly, but peers reference us by NodeId.
69    // The scheduler on remote nodes uses the peer's NodeId.
70    // So we need to know our own NodeId as seen by peers.
71    // This is stored in the peer list of OTHER nodes — we can't access it.
72    // Workaround: also check by mesh_ipv6 match, or just match all VMs
73    // assigned to any ID containing our name.
74
75    let ctx = ReconcileContext {
76        db,
77        hypervisor_id: state.hypervisor.id.as_str().to_string(),
78        node_ids,
79        node_name: state.hypervisor.name.clone(),
80        mesh_ipv6: state.hypervisor.mesh_ipv6,
81        runtime,
82        cycle,
83    };
84
85    tracing::debug!(
86        cycle,
87        node = ctx.node_name.as_str(),
88        "reconcile cycle start"
89    );
90
91    let results = reconciler::run_all(&ctx).await;
92
93    let total_actions: usize = results
94        .iter()
95        .map(|r| r.created + r.deleted + r.updated)
96        .sum();
97    let total_failures: usize = results.iter().map(|r| r.failed).sum();
98
99    if total_actions > 0 || total_failures > 0 {
100        tracing::info!(
101            cycle,
102            actions = total_actions,
103            failures = total_failures,
104            "reconcile cycle done"
105        );
106    } else {
107        tracing::debug!(cycle, "reconcile cycle done — in sync");
108    }
109
110    Ok(results)
111}
