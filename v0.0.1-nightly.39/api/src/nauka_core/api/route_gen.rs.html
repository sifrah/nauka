21/// Build an axum Router from resource registrations.
22///
23/// For resources with scope parents, generates nested routes:
24/// `/{prefix}/{parent}/{parent_id}/{resource}` etc.
25pub fn build_router(registrations: Vec<ResourceRegistration>, prefix: &str) -> Router {
26    let mut router = Router::new();
27
28    let shared: Vec<Arc<ResourceRegistration>> = registrations.into_iter().map(Arc::new).collect();
29
30    for reg in &shared {
31        let kind = reg.def.identity.cli_name;
32
33        // #10: Build route path from scope parents
34        let base = build_base_path(prefix, kind, &reg.def.scope.parents);
35
36        // Collect which methods we need on /{base} and /{base}/{id}
37        let mut collection_methods: Option<MethodRouter> = None;
38        let mut instance_methods: Option<MethodRouter> = None;
39
40        for op in &reg.def.operations {
41            let r = Arc::clone(reg);
42            let op_name = op.name.to_string();
43
44            match &op.semantics {
45                // #3: List with pagination query params
46                OperationSemantics::List => {
47                    let name = op_name.clone();
48                    let handler = get(move |Query(params): Query<HashMap<String, String>>| {
49                        let r = Arc::clone(&r);
50                        let name = name.clone();
51                        async move { handle_operation(&r, &name, None, params).await }
52                    });
53                    collection_methods = Some(match collection_methods {
54                        Some(m) => m.merge(handler),
55                        None => handler,
56                    });
57                }
58                OperationSemantics::Create => {
59                    let name = op_name.clone();
60                    let handler = post(move |Json(body): Json<HashMap<String, String>>| {
61                        let r = Arc::clone(&r);
62                        let name = name.clone();
63                        async move {
64                            let resource_name = body.get("name").cloned();
65                            handle_operation(&r, &name, resource_name, body).await
66                        }
67                    });
68                    collection_methods = Some(match collection_methods {
69                        Some(m) => m.merge(handler),
70                        None => handler,
71                    });
72                }
73                // #11: GET and DELETE on same path — merged into one MethodRouter
74                OperationSemantics::Get => {
75                    let name = op_name.clone();
76                    let handler = get(move |Path(id): Path<String>| {
77                        let r = Arc::clone(&r);
78                        let name = name.clone();
79                        async move { handle_operation(&r, &name, Some(id), HashMap::new()).await }
80                    });
81                    instance_methods = Some(match instance_methods {
82                        Some(m) => m.merge(handler),
83                        None => handler,
84                    });
85                }
86                OperationSemantics::Delete => {
87                    let name = op_name.clone();
88                    let handler = axum::routing::delete(move |Path(id): Path<String>| {
89                        let r = Arc::clone(&r);
90                        let name = name.clone();
91                        async move { handle_operation(&r, &name, Some(id), HashMap::new()).await }
92                    });
93                    instance_methods = Some(match instance_methods {
94                        Some(m) => m.merge(handler),
95                        None => handler,
96                    });
97                }
98                OperationSemantics::Update { .. } => {
99                    let name = op_name.clone();
100                    let handler = axum::routing::patch(
101                        move |Path(id): Path<String>, Json(body): Json<HashMap<String, String>>| {
102                            let r = Arc::clone(&r);
103                            let name = name.clone();
104                            async move { handle_operation(&r, &name, Some(id), body).await }
105                        },
106                    );
107                    instance_methods = Some(match instance_methods {
108                        Some(m) => m.merge(handler),
109                        None => handler,
110                    });
111                }
112                OperationSemantics::Action => {
113                    let name = op_name.clone();
114                    let route = format!("{base}/{}", op.name);
115                    router = router.route(
116                        &route,
117                        post(move |Json(body): Json<HashMap<String, String>>| {
118                            let r = Arc::clone(&r);
119                            let name = name.clone();
120                            async move {
121                                let resource_name = body.get("name").cloned();
122                                handle_operation(&r, &name, resource_name, body).await
123                            }
124                        }),
125                    );
126                }
127            }
128        }
129
130        // Register merged routes
131        if let Some(methods) = collection_methods {
132            router = router.route(&base, methods);
133        }
134        if let Some(methods) = instance_methods {
135            let instance_path = format!("{base}/{{id}}");
136            router = router.route(&instance_path, methods);
137        }
138    }
139
140    router
141}

143/// #10: Build a route path incorporating scope parents.
144///
145/// No parents: `/admin/v1/org`
146/// With parents: `/admin/v1/org/{org_id}/project/{project_id}/vpc`
147fn build_base_path(prefix: &str, kind: &str, parents: &[crate::resource::ParentRef]) -> String {
148    if parents.is_empty() {
149        return format!("{prefix}/{kind}");
150    }
151
152    let mut path = prefix.to_string();
153    for parent in parents {
154        let parent_kind = parent.kind;
155        let param = format!("{{{parent_kind}_id}}");
156        path = format!("{path}/{parent_kind}/{param}");
157    }
158    format!("{path}/{kind}")
159}

161async fn handle_operation(
162    reg: &ResourceRegistration,
163    operation: &str,
164    name: Option<String>,
165    fields: HashMap<String, String>,
166) -> impl IntoResponse {
167    let op_def = reg.def.operations.iter().find(|o| o.name == operation);
168    if let Some(op_def) = op_def {
169        for constraint in &op_def.constraints {
170            if let Err(msg) = constraint.validate(&fields) {
171                return Err(ApiError(NaukaError::validation(msg)));
172            }
173        }
174    }
175
176    let request = OperationRequest {
177        operation: operation.to_string(),
178        name,
179        scope: ScopeValues::default(),
180        fields,
181    };
182
183    let response = (reg.handler)(request)
184        .await
185        .map_err(|e: anyhow::Error| ApiError(NaukaError::internal(e.to_string())))?;
186
187    match response {
188        OperationResponse::Resource(v) => Ok(axum::Json(v).into_response()),
189        OperationResponse::ResourceList(items) => Ok(axum::Json(serde_json::json!({
190            "items": items,
191            "count": items.len(),
192        }))
193        .into_response()),
194        OperationResponse::Message(msg) => {
195            Ok(axum::Json(serde_json::json!({"message": msg})).into_response())
196        }
197        OperationResponse::None => Ok(axum::http::StatusCode::NO_CONTENT.into_response()),
198    }
199}

201/// Generate a list of all routes for documentation.
202pub fn list_routes(registrations: &[ResourceRegistration], prefix: &str) -> Vec<RouteInfo> {
203    let mut routes = Vec::new();
204
205    for reg in registrations {
206        let kind = reg.def.identity.cli_name;
207        let base = build_base_path(prefix, kind, &reg.def.scope.parents);
208
209        for op in &reg.def.operations {
210            let (method, path) = match &op.semantics {
211                OperationSemantics::List => ("GET", base.clone()),
212                OperationSemantics::Create => ("POST", base.clone()),
213                OperationSemantics::Get => ("GET", format!("{base}/{{id}}")),
214                OperationSemantics::Delete => ("DELETE", format!("{base}/{{id}}")),
215                OperationSemantics::Update { .. } => ("PATCH", format!("{base}/{{id}}")),
216                OperationSemantics::Action => ("POST", format!("{base}/{}", op.name)),
217            };
218
219            routes.push(RouteInfo {
220                method: method.to_string(),
221                path,
222                operation: op.name.to_string(),
223                resource: kind.to_string(),
224                description: op.description.to_string(),
225            });
226        }
227    }
228
229    routes
230}

232/// #7: OpenAPI-compatible route listing.
233#[derive(Debug, Clone, serde::Serialize)]
234pub struct RouteInfo {
235    pub method: String,
236    pub path: String,
237    pub operation: String,
238    pub resource: String,
239    pub description: String,
240}

242/// #7: Generate a minimal OpenAPI-style spec from routes.
243pub fn openapi_spec(registrations: &[ResourceRegistration], prefix: &str) -> serde_json::Value {
244    let routes = list_routes(registrations, prefix);
245
246    let mut paths = serde_json::Map::new();
247    for route in &routes {
248        let path_entry = paths
249            .entry(route.path.clone())
250            .or_insert_with(|| serde_json::json!({}));
251        let method_lower = route.method.to_lowercase();
252        path_entry[method_lower] = serde_json::json!({
253            "summary": route.description,
254            "operationId": format!("{}_{}", route.resource, route.operation),
255            "tags": [route.resource],
256        });
257    }
258
259    serde_json::json!({
260        "openapi": "3.0.0",
261        "info": {
262            "title": "Nauka API",
263            "version": env!("CARGO_PKG_VERSION"),
264        },
265        "paths": paths,
266    })
267}
