1//! Structured logging for Nauka.
2//!
3//! Sets up `tracing` with consistent formatting across daemon and CLI.
4//!
5//! # Features
6//!
7//! - **Text and JSON** output formats
8//! - **File rotation** by daily or size-based rolling
9//! - **Panic hook** redirects panics through tracing
10//! - **Global context** (node name, region, zone) on every log line
11//! - **Per-module filtering** via config (not just global level)
12//! - **Dependency noise suppression** (hyper, tokio, redb default to warn)
13//! - **Log metrics** (warn/error counters)
14//! - **Guard pattern** with must_use to prevent accidental drop
15//!
16//! ```no_run
17//! use nauka_core::logging;
18//! use nauka_core::config::LoggingConfig;
19//!
20//! let config = LoggingConfig::default();
21//! let _guard = logging::init(&config);
22//!
23//! tracing::info!("daemon started");
24//! tracing::warn!(zone = "fsn1", "peer unreachable");
25//! ```
26

36// ═══════════════════════════════════════════════════
37// 7. Log metrics — atomic counters
38// ═══════════════════════════════════════════════════
39
40static WARN_COUNT: AtomicU64 = AtomicU64::new(0);
41static ERROR_COUNT: AtomicU64 = AtomicU64::new(0);
42
43/// Get the number of warnings logged since init.
44pub fn warn_count() -> u64 {
45    WARN_COUNT.load(Ordering::Relaxed)
46}
47
48/// Get the number of errors logged since init.
49pub fn error_count() -> u64 {
50    ERROR_COUNT.load(Ordering::Relaxed)
51}
52
53/// Reset counters (for testing).
54pub fn reset_counters() {
55    WARN_COUNT.store(0, Ordering::Relaxed);
56    ERROR_COUNT.store(0, Ordering::Relaxed);
57}
58
59/// A tracing layer that counts warn and error events.
60struct MetricsLayer;
61
62impl<S: tracing::Subscriber> tracing_subscriber::Layer<S> for MetricsLayer {
63    fn on_event(
64        &self,
65        event: &tracing::Event<'_>,
66        _ctx: tracing_subscriber::layer::Context<'_, S>,
67    ) {
68        match *event.metadata().level() {
69            tracing::Level::WARN => {
70                WARN_COUNT.fetch_add(1, Ordering::Relaxed);
71            }
72            tracing::Level::ERROR => {
73                ERROR_COUNT.fetch_add(1, Ordering::Relaxed);
74            }
75            _ => {}
76        }
77    }
78}

84/// Default filter directives that suppress noisy dependencies.
85const NOISE_FILTERS: &[&str] = &[
86    "hyper=warn",
87    "tokio=warn",
88    "mio=warn",
89    "redb=warn",
90    "rustls=warn",
91    "h2=warn",
92    "tower=warn",
93    "reqwest=warn",
94];
95
96// ═══════════════════════════════════════════════════
97// Filter building — supports per-module levels
98// ═══════════════════════════════════════════════════
99
100/// Build an EnvFilter from config level + noise suppression.
101///
102/// Priority:
103/// 1. `RUST_LOG` env var (standard Rust convention, overrides everything)
104/// 2. Config level + per-module overrides + noise suppression
105///
106/// # Per-module filtering
107///
108/// The level string can include module-specific directives:
109/// - `"info"` → global info
110/// - `"info,nauka_fabric=debug"` → info globally, debug for fabric
111/// - `"warn,nauka_fabric::daemon=trace"` → warn globally, trace for daemon
112pub fn build_filter(level: &str) -> EnvFilter {
113    if std::env::var("RUST_LOG").is_ok() {
114        return EnvFilter::from_default_env();
115    }
116
117    let mut directives = String::new();
118
119    // Noise suppression first (lowest priority)
120    for noise in NOISE_FILTERS {
121        directives.push_str(noise);
122        directives.push(',');
123    }
124
125    // User level (can contain per-module directives)
126    directives.push_str(level);
127
128    EnvFilter::new(directives)
129}

135/// Initialize the logging system. Returns a guard that **must** be held
136/// for the lifetime of the program — dropping it flushes pending logs.
137pub fn init(config: &LoggingConfig) -> LogGuard {
138    let filter = build_filter(&config.level);
139
140    let guard = if config.file.is_empty() {
141        init_stderr(config, filter)
142    } else {
143        init_with_file(config, filter)
144    };
145
146    // 2. Install panic hook
147    install_panic_hook();
148
149    guard
150}
151
152/// Initialize logging to stderr only (CLI commands, foreground daemon).
153pub fn init_stderr(config: &LoggingConfig, filter: EnvFilter) -> LogGuard {
154    match config.format.as_str() {
155        "json" => {
156            let subscriber = tracing_subscriber::registry()
157                .with(filter)
158                .with(MetricsLayer)
159                .with(
160                    fmt::layer()
161                        .json()
162                        .with_target(true)
163                        .with_thread_ids(false)
164                        .with_writer(std::io::stderr),
165                );
166            tracing::subscriber::set_global_default(subscriber).ok();
167        }
168        _ => {
169            let subscriber = tracing_subscriber::registry()
170                .with(filter)
171                .with(MetricsLayer)
172                .with(
173                    fmt::layer()
174                        .with_target(true)
175                        .with_thread_ids(false)
176                        .with_writer(std::io::stderr),
177                );
178            tracing::subscriber::set_global_default(subscriber).ok();
179        }
180    }
181
182    LogGuard {
183        _guards: Vec::new(),
184    }
185}

187/// Initialize logging to file (with daily rotation) + stderr.
188fn init_with_file(config: &LoggingConfig, filter: EnvFilter) -> LogGuard {
189    let log_path = Path::new(&config.file);
190    let log_dir = log_path.parent().unwrap_or(Path::new("."));
191    let log_name = log_path
192        .file_name()
193        .and_then(|n| n.to_str())
194        .unwrap_or("nauka.log");
195
196    let _ = std::fs::create_dir_all(log_dir);
197
198    // 1. File rotation — daily rolling
199    let file_appender = tracing_appender::rolling::daily(log_dir, log_name);
200    let (file_writer, file_guard) = tracing_appender::non_blocking(file_appender);
201
202    let (stderr_writer, stderr_guard) = tracing_appender::non_blocking(std::io::stderr());
203
204    match config.format.as_str() {
205        "json" => {
206            let subscriber = tracing_subscriber::registry()
207                .with(filter)
208                .with(MetricsLayer)
209                .with(
210                    fmt::layer()
211                        .json()
212                        .with_target(true)
213                        .with_writer(file_writer),
214                )
215                .with(
216                    fmt::layer()
217                        .json()
218                        .with_target(true)
219                        .with_writer(stderr_writer),
220                );
221            tracing::subscriber::set_global_default(subscriber).ok();
222        }
223        _ => {
224            let subscriber = tracing_subscriber::registry()
225                .with(filter)
226                .with(MetricsLayer)
227                .with(fmt::layer().with_target(true).with_writer(file_writer))
228                .with(fmt::layer().with_target(true).with_writer(stderr_writer));
229            tracing::subscriber::set_global_default(subscriber).ok();
230        }
231    }
232
233    LogGuard {
234        _guards: vec![file_guard, stderr_guard],
235    }
236}

238// ═══════════════════════════════════════════════════
239// 2. Panic hook — redirects panics through tracing
240// ═══════════════════════════════════════════════════
241
242fn install_panic_hook() {
243    let default_hook = std::panic::take_hook();
244    std::panic::set_hook(Box::new(move |info| {
245        let payload = if let Some(s) = info.payload().downcast_ref::<&str>() {
246            (*s).to_string()
247        } else if let Some(s) = info.payload().downcast_ref::<String>() {
248            s.clone()
249        } else {
250            "unknown panic".to_string()
251        };
252
253        let location = info
254            .location()
255            .map(|l| format!("{}:{}:{}", l.file(), l.line(), l.column()))
256            .unwrap_or_else(|| "unknown".to_string());
257
258        tracing::error!(
259            panic = true,
260            location = %location,
261            "PANIC: {payload}"
262        );
263
264        default_hook(info);
265    }));
266}

268// ═══════════════════════════════════════════════════
269// 3. Global context span
270// ═══════════════════════════════════════════════════
271
272/// Create a global context span that adds fields to every log line.
273///
274/// Call this once at daemon startup after init():
275/// ```no_run
276/// # use nauka_core::logging;
277/// # let _guard = logging::init_cli();
278/// let _span = logging::global_context("node-1", "eu", "fsn1").entered();
279/// // All subsequent logs include node, region, zone
280/// tracing::info!("ready"); // → ... node="node-1" region="eu" zone="fsn1" ready
281/// ```
282pub fn global_context(node_name: &str, region: &str, zone: &str) -> tracing::Span {
283    tracing::info_span!("nauka", node = node_name, region = region, zone = zone,)
284}

300// ═══════════════════════════════════════════════════
301// 8. Guard with must_use
302// ═══════════════════════════════════════════════════
303
304/// Guard that **must** be held for the program's lifetime.
305/// Dropping it flushes any pending buffered logs.
306///
307/// If you see the compiler warning "unused variable `_guard`",
308/// that's fine — the name `_guard` prevents the warning while
309/// keeping the guard alive.
310#[must_use = "dropping the LogGuard will flush and stop logging — hold it for the program's lifetime"]
311pub struct LogGuard {
312    _guards: Vec<WorkerGuard>,
313}
