1//! VPC reconciler — ensures VXLAN bridges exist for VPCs with local VMs.
2
3use nauka_network::vpc::provision;
4
5use crate::types::{ReconcileContext, ReconcileResult};
6
7pub struct VpcReconciler;
8
9#[async_trait::async_trait]
10impl super::Reconciler for VpcReconciler {
11    fn name(&self) -> &str {
12        "vpc"
13    }
14
15    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
16        let mut result = ReconcileResult::new("vpc");
17
18        // 1. Find which VMs are on this node
19        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
20        let all_vms = vm_store.list(None, None, None).await?;
21        let local_vms: Vec<_> = all_vms
22            .iter()
23            .filter(|vm| vm.hypervisor_id.as_deref() == Some(&ctx.hypervisor_id))
24            .collect();
25
26        // 2. Collect unique VPC IDs needed on this node + resolve their VNIs
27        let vpc_store = nauka_network::vpc::store::VpcStore::new(ctx.db.clone());
28        let mut needed_vpcs: Vec<(String, u32)> = Vec::new(); // (vpc_id, vni)
29        for vm in &local_vms {
30            let vpc_id = vm.vpc_id.as_str().to_string();
31            if !needed_vpcs.iter().any(|(id, _)| id == &vpc_id) {
32                if let Some(vpc) = vpc_store.get(&vpc_id, None).await? {
33                    needed_vpcs.push((vpc_id, vpc.vni));
34                }
35            }
36        }
37
38        result.desired = needed_vpcs.len();
39
40        // 3. Check actual state — scan for existing bridges
41        let actual_bridges = crate::observer::network::list_bridges();
42        result.actual = actual_bridges.len();
43
44        // 4. Create missing bridges
45        for (vpc_id, vni) in &needed_vpcs {
46            let expected_bridge = provision::bridge_name(vpc_id);
47            if !actual_bridges.iter().any(|b| b == &expected_bridge) {
48                match provision::ensure_bridge(vpc_id, *vni, &ctx.mesh_ipv6) {
49                    Ok(()) => result.created += 1,
50                    Err(e) => {
51                        tracing::error!(vpc_id, error = %e, "failed to create bridge");
52                        result.failed += 1;
53                        result.errors.push(format!("vpc {vpc_id}: {e}"));
54                    }
55                }
56            }
57        }
58
59        // 5. Remove orphaned bridges (exist locally but no VMs need them)
60        let needed_bridge_names: Vec<String> = needed_vpcs
61            .iter()
62            .map(|(id, _)| provision::bridge_name(id))
63            .collect();
64        for bridge in &actual_bridges {
65            if !needed_bridge_names.contains(bridge) {
66                tracing::info!(bridge, "removing orphaned bridge");
67                // Derive VXLAN name from bridge name (nkb-HASH → nkx-HASH)
68                let vxlan = bridge.replace("nkb-", "nkx-");
69                let _ = std::process::Command::new("ip")
70                    .args(["link", "del", &vxlan])
71                    .status();
72                let _ = std::process::Command::new("ip")
73                    .args(["link", "del", bridge])
74                    .status();
75                result.deleted += 1;
76            }
77        }
78
79        Ok(result)
80    }
81}
