1//! Process utilities — paths, directories.
2//!
3//! No daemon, no fork, no PID file. Nauka is a CLI orchestrator,
4//! not a daemon. WireGuard runs in the kernel.
5
6use std::path::PathBuf;
7
8use crate::error::NaukaError;
9
10/// Default nauka directory.
11pub fn nauka_dir() -> PathBuf {
12    dirs::home_dir()
13        .unwrap_or_else(|| PathBuf::from("."))
14        .join(".nauka")
15}
16
17/// Ensure ~/.nauka exists with 0o700 permissions.
18pub fn ensure_nauka_dir() -> Result<(), NaukaError> {
19    let dir = nauka_dir();
20    std::fs::create_dir_all(&dir).map_err(NaukaError::from)?;
21    #[cfg(unix)]
22    {
23        use std::os::unix::fs::PermissionsExt;
24        let _ = std::fs::set_permissions(&dir, std::fs::Permissions::from_mode(0o700));
25    }
26    Ok(())
27}
28
29/// Default control socket path (for future API server).
30pub fn socket_path() -> PathBuf {
31    nauka_dir().join("control.sock")
32}
