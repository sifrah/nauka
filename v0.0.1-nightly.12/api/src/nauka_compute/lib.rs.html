20/// Connect to ClusterDb via PD endpoints from fabric state.
21pub(crate) async fn connect_cluster_db() -> anyhow::Result<controlplane::ClusterDb> {
22    let dir = nauka_core::process::nauka_dir();
23    let _ = std::fs::create_dir_all(&dir);
24    let db = LocalDb::open("hypervisor")?;
25
26    let state = fabric::state::FabricState::load(&db)
27        .map_err(|e| anyhow::anyhow!("{e}"))?
28        .ok_or_else(|| {
29            anyhow::anyhow!(
30                "cluster not initialized.\n\n\
31                 Initialize a cluster first with:\n\
32                 \x20 nauka hypervisor init"
33            )
34        })?;
35
36    let self_endpoint = format!(
37        "http://[{}]:{}",
38        state.hypervisor.mesh_ipv6,
39        controlplane::PD_CLIENT_PORT,
40    );
41    let mut endpoints = vec![self_endpoint];
42    for peer in &state.peers.peers {
43        endpoints.push(format!(
44            "http://[{}]:{}",
45            peer.mesh_ipv6,
46            controlplane::PD_CLIENT_PORT,
47        ));
48    }
49    let refs: Vec<&str> = endpoints.iter().map(|s| s.as_str()).collect();
50
51    controlplane::ClusterDb::connect(&refs)
52        .await
53        .map_err(Into::into)
54}
