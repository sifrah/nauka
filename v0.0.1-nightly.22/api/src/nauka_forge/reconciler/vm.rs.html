1//! VM reconciler — ensures VMs assigned to this node have TAPs and processes.
2
3use nauka_compute::runtime::{
4    gvisor::GVisorRuntime, kvm::KvmRuntime, Runtime, RuntimeMode, VmRunConfig,
5};
6use nauka_compute::vm::provision;
7use nauka_compute::vm::types::VmState;
8use nauka_network::vpc::provision as vpc_provision;
9
10use crate::types::{ReconcileContext, ReconcileResult};
11
12pub struct VmReconciler;
13
14impl VmReconciler {
15    /// Get the right runtime for this node.
16    fn runtime(ctx: &ReconcileContext) -> Box<dyn Runtime> {
17        if ctx.runtime == RuntimeMode::Kvm {
18            Box::new(KvmRuntime)
19        } else {
20            Box::new(GVisorRuntime)
21        }
22    }
23}
24
25#[async_trait::async_trait]
26impl super::Reconciler for VmReconciler {
27    fn name(&self) -> &str {
28        "vm"
29    }
30
31    async fn reconcile(&self, ctx: &ReconcileContext) -> anyhow::Result<ReconcileResult> {
32        let mut result = ReconcileResult::new("vm");
33        let rt = Self::runtime(ctx);
34
35        // 1. Get VMs assigned to this node
36        let vm_store = nauka_compute::vm::store::VmStore::new(ctx.db.clone());
37        let all_vms = vm_store.list(None, None, None).await?;
38        let local_vms: Vec<_> = all_vms
39            .iter()
40            .filter(|vm| {
41                vm.hypervisor_id
42                    .as_ref()
43                    .map(|hid| ctx.node_ids.iter().any(|nid| nid == hid))
44                    .unwrap_or(false)
45            })
46            .collect();
47
48        // VMs that should be running (pending = needs starting, running = should be alive)
49        let should_exist: Vec<_> = local_vms
50            .iter()
51            .filter(|vm| vm.state == VmState::Pending || vm.state == VmState::Running)
52            .collect();
53
54        result.desired = should_exist.len();
55
56        // 2. Check actual state
57        let actual_processes = crate::observer::process::list_vms();
58        result.actual = actual_processes.len();
59
60        // 3. Create TAPs + start processes for missing VMs
61        for vm in &should_exist {
62            let expected_tap = provision::tap_name(&vm.meta.id);
63            let actual_taps = provision::list_taps();
64
65            // Ensure TAP exists
66            if !actual_taps.iter().any(|t| t == &expected_tap) {
67                let bridge = vpc_provision::bridge_name(vm.vpc_id.as_str());
68                if let Err(e) = provision::ensure_tap(&vm.meta.id, &bridge) {
69                    tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to create TAP");
70                    result.failed += 1;
71                    result.errors.push(format!("tap {}: {e}", vm.meta.id));
72                    continue;
73                }
74            }
75
76            // Ensure process is running
77            if !actual_processes.contains(&vm.meta.id) {
78                let subnet_store =
79                    nauka_network::vpc::subnet::store::SubnetStore::new(ctx.db.clone());
80                let subnet = subnet_store.get(vm.subnet_id.as_str(), None, None).await?;
81
82                let (gateway, cidr) = match &subnet {
83                    Some(s) => (s.gateway.clone(), s.cidr.clone()),
84                    None => ("0.0.0.0".to_string(), "0.0.0.0/0".to_string()),
85                };
86
87                let config = VmRunConfig {
88                    vm_id: vm.meta.id.clone(),
89                    vm_name: vm.meta.name.clone(),
90                    vcpus: vm.vcpus,
91                    memory_mb: vm.memory_mb,
92                    disk_gb: vm.disk_gb,
93                    image: vm.image.clone(),
94                    tap_name: provision::tap_name(&vm.meta.id),
95                    private_ip: vm.private_ip.clone().unwrap_or_default(),
96                    gateway,
97                    subnet_cidr: cidr,
98                };
99
100                match rt.start(&config) {
101                    Ok(pid) => {
102                        tracing::info!(
103                            vm_id = vm.meta.id.as_str(),
104                            pid,
105                            runtime = %ctx.runtime,
106                            "VM process started"
107                        );
108                        result.created += 1;
109                    }
110                    Err(e) => {
111                        tracing::error!(vm_id = vm.meta.id.as_str(), error = %e, "failed to start VM");
112                        result.failed += 1;
113                        result.errors.push(format!("start {}: {e}", vm.meta.id));
114                    }
115                }
116            }
117        }
118
119        // 4. Remove orphaned processes + TAPs
120        let needed_ids: Vec<&str> = should_exist.iter().map(|vm| vm.meta.id.as_str()).collect();
121        for vm_id in &actual_processes {
122            if !needed_ids.contains(&vm_id.as_str()) {
123                tracing::info!(vm_id, "stopping orphaned VM");
124                let _ = rt.stop(vm_id);
125                let _ = provision::remove_tap(vm_id);
126                result.deleted += 1;
127            }
128        }
129
130        Ok(result)
131    }
132}
