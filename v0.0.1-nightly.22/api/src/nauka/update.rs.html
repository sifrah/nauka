1//! Self-update command — downloads and replaces the running binary.
2//!
3//! Detects the current channel (nightly, beta, stable) from the embedded
4//! build version and stays on it by default. `--channel` overrides.
5
6use anyhow::{bail, Context, Result};
7use sha2::{Digest, Sha256};
8use std::io::Read;
9
10use nauka_core::ui;
11use nauka_core::version::{backup_current_binary, Channel, Version};
12
13const REPO: &str = "sifrah/nauka";
14const GITHUB_API: &str = "https://api.github.com";
15
16/// Build the `nauka update` clap command.
17pub fn command() -> clap::Command {
18    clap::Command::new("update")
19        .about("Update nauka to the latest version")
20        .arg(
21            clap::Arg::new("channel")
22                .long("channel")
23                .short('c')
24                .help("Target channel (nightly, beta, stable). Defaults to current channel")
25                .value_name("CHANNEL"),
26        )
27        .arg(
28            clap::Arg::new("force")
29                .long("force")
30                .short('f')
31                .help("Force update even if already up to date")
32                .action(clap::ArgAction::SetTrue),
33        )
34        .arg(
35            clap::Arg::new("yes")
36                .long("yes")
37                .short('y')
38                .help("Skip confirmation prompt")
39                .action(clap::ArgAction::SetTrue),
40        )
41}

43/// Run the self-update flow.
44pub async fn run(matches: &clap::ArgMatches) -> Result<()> {
45    let current = Version::current();
46    let force = matches.get_flag("force");
47    let skip_confirm = matches.get_flag("yes");
48
49    // Determine target channel
50    let channel: Channel = match matches.get_one::<String>("channel") {
51        Some(ch) => ch.parse().map_err(|e| anyhow::anyhow!("{e}"))?,
52        None => current.channel,
53    };
54
55    eprintln!("  Current version: {current} ({channel})");
56
57    // Find latest release for channel
58    let (tag, version) = find_latest_release(channel).await?;
59    eprintln!("  Latest {channel}: {version}");
60
61    // Check if update is needed
62    if !force && !version.is_newer_than(&current) && channel == current.channel {
63        eprintln!("  Already up to date.");
64        return Ok(());
65    }
66
67    // Confirm
68    if !skip_confirm {
69        eprint!("  Update {current} -> {version}? [y/N] ");
70        let mut buf = String::new();
71        std::io::stdin().read_line(&mut buf)?;
72        if !buf.trim().eq_ignore_ascii_case("y") {
73            eprintln!("  Aborted.");
74            return Ok(());
75        }
76    }
77
78    // Detect platform
79    let target = detect_target()?;
80    let archive_name = format!("nauka-{tag}-{target}.tar.gz");
81    let archive_url = format!("https://github.com/{REPO}/releases/download/{tag}/{archive_name}");
82    let checksums_url = format!("https://github.com/{REPO}/releases/download/{tag}/SHA256SUMS.txt");
83
84    let client = reqwest::Client::builder()
85        .user_agent("nauka-self-update")
86        .build()?;
87
88    let steps = ui::Steps::new(4);
89
90    // Download archive
91    steps.set(&format!("Downloading {archive_name}"));
92    let archive_bytes = client
93        .get(&archive_url)
94        .send()
95        .await?
96        .error_for_status()
97        .context("failed to download release archive")?
98        .bytes()
99        .await?;
100    steps.inc();
101
102    // Download and verify checksum
103    steps.set("Verifying checksum");
104    let checksums_text = client
105        .get(&checksums_url)
106        .send()
107        .await?
108        .error_for_status()
109        .context("failed to download SHA256SUMS.txt")?
110        .text()
111        .await?;
112
113    let expected_hash = checksums_text
114        .lines()
115        .find(|line| line.contains(&archive_name))
116        .and_then(|line| line.split_whitespace().next())
117        .ok_or_else(|| anyhow::anyhow!("no checksum found for {archive_name}"))?;
118
119    let mut hasher = Sha256::new();
120    hasher.update(&archive_bytes);
121    let actual_hash = format!("{:x}", hasher.finalize());
122
123    if actual_hash != expected_hash {
124        steps.finish_err("Checksum mismatch");
125        bail!("checksum mismatch:\n  expected: {expected_hash}\n  actual:   {actual_hash}");
126    }
127    steps.inc();
128
129    // Extract binary from tarball
130    steps.set("Extracting");
131    let decoder = flate2::read::GzDecoder::new(&archive_bytes[..]);
132    let mut archive = tar::Archive::new(decoder);
133    let mut new_binary = Vec::new();
134
135    for entry in archive.entries()? {
136        let mut entry = entry?;
137        let path = entry.path()?;
138        if path.file_name().and_then(|n| n.to_str()) == Some("nauka") {
139            entry.read_to_end(&mut new_binary)?;
140            break;
141        }
142    }
143
144    if new_binary.is_empty() {
145        steps.finish_err("Binary not found in archive");
146        bail!("archive does not contain 'nauka' binary");
147    }
148
149    // Backup current binary
150    backup_current_binary().map_err(|e| anyhow::anyhow!("{e}"))?;
151    steps.inc();
152
153    // Replace current binary
154    steps.set("Installing");
155    let current_exe = std::env::current_exe()?;
156
157    // On Unix: remove then write (avoids "Text file busy")
158    std::fs::remove_file(&current_exe).context("failed to remove current binary")?;
159    std::fs::write(&current_exe, &new_binary).context("failed to write new binary")?;
160
161    #[cfg(unix)]
162    {
163        use std::os::unix::fs::PermissionsExt;
164        std::fs::set_permissions(&current_exe, std::fs::Permissions::from_mode(0o755))?;
165    }
166    steps.inc();
167
168    steps.finish(&format!("Updated to {version}"));
169
170    Ok(())
171}

173/// Find the latest GitHub release for a channel.
174/// Returns (tag, parsed version).
175async fn find_latest_release(channel: Channel) -> Result<(String, Version)> {
176    let client = reqwest::Client::builder()
177        .user_agent("nauka-self-update")
178        .build()?;
179
180    match channel {
181        Channel::Stable => {
182            // Use the /releases/latest endpoint (returns latest non-prerelease)
183            let url = format!("{GITHUB_API}/repos/{REPO}/releases/latest");
184            let release: serde_json::Value = client
185                .get(&url)
186                .header("Accept", "application/vnd.github+json")
187                .send()
188                .await?
189                .error_for_status()
190                .context("failed to fetch latest stable release")?
191                .json()
192                .await?;
193
194            let tag = release["tag_name"]
195                .as_str()
196                .ok_or_else(|| anyhow::anyhow!("no tag_name in release response"))?
197                .to_string();
198            let version = Version::parse(&tag)?;
199            Ok((tag, version))
200        }
201        _ => {
202            // List recent releases and find the highest version matching channel.
203            // GitHub API ordering is not guaranteed to be by version, so we
204            // parse all matching releases and pick the maximum.
205            let url = format!("{GITHUB_API}/repos/{REPO}/releases?per_page=30");
206            let releases: Vec<serde_json::Value> = client
207                .get(&url)
208                .header("Accept", "application/vnd.github+json")
209                .send()
210                .await?
211                .error_for_status()
212                .context("failed to fetch releases")?
213                .json()
214                .await?;
215
216            let channel_str = channel.as_str();
217            let mut best: Option<(String, Version)> = None;
218            for release in &releases {
219                if let Some(tag) = release["tag_name"].as_str() {
220                    if tag.contains(channel_str) {
221                        if let Ok(version) = Version::parse(tag) {
222                            if version.channel == channel {
223                                let is_newer =
224                                    best.as_ref().map(|(_, v)| &version > v).unwrap_or(true);
225                                if is_newer {
226                                    best = Some((tag.to_string(), version));
227                                }
228                            }
229                        }
230                    }
231                }
232            }
233
234            best.ok_or_else(|| anyhow::anyhow!("no {channel} release found"))
235        }
236    }
237}

239/// Detect the target triple for the current platform.
240fn detect_target() -> Result<String> {
241    let arch = if cfg!(target_arch = "x86_64") {
242        "x86_64"
243    } else if cfg!(target_arch = "aarch64") {
244        "aarch64"
245    } else {
246        bail!("unsupported architecture");
247    };
248
249    let os = if cfg!(target_os = "linux") {
250        "unknown-linux-musl"
251    } else if cfg!(target_os = "macos") {
252        "apple-darwin"
253    } else {
254        bail!("unsupported operating system");
255    };
256
257    Ok(format!("{arch}-{os}"))
258}
